5/// Smoothstep - quintic Hermite interpolation (NOT cubic!)
6///
7/// Formula: 6x^5 - 15x^4 + 10x^3
8///
9/// This is the standard smoothstep used in Perlin noise for smooth transitions.
10/// Java reference: `Mth.smoothstep(double)`
11#[inline]
12#[must_use]
13pub fn smoothstep(x: f64) -> f64 {
14    x * x * x * (x * (x * 6.0 - 15.0) + 10.0)
15}
16
17/// Smoothstep derivative for noise with derivatives.
18///
19/// Formula: 30x^2(x-1)^2
20///
21/// Java reference: `Mth.smoothstepDerivative(double)`
22#[inline]
23#[must_use]
24pub fn smoothstep_derivative(x: f64) -> f64 {
25    30.0 * x * x * (x - 1.0) * (x - 1.0)
26}

28/// Floor function that matches Java behavior.
29///
30/// In Java, `(int)v` truncates toward zero, but we need floor behavior.
31/// For negative values, we need to subtract 1 if there's a fractional part.
32///
33/// Java reference: `Mth.floor(double)`
34#[inline]
35#[must_use]
36pub fn floor(v: f64) -> i32 {
37    let i = v as i32;
38    if v < f64::from(i) { i - 1 } else { i }
39}
40
41/// Long floor function matching Java behavior.
42///
43/// Java reference: `Mth.lfloor(double)`
44#[inline]
45#[must_use]
46pub fn lfloor(v: f64) -> i64 {
47    let i = v as i64;
48    if v < i as f64 { i - 1 } else { i }
49}

51/// Linear interpolation.
52///
53/// Formula: a + alpha * (b - a)
54///
55/// Java reference: `Mth.lerp(double, double, double)`
56#[inline]
57#[must_use]
58pub fn lerp(alpha: f64, a: f64, b: f64) -> f64 {
59    a + alpha * (b - a)
60}
61
62/// Bilinear interpolation.
63///
64/// Interpolates between 4 values in a 2D grid.
65///
66/// Java reference: `Mth.lerp2(double, double, double, double, double, double)`
67#[inline]
68#[must_use]
69pub fn lerp2(a1: f64, a2: f64, x00: f64, x10: f64, x01: f64, x11: f64) -> f64 {
70    lerp(a2, lerp(a1, x00, x10), lerp(a1, x01, x11))
71}
72
73/// Trilinear interpolation.
74///
75/// Interpolates between 8 values in a 3D grid.
76///
77/// Java reference: `Mth.lerp3(...)`
78#[inline]
79#[must_use]
80#[expect(
81    clippy::too_many_arguments,
82    reason = "matches vanilla's Mth.lerp3 signature with 8 grid corner values"
83)]
84pub fn lerp3(
85    a1: f64,
86    a2: f64,
87    a3: f64,
88    x000: f64,
89    x100: f64,
90    x010: f64,
91    x110: f64,
92    x001: f64,
93    x101: f64,
94    x011: f64,
95    x111: f64,
96) -> f64 {
97    lerp(
98        a3,
99        lerp2(a1, a2, x000, x100, x010, x110),
100        lerp2(a1, a2, x001, x101, x011, x111),
101    )
102}

104/// Clamped linear interpolation.
105///
106/// Clamps the interpolation factor to [0, 1] before interpolating.
107///
108/// Java reference: `Mth.clampedLerp(double, double, double)`.
109/// Note: Vanilla's parameter order is `(factor, min, max)`, ours is `(min, max, factor)`.
110#[inline]
111#[must_use]
112pub fn clamped_lerp(min: f64, max: f64, factor: f64) -> f64 {
113    if factor < 0.0 {
114        min
115    } else if factor > 1.0 {
116        max
117    } else {
118        lerp(factor, min, max)
119    }
120}

122/// Clamp a value to the range [min, max].
123///
124/// Java reference: `Mth.clamp(double, double, double)`
125#[inline]
126#[must_use]
127pub fn clamp(value: f64, min: f64, max: f64) -> f64 {
128    if value < min {
129        min
130    } else if value > max {
131        max
132    } else {
133        value
134    }
135}

137/// Clamp a value to the range [min, max] (i32 version).
138#[inline]

140pub const fn clamp_i32(value: i32, min: i32, max: i32) -> i32 {
141    if value < min {
142        min
143    } else if value > max {
144        max
145    } else {
146        value

150/// Map a value from one range to another with clamped lerp.
151///
152/// Used for Y-clamped gradients in density functions.
153#[inline]
154#[must_use]
155pub fn map_clamped(value: f64, from_min: f64, from_max: f64, to_min: f64, to_max: f64) -> f64 {
156    let t = (value - from_min) / (from_max - from_min);
157    clamped_lerp(to_min, to_max, t)
158}
159
160/// Inverse linear interpolation (find the factor t such that lerp(t, a, b) == value).
161///
162/// Java reference: `Mth.inverseLerp(double, double, double)`
163#[inline]
164#[must_use]
165pub fn inverse_lerp(value: f64, a: f64, b: f64) -> f64 {
166    (value - a) / (b - a)
167}
168
169/// Map a value from one range to another (unclamped).
170///
171/// Unlike [`map_clamped`], the result can extrapolate outside `[to_min, to_max]`.
172///
173/// Java reference: `Mth.map(double, double, double, double, double)`
174#[inline]
175#[must_use]
176pub fn map(value: f64, from_min: f64, from_max: f64, to_min: f64, to_max: f64) -> f64 {
177    lerp(inverse_lerp(value, from_min, from_max), to_min, to_max)
178}

194/// Bias a noise value towards extremes (-1 or 1) using a sine curve.
195///
196/// Java reference: `NoiseUtils.biasTowardsExtreme(double, double)`
197#[inline]
198#[must_use]
199pub fn bias_towards_extreme(noise: f64, factor: f64) -> f64 {
200    noise + (PI * noise).sin() * factor / PI
201}
