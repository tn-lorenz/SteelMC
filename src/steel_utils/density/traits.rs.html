14/// Noise settings for a dimension, parsed from the datapack.
15///
16/// These are compile-time constants generated from `noise_settings` JSON files.
17pub trait NoiseSettings: Send + Sync {
18    /// Minimum Y coordinate for this dimension.
19    const MIN_Y: i32;
20    /// Total height of the world in blocks.
21    const HEIGHT: i32;
22    /// Sea level Y coordinate.
23    const SEA_LEVEL: i32;
24    /// Cell width in blocks (XZ direction).
25    const CELL_WIDTH: i32;
26    /// Cell height in blocks (Y direction).
27    const CELL_HEIGHT: i32;
28    /// Whether aquifers are enabled for this dimension.
29    const AQUIFERS_ENABLED: bool;
30    /// Whether ore veins are enabled for this dimension.
31    const ORE_VEINS_ENABLED: bool;
32    /// Whether this dimension uses Java's LCG random (true) or Xoroshiro (false).
33    const LEGACY_RANDOM_SOURCE: bool;
34
35    /// Get the default block state ID for this dimension.
36    fn default_block_id() -> BlockStateId;
37
38    /// Get the default fluid state ID for this dimension.
39    fn default_fluid_id() -> BlockStateId;
40}

42/// Column cache for a dimension's flat-cached density function results.
43///
44/// Stores Y-independent values that only need to be computed once per (x, z) column.
45pub trait ColumnCache: Clone + Default + Send + Sync {
46    /// The associated noises type for this cache.
47    type Noises: DimensionNoises<ColumnCache = Self>;
48
49    /// Ensure the cache is populated for the given block coordinates.
50    ///
51    /// If the cache already holds values for this column, this is a no-op.
52    fn ensure(&mut self, x: i32, z: i32, noises: &Self::Noises);
53
54    /// Pre-compute flat-cached values for all quart positions in a chunk.
55    ///
56    /// Matches vanilla's `NoiseChunk.FlatCache`: eagerly fills a 2D grid of
57    /// `(quart_size+1)²` entries (size baked in per dimension at compile time).
58    /// After this call, `ensure()` for in-bounds positions is an O(1) grid
59    /// lookup. Out-of-bounds positions fall back to on-the-fly evaluation at
60    /// raw (non-quantized) coordinates.
61    fn init_grid(&mut self, chunk_block_x: i32, chunk_block_z: i32, noises: &Self::Noises);
62}

64/// All noise generators and density functions for a dimension.
65///
66/// This trait abstracts over dimension-specific noise types (`OverworldNoises`,
67/// `NetherNoises`, etc.) allowing generic code to work with any dimension.
68pub trait DimensionNoises: Sized + Send + Sync {
69    /// The column cache type for this dimension.
70    type ColumnCache: ColumnCache<Noises = Self>;
71    /// The noise settings type for this dimension.
72    type Settings: NoiseSettings;
73
74    /// Create all noise generators from a world seed and its positional splitter.
75    fn create(
76        seed: u64,
77        splitter: &RandomSplitter,
78        params: &FxHashMap<String, NoiseParameters>,
79    ) -> Self;
80
81    // ── Router functions ────────────────────────────────────────────────────
82
83    /// Final density for terrain generation (positive = solid, negative = air).
84    fn router_final_density(&self, cache: &mut Self::ColumnCache, x: i32, y: i32, z: i32) -> f64;
85
86    /// Depth from surface (used for terrain shaping).
87    fn router_depth(&self, cache: &mut Self::ColumnCache, x: i32, y: i32, z: i32) -> f64;
88
89    // ── Aquifer router functions ────────────────────────────────────────────
90
91    /// Barrier noise for aquifer boundaries.
92    fn router_barrier(&self, cache: &mut Self::ColumnCache, x: i32, y: i32, z: i32) -> f64;
93
94    /// Fluid level floodedness for aquifers.
95    fn router_fluid_level_floodedness(
96        &self,
97        cache: &mut Self::ColumnCache,
98        x: i32,
99        y: i32,
100        z: i32,
101    ) -> f64;
102
103    /// Fluid level spread for aquifers.
104    fn router_fluid_level_spread(
105        &self,
106        cache: &mut Self::ColumnCache,
107        x: i32,
108        y: i32,
109        z: i32,
110    ) -> f64;
111
112    /// Lava placement noise for aquifers.
113    fn router_lava(&self, cache: &mut Self::ColumnCache, x: i32, y: i32, z: i32) -> f64;
114
115    // ── Ore vein router functions ───────────────────────────────────────────
116
117    /// Vein toggle (sign determines copper vs iron).
118    fn router_vein_toggle(&self, cache: &mut Self::ColumnCache, x: i32, y: i32, z: i32) -> f64;
119
120    /// Vein ridged noise for ore placement.
121    fn router_vein_ridged(&self, cache: &mut Self::ColumnCache, x: i32, y: i32, z: i32) -> f64;
122
123    /// Vein gap noise for ore vs filler placement.
124    fn router_vein_gap(&self, cache: &mut Self::ColumnCache, x: i32, y: i32, z: i32) -> f64;
125
126    // ── Climate/biome router functions (Y-independent, cached) ──────────────
127
128    /// Erosion value (cached in column cache).
129    fn router_erosion(&self, cache: &mut Self::ColumnCache, x: i32, y: i32, z: i32) -> f64;
130
131    /// Continentalness value (cached in column cache).
132    fn router_continentalness(&self, cache: &mut Self::ColumnCache, x: i32, y: i32, z: i32) -> f64;
133
134    /// Temperature value (cached in column cache).
135    fn router_temperature(&self, cache: &mut Self::ColumnCache, x: i32, y: i32, z: i32) -> f64;
136
137    /// Vegetation/humidity value (cached in column cache).
138    fn router_vegetation(&self, cache: &mut Self::ColumnCache, x: i32, y: i32, z: i32) -> f64;
139
140    /// Ridges/weirdness value (cached in column cache).
141    fn router_ridges(&self, cache: &mut Self::ColumnCache, x: i32, y: i32, z: i32) -> f64;
142
143    /// Preliminary surface level (cached in column cache).
144    fn router_preliminary_surface_level(
145        &self,
146        cache: &mut Self::ColumnCache,
147        x: i32,
148        y: i32,
149        z: i32,
150    ) -> f64;
151
152    // ── Interpolation functions ─────────────────────────────────────────────
153
154    /// Total number of independently interpolated channels across all router
155    /// entries (`final_density` + `vein_toggle` + `vein_ridged`).
156    fn interpolated_count() -> usize;
157
158    /// Whether vein functions have interpolation channels.
159    fn vein_interp_enabled() -> bool;
160
161    /// Evaluate the inner functions of all `Interpolated` markers at a cell corner.
162    ///
163    /// `out` must have length [`interpolated_count()`]. Each element receives
164    /// the value of one `Interpolated` marker's inner function at `(x, y, z)`.
165    fn fill_cell_corner_densities(
166        &self,
167        cache: &mut Self::ColumnCache,
168        x: i32,
169        y: i32,
170        z: i32,
171        out: &mut [f64],
172    );
173
174    /// Combine trilinearly interpolated values for `final_density`.
175    fn combine_interpolated(
176        &self,
177        cache: &mut Self::ColumnCache,
178        interpolated: &[f64],
179        x: i32,
180        y: i32,
181        z: i32,
182    ) -> f64;
183
184    /// Combine trilinearly interpolated values for `vein_toggle`.
185    fn combine_vein_toggle(
186        &self,
187        cache: &mut Self::ColumnCache,
188        interpolated: &[f64],
189        x: i32,
190        y: i32,
191        z: i32,
192    ) -> f64;
193
194    /// Combine trilinearly interpolated values for `vein_ridged`.
195    fn combine_vein_ridged(
196        &self,
197        cache: &mut Self::ColumnCache,
198        interpolated: &[f64],
199        x: i32,
200        y: i32,
201        z: i32,
202    ) -> f64;
203
204    // ── Surface rules ───────────────────────────────────────────────────────
205
206    /// Noise IDs referenced by this dimension's surface rule `NoiseThreshold`
207    /// conditions. Used to construct the `SurfaceSystem`'s condition noises.
208    fn surface_noise_ids() -> &'static [&'static str];
209
210    /// Apply the transpiled surface rule at the given context position.
211    fn try_apply_surface_rule(ctx: &SurfaceRuleContext<'_>) -> Option<BlockStateId>;
212}
