5/// A variable-length long integer.
6#[derive(Debug, Copy, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
7pub struct VarLong(pub i64);
8
9impl VarLong {
10    /// The maximum number of bytes a `VarLong` can be.
11    pub const MAX_SIZE: usize = 10;
12}
13
14#[allow(missing_docs)]
15impl ReadFrom for VarLong {
16    fn read(read: &mut Cursor<&[u8]>) -> Result<Self, Error> {
17        let mut val = 0i64;
18        for i in 0..Self::MAX_SIZE {
19            let byte = u8::read(read)?;
20            val |= (i64::from(byte) & 0x7F) << (i * 7);
21            if byte & 0x80 == 0 {
22                return Ok(Self(val));
23            }
24        }
25        Err(Error::other("VarLong too long"))
26    }
27}
28
29#[allow(missing_docs)]
30impl WriteTo for VarLong {
31    fn write(&self, writer: &mut impl Write) -> Result<(), Error> {
32        let mut val = self.0 as u64;
33        loop {
34            let b: u8 = val as u8 & 0x7F;
35            val >>= 7;
36            if val == 0 {
37                b.write(writer)?;
38                break;
39            }
40            (b | 0x80).write(writer)?;
41        }
42        Ok(())
43    }
44}

46#[allow(missing_docs)]
47impl From<i64> for VarLong {
48    fn from(value: i64) -> Self {
49        Self(value)
50    }
51}
52
53#[allow(missing_docs)]
54impl From<VarLong> for i64 {
55    fn from(value: VarLong) -> i64 {
56        value.0
57    }
58}

65    #[test]
66    fn test_varlong_read_write() {
67        let test_values = vec![
68            0i64,
69            1i64,
70            127i64,
71            128i64,
72            255i64,
73            2_147_483_647_i64,
74            9_223_372_036_854_775_807_i64,
75            -1i64,
76            -2_147_483_648_i64,
77        ];
78
79        for val in test_values {
80            let var_long = VarLong(val);
81            let mut buf = Vec::new();
82            var_long.write(&mut buf).expect("write failed");
83
84            let mut cursor = Cursor::new(buf.as_slice());
85            let read_val = VarLong::read(&mut cursor).expect("read failed");
86            assert_eq!(read_val, var_long, "Failed for value {val}");
87        }
