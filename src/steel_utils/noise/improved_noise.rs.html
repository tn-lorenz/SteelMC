9/// Improved Perlin noise generator.
10///
11/// This implements the improved Perlin noise algorithm as used in Minecraft.
12/// Each instance has a permutation table and offset values initialized from
13/// a random source.
14#[derive(Debug, Clone)]
15pub struct ImprovedNoise {
16    /// Permutation table (256 bytes)
17    p: [u8; 256],
18    /// X offset for the noise coordinates
19    pub xo: f64,
20    /// Y offset for the noise coordinates
21    pub yo: f64,
22    /// Z offset for the noise coordinates
23    pub zo: f64,
24}
25
26impl ImprovedNoise {
27    /// Creates a new `ImprovedNoise` from a random source.
28    ///
29    /// Initializes the permutation table using Fisher-Yates shuffle
30    /// and sets random offsets.
31    pub fn new<R: Random>(random: &mut R) -> Self {
32        let xo = random.next_f64() * 256.0;
33        let yo = random.next_f64() * 256.0;
34        let zo = random.next_f64() * 256.0;
35
36        let mut p = [0u8; 256];
37        #[expect(
38            clippy::needless_range_loop,
39            reason = "index is used as the initial permutation value"
40        )]
41        for i in 0..256 {
42            p[i] = i as u8;
43        }
44
45        // Fisher-Yates shuffle matching vanilla's implementation
46        for i in 0..256 {
47            let offset = random.next_i32_bounded((256 - i) as i32) as usize;
48            p.swap(i, i + offset);
49        }
50
51        Self { p, xo, yo, zo }
52    }
53
54    /// Sample noise at the given coordinates.
55    ///
56    /// This is the standard 3D Perlin noise sampling without Y scaling.
57    #[inline]
58    #[must_use]
59    pub fn noise(&self, x: f64, y: f64, z: f64) -> f64 {
60        self.noise_with_y_scale(x, y, z, 0.0, 0.0)
61    }
62
63    /// Sample noise at the given coordinates, accumulating partial derivatives.
64    ///
65    /// Returns the noise value and adds the partial derivatives (dx, dy, dz)
66    /// into `derivative_out`. Used by `BlendedNoise` for terrain generation.
67    #[must_use]
68    pub fn noise_with_derivative(
69        &self,
70        x: f64,
71        y: f64,
72        z: f64,
73        derivative_out: &mut [f64; 3],
74    ) -> f64 {
75        let x = x + self.xo;
76        let y = y + self.yo;
77        let z = z + self.zo;
78
79        let xf = floor(x);
80        let yf = floor(y);
81        let zf = floor(z);
82
83        let xr = x - f64::from(xf);
84        let yr = y - f64::from(yf);
85        let zr = z - f64::from(zf);
86
87        self.sample_with_derivative(xf, yf, zf, xr, yr, zr, derivative_out)
88    }
89
90    /// Sample noise with Y scale and fudge parameters.
91    ///
92    /// The `y_scale` and `y_fudge` parameters are used for terrain generation
93    /// where vertical noise needs special handling.
94    ///
95    /// # Arguments
96    /// * `x`, `y`, `z` - The coordinates to sample
97    /// * `y_scale` - Y scaling factor (0.0 to disable)
98    /// * `y_fudge` - Y fudge factor for floor snapping
99    #[must_use]
100    #[expect(
101        clippy::similar_names,
102        reason = "yr_fudge and y_fudge match vanilla naming"
103    )]
104    pub fn noise_with_y_scale(&self, x: f64, y: f64, z: f64, y_scale: f64, y_fudge: f64) -> f64 {
105        let x = x + self.xo;
106        let y = y + self.yo;
107        let z = z + self.zo;
108
109        let xf = floor(x);
110        let yf = floor(y);
111        let zf = floor(z);
112
113        let xr = x - f64::from(xf);
114        let yr = y - f64::from(yf);
115        let zr = z - f64::from(zf);
116
117        // Calculate Y fudge for terrain generation
118        #[expect(
119            clippy::if_not_else,
120            reason = "matches vanilla's conditional structure"
121        )]
122        let yr_fudge = if y_scale != 0.0 {
123            let fudge_limit = if y_fudge >= 0.0 && y_fudge < yr {
124                y_fudge
125            } else {
126                yr
127            };
128            // SHIFT_UP_EPSILON = 1.0E-7F in Java (float literal promoted to double)
129            (fudge_limit / y_scale + f64::from(1.0e-7_f32)).floor() * y_scale
130        } else {
131            0.0
132        };
133
134        self.sample_and_lerp(xf, yf, zf, xr, yr - yr_fudge, zr, yr)
135    }

137    /// Look up the permutation value at index x.
138    #[inline]
139    const fn p(&self, x: i32) -> usize {
140        self.p[(x & 255) as usize] as usize
141    }

143    /// Sample noise at grid point and interpolate.
144    #[expect(clippy::too_many_arguments, reason = "matches vanilla signature")]
145    fn sample_and_lerp(
146        &self,
147        x: i32,
148        y: i32,
149        z: i32,
150        xr: f64,
151        yr: f64,
152        zr: f64,
153        yr_original: f64,
154    ) -> f64 {
155        // Get permutation indices for the 8 corners
156        let x0 = self.p(x);
157        let x1 = self.p(x + 1);
158        let xy00 = self.p(x0 as i32 + y);
159        let xy01 = self.p(x0 as i32 + y + 1);
160        let xy10 = self.p(x1 as i32 + y);
161        let xy11 = self.p(x1 as i32 + y + 1);
162
163        // Calculate gradient dot products at each corner
164        let d000 = grad_dot(self.p(xy00 as i32 + z), xr, yr, zr);
165        let d100 = grad_dot(self.p(xy10 as i32 + z), xr - 1.0, yr, zr);
166        let d010 = grad_dot(self.p(xy01 as i32 + z), xr, yr - 1.0, zr);
167        let d110 = grad_dot(self.p(xy11 as i32 + z), xr - 1.0, yr - 1.0, zr);
168        let d001 = grad_dot(self.p(xy00 as i32 + z + 1), xr, yr, zr - 1.0);
169        let d101 = grad_dot(self.p(xy10 as i32 + z + 1), xr - 1.0, yr, zr - 1.0);
170        let d011 = grad_dot(self.p(xy01 as i32 + z + 1), xr, yr - 1.0, zr - 1.0);
171        let d111 = grad_dot(self.p(xy11 as i32 + z + 1), xr - 1.0, yr - 1.0, zr - 1.0);
172
173        // Apply smoothstep interpolation
174        let x_alpha = smoothstep(xr);
175        let y_alpha = smoothstep(yr_original);
176        let z_alpha = smoothstep(zr);
177
178        lerp3(
179            x_alpha, y_alpha, z_alpha, d000, d100, d010, d110, d001, d101, d011, d111,
180        )
181    }

183    /// Sample noise at grid point, interpolate, and accumulate derivatives.
184    #[expect(clippy::too_many_arguments, reason = "matches vanilla signature")]
185    fn sample_with_derivative(
186        &self,
187        x: i32,
188        y: i32,
189        z: i32,
190        xr: f64,
191        yr: f64,
192        zr: f64,
193        derivative_out: &mut [f64; 3],
194    ) -> f64 {
195        let x0 = self.p(x);
196        let x1 = self.p(x + 1);
197        let xy00 = self.p(x0 as i32 + y);
198        let xy01 = self.p(x0 as i32 + y + 1);
199        let xy10 = self.p(x1 as i32 + y);
200        let xy11 = self.p(x1 as i32 + y + 1);
201
202        // Get hashes and gradient vectors for all 8 corners
203        let h000 = self.p(xy00 as i32 + z);
204        let h100 = self.p(xy10 as i32 + z);
205        let h010 = self.p(xy01 as i32 + z);
206        let h110 = self.p(xy11 as i32 + z);
207        let h001 = self.p(xy00 as i32 + z + 1);
208        let h101 = self.p(xy10 as i32 + z + 1);
209        let h011 = self.p(xy01 as i32 + z + 1);
210        let h111 = self.p(xy11 as i32 + z + 1);
211
212        let g000 = &GRADIENT[h000 & 15];
213        let g100 = &GRADIENT[h100 & 15];
214        let g010 = &GRADIENT[h010 & 15];
215        let g110 = &GRADIENT[h110 & 15];
216        let g001 = &GRADIENT[h001 & 15];
217        let g101 = &GRADIENT[h101 & 15];
218        let g011 = &GRADIENT[h011 & 15];
219        let g111 = &GRADIENT[h111 & 15];
220
221        // Gradient dot products at each corner
222        let d000 = grad_dot(h000, xr, yr, zr);
223        let d100 = grad_dot(h100, xr - 1.0, yr, zr);
224        let d010 = grad_dot(h010, xr, yr - 1.0, zr);
225        let d110 = grad_dot(h110, xr - 1.0, yr - 1.0, zr);
226        let d001 = grad_dot(h001, xr, yr, zr - 1.0);
227        let d101 = grad_dot(h101, xr - 1.0, yr, zr - 1.0);
228        let d011 = grad_dot(h011, xr, yr - 1.0, zr - 1.0);
229        let d111 = grad_dot(h111, xr - 1.0, yr - 1.0, zr - 1.0);
230
231        let x_alpha = smoothstep(xr);
232        let y_alpha = smoothstep(yr);
233        let z_alpha = smoothstep(zr);
234
235        // Interpolate gradient components for direct derivative contribution
236        let d1x = lerp3(
237            x_alpha, y_alpha, z_alpha, g000[0], g100[0], g010[0], g110[0], g001[0], g101[0],
238            g011[0], g111[0],
239        );
240        let d1y = lerp3(
241            x_alpha, y_alpha, z_alpha, g000[1], g100[1], g010[1], g110[1], g001[1], g101[1],
242            g011[1], g111[1],
243        );
244        let d1z = lerp3(
245            x_alpha, y_alpha, z_alpha, g000[2], g100[2], g010[2], g110[2], g001[2], g101[2],
246            g011[2], g111[2],
247        );
248
249        // Smoothstep correction terms via differences
250        let d2x = lerp2(
251            y_alpha,
252            z_alpha,
253            d100 - d000,
254            d110 - d010,
255            d101 - d001,
256            d111 - d011,
257        );
258        let d2y = lerp2(
259            z_alpha,
260            x_alpha,
261            d010 - d000,
262            d011 - d001,
263            d110 - d100,
264            d111 - d101,
265        );
266        let d2z = lerp2(
267            x_alpha,
268            y_alpha,
269            d001 - d000,
270            d101 - d100,
271            d011 - d010,
272            d111 - d110,
273        );
274
275        let x_sd = smoothstep_derivative(xr);
276        let y_sd = smoothstep_derivative(yr);
277        let z_sd = smoothstep_derivative(zr);
278
279        // Accumulate derivatives (vanilla uses +=)
280        derivative_out[0] += d1x + x_sd * d2x;
281        derivative_out[1] += d1y + y_sd * d2y;
282        derivative_out[2] += d1z + z_sd * d2z;
283
284        lerp3(
285            x_alpha, y_alpha, z_alpha, d000, d100, d010, d110, d001, d101, d011, d111,
286        )
287    }
288}

290/// Calculate the dot product of a gradient vector and the position vector.
291#[inline]
292fn grad_dot(hash: usize, x: f64, y: f64, z: f64) -> f64 {
293    let g = &GRADIENT[hash & 15];
294    g[0] * x + g[1] * y + g[2] * z
295}

302    #[test]
303    fn test_improved_noise_deterministic() {
304        let mut rng1 = Xoroshiro::from_seed(12345);
305        let mut rng2 = Xoroshiro::from_seed(12345);
306
307        let noise1 = ImprovedNoise::new(&mut rng1);
308        let noise2 = ImprovedNoise::new(&mut rng2);
309
310        // Same seed should produce same noise
311        #[expect(
312            clippy::float_cmp,
313            reason = "determinism test: identical seeds must produce bit-identical offsets"
314        )]
315        {
316            assert_eq!(noise1.xo, noise2.xo);
317            assert_eq!(noise1.yo, noise2.yo);
318            assert_eq!(noise1.zo, noise2.zo);
319        }
320        assert_eq!(noise1.p, noise2.p);
321
322        // Same coordinates should produce same values
323        let v1 = noise1.noise(100.0, 64.0, 100.0);
324        let v2 = noise2.noise(100.0, 64.0, 100.0);
325        assert!((v1 - v2).abs() < 1e-15);
326    }
