1//! `BlendedNoise` implementation matching vanilla Minecraft's `BlendedNoise.java`
2//!
3//! Combines three `PerlinNoise` instances (min limit, max limit, main) for terrain generation.
4//! The main noise determines the blend factor between the min and max limit noises.
5
6use crate::math::clamped_lerp;
7use crate::noise::PerlinNoise;
8use crate::noise::perlin_noise::wrap;
9use crate::random::RandomSource;
10
11/// Base frequency multiplier for all `BlendedNoise` coordinate transforms.
12const COORDINATE_SCALE: f64 = 684.412;
13
14/// Runtime `BlendedNoise` sampler with three seeded `PerlinNoise` instances.
15///
16/// Matches vanilla's `BlendedNoise` density function.
17#[derive(Debug, Clone)]
18pub struct BlendedNoise {
19    min_limit_noise: PerlinNoise,
20    max_limit_noise: PerlinNoise,
21    main_noise: PerlinNoise,
22    xz_multiplier: f64,
23    y_multiplier: f64,
24    xz_factor: f64,
25    y_factor: f64,
26    smear_scale_multiplier: f64,
27    max_value: f64,
28}
29
30impl BlendedNoise {
31    /// Create a new `BlendedNoise` from a random source and scale parameters.
32    ///
33    /// This matches vanilla's `BlendedNoise(RandomSource, ...)` constructor which
34    /// uses the legacy initialization path with `createLegacyForBlendedNoise`.
35    #[must_use]
36    pub fn new(
37        random: &mut RandomSource,
38        xz_scale: f64,
39        y_scale: f64,
40        xz_factor: f64,
41        y_factor: f64,
42        smear_scale_multiplier: f64,
43    ) -> Self {
44        // min/max limit: 16 octaves (-15 to 0), main: 8 octaves (-7 to 0)
45        let min_limit_noise = PerlinNoise::create_legacy_for_nether(random, -15, &[1.0; 16]);
46        let max_limit_noise = PerlinNoise::create_legacy_for_nether(random, -15, &[1.0; 16]);
47        let main_noise = PerlinNoise::create_legacy_for_nether(random, -7, &[1.0; 8]);
48
49        let xz_multiplier = COORDINATE_SCALE * xz_scale;
50        let y_multiplier = COORDINATE_SCALE * y_scale;
51        let max_value = min_limit_noise.max_broken_value(y_multiplier);
52
53        Self {
54            min_limit_noise,
55            max_limit_noise,
56            main_noise,
57            xz_multiplier,
58            y_multiplier,
59            xz_factor,
60            y_factor,
61            smear_scale_multiplier,
62            max_value,
63        }
64    }
65
66    /// Compute the blended noise value at the given block coordinates.
67    ///
68    /// This is the core terrain density computation. It:
69    /// 1. Samples the main noise (8 octaves) to get a blend factor
70    /// 2. Conditionally samples min/max limit noises (16 octaves each)
71    /// 3. Interpolates between min and max based on the blend factor
72    #[must_use]
73    pub fn compute(&self, block_x: i32, block_y: i32, block_z: i32) -> f64 {
74        let limit_x = f64::from(block_x) * self.xz_multiplier;
75        let limit_y = f64::from(block_y) * self.y_multiplier;
76        let limit_z = f64::from(block_z) * self.xz_multiplier;
77        let main_x = limit_x / self.xz_factor;
78        let main_y = limit_y / self.y_factor;
79        let main_z = limit_z / self.xz_factor;
80        let limit_smear = self.y_multiplier * self.smear_scale_multiplier;
81        let main_smear = limit_smear / self.y_factor;
82
83        // Sample main noise (8 octaves, highest frequency first)
84        let mut main_noise_value = 0.0;
85        let mut pow = 1.0;
86        for i in 0..8 {
87            if let Some(noise) = self.main_noise.get_octave_noise(i) {
88                main_noise_value += noise.noise_with_y_scale(
89                    wrap(main_x * pow),
90                    wrap(main_y * pow),
91                    wrap(main_z * pow),
92                    main_smear * pow,
93                    main_y * pow,
94                ) / pow;
95            }
96            pow /= 2.0;
97        }
98
99        // Determine blend factor and which limit noises to sample
100        let factor = f64::midpoint(main_noise_value / 10.0, 1.0);
101        let is_max = factor >= 1.0;
102        let is_min = factor <= 0.0;
103
104        // Sample limit noises (16 octaves each, highest frequency first)
105        let mut blend_min = 0.0;
106        let mut blend_max = 0.0;
107        pow = 1.0;
108        for i in 0..16 {
109            let wx = wrap(limit_x * pow);
110            let wy = wrap(limit_y * pow);
111            let wz = wrap(limit_z * pow);
112            let y_scale_pow = limit_smear * pow;
113
114            if !is_max && let Some(noise) = self.min_limit_noise.get_octave_noise(i) {
115                blend_min += noise.noise_with_y_scale(wx, wy, wz, y_scale_pow, limit_y * pow) / pow;
116            }
117
118            if !is_min && let Some(noise) = self.max_limit_noise.get_octave_noise(i) {
119                blend_max += noise.noise_with_y_scale(wx, wy, wz, y_scale_pow, limit_y * pow) / pow;
120            }
121
122            pow /= 2.0;
123        }
124
125        clamped_lerp(blend_min / 512.0, blend_max / 512.0, factor) / 128.0
126    }
127
128    /// Maximum possible output value.
129    #[inline]
130    #[must_use]
131    pub const fn max_value(&self) -> f64 {
132        self.max_value
133    }
134
135    /// Minimum possible output value (negative of max).
136    #[inline]
137    #[must_use]
138    pub fn min_value(&self) -> f64 {
139        -self.max_value
140    }
141}

143#[cfg(test)]
144mod tests {
145    use super::*;
146    use crate::random::xoroshiro::Xoroshiro;
147
148    fn make_source(seed: u64) -> RandomSource {
149        RandomSource::Xoroshiro(Xoroshiro::from_seed(seed))
150    }
151
152    #[test]
153    fn test_blended_noise_deterministic() {
154        let bn1 = BlendedNoise::new(&mut make_source(12345), 1.0, 1.0, 80.0, 160.0, 8.0);
155        let bn2 = BlendedNoise::new(&mut make_source(12345), 1.0, 1.0, 80.0, 160.0, 8.0);
156
157        let v1 = bn1.compute(0, 64, 0);
158        let v2 = bn2.compute(0, 64, 0);
159        assert!(
160            (v1 - v2).abs() < 1e-15,
161            "BlendedNoise not deterministic: {v1} vs {v2}",
162        );
163    }
164
165    #[test]
166    fn test_blended_noise_spatial_variation() {
167        let bn = BlendedNoise::new(&mut make_source(42), 1.0, 1.0, 80.0, 160.0, 8.0);
168
169        let values: Vec<f64> = (-5..5).map(|x| bn.compute(x * 16, 64, 0)).collect();
170
171        let min = values.iter().copied().fold(f64::INFINITY, f64::min);
172        let max = values.iter().copied().fold(f64::NEG_INFINITY, f64::max);
173        assert!(
174            max - min > 1e-6,
175            "BlendedNoise should have variation: {values:?}"
176        );
177    }
178
179    #[test]
180    fn test_blended_noise_range() {
181        let bn = BlendedNoise::new(&mut make_source(42), 1.0, 1.0, 80.0, 160.0, 8.0);
182
183        for x in -10..10 {
184            for y in (-4..20).step_by(4) {
185                let v = bn.compute(x * 16, y * 4, x * 16);
186                assert!(
187                    v.abs() <= bn.max_value() + 0.01,
188                    "BlendedNoise value {v} exceeds max {} at ({}, {}, {})",
189                    bn.max_value(),
190                    x * 16,
191                    y * 4,
192                    x * 16,
193                );
194            }
195        }
196    }
