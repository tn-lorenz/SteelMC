26/// A mutex wrapper that allows debug checks to prevent deadlocks.
27///
28/// Use [`SteelMutex::new_sync`] for synchronous contexts (`parking_lot::Mutex`)
29/// or [`SteelMutex::new_async`] for async contexts (`tokio::sync::Mutex`).
30pub struct SteelMutex<Mutex: GenericLock<T>, T: ?Sized + Send + Sync> {
31    mutex: Mutex,
32    _marker: PhantomData<T>,
33}
34
35impl<T: Sized + Send + Sync> SteelMutex<parking_lot::Mutex<T>, T> {
36    /// Creates a new synchronous mutex backed by `parking_lot::Mutex`.
37    pub fn new(data: T) -> Self {
38        Self {
39            mutex: parking_lot::Mutex::new(data),
40            _marker: PhantomData,
41        }
42    }
43
44    /// Acquires the lock, blocking the current thread until it is available.
45    pub fn lock(&self) -> parking_lot::MutexGuard<'_, T> {
46        self.mutex.lock()
47    }
48
49    /// Attempts to acquire the lock without blocking.
50    ///
51    /// Returns `None` if the lock is already held.
52    pub fn try_lock(&self) -> Option<parking_lot::MutexGuard<'_, T>> {
53        self.mutex.try_lock()
54    }
55}
56
57impl<T: Sized + Send + Sync> SteelMutex<tokio::sync::Mutex<T>, T> {
58    /// Creates a new asynchronous mutex backed by `tokio::sync::Mutex`.
59    pub fn new(data: T) -> Self {
60        Self {
61            mutex: tokio::sync::Mutex::new(data),
62            _marker: PhantomData,
63        }
64    }
65
66    /// Acquires the lock, blocking the current thread until it is available.
67    ///
68    /// # Panics
69    /// Panics if called from within an async runtime (use [`lock_async`](Self::lock_async) instead).
70    pub fn lock_blocking(&self) -> tokio::sync::MutexGuard<'_, T> {
71        self.mutex.blocking_lock()
72    }
73
74    /// Acquires the lock asynchronously.
75    pub async fn lock_async(&self) -> tokio::sync::MutexGuard<'_, T> {
76        self.mutex.lock().await
77    }
78}

80/// An `RwLock` wrapper that allows debug checks to prevent deadlocks.
81///
82/// Use [`SteelRwLock::new_sync`] for synchronous contexts (`parking_lot::RwLock`)
83/// or [`SteelRwLock::new_async`] for async contexts (`tokio::sync::RwLock`).
84pub struct SteelRwLock<RwLock: GenericLock<T>, T: ?Sized + Send + Sync> {
85    rwlock: RwLock,
86    _marker: PhantomData<T>,
87}

89impl<T: Sized + Send + Sync> SteelRwLock<parking_lot::RwLock<T>, T> {
90    /// Creates a new synchronous `RwLock` backed by `parking_lot::RwLock`.
91    pub fn new(data: T) -> Self {
92        Self {
93            rwlock: parking_lot::RwLock::new(data),
94            _marker: PhantomData,
95        }
96    }
97
98    /// Acquires a read lock, blocking until available.
99    ///
100    /// Multiple readers can hold the lock simultaneously.
101    pub fn read(&self) -> parking_lot::RwLockReadGuard<'_, T> {
102        self.rwlock.read()
103    }
104
105    /// Attempts to acquire a read lock without blocking.
106    ///
107    /// Returns `None` if a write lock is currently held.
108    pub fn try_read(&self) -> Option<parking_lot::RwLockReadGuard<'_, T>> {
109        self.rwlock.try_read()
110    }
111
112    /// Acquires a write lock, blocking until available.
113    ///
114    /// Only one writer can hold the lock at a time.
115    pub fn write(&self) -> parking_lot::RwLockWriteGuard<'_, T> {
116        self.rwlock.write()
117    }
118
119    /// Attempts to acquire a write lock without blocking.
120    ///
121    /// Returns `None` if any lock (read or write) is currently held.
122    pub fn try_write(&self) -> Option<parking_lot::RwLockWriteGuard<'_, T>> {
123        self.rwlock.try_write()
124    }
125}
