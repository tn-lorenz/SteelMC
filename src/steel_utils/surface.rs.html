1//! Surface rule context types used by both generated code and runtime.

3use crate::BlockStateId;
4use crate::random::name_hash::NameHash;

6/// Context data passed to transpiled surface rule functions.
7///
8/// This is a flat struct holding all the values a surface rule condition might need.
9/// The `SurfaceContext` in steel-core populates this and passes it to the generated
10/// `try_apply_surface_rule()` function.
11pub struct SurfaceRuleContext<'a> {
12    /// World X coordinate.
13    pub block_x: i32,
14    /// World Z coordinate.
15    pub block_z: i32,
16    /// Noise-based surface layer thickness (typically 3-6 blocks).
17    pub surface_depth: i32,
18    /// Surface secondary noise value for depth variation.
19    pub surface_secondary: f64,
20    /// Minimum surface level from preliminary surface interpolation.
21    pub min_surface_level: i32,
22    /// Whether this column has a steep slope.
23    pub steep: bool,
24    /// World Y coordinate.
25    pub block_y: i32,
26    /// How many solid blocks above the current position.
27    pub stone_depth_above: i32,
28    /// How many solid blocks below until the next cavity.
29    pub stone_depth_below: i32,
30    /// Y of water surface above this block, or `i32::MIN` if no water.
31    pub water_height: i32,
32    /// Numeric biome ID at the current position.
33    pub biome_id: u16,
34    /// Whether the current biome is cold enough to snow at this position.
35    pub cold_enough_to_snow: bool,
36    /// Reference to the surface system for noise lookups and band generation.
37    pub system: &'a dyn SurfaceNoiseProvider,
38}
39
40/// Trait for providing noise values and clay band data to surface rules.
41///
42/// Implemented by `SurfaceSystem` in steel-core. The transpiled code calls these
43/// methods through the `SurfaceRuleContext.system` field.
44pub trait SurfaceNoiseProvider {
45    /// Sample a surface condition noise at (x, z). The noise is identified by
46    /// its index in the dimension's `surface_noise_ids()` list.
47    fn get_noise(&self, noise_index: usize, x: i32, z: i32) -> f64;
48
49    /// Get the badlands clay band block at position (x, y, z).
50    fn get_band(&self, x: i32, y: i32, z: i32) -> BlockStateId;
51
52    /// Evaluate a vertical gradient condition using positional random.
53    ///
54    /// Returns true if the random value at `(block_x, block_y, block_z)` falls
55    /// within the gradient between `true_at_and_below` and `false_at_and_above`.
56    fn vertical_gradient(
57        &self,
58        random_name: &NameHash,
59        block_x: i32,
60        block_y: i32,
61        block_z: i32,
62        true_at_and_below: i32,
63        false_at_and_above: i32,
64    ) -> bool;
65}
