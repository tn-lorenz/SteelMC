16/// A trait for random number generators.
17#[enum_dispatch]
18#[allow(missing_docs)]
19pub trait Random {
20    #[must_use]
21    fn fork(&mut self) -> Self;
22
23    fn next_i32(&mut self) -> i32;
24
25    fn next_i32_bounded(&mut self, bound: i32) -> i32;
26
27    fn next_i32_between(&mut self, min: i32, max: i32) -> i32 {
28        self.next_i32_bounded(max - min + 1) + min
29    }
30
31    fn next_i32_between_exclusive(&mut self, min: i32, max: i32) -> i32 {
32        min + self.next_i32_bounded(max - min)
33    }
34
35    fn next_i64(&mut self) -> i64;
36
37    fn next_f32(&mut self) -> f32;
38
39    fn next_f64(&mut self) -> f64;
40
41    fn next_bool(&mut self) -> bool;
42
43    fn next_gaussian(&mut self) -> f64;
44
45    fn triangle(&mut self, min: f64, max: f64) -> f64 {
46        min + max * (self.next_f64() - self.next_f64())
47    }
48
49    fn triangle_f32(&mut self, min: f32, max: f32) -> f32 {
50        min + max * (self.next_f32() - self.next_f32())
51    }
52
53    fn next_positional(&mut self) -> RandomSplitter;
54
55    fn consume_count(&mut self, count: i32) {
56        for _ in 0..count {
57            self.next_i64();
58        }
59    }
60}

91/// Gets a seed from a position.
92#[allow(clippy::cast_sign_loss)]
93#[must_use]
94pub fn get_seed(x: i32, y: i32, z: i32) -> i64 {
95    let l = i64::from(x.wrapping_mul(3_129_871))
96        ^ (i64::from(z).wrapping_mul(116_129_781_i64))
97        ^ i64::from(y);
98    let l = l
99        .wrapping_mul(l)
100        .wrapping_mul(42_317_861_i64)
101        .wrapping_add(l.wrapping_mul(11));
102    l >> 16
103}
