25/// A chunk position.
26#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
27pub struct ChunkPos(pub Vector2<i32>);
28
29impl std::hash::Hash for ChunkPos {
30    fn hash<H: std::hash::Hasher>(&self, state: &mut H) {
31        state.write_u64(self.as_i64() as u64);
32    }
33}
34
35impl ChunkPos {
36    const OFFSETS: [(i32, i32); 8] = [
37        (-1, -1),
38        (0, -1),
39        (1, -1),
40        (-1, 0),
41        (1, 0),
42        (-1, 1),
43        (0, 1),
44        (1, 1),
45    ];
46
47    /// Returns all 8 neighbors of this chunk position.
48    #[must_use]
49    pub fn neighbors(&self) -> [ChunkPos; 8] {
50        Self::OFFSETS.map(|(dx, dy)| ChunkPos::new(self.0.x + dx, self.0.y + dy))
51    }
52
53    #[must_use]
54    #[inline]
55    /// Creates a new `ChunkPos` with the given x and y coordinates.
56    pub const fn new(x: i32, y: i32) -> Self {
57        Self(Vector2::new(x, y))
58    }
59
60    /// Converts the `ChunkPos` to an `i64`.
61    #[must_use]
62    #[inline]
63    pub fn as_i64(&self) -> i64 {
64        (i64::from(self.0.x) & 0xFFFF_FFFF) | ((i64::from(self.0.y) & 0xFFFF_FFFF) << 32)
65    }
66
67    /// Creates a new `ChunkPos` from an `i64`.
68    #[must_use]
69    #[inline]
70    pub fn from_i64(value: i64) -> Self {
71        Self(Vector2::new(
72            (value & 0xFFFF_FFFF) as i32,
73            (value >> 32) as i32,
74        ))
75    }
76}

92/// A block position.
93#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
94pub struct BlockPos(pub Vector3<i32>);
95
96impl BlockPos {
97    // Define constants as per the Java logic, but in Rust style
98    const PACKED_HORIZONTAL_LEN: u32 = 26;
99    const PACKED_Y_LEN: u32 = 64 - 2 * Self::PACKED_HORIZONTAL_LEN;
100    const X_OFFSET: u32 = Self::PACKED_Y_LEN + Self::PACKED_HORIZONTAL_LEN;
101    const Z_OFFSET: u32 = 0;
102    const PACKED_X_MASK: i64 = (1i64 << Self::PACKED_HORIZONTAL_LEN) - 1;
103    const PACKED_Y_MASK: i64 = (1i64 << Self::PACKED_Y_LEN) - 1;
104    const PACKED_Z_MASK: i64 = (1i64 << Self::PACKED_HORIZONTAL_LEN) - 1;
105
106    /// Converts the `BlockPos` to an `i64`.
107    #[must_use]
108    pub fn as_i64(&self) -> i64 {
109        let x = i64::from(self.0.x);
110        let y = i64::from(self.0.y);
111        let z = i64::from(self.0.z);
112        ((x & Self::PACKED_X_MASK) << Self::X_OFFSET)
113            | (y & Self::PACKED_Y_MASK)
114            | ((z & Self::PACKED_Z_MASK) << Self::Z_OFFSET)
115    }
116}

135/// An identifier used by Minecraft.
136#[derive(Debug, Clone, PartialEq, Eq, Hash, Default)]
137pub struct Identifier {
138    /// The namespace of the identifier.
139    pub namespace: Cow<'static, str>,
140    /// The path of the identifier.
141    pub path: Cow<'static, str>,
142}
143
144impl Identifier {
145    /// The vanilla namespace.
146    pub const VANILLA_NAMESPACE: &'static str = "minecraft";
147
148    /// Creates a new `Identifier` with the vanilla namespace.
149    #[must_use]
150    pub fn vanilla(path: String) -> Self {
151        Identifier {
152            namespace: Cow::Borrowed(Self::VANILLA_NAMESPACE),
153            path: Cow::Owned(path),
154        }
155    }
156
157    /// Creates a new `Identifier` with the vanilla namespace and a static path.
158    #[must_use]
159    pub const fn vanilla_static(path: &'static str) -> Self {
160        Identifier {
161            namespace: Cow::Borrowed(Self::VANILLA_NAMESPACE),
162            path: Cow::Borrowed(path),
163        }
164    }
165
166    /// Returns whether the character is a valid namespace character.
167    #[must_use]
168    pub fn valid_namespace_char(char: char) -> bool {
169        char == '_'
170            || char == '-'
171            || char.is_ascii_lowercase()
172            || char.is_ascii_digit()
173            || char == '.'
174    }
175
176    /// Returns whether the character is a valid path character.
177    #[must_use]
178    pub fn valid_char(char: char) -> bool {
179        Self::valid_namespace_char(char) || char == '/'
180    }
181
182    /// Returns whether the namespace is valid.
183    pub fn validate_namespace(namespace: &str) -> bool {
184        namespace.chars().all(Self::valid_namespace_char)
185    }
186
187    /// Returns whether the path is valid.
188    pub fn validate_path(path: &str) -> bool {
189        path.chars().all(Self::valid_char)
190    }
191
192    /// Returns whether the namespace and path are valid.
193    #[must_use]
194    pub fn validate(namespace: &str, path: &str) -> bool {
195        Self::validate_namespace(namespace) && Self::validate_path(path)
196    }
197}

206#[allow(missing_docs)]
207impl FromStr for Identifier {
208    type Err = &'static str;
209
210    fn from_str(s: &str) -> Result<Self, Self::Err> {
211        let parts: Vec<&str> = s.split(':').collect();
212        if parts.len() != 2 {
213            return Err("Invalid resource location");
214        }
215
216        if !Identifier::validate_namespace(parts[0]) {
217            return Err("Invalid namespace");
218        }
219
220        if !Identifier::validate_path(parts[1]) {
221            return Err("Invalid path");
222        }
223
224        Ok(Identifier {
225            namespace: Cow::Owned(parts[0].to_string()),
226            path: Cow::Owned(parts[1].to_string()),
227        })
228    }
229}
