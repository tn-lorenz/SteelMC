9/// The six cardinal directions in Minecraft.
10#[derive(Clone, Copy, Debug)]
11#[derive_const(PartialEq)]
12pub enum Direction {
13    /// Negative Y direction.
14    Down,
15    /// Positive Y direction.
16    Up,
17    /// Negative Z direction.
18    North,
19    /// Positive Z direction.
20    South,
21    /// Negative X direction.
22    West,
23    /// Positive X direction.
24    East,
25}
26
27impl ReadFrom for Direction {
28    fn read(data: &mut Cursor<&[u8]>) -> io::Result<Self> {
29        let id = VarInt::read(data)?.0;
30        match id {
31            0 => Ok(Direction::Down),
32            1 => Ok(Direction::Up),
33            2 => Ok(Direction::North),
34            3 => Ok(Direction::South),
35            4 => Ok(Direction::West),
36            5 => Ok(Direction::East),
37            _ => Err(io::Error::other("Invalid Direction id")),
38        }
39    }
40}

42impl Direction {
43    /// Returns the block position offset for this direction.
44    #[must_use]
45    pub const fn offset(&self) -> (i32, i32, i32) {
46        match self {
47            Direction::Down => (0, -1, 0),
48            Direction::Up => (0, 1, 0),
49            Direction::North => (0, 0, -1),
50            Direction::South => (0, 0, 1),
51            Direction::West => (-1, 0, 0),
52            Direction::East => (1, 0, 0),
53        }
54    }

56    /// Returns the block position relative to the given position in this direction.
57    #[must_use]
58    pub const fn relative(&self, pos: &BlockPos) -> BlockPos {
59        let (dx, dy, dz) = self.offset();
60        pos.offset(dx, dy, dz)
61    }

63    /// Returns the axis this direction is on.
64    #[must_use]
65    pub const fn get_axis(&self) -> Axis {
66        match self {
67            Direction::Down | Direction::Up => Axis::Y,
68            Direction::North | Direction::South => Axis::Z,
69            Direction::West | Direction::East => Axis::X,
70        }
71    }

73    /// Returns the opposite direction.
74    #[must_use]
75    pub const fn opposite(&self) -> Direction {
76        match self {
77            Direction::Down => Direction::Up,
78            Direction::Up => Direction::Down,
79            Direction::North => Direction::South,
80            Direction::South => Direction::North,
81            Direction::West => Direction::East,
82            Direction::East => Direction::West,
83        }
84    }

86    /// Returns the horizontal direction from a yaw rotation.
87    ///
88    /// Yaw values follow Minecraft's convention:
89    /// - 0째 = South (+Z)
90    /// - 90째 = West (-X)
91    /// - 180째 = North (-Z)
92    /// - 270째 = East (+X)
93    #[must_use]
94    pub fn from_yaw(yaw: f32) -> Direction {
95        let adjusted = yaw.rem_euclid(360.0);
96        match adjusted {
97            y if !(45.0..315.0).contains(&y) => Direction::South,
98            y if y < 135.0 => Direction::West,
99            y if y < 225.0 => Direction::North,
100            _ => Direction::East,
101        }
102    }

104    /// Returns the yaw rotation for this direction.
105    ///
106    /// Only meaningful for horizontal directions.
107    /// Vertical directions return 0.
108    #[must_use]
109    pub const fn to_yaw(&self) -> f32 {
110        match self {
111            Direction::North => 180.0,
112            Direction::South | Direction::Up | Direction::Down => 0.0,
113            Direction::West => 90.0,
114            Direction::East => 270.0,
115        }
116    }

125    #[must_use]

127        matches!(
128            self,
129            Direction::North | Direction::South | Direction::East | Direction::West

133    /// Rotates this direction 90 degrees clockwise around the Y axis.
134    ///
135    /// Vertical directions are unchanged.
136    #[must_use]
137    pub const fn rotate_y_clockwise(&self) -> Direction {
138        match self {
139            Direction::North => Direction::East,
140            Direction::East => Direction::South,
141            Direction::South => Direction::West,
142            Direction::West => Direction::North,
143            other => *other,

151    pub const fn rotate_y_counter_clockwise(&self) -> Direction {
152        match self {
153            Direction::North => Direction::West,
154            Direction::West => Direction::South,
155            Direction::South => Direction::East,
156            Direction::East => Direction::North,

172    /// Returns all directions ordered by how closely they match the player's look direction.
173    ///
174    /// This matches vanilla's `Direction.orderedByNearest(Entity)`.
175    /// - `yaw`: Player's yaw rotation in degrees (0 = South, 90 = West, 180 = North, 270 = East)
176    /// - `pitch`: Player's pitch rotation in degrees (negative = looking up, positive = looking down)
177    #[must_use]
178    pub fn ordered_by_nearest(yaw: f32, pitch: f32) -> [Direction; 6] {
179        // Convert to radians and negate yaw to match vanilla's coordinate system
180        let pitch_rad = pitch.to_radians();
181        let yaw_rad = (-yaw).to_radians();
182
183        let pitch_sin = pitch_rad.sin();
184        let pitch_cos = pitch_rad.cos();
185        let yaw_sin = yaw_rad.sin();
186        let yaw_cos = yaw_rad.cos();
187
188        // Determine which direction on each axis the player is looking
189        let x_pos = yaw_sin > 0.0;
190        let y_pos = pitch_sin < 0.0; // Negative pitch = looking up
191        let z_pos = yaw_cos > 0.0;
192
193        // Calculate magnitude of look direction on each axis
194        let x_yaw = if x_pos { yaw_sin } else { -yaw_sin };
195        let y_mag = if y_pos { -pitch_sin } else { pitch_sin };
196        let z_yaw = if z_pos { yaw_cos } else { -yaw_cos };
197        let x_mag = x_yaw * pitch_cos;
198        let z_mag = z_yaw * pitch_cos;
199
200        // Determine the primary direction on each axis
201        let axis_x = if x_pos {
202            Direction::East
203        } else {
204            Direction::West
205        };
206        let axis_y = if y_pos {
207            Direction::Up
208        } else {
209            Direction::Down
210        };
211        let axis_z = if z_pos {
212            Direction::South
213        } else {
214            Direction::North
215        };
216
217        // Sort axes by magnitude and build the direction array
218        if x_yaw > z_yaw {
219            if y_mag > x_mag {
220                Self::make_direction_array(axis_y, axis_x, axis_z)
221            } else if z_mag > y_mag {
222                Self::make_direction_array(axis_x, axis_z, axis_y)
223            } else {
224                Self::make_direction_array(axis_x, axis_y, axis_z)
225            }
226        } else if y_mag > z_mag {
227            Self::make_direction_array(axis_y, axis_z, axis_x)
228        } else if x_mag > y_mag {
229            Self::make_direction_array(axis_z, axis_x, axis_y)
230        } else {
231            Self::make_direction_array(axis_z, axis_y, axis_x)
232        }
233    }
234
235    /// Creates an array of all 6 directions ordered by magnitude.
236    ///
237    /// The order is: 3 primary directions by magnitude, then their opposites in reverse order.
238    /// This matches vanilla's `Direction.makeDirectionArray()`.
239    const fn make_direction_array(
240        axis1: Direction,
241        axis2: Direction,
242        axis3: Direction,
243    ) -> [Direction; 6] {
244        [
245            axis1,
246            axis2,
247            axis3,
248            axis3.opposite(),
249            axis2.opposite(),
250            axis1.opposite(),
251        ]
252    }
