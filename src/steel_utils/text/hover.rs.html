1use std::{borrow::Cow, vec};
2
3use serde::{Deserialize, Serialize};
4
5use super::TextComponent;
6
7/// A hover event.
8#[derive(Clone, Debug, Serialize, Deserialize, PartialEq, Eq, Hash)]
9#[serde(tag = "action", rename_all = "snake_case")]
10pub enum HoverEvent {
11    /// Displays a tooltip with the given text.
12    #[allow(missing_docs)]
13    ShowText { value: Vec<TextComponent> },
14    /// Shows an item.
15    #[allow(missing_docs)]
16    ShowItem {
17        /// Resource identifier of the item
18        id: Cow<'static, str>,
19        /// Number of the items in the stack
20        #[serde(default, skip_serializing_if = "Option::is_none")]
21        count: Option<i32>,
22        // #[serde(default, skip_serializing_if = "Option::is_none")]
23        // components: Option<Cow<'static, str>>,
24    },
25    /// Shows an entity.
26    #[allow(missing_docs)]
27    ShowEntity {
28        /// The entity's ID Entity Type
29        id: Cow<'static, str>,
30        /// The entity's UUID
31        /// The UUID cannot use `uuid::Uuid` because its serialization parses it into bytes, so its double bytes serialized
32        uuid: Cow<'static, str>,
33        /// Optional custom name for the entity
34        #[serde(default, skip_serializing_if = "Option::is_none")]
35        name: Option<Vec<TextComponent>>,
36    },
37}
38
39#[allow(missing_docs)]
40impl HoverEvent {
41    #[must_use]
42    pub fn show_text(text: TextComponent) -> Self {
43        Self::ShowText { value: vec![text] }
44    }
45    pub fn show_entity<P: Into<Cow<'static, str>>>(
46        uuid: P,
47        kind: P,
48        name: Option<TextComponent>,
49    ) -> Self {
50        Self::ShowEntity {
51            id: kind.into(),
52            uuid: uuid.into(),
53            name: match name {
54                Some(name) => Some(vec![name]),
55                None => None,
56            },
57        }
58    }
59}
