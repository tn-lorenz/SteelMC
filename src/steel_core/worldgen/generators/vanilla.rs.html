43/// A chunk generator for vanilla (normal) world generation.
44///
45/// Matches vanilla's `NoiseBasedChunkGenerator`. The biome source is pluggable
46/// per-dimension — overworld, nether, and end each provide a different
47/// [`BiomeSourceKind`] variant.
48///
49/// Generic over `N: DimensionNoises` to support different dimensions with
50/// their own transpiled density functions and noise settings.
51pub struct VanillaGenerator<N: DimensionNoises> {
52    /// Biome source for this dimension. Determines biomes at each quart position.
53    biome_source: BiomeSourceKind,
54    /// Representative biome for source-carver lookup when every possible
55    /// biome from `biome_source` has the same carver list.
56    ///
57    /// Vanilla still samples each source biome in `apply_carvers`; Steel skips
58    /// that sampling only when the source's full possible-biome set proves the
59    /// carver list is uniform. If future biome sources can produce mixed
60    /// carver lists this remains `None` and the vanilla per-source lookup is
61    /// used.
62    uniform_carver_biome: Option<BiomeRef>,
63    /// Noise generators for this dimension's density functions.
64    /// Boxed because noise structs can be large.
65    noises: Box<N>,
66    /// Seed positional splitter for per-chunk construction of aquifers.
67    splitter: RandomSplitter,
68    /// Ore vein generator for replacing stone with ore blocks.
69    ore_veinifier: Option<OreVeinifier>,
70    /// Surface system for biome-specific block replacement.
71    surface_system: SurfaceSystem,
72    /// Which vanilla surface extension biomes this source can produce.
73    surface_extension_biomes: SurfaceExtensionBiomes,
74    /// Block state ID for the default block, cached at construction time.
75    default_block_id: BlockStateId,
76    /// Obfuscated seed for `BiomeManager` biome zoom fuzzing.
77    biome_zoom_seed: i64,
78    /// World seed as i64 (matching Java's long), used for structures and carver seeding.
79    seed: i64,
80    /// Shared structure placement/selection engine.
81    structure_generator: StructureGenerator,
82    /// Cached placed-feature order for biome decoration.
83    feature_runner: FeatureDecorationRunner,
84    _phantom: PhantomData<N>,
85}

107impl<N: DimensionNoises> VanillaGenerator<N> {
108    /// Creates a new `VanillaGenerator` with the given biome source and seed.
109    ///
110    /// # Panics
111    /// Panics if SHA-256 hash output is shorter than 8 bytes (cannot happen).
112    #[must_use]
113    pub fn new(
114        world_path: Option<&Path>,
115        biome_source: BiomeSourceKind,
116        seed: u64,
117        thread_pool: &rayon::ThreadPool,
118    ) -> Self {
119        // Nether uses Java's LCG; overworld/end use Xoroshiro.
120        let splitter = if N::Settings::LEGACY_RANDOM_SOURCE {
121            LegacyRandom::from_seed(seed).next_positional()
122        } else {
123            Xoroshiro::from_seed(seed).next_positional()
124        };
125        let noise_params = get_noise_parameters();
126        let noises = N::create(seed, &splitter, &noise_params);
127
128        let ore_veinifier = if N::Settings::ORE_VEINS_ENABLED {
129            Some(OreVeinifier::new(&splitter))
130        } else {
131            None
132        };
133
134        let default_block_id = N::Settings::default_block_id();
135        let surface_system = SurfaceSystem::new(
136            &splitter,
137            &noise_params,
138            N::surface_noise_ids(),
139            N::surface_gradient_ids(),
140            default_block_id,
141            N::Settings::SEA_LEVEL,
142        );
143
144        let biome_zoom_seed = obfuscate_biome_seed(seed as i64);
145
146        // Force the lazy parameter-list R-tree inside the configured generation
147        // pool so its parallel construction does not initialize Rayon's global pool.
148        let possible_biome_refs = thread_pool.install(|| biome_source.possible_biome_refs());
149        let possible_biomes = biome_source.possible_biomes();
150        let surface_extension_biomes = SurfaceExtensionBiomes::from_possible(&possible_biomes);
151        let structure_generator =
152            StructureGenerator::vanilla(seed as i64, world_path, &biome_source, thread_pool);
153        let uniform_carver_biome = Self::uniform_carver_biome(&possible_biomes);
154        let feature_runner = FeatureDecorationRunner::new(&possible_biome_refs, &REGISTRY);
155
156        Self {
157            biome_source,
158            uniform_carver_biome,
159            noises: Box::new(noises),
160            splitter,
161            ore_veinifier,
162            surface_system,
163            surface_extension_biomes,
164            default_block_id,
165            biome_zoom_seed,
166            seed: seed as i64,
167            structure_generator,
168            feature_runner,
169            _phantom: PhantomData,
170        }
171    }
172
173    fn uniform_carver_biome(possible_biomes: &FxHashSet<Identifier>) -> Option<BiomeRef> {
174        let mut possible_biomes = possible_biomes.iter();
175        let first_key = possible_biomes.next()?;
176        let first = REGISTRY.biomes.by_key(first_key)?;
177
178        possible_biomes
179            .all(|key| {
180                REGISTRY
181                    .biomes
182                    .by_key(key)
183                    .is_some_and(|biome| biome.carvers == first.carvers)
184            })
185            .then_some(first)
186    }
187}

212    fn create_structures(&self, chunk: &ChunkAccess) {
213        let pos = chunk.pos();
214        let chunk_x = pos.0.x;
215        let chunk_z = pos.0.y;
216
217        let mut sampler = self.biome_source.chunk_sampler();
218        let chunk_min_x = chunk_x * 16;
219        let chunk_min_z = chunk_z * 16;
220
221        let mut height_cache = N::ColumnCache::default();
222        let sea_level = N::Settings::SEA_LEVEL;
223
224        // No eager `init_grid`: most chunks' structures (mineshaft, village)
225        // use their own caches, and the 1–4 column probes of the remainder
226        // hit this cache's lazy single-entry mode cheaply. Eager 5×5 quart
227        // init cost ~36µs per chunk with no payoff.
228        let mut aquifer = LazyAquifer::new(chunk_min_x, chunk_min_z, &self.splitter, &*self.noises);
229        let mut surface_y_cache: Option<i32> = None;
230        let mut height_cache_grid_ready = false;
231        let mut ctx = GenerationContext::<'_, '_, N>::new(
232            self.seed,
233            chunk_x,
234            chunk_z,
235            sea_level,
236            &self.noises,
237            &self.splitter,
238            self.structure_generator.template_pools(),
239            self.structure_generator.templates(),
240            &mut sampler,
241            &mut height_cache,
242            &mut aquifer,
243            &mut surface_y_cache,
244            &mut height_cache_grid_ready,
245        );
246
247        create_structures(&self.structure_generator, chunk, &mut ctx);
248    }

250    fn create_biomes(&self, chunk: &ChunkAccess) {
251        let pos = chunk.pos();
252        let min_y = chunk.min_y();
253        let section_count = chunk.sections().sections.len();
254
255        let chunk_x = pos.0.x;
256        let chunk_z = pos.0.y;
257
258        let mut sampler = self.biome_source.chunk_sampler();
259        // Pre-compute the flat (xz-only) climate-noise grid for this chunk so the
260        // per-cell sampling below does O(1) column lookups instead of recomputing
261        // the flat noise for all 1536 cells (the noise stage's `fill_from_noise`
262        // already does this). Values are bit-identical — same functions, same quart
263        // coordinates — so biome selection is unchanged.
264        sampler.init_grid(chunk_x * 16, chunk_z * 16);
265
266        // Match vanilla's iteration order: Section(Y) → X → Y → Z.
267        // This is critical because the R-tree biome cache (persistent warm-start)
268        // determines tie-breaking for equal-distance entries, and the cache state
269        // depends on the order of biome lookups.
270        for section_index in 0..section_count {
271            let section_y = (min_y / 16) + section_index as i32;
272            let section = &chunk.sections().sections[section_index];
273            let mut section_guard = section.write();
274
275            for local_quart_x in 0..4i32 {
276                let quart_x = chunk_x * 4 + local_quart_x;
277
278                for local_quart_y in 0..4i32 {
279                    let quart_y = section_y * 4 + local_quart_y;
280
281                    for local_quart_z in 0..4i32 {
282                        let quart_z = chunk_z * 4 + local_quart_z;
283
284                        let biome = sampler.sample(quart_x, quart_y, quart_z);
285                        let biome_id = biome.id() as u16;
286
287                        section_guard.biomes.set(
288                            local_quart_x as usize,
289                            local_quart_y as usize,
290                            local_quart_z as usize,
291                            biome_id,
292                        );
293                    }
294                }
295            }
296        }
297
298        chunk.mark_dirty();
299    }

301    fn fill_from_noise(&self, chunk: &ChunkAccess, beardifier: Option<&Beardifier>) {
302        let pos = chunk.pos();
303        let chunk_min_x = pos.0.x * 16;
304        let chunk_min_z = pos.0.y * 16;
305
306        let min_y = N::Settings::MIN_Y;
307        let height = N::Settings::HEIGHT;
308
309        let mut noise_chunk = NoiseChunk::<N>::new(chunk_min_x, chunk_min_z);
310        let noises = &*self.noises;
311
312        let mut column_cache = N::ColumnCache::default();
313        column_cache.init_grid(chunk_min_x, chunk_min_z, noises);
314
315        let default_block_id = self.default_block_id;
316        let ore_veinifier = &self.ore_veinifier;
317        let mut aquifer = Aquifer::<N>::new(
318            chunk_min_x,
319            chunk_min_z,
320            min_y,
321            height,
322            &self.splitter,
323            noises,
324            // Aquifer samples at arbitrary (x,z) outside the chunk, so it needs its own cache
325            column_cache.clone(),
326        );
327
328        // Collect writes per (x,z) column and flush in batch to avoid per-block
329        // write lock acquisition on sections.
330        let mut pending_writes: Vec<(usize, usize, usize, BlockStateId)> = Vec::new();
331        let mut prev_x: usize = usize::MAX;
332        let mut prev_z: usize = usize::MAX;
333        let mut ocean_floor_wg =
334            Heightmap::new(HeightmapType::OceanFloorWg, min_y, N::Settings::HEIGHT);
335        let mut world_surface_wg =
336            Heightmap::new(HeightmapType::WorldSurfaceWg, min_y, N::Settings::HEIGHT);
337
338        noise_chunk.fill(
339            noises,
340            &mut column_cache,
341            beardifier,
342            |local_x, world_y, local_z, density, interpolated, cache| {
343                // Flush when we move to a new column
344                if local_x != prev_x || local_z != prev_z {
345                    if !pending_writes.is_empty() {
346                        chunk.write_block_batch_for_generation(&pending_writes);
347                        pending_writes.clear();
348                    }
349                    prev_x = local_x;
350                    prev_z = local_z;
351                }
352
353                let relative_y = (world_y - min_y) as usize;
354                let world_x = chunk_min_x + local_x as i32;
355                let world_z = chunk_min_z + local_z as i32;
356
357                match aquifer.compute_substance(noises, world_x, world_y, world_z, density) {
358                    AquiferResult::Solid => {
359                        let block = ore_veinifier
360                            .as_ref()
361                            .and_then(|ov| {
362                                ov.compute_interpolated(
363                                    noises,
364                                    cache,
365                                    interpolated,
366                                    world_x,
367                                    world_y,
368                                    world_z,
369                                )
370                            })
371                            .unwrap_or(default_block_id);
372                        pending_writes.push((local_x, relative_y, local_z, block));
373                        ocean_floor_wg.update_for_initial_fill(local_x, world_y, local_z, block);
374                        world_surface_wg.update_for_initial_fill(local_x, world_y, local_z, block);
375                    }
376                    AquiferResult::Fluid(id) => {
377                        pending_writes.push((local_x, relative_y, local_z, id));
378                        ocean_floor_wg.update_for_initial_fill(local_x, world_y, local_z, id);
379                        world_surface_wg.update_for_initial_fill(local_x, world_y, local_z, id);
380                        if aquifer.should_schedule_fluid_update() && id.has_fluid() {
381                            chunk.mark_pos_for_postprocessing(BlockPos::new(
382                                world_x, world_y, world_z,
383                            ));
384                        }
385                    }
386                    AquiferResult::Air => {}
387                }
388            },
389        );
390
391        // Flush remaining writes
392        if !pending_writes.is_empty() {
393            chunk.write_block_batch_for_generation(&pending_writes);
394        }
395
396        let ChunkAccess::Proto(proto) = chunk else {
397            return;
398        };
399        let mut heightmaps = proto.heightmaps.write();
400        heightmaps.replace(ocean_floor_wg);
401        heightmaps.replace(world_surface_wg);
402    }

404    #[expect(clippy::too_many_lines, reason = "splitting would hurt readability")]
405    fn build_surface(&self, chunk: &ChunkAccess, neighbor_biomes: &dyn Fn(IVec3) -> u16) {
406        let min_y = N::Settings::MIN_Y;
407        let pos = chunk.pos();
408        let chunk_min_x = pos.0.x * 16;
409        let chunk_min_z = pos.0.y * 16;
410        let default_block_id = self.default_block_id;
411        let noises = &*self.noises;
412        let surface_rule_block_states = N::surface_rule_block_states();
413        let surface_rule_uses_biome = N::surface_rule_uses_biome();
414        let surface_rule_uses_preliminary_surface = N::surface_rule_uses_preliminary_surface();
415        let surface_rule_uses_surface_secondary = N::surface_rule_uses_surface_secondary();
416        let surface_rule_uses_steep = N::surface_rule_uses_steep();
417        let lazy_surface_rule_biome =
418            surface_rule_uses_biome && surface_rule_uses_preliminary_surface;
419        let surface_needs_min_surface_level =
420            surface_rule_uses_preliminary_surface || self.surface_extension_biomes.frozen_ocean;
421        let surface_needs_biomes =
422            surface_rule_uses_biome || self.surface_extension_biomes.needs_surface_biome();
423        let chunk_quart_x = pos.0.x * 4;
424        let chunk_quart_z = pos.0.y * 4;
425
426        chunk.prime_heightmaps(&[HeightmapType::WorldSurfaceWg]);
427
428        // Pre-compute preliminary surface corners only for rules/extensions that read them.
429        let preliminary_surface_corners = surface_needs_min_surface_level.then(|| {
430            let mut psl_cache = N::ColumnCache::default();
431            let p00 =
432                preliminary_surface_level::<N>(noises, &mut psl_cache, chunk_min_x, chunk_min_z);
433            let p10 = preliminary_surface_level::<N>(
434                noises,
435                &mut psl_cache,
436                chunk_min_x + 16,
437                chunk_min_z,
438            );
439            let p01 = preliminary_surface_level::<N>(
440                noises,
441                &mut psl_cache,
442                chunk_min_x,
443                chunk_min_z + 16,
444            );
445            let p11 = preliminary_surface_level::<N>(
446                noises,
447                &mut psl_cache,
448                chunk_min_x + 16,
449                chunk_min_z + 16,
450            );
451            (p00, p10, p01, p11)
452        });
453
454        let eroded_badlands_id = (*vanilla_biomes::ERODED_BADLANDS).id() as u16;
455        let frozen_ocean_id = (*vanilla_biomes::FROZEN_OCEAN).id() as u16;
456        let deep_frozen_ocean_id = (*vanilla_biomes::DEEP_FROZEN_OCEAN).id() as u16;
457
458        // Pre-extract biome palette values only if surface rules/extensions need them.
459        let biome_data = surface_needs_biomes.then(|| chunk.sections().read_all_biomes());
460        let section_count = chunk.sections().sections.len();
461
462        let mut pending_writes: Vec<(usize, BlockStateId)> = Vec::new();
463        let mut column_buf: Vec<BlockStateId> = Vec::new();
464        let condition_noise_values = N::surface_noise_ids()
465            .iter()
466            .map(|_| Cell::new(0.0))
467            .collect::<Vec<_>>();
468        let condition_noise_initialized = N::surface_noise_ids()
469            .iter()
470            .map(|_| Cell::new(false))
471            .collect::<Vec<_>>();
472        let condition_noise_cache =
473            SurfaceConditionNoiseCache::new(&condition_noise_values, &condition_noise_initialized);
474
475        for local_x in 0..16usize {
476            for local_z in 0..16usize {
477                let block_x = chunk_min_x + local_x as i32;
478                let block_z = chunk_min_z + local_z as i32;
479
480                // Start scanning from one above the highest non-air block
481                let mut start_height =
482                    chunk.height_at(HeightmapType::WorldSurfaceWg, local_x, local_z);
483
484                // Column-local Voronoi cache for fuzzed biome lookups.
485                let mut biome_col = biome_data.as_deref().map(|biome_data| {
486                    FuzzedBiomeColumn::new(
487                        biome_data,
488                        section_count,
489                        self.biome_zoom_seed,
490                        block_x,
491                        block_z,
492                        min_y,
493                        chunk_quart_x,
494                        chunk_quart_z,
495                        neighbor_biomes,
496                    )
497                });
498
499                // Eroded badlands extension: add terracotta pillars above surface
500                let surface_biome_id = if self.surface_extension_biomes.needs_surface_biome() {
501                    biome_col
502                        .as_mut()
503                        .map(|biome_col| biome_col.get(start_height))
504                } else {
505                    None
506                };
507                if self.surface_extension_biomes.eroded_badlands
508                    && surface_biome_id == Some(eroded_badlands_id)
509                {
510                    start_height = self.surface_system.eroded_badlands_extension(
511                        chunk,
512                        local_x,
513                        local_z,
514                        block_x,
515                        block_z,
516                        start_height,
517                        min_y,
518                    );
519                }
520
521                // Snapshot the column once — avoids per-block section locking in the Y scan.
522                // Taken after eroded_badlands_extension which may write blocks above the surface.
523                chunk
524                    .sections()
525                    .read_column_into(local_x, local_z, &mut column_buf);
526
527                // Surface depth for this column
528                let surface_depth = self.surface_system.get_surface_depth(block_x, block_z);
529
530                let surface_secondary = if surface_rule_uses_surface_secondary {
531                    self.surface_system.get_surface_secondary(block_x, block_z)
532                } else {
533                    0.0
534                };
535                condition_noise_cache.reset();
536
537                let min_surface_level =
538                    if let Some((p00, p10, p01, p11)) = preliminary_surface_corners {
539                        // Vanilla: (float)(blockX & 15) / 16.0F — exact for 0-15.
540                        let t_x = f64::from(local_x as u8) / 16.0;
541                        let t_z = f64::from(local_z as u8) / 16.0;
542                        let interp = lerp2(
543                            t_x,
544                            t_z,
545                            f64::from(p00),
546                            f64::from(p10),
547                            f64::from(p01),
548                            f64::from(p11),
549                        );
550                        interp.floor() as i32 + surface_depth - 8
551                    } else {
552                        0
553                    };
554
555                // Steep condition: vanilla only checks south >= north + 4 and
556                // west >= east + 4 (asymmetric, not absolute difference).
557                let steep = surface_rule_uses_steep && {
558                    let z_north = local_z.saturating_sub(1);
559                    let z_south = (local_z + 1).min(15);
560                    let h_north =
561                        chunk.height_at(HeightmapType::WorldSurfaceWg, local_x, z_north) - 1;
562                    let h_south =
563                        chunk.height_at(HeightmapType::WorldSurfaceWg, local_x, z_south) - 1;
564                    if h_south >= h_north + 4 {
565                        true
566                    } else {
567                        let x_west = local_x.saturating_sub(1);
568                        let x_east = (local_x + 1).min(15);
569                        let h_west =
570                            chunk.height_at(HeightmapType::WorldSurfaceWg, x_west, local_z) - 1;
571                        let h_east =
572                            chunk.height_at(HeightmapType::WorldSurfaceWg, x_east, local_z) - 1;
573                        h_west >= h_east + 4
574                    }
575                };
576
577                let mut stone_depth_above: i32 = 0;
578                let mut water_height: i32 = i32::MIN;
579                let mut next_ceiling_stone_y: i32 = i32::MAX;
580                pending_writes.clear();
581
582                for y in (min_y..=start_height).rev() {
583                    let relative_y = (y - min_y) as usize;
584                    let state = column_buf[relative_y];
585
586                    if state.is_air() {
587                        stone_depth_above = 0;
588                        water_height = i32::MIN;
589                        continue;
590                    }
591
592                    if state.get_block().config.liquid {
593                        if water_height == i32::MIN {
594                            water_height = y + 1;
595                        }
596                        continue;
597                    }
598
599                    // Solid block — scan for stone_depth_below (lookahead)
600                    if next_ceiling_stone_y >= y {
601                        next_ceiling_stone_y = i32::MIN;
602                        for la_y in (min_y - 1..y).rev() {
603                            if la_y < min_y {
604                                next_ceiling_stone_y = la_y + 1;
605                                break;
606                            }
607                            let la_rel = (la_y - min_y) as usize;
608                            let la_state = column_buf[la_rel];
609                            // isStone = !isAir && !isLiquid
610                            if la_state.is_air() || la_state.get_block().config.liquid {
611                                next_ceiling_stone_y = la_y + 1;
612                                break;
613                            }
614                        }
615                    }
616
617                    stone_depth_above += 1;
618                    let stone_depth_below = y - next_ceiling_stone_y + 1;
619
620                    // Only apply surface rules to the default block
621                    if state == default_block_id {
622                        let eager_biome_id = if surface_rule_uses_biome && !lazy_surface_rule_biome
623                        {
624                            biome_col.as_mut().map(|biome_col| biome_col.get(y))
625                        } else {
626                            None
627                        };
628                        let biome_provider = if lazy_surface_rule_biome {
629                            biome_col
630                                .as_mut()
631                                .map(|biome_col| biome_col as &mut dyn SurfaceBiomeProvider)
632                        } else {
633                            None
634                        };
635
636                        let mut ctx = SurfaceRuleContext::new(
637                            block_x,
638                            block_z,
639                            surface_depth,
640                            surface_secondary,
641                            min_surface_level,
642                            steep,
643                            y,
644                            stone_depth_above,
645                            stone_depth_below,
646                            water_height,
647                            eager_biome_id,
648                            biome_provider,
649                            &self.surface_system,
650                            &condition_noise_cache,
651                            surface_rule_block_states,
652                        );
653
654                        let rule_result = N::try_apply_surface_rule(&mut ctx);
655
656                        if let Some(new_block) = rule_result {
657                            pending_writes.push((relative_y, new_block));
658                        }
659                    }
660                }
661
662                // Flush batched writes — holds each section's write guard once
663                if !pending_writes.is_empty() {
664                    chunk.write_column_blocks_for_generation(local_x, local_z, &pending_writes);
665                    for &(relative_y, state) in &pending_writes {
666                        column_buf[relative_y] = state;
667                    }
668                    chunk.update_heightmaps_after_direct_column_writes(
669                        local_x,
670                        local_z,
671                        &pending_writes,
672                    );
673                    chunk.mark_dirty();
674                }
675
676                // Frozen ocean iceberg extension: add packed ice and snow
677                if self.surface_extension_biomes.frozen_ocean
678                    && let Some(surface_biome_id) = surface_biome_id
679                        .filter(|id| *id == frozen_ocean_id || *id == deep_frozen_ocean_id)
680                {
681                    pending_writes.clear();
682                    self.surface_system.collect_frozen_ocean_extension_writes(
683                        surface_biome_id,
684                        block_x,
685                        block_z,
686                        start_height,
687                        min_surface_level,
688                        min_y,
689                        &column_buf,
690                        &mut pending_writes,
691                    );
692                    if !pending_writes.is_empty() {
693                        chunk.write_column_blocks_for_generation(local_x, local_z, &pending_writes);
694                        chunk.update_heightmaps_after_direct_column_writes(
695                            local_x,
696                            local_z,
697                            &pending_writes,
698                        );
699                        chunk.mark_dirty();
700                    }
701                }
702            }
703        }
704    }

706    #[expect(clippy::too_many_lines, reason = "matches vanilla carver setup flow")]
707    fn apply_carvers(&self, chunk: &ChunkAccess) {
708        // Carvers only run on proto chunks.
709        let ChunkAccess::Proto(proto) = chunk else {
710            return;
711        };
712
713        if self
714            .uniform_carver_biome
715            .is_some_and(|biome| biome.carvers.is_empty())
716        {
717            return;
718        }
719
720        chunk.prime_heightmaps(&[HeightmapType::WorldSurfaceWg]);
721
722        let pos = chunk.pos();
723        let chunk_min_x = pos.0.x * 16;
724        let chunk_min_z = pos.0.y * 16;
725        let min_y = N::Settings::MIN_Y;
726        let height = N::Settings::HEIGHT;
727        let noises = &*self.noises;
728
729        // Fresh aquifer (vanilla caches NoiseChunk across stages; see TODO on
730        // ProtoChunk::carving_mask for why we rebuild instead).
731        let mut column_cache = N::ColumnCache::default();
732        if N::Settings::AQUIFERS_ENABLED {
733            column_cache.init_grid(chunk_min_x, chunk_min_z, noises);
734        }
735        let aquifer = Aquifer::<N>::new(
736            chunk_min_x,
737            chunk_min_z,
738            min_y,
739            height,
740            &self.splitter,
741            noises,
742            column_cache,
743        );
744
745        // Preliminary surface level at the chunk's 4 corners — used by
746        // top_material min_surface_level interpolation.
747        let mut psl_cache = N::ColumnCache::default();
748        let psl_corners = PreliminarySurfaceCorners {
749            nw: preliminary_surface_level::<N>(noises, &mut psl_cache, chunk_min_x, chunk_min_z),
750            ne: preliminary_surface_level::<N>(
751                noises,
752                &mut psl_cache,
753                chunk_min_x + 16,
754                chunk_min_z,
755            ),
756            sw: preliminary_surface_level::<N>(
757                noises,
758                &mut psl_cache,
759                chunk_min_x,
760                chunk_min_z + 16,
761            ),
762            se: preliminary_surface_level::<N>(
763                noises,
764                &mut psl_cache,
765                chunk_min_x + 16,
766                chunk_min_z + 16,
767            ),
768        };
769
770        let mut ctx = CarvingContext {
771            min_y,
772            gen_depth: height,
773            surface_system: &self.surface_system,
774            aquifer,
775            default_block_id: self.default_block_id,
776            psl_corners,
777            chunk_min_x,
778            chunk_min_z,
779        };
780
781        let ids = CarverBlockIds::load();
782
783        // Pre-fetch the 17×17 source-chunk carver lists. Done up front so we
784        // can later close over `biome_sampler` mutably inside `biome_getter`.
785        // Vanilla samples every source biome here; when this generator's full
786        // possible-biome set has a uniform carver list, the representative
787        // biome gives the same carver keys without 289 climate lookups.
788        let mut biome_sampler = self.biome_source.chunk_sampler();
789        let mut source_biomes: SmallVec<[SourceChunk; CARVER_SOURCE_CHUNK_COUNT]> = SmallVec::new();
790        for dx in -8i32..=8 {
791            for dz in -8i32..=8 {
792                let sx = pos.0.x + dx;
793                let sz = pos.0.y + dz;
794                let biome = if let Some(biome) = self.uniform_carver_biome {
795                    biome
796                } else {
797                    let qx = (sx * 16) >> 2;
798                    let qz = (sz * 16) >> 2;
799                    biome_sampler.sample(qx, 0, qz)
800                };
801                source_biomes.push(SourceChunk {
802                    pos: ChunkPos::new(sx, sz),
803                    biome,
804                });
805            }
806        }
807
808        // Grab (and lazily create) the carving mask on the proto chunk.
809        let mut mask_guard = proto.get_or_create_carving_mask();
810        let mask = &mut *mask_guard;
811
812        // `WorldgenRandom(LegacyRandomSource(generateUniqueSeed()))` — initial
813        // seed is irrelevant; every carver overwrites it via
814        // `set_large_feature_seed` before its probability check.
815        let mut random = LegacyRandom::from_seed(0);
816        let seed_i64 = self.seed;
817
818        let biome_zoom_seed = self.biome_zoom_seed;
819        // BiomeManager-fuzzed lookup — matches vanilla's `BiomeManager.getBiome`
820        // used by the carver's top-material path. An unfuzzed quart lookup
821        // would mismatch vanilla at quart-cell boundaries.
822        let mut biome_getter = |pos: BlockPos| -> u16 {
823            fuzzed_biome_at_block(biome_zoom_seed, pos, |q_pos| {
824                biome_sampler.sample(q_pos.x, q_pos.y, q_pos.z).id() as u16
825            })
826        };
827
828        let mut run = CarveRun {
829            ctx: &mut ctx,
830            noises,
831            chunk,
832            chunk_min_x,
833            chunk_min_z,
834            biome_getter: &mut biome_getter,
835            mask,
836            ids,
837        };
838
839        run.run_all(&source_biomes, seed_i64, &mut random);
840    }

843        worldgen_region_random_from_splitter(&self.splitter, center)

852impl<N, F> CarveRun<'_, '_, N, F>
853where
854    N: DimensionNoises,
855    F: FnMut(BlockPos) -> u16,
856{
857    /// Drive the 17×17 source-chunk carver loop. Each carver in each source
858    /// biome is seeded via `set_large_feature_seed`, probability-checked,
859    /// then dispatched to the appropriate `carve_*` method.
860    fn run_all(&mut self, source_biomes: &[SourceChunk], seed_i64: i64, random: &mut LegacyRandom) {
861        for source in source_biomes {
862            for (index, carver_key) in source.biome.carvers.iter().enumerate() {
863                let Some(carver) = REGISTRY.configured_carvers.by_key(carver_key) else {
864                    panic!(
865                        "biome {} references unknown configured carver {}",
866                        source.biome.key, carver_key
867                    );
868                };
869                let index_i64 = index as i64;
870                random.set_large_feature_seed(
871                    seed_i64.wrapping_add(index_i64),
872                    source.pos.0.x,
873                    source.pos.0.y,
874                );
875
876                let probability = carver.base().probability;
877                if random.next_f32() > probability {
878                    continue;
879                }
880
881                match &carver.kind {
882                    ConfiguredCarverKind::Cave(cfg) => {
883                        self.carve_cave(cfg, cave::CaveKind::Overworld, source.pos, random);
884                    }
885                    ConfiguredCarverKind::NetherCave(cfg) => {
886                        self.carve_cave(cfg, cave::CaveKind::Nether, source.pos, random);
887                    }
888                    ConfiguredCarverKind::Canyon(cfg) => {
889                        self.carve_canyon(cfg, source.pos, random);
890                    }
891                }
892            }
893        }
894    }
895}

899/// Vanilla's `LinearCongruentialGenerator.next()`.
900#[inline]
901const fn lcg_next(mut rval: i64, c: i64) -> i64 {
902    rval = rval.wrapping_mul(
903        rval.wrapping_mul(6_364_136_223_846_793_005)
904            .wrapping_add(1_442_695_040_888_963_407),
905    );
906    rval = rval.wrapping_add(c);
907    rval
908}

910/// Vanilla's `BiomeManager.getFiddle()`.
911#[inline]
912fn get_fiddle(rval: i64) -> f64 {
913    let uniform = ((rval >> 24).rem_euclid(1024)) as f64 / 1024.0;
914    (uniform - 0.5) * 0.9
915}
916
917/// Single-shot fuzzed biome lookup at a block position. Matches vanilla's
918/// `BiomeManager.getBiome(BlockPos)`: the block is shifted by `-2`, snapped to
919/// the enclosing quart cell, and the winning biome is chosen from the 8
920/// corners of that cell by `get_fiddle`-perturbed squared distance.
921///
922/// `quart_biome` returns the unfuzzed biome at a quart-coordinate — typically
923/// `biome_sampler.sample(qx, qy, qz).id()`.
924///
925/// Used by carver top-material lookups where a simple unfuzzed lookup would
926/// differ from vanilla at the quart-cell boundaries.
927pub(crate) fn fuzzed_biome_at_block<F: FnMut(IVec3) -> u16>(
928    biome_zoom_seed: i64,
929    pos: BlockPos,
930    mut quart_biome: F,
931) -> u16 {
932    let abs = pos.0 - IVec3::splat(2);
933    let parent = IVec3::new(abs.x >> 2, abs.y >> 2, abs.z >> 2);
934    let fract = DVec3::new(
935        f64::from(abs.x & 3),
936        f64::from(abs.y & 3),
937        f64::from(abs.z & 3),
938    ) / 4.0;
939
940    let mut min_i = 0usize;
941    let mut min_dist = f64::INFINITY;
942
943    for i in 0..8usize {
944        let x_even = (i & 4) == 0;
945        let y_even = (i & 2) == 0;
946        let z_even = (i & 1) == 0;
947        let cx = if x_even { parent.x } else { parent.x + 1 };
948        let cy = if y_even { parent.y } else { parent.y + 1 };
949        let cz = if z_even { parent.z } else { parent.z + 1 };
950        let dx = if x_even { fract.x } else { fract.x - 1.0 };
951        let dy = if y_even { fract.y } else { fract.y - 1.0 };
952        let dz = if z_even { fract.z } else { fract.z - 1.0 };
953
954        // BiomeManager.getFiddledDistance — identical sequence to
955        // FuzzedBiomeColumn::compute_cy_group but without the column cache.
956        let mut rval = lcg_next(biome_zoom_seed, i64::from(cx));
957        rval = lcg_next(rval, i64::from(cy));
958        rval = lcg_next(rval, i64::from(cz));
959        rval = lcg_next(rval, i64::from(cx));
960        rval = lcg_next(rval, i64::from(cy));
961        rval = lcg_next(rval, i64::from(cz));
962        let fx = get_fiddle(rval);
963        rval = lcg_next(rval, biome_zoom_seed);
964        let fy = get_fiddle(rval);
965        rval = lcg_next(rval, biome_zoom_seed);
966        let fz = get_fiddle(rval);
967
968        let dist = (dx + fx).powi(2) + (dy + fy).powi(2) + (dz + fz).powi(2);
969        if min_dist > dist {
970            min_i = i;
971            min_dist = dist;
972        }
973    }
974
975    let b = IVec3::new(
976        if (min_i & 4) == 0 {
977            parent.x
978        } else {
979            parent.x + 1
980        },
981        if (min_i & 2) == 0 {
982            parent.y
983        } else {
984            parent.y + 1
985        },
986        if (min_i & 1) == 0 {
987            parent.z
988        } else {
989            parent.z + 1
990        },
991    );
992    quart_biome(b)
993}

995/// Column-local cache for fuzzed biome lookups (vanilla `BiomeManager.getBiome()`).
996///
997/// Within a column, `parent_x`, `parent_z`, `fract_x`, `fract_z` are constant.
998/// The 8 Voronoi candidate fiddle values (computed via 8 serial LCG calls each)
999/// only change when `parent_y` changes (every 4 blocks). This cache precomputes
1000/// the fiddle values and X/Z distance components per `parent_y` group, reducing
1001/// per-block work to 8 additions + 8 multiplies + 8 comparisons.
1002struct FuzzedBiomeColumn<'a> {
1003    biome_data: &'a [u16],
1004    section_count: usize,
1005    biome_zoom_seed: i64,
1006    parent_x: i32,
1007    parent_z: i32,
1008    fract_x: f64,
1009    fract_z: f64,
1010    min_y: i32,
1011    chunk_quart_x: i32,
1012    chunk_quart_z: i32,
1013    neighbor_biomes: &'a dyn Fn(IVec3) -> u16,
1014    cached_parent_y: i32,
1015    /// Per-candidate cached values: (`fy`, `xz_partial_distance`).
1016    candidates: [(f64, f64); 8],
1017    /// Precomputed `lcg_next(seed, parent_x)` and `lcg_next(seed, parent_x + 1)`.
1018    rval_after_cx: [i64; 2],
1019}
1020
1021impl<'a> FuzzedBiomeColumn<'a> {
1022    #[expect(
1023        clippy::too_many_arguments,
1024        reason = "matches vanilla BiomeManager.getBiome signature"
1025    )]
1026    fn new(
1027        biome_data: &'a [u16],
1028        section_count: usize,
1029        biome_zoom_seed: i64,
1030        block_x: i32,
1031        block_z: i32,
1032        min_y: i32,
1033        chunk_quart_x: i32,
1034        chunk_quart_z: i32,
1035        neighbor_biomes: &'a dyn Fn(IVec3) -> u16,
1036    ) -> Self {
1037        let abs_x = block_x - 2;
1038        let abs_z = block_z - 2;
1039        let parent_x = abs_x >> 2;
1040        let parent_z = abs_z >> 2;
1041        Self {
1042            biome_data,
1043            section_count,
1044            biome_zoom_seed,
1045            parent_x,
1046            parent_z,
1047            fract_x: f64::from(abs_x & 3) / 4.0,
1048            fract_z: f64::from(abs_z & 3) / 4.0,
1049            min_y,
1050            chunk_quart_x,
1051            chunk_quart_z,
1052            neighbor_biomes,
1053            cached_parent_y: i32::MIN,
1054            candidates: [(0.0, 0.0); 8],
1055            rval_after_cx: [
1056                lcg_next(biome_zoom_seed, i64::from(parent_x)),
1057                lcg_next(biome_zoom_seed, i64::from(parent_x + 1)),
1058            ],
1059        }
1060    }
1061
1062    /// Compute candidates for a given `cy`, writing to either the low (bit1=0)
1063    /// or high (bit1=1) slots. Shares the `lcg_next(seed, cx)` precomputation
1064    /// and the `lcg_next(_, cy)` step within each cx group.
1065    #[inline]
1066    fn compute_cy_group(&mut self, cy: i32, high: bool) {
1067        let base_idx = if high { 2 } else { 0 };
1068        for cx_idx in 0..2usize {
1069            let cx = self.parent_x + cx_idx as i32;
1070            let dx = if cx_idx == 0 {
1071                self.fract_x
1072            } else {
1073                self.fract_x - 1.0
1074            };
1075            let rval_cy = lcg_next(self.rval_after_cx[cx_idx], i64::from(cy));
1076            for cz_off in 0..2usize {
1077                let cz = self.parent_z + cz_off as i32;
1078                let dz = if cz_off == 0 {
1079                    self.fract_z
1080                } else {
1081                    self.fract_z - 1.0
1082                };
1083
1084                let mut rval = lcg_next(rval_cy, i64::from(cz));
1085                rval = lcg_next(rval, i64::from(cx));
1086                rval = lcg_next(rval, i64::from(cy));
1087                rval = lcg_next(rval, i64::from(cz));
1088                let fx = get_fiddle(rval);
1089                rval = lcg_next(rval, self.biome_zoom_seed);
1090                let fy = get_fiddle(rval);
1091                rval = lcg_next(rval, self.biome_zoom_seed);
1092                let fz = get_fiddle(rval);
1093
1094                let xz_partial = (dx + fx) * (dx + fx) + (dz + fz) * (dz + fz);
1095                self.candidates[cx_idx * 4 + base_idx + cz_off] = (fy, xz_partial);
1096            }
1097        }
1098    }
1099
1100    /// Recompute the 8 candidate fiddle values and X/Z distance for a new `parent_y`.
1101    ///
1102    /// When scanning downward (`parent_y` decreases by 1), the old low-cy candidates
1103    /// (`cy=old_parent_y`) match the new high-cy slots (`cy=new_parent_y+1`), so only
1104    /// the 4 new low-cy candidates need fresh LCG computation.
1105    fn recompute_candidates(&mut self, parent_y: i32) {
1106        if self.cached_parent_y != i32::MIN && parent_y == self.cached_parent_y - 1 {
1107            // Reuse: old low-cy group → new high-cy group
1108            self.candidates[2] = self.candidates[0];
1109            self.candidates[3] = self.candidates[1];
1110            self.candidates[6] = self.candidates[4];
1111            self.candidates[7] = self.candidates[5];
1112            self.compute_cy_group(parent_y, false);
1113        } else {
1114            self.compute_cy_group(parent_y, false);
1115            self.compute_cy_group(parent_y + 1, true);
1116        }
1117        self.cached_parent_y = parent_y;
1118    }
1119
1120    /// Fuzzed biome lookup for a given `block_y`.
1121    #[expect(
1122        clippy::similar_names,
1123        reason = "matches vanilla variable names: fract_x/y/z, parent_x/y/z"
1124    )]
1125    #[inline]
1126    fn get(&mut self, block_y: i32) -> u16 {
1127        let abs_y = block_y - 2;
1128        let parent_y = abs_y >> 2;
1129        let fract_y = f64::from(abs_y & 3) / 4.0;
1130
1131        if parent_y != self.cached_parent_y {
1132            self.recompute_candidates(parent_y);
1133        }
1134
1135        let mut min_i = 0usize;
1136        let mut min_dist = f64::INFINITY;
1137        for i in 0..8usize {
1138            let (fy, xz_partial) = self.candidates[i];
1139            let dy = if (i & 2) == 0 { fract_y } else { fract_y - 1.0 };
1140            let dist = xz_partial + (dy + fy) * (dy + fy);
1141            if min_dist > dist {
1142                min_i = i;
1143                min_dist = dist;
1144            }
1145        }
1146
1147        let biome_quart = IVec3::new(
1148            if (min_i & 4) == 0 {
1149                self.parent_x
1150            } else {
1151                self.parent_x + 1
1152            },
1153            if (min_i & 2) == 0 {
1154                parent_y
1155            } else {
1156                parent_y + 1
1157            },
1158            if (min_i & 1) == 0 {
1159                self.parent_z
1160            } else {
1161                self.parent_z + 1
1162            },
1163        );
1164
1165        let in_chunk = biome_quart.x >= self.chunk_quart_x
1166            && biome_quart.x < self.chunk_quart_x + 4
1167            && biome_quart.z >= self.chunk_quart_z
1168            && biome_quart.z < self.chunk_quart_z + 4;
1169
1170        if in_chunk {
1171            let min_qy = self.min_y >> 2;
1172            let total_quarts_y = self.section_count * 4;
1173            let local_qx = (biome_quart.x - self.chunk_quart_x) as usize;
1174            let local_qz = (biome_quart.z - self.chunk_quart_z) as usize;
1175            let qy_in_chunk = (biome_quart.y - min_qy).clamp(0, total_quarts_y as i32 - 1) as usize;
