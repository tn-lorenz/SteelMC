23    pub fn empty(
24        context: Arc<WorldGenContext>,
25        _step: &ChunkStep,
26        _cache: &Arc<StaticCache2D<Arc<ChunkHolder>>>,
27        holder: Arc<ChunkHolder>,
28    ) {
29        let sections = (0..context.section_count())
30            .map(|_| ChunkSection::new_empty())
31            .collect::<Vec<_>>()
32            .into_boxed_slice();
33
34        let proto_chunk = ProtoChunk::new(
35            Sections::from_owned(sections),
36            holder.get_pos(),
37            context.min_y(),
38            context.height(),
39        );
40
41        // Use no_notify variant - the caller (apply_step) will notify via the completion channel
42        // to avoid rayon threads contending on tokio's scheduler mutex
43        holder.insert_chunk_no_notify(ChunkAccess::Proto(proto_chunk));
44    }

105    /// # Panics
106    /// Panics if the chunk has not reached `ChunkStatus::Noise`.
107    #[expect(
108        clippy::similar_names,
109        reason = "chunk_x/chunk_z and local_qx/local_qz are intentionally similar"
110    )]
111    pub fn generate_surface(
112        context: Arc<WorldGenContext>,
113        _step: &ChunkStep,
114        cache: &Arc<StaticCache2D<Arc<ChunkHolder>>>,
115        holder: Arc<ChunkHolder>,
116    ) {
117        let chunk = holder
118            .try_chunk(ChunkStatus::Noise)
119            .expect("Chunk not found at status Noise");
120
121        let min_qy = chunk.min_y() >> 2;
122        let total_quarts_y = (chunk.sections().sections.len() * 4) as i32;
123
124        let neighbor_biomes = |qx: i32, qy: i32, qz: i32| -> u16 {
125            let chunk_x = qx >> 2;
126            let chunk_z = qz >> 2;
127            let neighbor = cache.get(chunk_x, chunk_z);
128            let neighbor_chunk = neighbor
129                .try_chunk(ChunkStatus::Biomes)
130                .expect("Neighbor not at Biomes status");
131            let sections = neighbor_chunk.sections();
132            let local_qx = (qx - chunk_x * 4) as usize;
133            let local_qz = (qz - chunk_z * 4) as usize;
134            let qy_clamped = (qy - min_qy).clamp(0, total_quarts_y - 1) as usize;
135            let section_idx = qy_clamped / 4;
136            let local_qy = qy_clamped % 4;
137            sections.sections[section_idx]
138                .read()
139                .biomes
140                .get(local_qx, local_qy, local_qz)
141        };
142
143        context.generator.build_surface(&chunk, &neighbor_biomes);
144    }
