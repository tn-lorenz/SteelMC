1//! This module contains the `WorldGenContext` struct, which is used to provide context for chunk generation.
2
3use std::sync::{Arc, Weak};
4
5use enum_dispatch::enum_dispatch;
6
7use crate::chunk::{
8    chunk_access::ChunkAccess, chunk_generator::ChunkGenerator,
9    flat_chunk_generator::FlatChunkGenerator,
10};
11use crate::world::World;
12
13#[allow(missing_docs)]
14#[enum_dispatch(ChunkGenerator)]
15pub enum ChunkGeneratorType {
16    Flat(FlatChunkGenerator),
17    //Custom(Box<dyn ChunkGenerator>),
18}
19
20/// Context for world generation.
21///
22/// Similar to vanilla's `WorldGenContext`, this provides access to the level/dimension
23/// and generation infrastructure.
24pub struct WorldGenContext {
25    /// The chunk generator to use.
26    pub generator: Arc<ChunkGeneratorType>,
27    /// Weak reference to the world (to avoid circular Arc reference).
28    /// Use `world()` to get a strong reference when needed.
29    world: Weak<World>,
30    // Add other fields as needed:
31    // pub structure_manager: StructureTemplateManager,
32    // pub light_engine: ThreadedLevelLightEngine,
33    // pub main_thread_executor: Executor,
34    // pub unsaved_listener: UnsavedListener,
35}
36
37impl WorldGenContext {
38    /// Creates a new `WorldGenContext`.
39    #[must_use]
40    pub fn new(generator: Arc<ChunkGeneratorType>, world: Weak<World>) -> Self {
41        Self { generator, world }
42    }
43
44    /// Gets a strong reference to the world.
45    ///
46    /// # Panics
47    /// Panics if the world has been dropped.
48    #[must_use]
49    pub fn world(&self) -> Arc<World> {
50        self.world.upgrade().expect("World has been dropped")
51    }
52
53    /// Gets a weak reference to the world.
54    ///
55    /// This is useful for passing to chunks without creating a strong reference cycle.
56    #[must_use]
57    pub fn weak_world(&self) -> Weak<World> {
58        self.world.clone()
59    }
60
61    /// Returns the minimum Y coordinate of the world.
62    #[must_use]
63    pub fn min_y(&self) -> i32 {
64        self.world().get_min_y()
65    }
66
67    /// Returns the total height of the world in blocks.
68    #[must_use]
69    pub fn height(&self) -> i32 {
70        self.world().get_height()
71    }
72
73    #[must_use]
74    /// How many sections this dimension has
75    pub fn section_count(&self) -> usize {
76        (self.height() / 16) as usize
77    }
78}
