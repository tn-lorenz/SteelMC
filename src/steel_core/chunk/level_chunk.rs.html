34/// A chunk that is ready to be sent to the client.
35///
36/// Similar to Java's `LevelChunk`, this holds a weak reference to the world
37/// (called `level` in Java) for callbacks during block state changes.
38pub struct LevelChunk {
39    /// The sections of the chunk.
40    pub sections: Sections,
41    /// The position of the chunk.
42    pub pos: ChunkPos,
43    /// Whether the chunk has been modified since last save.
44    pub dirty: AtomicBool,
45    /// The heightmaps for this chunk (wrapped in `RwLock` for interior mutability).
46    pub heightmaps: SyncRwLock<ChunkHeightmaps>,
47    /// The minimum Y coordinate of the world this chunk belongs to.
48    min_y: i32,
49    /// The total height of the world.
50    height: i32,
51    /// Weak reference to the world (called `level` in Java).
52    /// This mirrors Java's `LevelChunk.level` field.
53    level: Weak<World>,
54    /// Block entities stored in this chunk.
55    block_entities: BlockEntityStorage,
56    /// Entities stored in this chunk.
57    pub entities: EntityStorage,
58    /// Scheduled block ticks pending in this chunk.
59    pub block_ticks: SyncMutex<BlockTickList>,
60    /// Scheduled fluid ticks pending in this chunk.
61    pub fluid_ticks: SyncMutex<FluidTickList>,
62}

64impl LevelChunk {
65    /// Ticks this chunk, processing random block ticks and block entity ticks.
66    ///
67    /// For each section that contains randomly-ticking blocks, selects
68    /// `random_tick_speed` random blocks and calls their `random_tick` behavior.
69    /// Also ticks all ticking block entities in this chunk.
70    ///
71    /// # Arguments
72    /// * `random_tick_speed` - Number of random blocks to tick per section per tick.
73    ///   This is controlled by the `randomTickSpeed` game rule.
74    /// * `tick_count` - Current server tick count (for entity sync timing).
75    ///
76    /// # Panics
77    /// Panics if the block behavior registry has not been initialized.
78    pub fn tick(
79        &self,
80        random_tick_speed: u32,
81        tick_count: i32,
82        ready_block_ticks: &mut Vec<BlockTick>,
83        ready_fluid_ticks: &mut Vec<FluidTick>,
84    ) {
85        // Drain ready scheduled ticks (decrement delays, collect those at 0)
86        ready_block_ticks.extend(self.block_ticks.lock().drain_ready());
87        ready_fluid_ticks.extend(self.fluid_ticks.lock().drain_ready());
88
89        // Tick block entities regardless of random tick speed
90        self.tick_block_entities();
91
92        // Tick entities in this chunk
93        if let Some(world) = self.get_level() {
94            let ticked_entities = self.entities.tick(&world, self.pos, tick_count);
95            if ticked_entities {
96                // Mark chunk dirty since entity state may have changed
97                self.dirty.store(true, Ordering::Release);
98            }
99        }
100
101        if random_tick_speed == 0 {
102            return;
103        }
104
105        for (section_index, section) in self.sections.sections.iter().enumerate() {
106            // Skip sections with no randomly-ticking blocks (lock-free check)
107            if !section.is_randomly_ticking() {
108                continue;
109            }
110
111            let Some(world) = self.get_level() else {
112                return;
113            };
114
115            let block_behaviors = &*BLOCK_BEHAVIORS;
116            let mut rng = rand::rng();
117            let chunk_base_x = self.pos.0.x * 16;
118            let chunk_base_z = self.pos.0.y * 16;
119
120            let section_base_y = self.min_y + (section_index as i32 * 16);
121
122            // Collect blocks to tick while holding the read lock, then release it
123            // before calling random_tick to avoid deadlock (random_tick may call set_block)
124            let blocks_to_tick: Vec<(BlockStateId, BlockPos)> = {
125                let section_guard = section.read();
126
127                let mut blocks = Vec::with_capacity(random_tick_speed as usize);
128
129                for _ in 0..random_tick_speed {
130                    let local_x = rng.random_range(0..16);
131                    let local_y = rng.random_range(0..16);
132                    let local_z = rng.random_range(0..16);
133
134                    let state = section_guard.states.get(local_x, local_y, local_z);
135
136                    if block_behaviors
137                        .get_behavior(state.get_block())
138                        .is_randomly_ticking(state)
139                    {
140                        let pos = BlockPos::new(
141                            chunk_base_x + local_x as i32,
142                            section_base_y + local_y as i32,
143                            chunk_base_z + local_z as i32,
144                        );
145                        blocks.push((state, pos));
146                    }
147                }
148
149                blocks
150            }; // section_guard dropped here
151
152            // Now process the collected blocks without holding any lock
153            for (state, pos) in blocks_to_tick {
154                let behavior = block_behaviors.get_behavior(state.get_block());
155                behavior.random_tick(state, &world, pos);
156            }
157        }
158    }

160    /// Creates a new `LevelChunk` from a `ProtoChunk`.
161    ///
162    /// Transfers final heightmaps from the proto chunk if available.
163    /// Recalculates section block counts for random tick optimization.
164    ///
165    /// # Arguments
166    /// * `proto_chunk` - The proto chunk to convert
167    /// * `min_y` - The minimum Y coordinate of the world
168    /// * `height` - The total height of the world
169    /// * `level` - Weak reference to the world (mirrors Java's `LevelChunk.level`)
170    ///
171    /// # Panics
172    /// Panics if the block behavior registry has not been initialized.
173    #[must_use]
174    pub fn from_proto(
175        proto_chunk: ProtoChunk,
176        min_y: i32,
177        height: i32,
178        level: Weak<World>,
179    ) -> Self {
180        // Transfer final heightmaps from proto chunk if available
181        let proto_heightmaps = proto_chunk.heightmaps.read();
182        let mut chunk_heightmaps = ChunkHeightmaps::new(min_y, height);
183
184        // Copy final heightmap data if available in proto chunk
185        for &hm_type in HeightmapType::final_types() {
186            if let Some(proto_hm) = proto_heightmaps.get(&hm_type) {
187                chunk_heightmaps
188                    .get_mut(hm_type)
189                    .set_raw_data(&proto_hm.get_raw_data());
190            }
191        }
192        drop(proto_heightmaps);
193
194        // Recalculate section counts for random tick optimization
195        for section in &proto_chunk.sections.sections {
196            section.write().recalculate_counts();
197        }
198
199        Self {
200            sections: proto_chunk.sections,
201            pos: proto_chunk.pos,
202            dirty: AtomicBool::new(proto_chunk.dirty.load(Ordering::Acquire)),
203            heightmaps: SyncRwLock::new(chunk_heightmaps),
204            min_y,
205            height,
206            level,
207            block_entities: BlockEntityStorage::new(),
208            entities: EntityStorage::new(),
209            block_ticks: SyncMutex::new(BlockTickList::new()),
210            fluid_ticks: SyncMutex::new(FluidTickList::new()),
211        }
212    }

329    /// Adds an entity to this chunk and registers it with all world systems.
330    ///
331    /// This is the main entry point for adding entities. It handles:
332    /// 1. Adding to chunk's entity storage
333    /// 2. Setting up the level callback for position tracking
334    /// 3. Registering in entity cache for fast lookups
335    /// 4. Adding to entity tracker and sending spawn packets to nearby players
336    /// 5. Marking the chunk dirty for persistence
337    ///
338    /// Returns `false` if the world reference is no longer valid.
339    pub fn add_and_register_entity(&self, entity: SharedEntity) -> bool {
340        use crate::entity::EntityChunkCallback;
341
342        let Some(world) = self.level.upgrade() else {
343            return false;
344        };
345
346        // Add to chunk storage
347        self.entities.add(entity.clone());
348
349        // Set up callback for chunk/section tracking
350        let callback = Arc::new(EntityChunkCallback::new(&entity, Arc::downgrade(&world)));
351        entity.set_level_callback(callback);
352
353        // Register in entity cache (for fast lookups)
354        world.entity_cache().register(&entity);
355
356        // Add to entity tracker and send spawn packets to nearby players
357        world.entity_tracker().add(
358            &entity,
359            |chunk| world.player_area_map.get_tracking_players(chunk),
360            |id| world.players.get_by_entity_id(id),
361        );
362
363        // Mark chunk dirty for persistence
364        self.mark_unsaved();
365
366        true
367    }

396    /// Ticks all ticking block entities in this chunk.
397    ///
398    /// Called each game tick for chunks that are in ticking range.
399    pub fn tick_block_entities(&self) {
400        let Some(world) = self.get_level() else {
401            return;
402        };
403
404        // Get entities to tick (already filters out removed)
405        let entities = self.block_entities.get_tickers();
406
407        // Tick each entity
408        for entity in entities {
409            let mut guard = entity.lock();
410            if guard.is_removed() {
411                continue;
412            }
413            guard.tick(&world);
414        }
415
416        // Clean up removed entities from the ticking list
417        self.block_entities.cleanup_tickers();
418    }

420    /// Sets a block state at the given position.
421    ///
422    /// Returns the old block state, or `None` if nothing changed.
423    ///
424    /// # Arguments
425    /// * `pos` - The absolute block position
426    /// * `state` - The new block state to set
427    /// * `flags` - Update flags controlling behavior
428    ///
429    /// # Panics
430    ///
431    /// Panics if the behavior registry has not been initialized.
432    #[must_use]
433    pub fn set_block_state(
434        &self,
435        pos: BlockPos,
436        state: BlockStateId,
437        flags: UpdateFlags,
438    ) -> Option<BlockStateId> {
439        let y = pos.0.y;
440
441        if y < self.min_y || y >= self.min_y + self.height {
442            return None;
443        }
444
445        let section_index = self.get_section_index(y);
446
447        if section_index >= self.sections.sections.len() {
448            return None;
449        }
450
451        let section = &self.sections.sections[section_index];
452
453        let was_empty = section.read().is_empty();
454        if was_empty && state.is_air() {
455            return None;
456        }
457
458        let local_x = (pos.0.x & 15) as usize;
459        let local_y = (y & 15) as usize;
460        let local_z = (pos.0.z & 15) as usize;
461
462        let old_state = section
463            .write()
464            .set_block_state(local_x, local_y, local_z, state);
465
466        if old_state == state {
467            return None;
468        }
469
470        let min_y = self.min_y;
471        let sections = &self.sections;
472        self.heightmaps
473            .write()
474            .update(local_x, y, local_z, state, |lx, scan_y, lz| {
475                let scan_section_index = ((scan_y - min_y) / 16) as usize;
476                let scan_local_y = ((scan_y - min_y) % 16) as usize;
477                sections.sections[scan_section_index]
478                    .read()
479                    .states
480                    .get(lx, scan_local_y, lz)
481            });
482
483        let old_block = old_state.get_block();
484        let new_block = state.get_block();
485
486        // TODO: Light updates
487        // In vanilla, light engine is notified when section emptiness changes:
488        // let is_empty = section.read().states.has_only_air();
489        // if was_empty != is_empty {
490        //     level.chunk_source.light_engine.update_section_status(pos, is_empty);
491        //     level.chunk_source.on_section_emptiness_changed(chunk_pos.x, section_y, chunk_pos.z, is_empty);
492        // }
493        //
494        // And when light properties change:
495        // if LightEngine::has_different_light_properties(old_state, state) {
496        //     self.sky_light_sources.update(self, local_x, y, local_z);
497        //     level.chunk_source.light_engine.check_block(pos);
498        // }
499
500        // Re-read the block to verify it wasn't changed concurrently
501        let current_block = section
502            .read()
503            .states
504            .get(local_x, local_y, local_z)
505            .get_block();
506        if !ptr::eq(current_block, new_block) {
507            return None;
508        }
509
510        if let Some(level) = self.get_level() {
511            let block_changed = !ptr::eq(old_block, new_block);
512            let moved_by_piston = flags.contains(UpdateFlags::UPDATE_MOVE_BY_PISTON);
513            let side_effects = !flags.contains(UpdateFlags::UPDATE_SKIP_BLOCK_ENTITY_SIDEEFFECTS);
514
515            let block_behaviors = &*BLOCK_BEHAVIORS;
516            let old_behavior = block_behaviors.get_behavior(old_block);
517            let new_behavior = block_behaviors.get_behavior(new_block);
518
519            // Block entity removal when block type changes
520            if block_changed && old_behavior.has_block_entity() {
521                let should_keep = new_behavior.should_keep_block_entity(old_state, state);
522                if !should_keep {
523                    if side_effects && let Some(block_entity) = self.get_block_entity(pos) {
524                        block_entity.lock().pre_remove_side_effects(pos, old_state);
525                    }
526                    self.remove_block_entity(pos);
527                }
528            }
529
530            // Notify neighbors that we were removed (for rails, etc.)
531            if block_changed && (flags.contains(UpdateFlags::UPDATE_NEIGHBORS) || moved_by_piston) {
532                old_behavior.affect_neighbors_after_removal(
533                    old_state,
534                    &level,
535                    pos,
536                    moved_by_piston,
537                );
538            }
539
540            // Call on_place for the new block
541            if !flags.contains(UpdateFlags::UPDATE_SKIP_ON_PLACE) {
542                new_behavior.on_place(state, &level, pos, old_state, moved_by_piston);
543            }
544
545            // Block entity creation after on_place
546            if new_behavior.has_block_entity() {
547                if let Some(existing) = self.get_block_entity(pos) {
548                    // Update existing block entity's state
549                    existing.lock().set_block_state(state);
550                    self.update_block_entity_ticker(&existing);
551                } else {
552                    // Create new block entity
553                    if let Some(entity) =
554                        new_behavior.new_block_entity(self.level.clone(), pos, state)
555                    {
556                        self.add_and_register_block_entity(entity);
557                    }
558                }
559            }
560        }
561
562        self.mark_unsaved();
563        Some(old_state)
564    }

566    /// Gets a block state at the given position.
567    #[must_use]
568    pub fn get_block_state(&self, pos: BlockPos) -> BlockStateId {
569        let y = pos.0.y;
570        let section_index = self.get_section_index(y);
571
572        // Bounds check - return air if out of range
573        if section_index >= self.sections.sections.len() {
574            return REGISTRY.blocks.get_base_state_id(vanilla_blocks::VOID_AIR);
575        }
576
577        let section = &self.sections.sections[section_index];
578        let section_guard = section.read();
579
580        if section_guard.is_empty() {
581            return REGISTRY.blocks.get_base_state_id(vanilla_blocks::VOID_AIR);
582        }
583
584        let local_x = (pos.0.x & 15) as usize;
585        let local_y = (y & 15) as usize;
586        let local_z = (pos.0.z & 15) as usize;
587
588        section_guard.states.get(local_x, local_y, local_z)
589    }

591    /// Extracts the chunk data for sending to the client.
592    #[must_use]
593    pub fn extract_chunk_data(&self) -> ChunkPacketData {
594        let data = Vec::new();
595
596        let mut cursor = Cursor::new(data);
597        self.sections.sections.iter().for_each(|section| {
598            section.read().write(&mut cursor);
599        });
600
601        let heightmaps_guard = self.heightmaps.read();
602
603        // Collect block entity data for client sync
604        let block_entities: Vec<BlockEntityInfo> = self
605            .block_entities
606            .get_all()
607            .iter()
608            .map(|entity| {
609                let guard = entity.lock();
610                let pos = guard.get_block_pos();
611                let type_id = *REGISTRY.block_entity_types.get_id(guard.get_type()) as i32;
612                let update_tag = guard.get_update_tag();
613
614                // Pack local X and Z coordinates into a single byte
615                let local_x = (pos.0.x & 15) as u8;
616                let local_z = (pos.0.z & 15) as u8;
617                let packed_xz = (local_x << 4) | local_z;
618
619                BlockEntityInfo {
620                    packed_xz,
621                    y: pos.0.y as i16,
622                    type_id,
623                    data: update_tag.into(),
624                }
625            })
626            .collect();
627
628        ChunkPacketData {
629            heightmaps: Heightmaps {
630                heightmaps: vec![
631                    (
632                        ProtocolHeightmapType::WorldSurface,
633                        heightmaps_guard
634                            .get(HeightmapType::WorldSurface)
635                            .get_raw_data(),
636                    ),
637                    (
638                        ProtocolHeightmapType::MotionBlocking,
639                        heightmaps_guard
640                            .get(HeightmapType::MotionBlocking)
641                            .get_raw_data(),
642                    ),
643                    (
644                        ProtocolHeightmapType::MotionBlockingNoLeaves,
645                        heightmaps_guard
646                            .get(HeightmapType::MotionBlockingNoLeaves)
647                            .get_raw_data(),
648                    ),
649                ],
650            },
651            data: cursor.into_inner(),
652            block_entities,
653        }
654    }

656    /// Extracts the light data for sending to the client.
657    #[must_use]
658    pub fn extract_light_data(&self) -> LightUpdatePacketData {
659        // Vanilla's light section count is sectionsCount + 2 (one below and one above the world)
660        let light_section_count = self.sections.sections.len() + 2;
661        let mut sky_y_mask = BitSet(vec![0; light_section_count.div_ceil(64)].into_boxed_slice());
662        let mut block_y_mask = BitSet(vec![0; light_section_count.div_ceil(64)].into_boxed_slice());
663        let empty_sky_y_mask = BitSet(vec![0; light_section_count.div_ceil(64)].into_boxed_slice());
664        let empty_block_y_mask =
665            BitSet(vec![0; light_section_count.div_ceil(64)].into_boxed_slice());
666
667        let mut sky_updates = Vec::new();
668        let mut block_updates = Vec::new();
669
670        for i in 0..light_section_count {
671            sky_y_mask.set(i, true);
672            block_y_mask.set(i, true);
