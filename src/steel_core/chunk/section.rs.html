1//! This module contains the `Sections` and `ChunkSection` structs.
2use std::{fmt::Debug, io::Cursor};
3
4use steel_utils::{BlockStateId, serial::WriteTo};

6use crate::chunk::paletted_container::{BiomePalette, BlockPalette};

8/// A collection of chunk sections.
9#[derive(Debug, Clone)]
10pub struct Sections {
11    /// The sections in the collection.
12    pub sections: Box<[ChunkSection]>,
13}
14
15impl Sections {
16    /// Gets the sections in the collection.
17    #[must_use]
18    pub const fn get(&self) -> &[ChunkSection] {
19        &self.sections
20    }
21
22    /// Gets a block at a relative position in the chunk.
23    #[must_use]
24    pub fn get_relative_block(
25        &self,
26        relative_x: usize,
27        relative_y: usize,
28        relative_z: usize,
29    ) -> Option<BlockStateId> {
30        debug_assert!(relative_x < BlockPalette::SIZE);
31        debug_assert!(relative_z < BlockPalette::SIZE);
32
33        let section_index = relative_y / BlockPalette::SIZE;
34        let relative_y = relative_y % BlockPalette::SIZE;
35        self.sections
36            .get(section_index)
37            .map(|section| section.states.get(relative_x, relative_y, relative_z))
38    }
39
40    /// Sets a block at a relative position in the chunk.
41    pub fn set_relative_block(
42        &mut self,
43        relative_x: usize,
44        relative_y: usize,
45        relative_z: usize,
46        value: BlockStateId,
47    ) {
48        debug_assert!(relative_x < BlockPalette::SIZE);
49        debug_assert!(relative_z < BlockPalette::SIZE);
50
51        let idx = relative_y / BlockPalette::SIZE;
52        let relative_y = relative_y % BlockPalette::SIZE;
53        //println!(
54        //    "setting block at {}, {}, {} to {}",
55        //    relative_x, relative_y, relative_z, value.0
56        //);
57        self.sections[idx]
58            .states
59            .set(relative_x, relative_y, relative_z, value);
60    }
61}

63/// A chunk section.
64#[derive(Debug, Clone)]
65pub struct ChunkSection {
66    /// The block states in the section.
67    pub states: BlockPalette,
68    /// The biomes in the section.
69    pub biomes: BiomePalette,
70}
71
72impl ChunkSection {
73    /// Creates a new chunk section.
74    #[must_use]
75    pub fn new(states: BlockPalette) -> Self {
76        Self {
77            states,
78            biomes: BiomePalette::Homogeneous(0),
79        }
80    }
81
82    /// Creates a new empty chunk section.
83    #[must_use]
84    pub fn new_empty() -> Self {
85        Self {
86            states: BlockPalette::Homogeneous(BlockStateId(0)),
87            biomes: BiomePalette::Homogeneous(0),
88        }
89    }
90
91    /// Writes the chunk section to a writer.
92    ///
93    /// # Panics
94    /// - If the writer fails to write.
95    pub fn write(&self, writer: &mut Cursor<Vec<u8>>) {
96        self.states
97            .non_empty_block_count()
98            .write(writer)
99            .expect("Failed to write block count");
100
101        self.states
102            .write(writer)
103            .expect("Failed to write block states");
104        self.biomes.write(writer).expect("Failed to write biomes");
105    }
106}
