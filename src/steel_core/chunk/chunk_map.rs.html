141    /// Records a block change at the given position.
142    /// This marks the chunk as having pending changes to broadcast.
143    pub fn block_changed(&self, pos: &BlockPos) {
144        let chunk_pos = ChunkPos::new(
145            SectionPos::block_to_section_coord(pos.0.x),
146            SectionPos::block_to_section_coord(pos.0.z),
147        );
148
149        if let Some(holder) = self.chunks.read_sync(&chunk_pos, |_, h| h.clone())
150            && holder.block_changed(pos)
151        {
152            // First change for this chunk - add to broadcast list
153            self.chunks_to_broadcast.lock().push(holder);
154        }
155    }

157    /// Broadcasts all pending block changes to nearby players.
158    ///
159    /// # Panics
160    /// Panics if a section has exactly one change (should never happen).
161    pub fn broadcast_changed_chunks(&self) {
162        let holders = {
163            let mut guard = self.chunks_to_broadcast.lock();
164            if guard.is_empty() {
165                return;
166            }
167            mem::take(&mut *guard)
168        };
169
170        let world = self.world_gen_context.world();
171
172        for holder in holders {
173            let chunk_pos = holder.get_pos();
174            let min_y = holder.min_y();
175
176            // Take all pending changes from this chunk holder
177            let changes_by_section = holder.take_changed_blocks();
178
179            if changes_by_section.is_empty() {
180                continue;
181            }
182
183            // Get players tracking this chunk
184            let tracking_players = world.player_area_map.get_tracking_players(chunk_pos);
185            if tracking_players.is_empty() {
186                continue;
187            }
188
189            // For each section with changes, send appropriate packet
190            for (section_index, changed_positions) in changes_by_section {
191                let section_y = min_y / 16 + section_index as i32;
192                let section_pos = SectionPos::new(chunk_pos.0.x, section_y, chunk_pos.0.y);
193
194                if changed_positions.len() == 1 {
195                    // Single block change - use CBlockUpdate
196                    let packed = *changed_positions.iter().next().expect("len == 1");
197                    let block_pos = section_pos.relative_to_block_pos(packed);
198                    let block_state = world.get_block_state(&block_pos);
199
200                    tracing::debug!(
201                        ?block_pos,
202                        ?block_state,
203                        player_count = tracking_players.len(),
204                        "Broadcasting single block update"
205                    );
206
207                    let update_packet = CBlockUpdate {
208                        pos: block_pos,
209                        block_state,
210                    };
211
212                    for entity_id in &tracking_players {
213                        if let Some(player) = world.players.get_by_entity_id(*entity_id) {
214                            player.connection.send_packet(update_packet.clone());
215                        }
216                    }
217                } else {
218                    // Multiple block changes - use CSectionBlocksUpdate
219                    let changes: Vec<BlockChange> = changed_positions
220                        .iter()
221                        .map(|&packed| {
222                            let block_pos = section_pos.relative_to_block_pos(packed);
223                            let block_state = world.get_block_state(&block_pos);
224                            BlockChange {
225                                pos: block_pos,
226                                block_state,
227                            }
228                        })
229                        .collect();
230
231                    tracing::debug!(
232                        change_count = changes.len(),
233                        ?section_pos,
234                        player_count = tracking_players.len(),
235                        "Broadcasting section block updates"
236                    );
237
238                    let packet = CSectionBlocksUpdate {
239                        section_pos,
240                        changes,
241                    };
242
243                    for entity_id in &tracking_players {
244                        if let Some(player) = world.players.get_by_entity_id(*entity_id) {
245                            player.connection.send_packet(packet.clone());
246                        }
247                    }
248                }
249            }
250        }
251    }

271    /// Runs queued generation tasks.
272    #[instrument(level = "trace", skip(self))]
273    pub fn run_generation_tasks_b(&self) {
274        let mut pending = self.pending_generation_tasks.lock();
275        if pending.is_empty() {
276            return;
277        }
278        let task_count = pending.len();
279        tracing::trace!(task_count, "Running generation tasks");
280        let tasks = pending.drain(..).collect::<Vec<_>>();
281        drop(pending); // Release lock before spawning
282
283        for task in tasks {
284            self.task_tracker
285                .spawn_on(async move { task.run().await }, self.chunk_runtime.handle());
286        }
287    }

289    /// Updates scheduling for a chunk based on its new level.
290    /// Returns the chunk holder if it is active.
291    #[inline]
292    #[allow(clippy::missing_panics_doc, clippy::unwrap_used)]
293    pub fn update_chunk_level(
294        self: &Arc<Self>,
295        pos: &ChunkPos,
296        new_level: Option<u8>,
297    ) -> Option<Arc<ChunkHolder>> {
298        // Recover from unloading if possible, else create new holder.
299        let chunk_holder =
300            if let Some(holder) = self.chunks.read_sync(pos, |_, holder| holder.clone()) {
301                holder
302            } else {
303                new_level?;
304
305                if let Some(entry) = self.unloading_chunks.remove_sync(pos) {
306                    let _ = self.chunks.insert_sync(*pos, entry.1.clone());
307                    entry.1
308                } else {
309                    let holder = Arc::new(ChunkHolder::new(
310                        *pos,
311                        new_level.unwrap(),
312                        self.world_gen_context.min_y(),
313                        self.world_gen_context.height(),
314                    ));
315                    let _ = self.chunks.insert_sync(*pos, holder.clone());
316                    holder
317                }
318            };
319
320        if let Some(level) = new_level {
321            let old = chunk_holder.ticket_level.swap(level, Ordering::Relaxed);
322            if old != level {
323                chunk_holder.update_highest_allowed_status(level);
324            }
325            Some(chunk_holder)
326        } else {
327            //log::info!("Unloading chunk at {pos:?}");
328            chunk_holder.cancel_generation_task();
329            chunk_holder.ticket_level.store(u8::MAX, Ordering::Relaxed);
330            chunk_holder.update_highest_allowed_status(u8::MAX);
331
332            // Move to unloading_chunks for deferred unload
333            if let Some((_, holder)) = self.chunks.remove_sync(pos) {
334                let _ = self.unloading_chunks.insert_sync(*pos, holder);
335            }
336            None
337        }
338    }

340    /// Processes chunk updates and ticks chunks.
341    ///
342    /// # Arguments
343    /// * `tick_count` - The current server tick count
344    /// * `random_tick_speed` - Number of random blocks to tick per section per tick
345    /// * `runs_normally` - Whether game elements should run (false when frozen)
346    ///
347    /// Returns timing information for each phase of the tick.
348    #[allow(clippy::too_many_lines)]
349    #[instrument(level = "trace", skip(self), name = "chunk_map_tick")]
350    pub fn tick_b(
351        self: &Arc<Self>,
352        tick_count: u64,
353        random_tick_speed: u32,
354        runs_normally: bool,
355    ) -> ChunkMapTickTimings {
356        let mut timings = ChunkMapTickTimings::default();
357
358        {
359            let mut ct = self.chunk_tickets.lock();
360
361            // Only process chunks that actually changed
362            let changes: Vec<LevelChange> = {
363                let _span = tracing::trace_span!("ticket_updates").entered();
364                let start = Instant::now();
365                let result = ct.run_all_updates().to_vec();
366                timings.ticket_updates = start.elapsed();
367                result
368            };
369
370            let holders_to_schedule: Vec<_> = {
371                let _span = tracing::trace_span!("holder_creation").entered();
372                let start = Instant::now();
373                let result = changes
374                    .iter()
375                    .filter_map(|change| {
376                        self.update_chunk_level(&change.pos, change.new_level)
377                            .map(|holder| (holder, change.new_level))
378                    })
379                    .collect();
380                timings.holder_creation = start.elapsed();
381                result
382            };
383
384            {
385                let _span = tracing::trace_span!("schedule_generation").entered();
386                let start = Instant::now();
387                let scheduled_count: usize = holders_to_schedule
388                    .into_par_iter()
389                    .filter(|(holder, level)| {
390                        level.is_some_and(is_full)
391                            && holder.schedule_chunk_generation_task_b(ChunkStatus::Full, self)
392                    })
393                    .count();
394                timings.schedule_generation = start.elapsed();
395                timings.scheduled_count = scheduled_count;
396            }
397        };
398
399        {
400            let _span = tracing::trace_span!("run_generation").entered();
401            let start = Instant::now();
402            self.run_generation_tasks_b();
403            timings.run_generation = start.elapsed();
404        }
405
406        {
407            let _span = tracing::trace_span!("broadcast_changes").entered();
408            let start = Instant::now();
409            self.broadcast_changed_chunks();
410            timings.broadcast_changes = start.elapsed();
411        }
412
413        {
414            let _span = tracing::trace_span!("process_unloads").entered();
415            let start = Instant::now();
416            self.process_unloads();
417            timings.process_unloads = start.elapsed();
418        }
419
420        if tick_count.is_multiple_of(100) {
421            tracing::debug!(
422                chunks = self.chunks.len(),
423                unloading = self.unloading_chunks.len(),
424                "Chunk map status"
425            );
426        }
427
428        // Chunk ticking - skip when frozen
429        if !runs_normally {
430            return timings;
431        }
432
433        {
434            let _span = tracing::trace_span!("collect_tickable").entered();
435            let start = Instant::now();
436            let mut total_chunks = 0;
437            let last_len = self.last_tickable_len.load(Ordering::Relaxed);
438            let mut tickable_chunks = Vec::with_capacity(last_len);
439            self.chunks.iter_sync(|_, holder| {
440                total_chunks += 1;
441                let level = holder.ticket_level.load(Ordering::Relaxed);
442                if is_ticked(level) {
443                    tickable_chunks.push(holder.clone());
444                }
445                true
446            });
447            self.last_tickable_len
448                .store(tickable_chunks.len(), Ordering::Relaxed);
449            timings.collect_tickable = start.elapsed();
450            timings.total_chunks = total_chunks;
451            timings.tickable_count = tickable_chunks.len();
452
453            if !tickable_chunks.is_empty() {
454                let _span = tracing::trace_span!(
455                    "tick_chunks",
456                    count = tickable_chunks.len(),
457                    total_chunks
458                )
459                .entered();
460                let start = Instant::now();
461                // TODO: In the future we might want to tick different regions/islands in parallel
462                for holder in &tickable_chunks {
463                    if let Some(chunk_guard) = holder.try_chunk(ChunkStatus::Full) {
464                        chunk_guard.tick(random_tick_speed);
465                    }
466                }
467                timings.tick_chunks = start.elapsed();
468            }
469        }
470
471        timings
472    }

474    /// Saves a chunk to disk. Does not remove from `unloading_chunks`.
475    #[allow(clippy::missing_panics_doc, clippy::unwrap_used)]
476    #[instrument(level = "trace", skip(self, chunk_holder), fields(chunk = ?chunk_holder.get_pos()))]
477    async fn save_chunk(&self, chunk_holder: &Arc<ChunkHolder>) {
478        // Prepare chunk data while holding the lock, then release before async I/O
479        let prepared = {
480            let Some(chunk_guard) = chunk_holder.try_chunk(ChunkStatus::StructureStarts) else {
481                // Chunk was at Empty stage so no need to save it
482                return;
483            };
484
485            let status = chunk_holder
486                .persisted_status()
487                .expect("The check above confirmed it exists");
488
489            let prepared = RegionManager::prepare_chunk_save(&chunk_guard);
490
491            // Clear dirty flag while we still have the lock (only if we're actually saving)
492            if prepared.is_some() {
493                chunk_guard.clear_dirty();
494            }
495
496            (prepared, status)
497        }; // chunk_guard dropped here
498
499        let (prepared, status) = prepared;
500
501        // Save chunk data if dirty
502        if let Some(prepared) = prepared {
503            let result = self.region_manager.save_chunk_data(prepared, status).await;
504
505            if let Err(e) = result {
506                tracing::error!("Error saving chunk: {e}");
507            }
508        }
509    }

511    /// Processes chunks that are pending unload.
512    ///
513    /// Iterates over `unloading_chunks`. For each chunk with `strong_count == 1`:
514    /// - If dirty: spawn save task (keep until saved and clean)
515    /// - If not dirty: release region handle and remove
516    #[instrument(level = "trace", skip(self))]
517    pub fn process_unloads(self: &Arc<Self>) {
518        self.unloading_chunks.retain_sync(|pos, holder| {
519            if Arc::strong_count(holder) == 1 {
520                // Check if dirty by trying to get chunk access
521                let is_dirty = holder
522                    .try_chunk(ChunkStatus::StructureStarts)
523                    .is_some_and(|chunk| chunk.is_dirty());
524
525                if is_dirty {
526                    // Save the chunk, keep until next tick when it's clean
527                    let holder_clone = holder.clone();
528                    let map_clone = self.clone();
529                    self.task_tracker.spawn(async move {
530                        map_clone.save_chunk(&holder_clone).await;
531                    });
532                    true // keep until clean
533                } else {
534                    // Clean and no refs - release region handle and remove
535                    let pos = *pos;
536                    let map_clone = self.clone();
537                    self.task_tracker.spawn(async move {
538                        if let Err(e) = map_clone.region_manager.release_chunk(pos).await {
539                            tracing::error!(?pos, "Error releasing chunk: {e}");
540                        }
541                    });
542                    false // remove
543                }
544            } else {
545                true // keep, still has refs
546            }
547        });
548    }

550    /// Updates the player's status in the chunk map.
551    pub fn update_player_status(&self, player: &Player) {
552        let current_chunk_pos = *player.last_chunk_pos.lock();
553        let view_distance = player.view_distance();
554
555        let new_view = PlayerChunkView::new(current_chunk_pos, view_distance);
556        let mut last_view_guard = player.last_tracking_view.lock();
557
558        if last_view_guard.as_ref() != Some(&new_view) {
559            let mut chunk_tickets = self.chunk_tickets.lock();
560
561            let connection = &player.connection;
562            let world = self.world_gen_context.world();
563
564            if let Some(last_view) = last_view_guard.as_ref() {
565                if last_view.center != new_view.center
566                    || last_view.view_distance != new_view.view_distance
567                {
568                    chunk_tickets.remove_ticket(
569                        last_view.center,
570                        MAX_VIEW_DISTANCE.saturating_sub(last_view.view_distance),
571                    );
572                    chunk_tickets.add_ticket(
573                        new_view.center,
574                        MAX_VIEW_DISTANCE.saturating_sub(new_view.view_distance),
575                    );
576
577                    connection.send_packet(CSetChunkCenter {
578                        x: new_view.center.0.x,
579                        y: new_view.center.0.y,
580                    });
581                }
582
583                // Track chunks for PlayerAreaMap update
584                let mut added_chunks = Vec::new();
585                let mut removed_chunks = Vec::new();
586
587                // We lock here to ensure we have unique access for the duration of the diff
588                let mut chunk_sender = player.chunk_sender.lock();
589                PlayerChunkView::difference(
590                    last_view,
591                    &new_view,
592                    |pos, ctx: &mut (&mut _, &mut Vec<_>, &mut Vec<_>)| {
593                        ctx.0.mark_chunk_pending_to_send(pos);
594                        ctx.1.push(pos);
595                    },
596                    |pos, ctx: &mut (&mut _, &mut Vec<_>, &mut Vec<_>)| {
597                        ctx.0.drop_chunk(connection, pos);
598                        ctx.2.push(pos);
599                    },
600                    &mut (&mut chunk_sender, &mut added_chunks, &mut removed_chunks),
601                );
602                drop(chunk_sender);
603
604                // Update the player area map with the diff
605                world.player_area_map.on_player_view_change(
606                    player.id,
607                    &added_chunks,
608                    &removed_chunks,
609                );
610            } else {
611                chunk_tickets.add_ticket(
612                    new_view.center,
613                    MAX_VIEW_DISTANCE.saturating_sub(new_view.view_distance),
614                );
615
616                let mut chunk_sender = player.chunk_sender.lock();
617                new_view.for_each(|pos| {
618                    chunk_sender.mark_chunk_pending_to_send(pos);
619                });
620                drop(chunk_sender);
621
622                // First time - add all chunks in view to player area map
623                world.player_area_map.on_player_join(player, &new_view);
624            }
625
626            *last_view_guard = Some(new_view);
627        }
628    }

630    /// Removes a player from the chunk map.
631    pub fn remove_player(&self, player: &Player) {
632        // Okay to lock sync lock here cause it has low contention
633        let mut last_view_guard = player.last_tracking_view.lock();
634        if let Some(last_view) = last_view_guard.take() {
635            drop(last_view_guard);
636            let mut chunk_tickets = self.chunk_tickets.lock();
637            chunk_tickets.remove_ticket(
638                last_view.center,
639                MAX_VIEW_DISTANCE.saturating_sub(last_view.view_distance),
640            );
641        }

644    /// Saves all dirty chunks to disk.
645    ///
646    /// This method should be called during graceful shutdown to ensure all
647    /// modified chunks are persisted. It saves:
648    /// 1. All dirty chunks in the active `chunks` map
649    /// 2. All chunks pending unload in the `unloading_chunks` map
650    /// 3. Closes all region file handles (flushing headers)
651    ///
652    /// Returns the number of chunks saved.
653    #[instrument(level = "info", skip(self), name = "save_all_chunks")]
654    pub async fn save_all_chunks(self: &Arc<Self>) -> io::Result<usize> {
655        let mut saved_count = 0;
656
657        // Collect all chunks from both maps
658        let all_chunks: Vec<Arc<ChunkHolder>> = {
659            let mut chunks = Vec::new();
660            self.chunks.iter_sync(|_, holder| {
661                chunks.push(holder.clone());
662                true
663            });
664            self.unloading_chunks.iter_sync(|_, holder| {
665                chunks.push(holder.clone());
666                true
667            });
668            chunks
669        };
670
671        tracing::info!(chunk_count = all_chunks.len(), "Saving chunks");
672
673        // Save all chunks that have data
674        for holder in &all_chunks {
675            let prepared = {
676                let Some(chunk) = holder.try_chunk(ChunkStatus::StructureStarts) else {
677                    continue;
678                };
679                let Some(status) = holder.persisted_status() else {
680                    continue;
681                };
682                let Some(prepared) = RegionManager::prepare_chunk_save(&chunk) else {
683                    continue; // Not dirty
684                };
685                chunk.clear_dirty();
686                (prepared, status)
687            };
688
689            let (prepared, status) = prepared;
690            match self.region_manager.save_chunk_data(prepared, status).await {
691                Ok(true) => saved_count += 1,
692                Ok(false) => {} // Not dirty
693                Err(e) => {
694                    tracing::error!(chunk = ?holder.get_pos(), "Failed to save chunk: {e}");
695                }
696            }
697        }
698
699        // Close all region files (flushes headers and releases file handles)
700        if let Err(e) = self.region_manager.close_all().await {
701            tracing::error!("Failed to close region files: {e}");
702        }
703
704        tracing::info!(
705            saved_count,
706            total_checked = all_chunks.len(),
707            "Chunk save complete"
708        );
709
710        Ok(saved_count)
711    }
712}
