5/// A chunk generator that generates a flat world.

9    /// The block state id for dirt.

11    /// The block state id for grass blocks.

13}

15impl FlatChunkGenerator {
16    /// Creates a new `FlatChunkGenerator`.

19        Self {
20            bedrock,
21            dirt,
22            grass,
23        }
24    }

32    fn fill_from_noise(&self, chunk: &ChunkAccess) {
33        // Layers:
34        // 0: Bedrock
35        // 1-2: Dirt
36        // 3: Grass Block
37        // (Relative to bottom of chunk, assuming 0 is bottom)
38
39        // TODO: Get actual height from level/config?
40        // For now assuming standard height behavior where 0 is bottom of the chunk data.
41
42        for x in 0..16 {
43            for z in 0..16 {
44                // Bedrock at bottom
45                chunk.set_relative_block(x, 0, z, self.bedrock);
46
47                // Dirt layers
48                chunk.set_relative_block(x, 1, z, self.dirt);
49                chunk.set_relative_block(x, 2, z, self.dirt);
50
51                // Grass block
52                chunk.set_relative_block(x, 3, z, self.grass);
53            }
54        }
55    }
