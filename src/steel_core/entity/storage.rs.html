1//! Per-chunk entity storage.
2//!
3//! Entities are stored in chunks similar to block entities.
4//! The chunk owns the `Arc<dyn Entity>` and is responsible for ticking.
5
6use std::sync::Arc;
7
8use rustc_hash::FxHashMap;
9use steel_protocol::packets::game::CSetEntityData;
10use steel_utils::ChunkPos;
11use steel_utils::locks::SyncRwLock;
12
13use super::SharedEntity;
14use crate::world::World;
15
16/// Storage for entities in a chunk.
17///
18/// This mirrors `BlockEntityStorage` - entities are keyed by their ID
19/// and ticked from the chunk's tick method.
20pub struct EntityStorage {
21    /// Entities in this chunk, keyed by entity ID.
22    entities: SyncRwLock<FxHashMap<i32, SharedEntity>>,
23}
24
25impl EntityStorage {
26    /// Creates a new empty entity storage.
27    #[must_use]
28    pub fn new() -> Self {
29        Self {
30            entities: SyncRwLock::new(FxHashMap::default()),
31        }
32    }
33
34    /// Adds an entity to this chunk's storage.
35    pub fn add(&self, entity: SharedEntity) {
36        let id = entity.id();
37        self.entities.write().insert(id, entity);
38    }
39
40    /// Removes an entity from this chunk's storage by ID.
41    ///
42    /// Returns the entity if it was present.
43    pub fn remove(&self, entity_id: i32) -> Option<SharedEntity> {
44        self.entities.write().remove(&entity_id)
45    }
46
47    /// Gets an entity by ID.
48    #[must_use]
49    pub fn get(&self, entity_id: i32) -> Option<SharedEntity> {
50        self.entities.read().get(&entity_id).cloned()
51    }
52
53    /// Returns all entities in this chunk.
54    #[must_use]
55    pub fn get_all(&self) -> Vec<SharedEntity> {
56        self.entities.read().values().cloned().collect()
57    }
58
59    /// Returns the number of entities in this chunk.
60    #[must_use]
61    pub fn len(&self) -> usize {
62        self.entities.read().len()
63    }
64
65    /// Returns whether there are no entities in this chunk.
66    #[must_use]
67    pub fn is_empty(&self) -> bool {
68        self.entities.read().is_empty()
69    }
70
71    /// Returns entities that should be saved when the chunk is persisted.
72    ///
73    /// Excludes:
74    /// - Removed entities
75    /// - Players (saved separately in playerdata)
76    /// - Entity types with `can_serialize = false`
77    #[must_use]
78    pub fn get_saveable_entities(&self) -> Vec<SharedEntity> {
79        self.entities
80            .read()
81            .values()
82            .filter(|e| {
83                !e.is_removed()
84                    && (*e).clone().as_player().is_none()
85                    && e.entity_type().can_serialize
86            })
87            .cloned()
88            .collect()
89    }
90
91    /// Ticks all entities in this chunk and broadcasts dirty entity data.
92    ///
93    /// Called from `LevelChunk::tick()`.
94    /// Returns `true` if any entities were ticked (chunk should be marked dirty).
95    ///
96    /// Uses `tick_count` to prevent double-ticking: if an entity moves to a
97    /// different chunk during its tick and that chunk is ticked later in the
98    /// same server tick, the entity will be skipped.
99    pub fn tick(&self, world: &Arc<World>, chunk_pos: ChunkPos, tick_count: i32) -> bool {
100        // Clone to avoid holding lock during tick
101        let entities: Vec<SharedEntity> = self.entities.read().values().cloned().collect();
102
103        let mut ticked_any = false;
104        for entity in entities {
105            if entity.is_removed() {
106                continue;
107            }
108
109            // Skip entities that were already ticked this server tick.
110            // This happens when an entity moves from a chunk that was already
111            // ticked to this chunk within the same server tick.
112            if entity.was_ticked_this_tick(tick_count) {
113                continue;
114            }
115
116            ticked_any = true;
117
118            // Mark as ticked before running tick() to prevent double-tick
119            // even if the entity moves during its own tick
120            entity.mark_ticked(tick_count);
121
122            // Entity-specific tick (entities access world via self.level())
123            entity.tick();
124
125            // Send position/velocity changes (mirrors vanilla's ServerEntity.sendChanges())
126            entity.send_changes(tick_count);
127
128            // Broadcast dirty entity data (base tick behavior)
129            if let Some(dirty_data) = entity.pack_dirty_entity_data() {
130                let packet = CSetEntityData::new(entity.id(), dirty_data);
131                world.broadcast_to_nearby(chunk_pos, packet, None);
132            }
133        }
134
135        // Cleanup removed entities
136        self.entities.write().retain(|_, e| !e.is_removed());
137
138        ticked_any
139    }
140
141    /// Clears all entities from storage.
142    pub fn clear(&self) {
143        self.entities.write().clear();
144    }
145}
146
147impl Default for EntityStorage {
148    fn default() -> Self {
149        Self::new()
150    }
