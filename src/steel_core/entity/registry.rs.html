17/// Factory function type for creating entities.
18///
19/// Takes the entity ID, spawn position, and world reference.
20/// Returns a new entity instance. The entity ID should be obtained from
21/// `next_entity_id()`.
22pub type EntityFactory = fn(i32, Vector3<f64>, Weak<World>) -> SharedEntity;
23
24/// Factory function type for loading entities from disk.
25///
26/// Takes all base entity fields needed for reconstruction:
27/// - `entity_id`: Fresh ID from `next_entity_id()` (not persisted)
28/// - position: Restored position
29/// - uuid: Persisted UUID
30/// - velocity: Restored velocity
31/// - rotation: Restored (yaw, pitch)
32/// - `on_ground`: Restored ground state
33/// - world: Reference to the world
34pub type EntityLoadFactory = fn(
35    i32,          // entity_id
36    Vector3<f64>, // position
37    Uuid,         // uuid
38    Vector3<f64>, // velocity
39    (f32, f32),   // rotation (yaw, pitch)
40    bool,         // on_ground
41    Weak<World>,  // world
42) -> SharedEntity;
43
44/// Registry entry for an entity type.
45struct EntityEntry {
46    /// Factory function to create new instances.
47    factory: Option<EntityFactory>,
48    /// Factory function to load instances from disk.
49    load_factory: Option<EntityLoadFactory>,
50}
51
52/// Registry for entity factories.
53///
54/// Maps `EntityType` to factory functions that can create entity instances.
55/// This is used when loading entities from disk or when entities are spawned.
56pub struct EntityRegistry {
57    entries: Vec<EntityEntry>,
58}
59
60impl EntityRegistry {
61    /// Creates a new empty registry with entries for all entity types.
62    #[must_use]
63    pub fn new() -> Self {
64        let count = REGISTRY.entity_types.len();
65        let entries = (0..count)
66            .map(|_| EntityEntry {
67                factory: None,
68                load_factory: None,
69            })
70            .collect();
71
72        Self { entries }
73    }
74
75    /// Registers a factory function for an entity type.
76    pub fn register(&mut self, entity_type: EntityTypeRef, factory: EntityFactory) {
77        let id = *REGISTRY.entity_types.get_id(entity_type);
78        self.entries[id].factory = Some(factory);
79    }
80
81    /// Registers a load factory function for an entity type.
82    ///
83    /// The load factory is used when loading entities from disk.
84    pub fn register_load(&mut self, entity_type: EntityTypeRef, factory: EntityLoadFactory) {
85        let id = *REGISTRY.entity_types.get_id(entity_type);
86        self.entries[id].load_factory = Some(factory);
87    }
88
89    /// Creates a new entity instance.
90    ///
91    /// Returns `None` if no factory is registered for the given type.
92    #[must_use]
93    pub fn create(
94        &self,
95        entity_type: EntityTypeRef,
96        entity_id: i32,
97        pos: Vector3<f64>,
98        world: Weak<World>,
99    ) -> Option<SharedEntity> {
100        let id = *REGISTRY.entity_types.get_id(entity_type);
101        self.entries
102            .get(id)?
103            .factory
104            .map(|f| f(entity_id, pos, world))
105    }
106
107    /// Creates an entity from persisted data and loads its type-specific NBT.
108    ///
109    /// Returns `None` if no load factory is registered for the entity type.
110    #[must_use]
111    #[allow(clippy::too_many_arguments)]
112    pub fn create_and_load(
113        &self,
114        entity_type: EntityTypeRef,
115        pos: Vector3<f64>,
116        uuid: Uuid,
117        velocity: Vector3<f64>,
118        rotation: (f32, f32),
119        on_ground: bool,
120        world: Weak<World>,
121        nbt: &BorrowedNbtCompound<'_>,
122    ) -> Option<SharedEntity> {
123        let id = *REGISTRY.entity_types.get_id(entity_type);
124        let load_factory = self.entries.get(id)?.load_factory?;
125
126        let entity_id = next_entity_id();
127        let entity = load_factory(entity_id, pos, uuid, velocity, rotation, on_ground, world);
128        entity.load_additional(nbt);
129        Some(entity)
130    }
131
132    /// Returns whether a factory is registered for the given type.
133    #[must_use]
134    pub fn has_factory(&self, entity_type: EntityTypeRef) -> bool {
135        let id = *REGISTRY.entity_types.get_id(entity_type);
136        self.entries.get(id).is_some_and(|e| e.factory.is_some())
137    }
138
139    /// Returns whether a load factory is registered for the given type.
140    #[must_use]
141    pub fn has_load_factory(&self, entity_type: EntityTypeRef) -> bool {
142        let id = *REGISTRY.entity_types.get_id(entity_type);
143        self.entries
144            .get(id)
145            .is_some_and(|e| e.load_factory.is_some())
146    }
147}
148
149impl Default for EntityRegistry {
150    fn default() -> Self {
151        Self::new()
152    }
153}

155/// Wrapper for the global entity registry that implements `Deref`.
156pub struct EntityRegistryLock(OnceLock<EntityRegistry>);
157
158impl Deref for EntityRegistryLock {
159    type Target = EntityRegistry;
160
161    fn deref(&self) -> &Self::Target {
162        self.0.get().expect("Entity registry not initialized")
163    }
164}
165
166impl EntityRegistryLock {
167    /// Sets the registry. Returns `Err` if already initialized.
168    pub fn set(&self, registry: EntityRegistry) -> Result<(), EntityRegistry> {
169        self.0.set(registry)
170    }
171}
172
173/// Global entity registry.
174///
175/// Access via deref: `ENTITIES.create(type, entity_id, pos)`
176pub static ENTITIES: EntityRegistryLock = EntityRegistryLock(OnceLock::new());
177
178/// Initializes the global entity registry.
179///
180/// This should be called once after the main registry is frozen.
181///
182/// # Panics
183///
184/// Panics if called more than once.
185pub fn init_entities() {
186    let mut registry = EntityRegistry::new();
187
188    // Register block display entity factory
189    registry.register(vanilla_entities::BLOCK_DISPLAY, |id, pos, world| {
190        Arc::new(BlockDisplayEntity::new(id, pos, world))
191    });
192    registry.register_load(
193        vanilla_entities::BLOCK_DISPLAY,
194        |id, pos, uuid, _velocity, _rotation, _on_ground, world| {
195            Arc::new(BlockDisplayEntity::from_saved(id, pos, uuid, world))
196        },
197    );
198
199    // Register item entity factory
200    registry.register(vanilla_entities::ITEM, |id, pos, world| {
201        Arc::new(ItemEntity::new(id, pos, world))
202    });
203    registry.register_load(
204        vanilla_entities::ITEM,
205        |id, pos, uuid, velocity, rotation, on_ground, world| {
206            Arc::new(ItemEntity::from_saved(
207                id, pos, uuid, velocity, rotation, on_ground, world,
208            ))
209        },
210    );
211
212    assert!(
213        ENTITIES.set(registry).is_ok(),
214        "Entity registry already initialized"
215    );
216}
