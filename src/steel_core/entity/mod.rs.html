23/// Global counter for allocating unique entity IDs.
24///
25/// Mirrors vanilla's `Entity.ENTITY_COUNTER`. Each new entity increments this
26/// counter to get a unique network ID. Starts at 1 (0 is reserved).
27static ENTITY_COUNTER: AtomicI32 = AtomicI32::new(1);
28
29/// Allocates a new unique entity ID.
30///
31/// This is the primary way to get entity IDs for spawning entities.
32/// Thread-safe and lock-free.
33#[must_use]
34pub fn next_entity_id() -> i32 {
35    ENTITY_COUNTER.fetch_add(1, Ordering::Relaxed)
36}

62/// A trait for entities.
63///
64/// This trait provides the core functionality for entities.
65/// It's based on Minecraft's `Entity` class.
66///
67/// # Using `EntityBase`
68///
69/// Entities that embed [`EntityBase`] can implement `base()` to get default
70/// implementations for common methods like `id()`, `uuid()`, `position()`, etc.
71///
72/// ```ignore
73/// impl Entity for MyEntity {
74///     fn base(&self) -> Option<&EntityBase> { Some(&self.base) }
75///     fn entity_type(&self) -> EntityTypeRef { vanilla_entities::MY_ENTITY }
76///     fn bounding_box(&self) -> AABBd { /* ... */ }
77///     // All other common methods use defaults from EntityBase!
78/// }
79/// ```
80pub trait Entity: Send + Sync {
81    /// Returns a reference to the entity's base fields, if available.
82    ///
83    /// Implement this to get default implementations for common methods.
84    /// Returns `None` by default (for entities like Player that don't use `EntityBase`).
85    fn base(&self) -> Option<&EntityBase> {
86        None
87    }
88
89    /// Gets the entity type containing tracking range, dimensions, etc.
90    fn entity_type(&self) -> EntityTypeRef;
91
92    /// Gets the entity's unique network ID (session-local).
93    fn id(&self) -> i32 {
94        self.base().map_or(0, EntityBase::id)
95    }
96
97    /// Gets the UUID of the entity (persistent identifier).
98    fn uuid(&self) -> Uuid {
99        self.base().map_or(Uuid::nil(), EntityBase::uuid)
100    }
101
102    /// Gets the entity's current position.
103    fn position(&self) -> Vector3<f64> {
104        self.base()
105            .map_or(Vector3::new(0.0, 0.0, 0.0), EntityBase::position)
106    }
107
108    /// Gets the entity's bounding box for collision queries.
109    fn bounding_box(&self) -> AABBd;
110
111    /// Called every game tick when the entity is in a ticked chunk.
112    ///
113    /// Use `self.level()` to access the world for physics, block queries, etc.
114    /// The caller (`EntityStorage`) handles base tick logic like dirty data sync.
115    fn tick(&self) {}
116
117    /// Sends position/velocity changes to tracking players.
118    ///
119    /// Called every tick by `EntityStorage` after `tick()`, mirrors vanilla's
120    /// `ServerEntity.sendChanges()`. Handles position sync based on `updateInterval`,
121    /// velocity sync when `needsSync` is set, and on-ground state changes.
122    ///
123    /// Default implementation does nothing. Override for entities that need
124    /// position/velocity synchronization (items, projectiles, etc.).
125    fn send_changes(&self, _tick_count: i32) {}

127    /// Gets the world this entity is in.
128    ///
129    /// Returns `None` if the entity is not in a world or the world was dropped.
130    /// Mirrors vanilla's `Entity.level()`.
131    fn level(&self) -> Option<Arc<World>> {
132        self.base().and_then(EntityBase::level)
133    }
134
135    /// Packs dirty entity data for network synchronization.
136    ///
137    /// Returns `Some(values)` if there are dirty values to sync, `None` otherwise.
138    /// Clears the dirty flags after packing.
139    fn pack_dirty_entity_data(&self) -> Option<Vec<DataValue>> {
140        None
141    }
142
143    /// Packs all non-default entity data for initial spawn.
144    ///
145    /// Used when sending entity data to a player who just started tracking this entity.
146    fn pack_all_entity_data(&self) -> Vec<DataValue> {
147        Vec::new()
148    }

150    /// Returns true if the entity has been marked for removal.
151    fn is_removed(&self) -> bool {
152        self.base().is_some_and(EntityBase::is_removed)
153    }
154
155    /// Marks the entity as removed with the given reason.
156    fn set_removed(&self, reason: RemovalReason) {
157        if let Some(base) = self.base() {
158            base.set_removed(reason);
159        }
160    }

165            base.set_level_callback(callback);
166        }
167    }

170    fn as_player(self: Arc<Self>) -> Option<Arc<Player>> {
171        None
172    }

174    /// Gets the entity as an `ItemEntity` if it is one.

176        None
177    }

179    /// Gets the entity's rotation as (yaw, pitch) in degrees.
180    ///
181    /// Yaw is horizontal rotation (0-360), pitch is vertical (-90 to 90).
182    fn rotation(&self) -> (f32, f32) {

184    }

186    /// Gets the eye height for this entity.
187    ///
188    /// Default implementation returns the eye height from the entity type dimensions.

192    }

198        self.position().y + self.get_eye_height()
199    }

201    /// Gets the entity's velocity in blocks per tick.
202    fn velocity(&self) -> Vector3<f64> {
203        Vector3::new(0.0, 0.0, 0.0)
204    }

206    /// Sets the entity's velocity.
207    fn set_velocity(&self, _velocity: Vector3<f64>) {}

209    /// Returns true if the entity is on the ground.
210    fn on_ground(&self) -> bool {
211        false
212    }

221        }
222    }

224    // === Physics Helper Methods ===
225    // These mirror vanilla's Entity class methods.
226
227    /// Gets the default gravity for this entity type.
228    ///
229    /// Override this to specify entity-specific gravity.
230    /// Vanilla values: `ItemEntity` = 0.04, `LivingEntity` = 0.08
231    fn get_default_gravity(&self) -> f64 {
232        0.0
233    }
234
235    /// Returns true if gravity is disabled for this entity.
236    ///
237    /// Override to read from entity data's `no_gravity` field.
238    fn is_no_gravity(&self) -> bool {
239        false
240    }
241
242    /// Gets the current gravity value.
243    ///
244    /// Returns 0 if `no_gravity` is set, otherwise returns `get_default_gravity()`.
245    fn get_gravity(&self) -> f64 {
246        if self.is_no_gravity() {
247            0.0
248        } else {
249            self.get_default_gravity()
250        }
251    }
252
253    /// Applies gravity to the entity's velocity.
254    ///
255    /// Mirrors vanilla's `Entity.applyGravity()`.
256    fn apply_gravity(&self) {
257        let gravity = self.get_gravity();
258        if gravity != 0.0 {
259            let mut vel = self.velocity();
260            vel.y -= gravity;
261            self.set_velocity(vel);
262        }
263    }

265    /// Moves the entity with collision detection.
266    ///
267    /// Mirrors vanilla's `Entity.move(MoverType, Vec3)`.
268    /// Updates position, `on_ground`, velocity (on collision), and returns collision info.
269    fn do_move(&self, mover_type: MoverType) -> Option<MoveResult> {
270        let world = self.level()?;
271        let velocity = self.velocity();
272
273        // Build physics state
274        let mut physics_state = EntityPhysicsState::new(self.position(), self.entity_type());
275        physics_state.velocity = velocity;
276        physics_state.on_ground = self.on_ground();
277        // Most entities don't step up; override for entities that do
278        physics_state.max_up_step = 0.0;
279        physics_state.is_crouching = false;
280
281        // Perform collision detection and movement
282        let collision_world = WorldCollisionProvider::new(&world);
283        let result = move_entity(&physics_state, velocity, mover_type, &collision_world);
284
285        // Update entity state
286        self.set_position(result.final_position);
287        self.set_on_ground(result.on_ground);
288
289        // Vanilla: Entity.move() zeros velocity components on collision.
290        // Horizontal collision zeros X/Z individually based on which axis collided.
291        // Vertical collision calls Block.updateEntityMovementAfterFallOn which by default zeros Y.
292        // (vanilla: Entity.move lines 776-785)
293        // TODO: Support block-specific behavior (slime bounce, etc.)
294        if result.horizontal_collision {
295            let vel = self.velocity();
296            self.set_velocity(Vector3::new(
297                if result.x_collision { 0.0 } else { vel.x },
298                vel.y,
299                if result.z_collision { 0.0 } else { vel.z },
300            ));
301        }
302        if result.vertical_collision {
303            // Default Block.updateEntityMovementAfterFallOn behavior: zero Y velocity
304            let vel = self.velocity();
305            self.set_velocity(Vector3::new(vel.x, 0.0, vel.z));
306        }
307
308        Some(result)
309    }

311    /// Spawns an item at this entity's location.
312    ///
313    /// Mirrors vanilla's `Entity.spawnAtLocation()`. The item spawns at the
314    /// entity's position with the given Y offset and has a default pickup delay.
315    ///
316    /// Returns `None` if the item stack is empty or the entity has no world.
317    fn spawn_at_location(
318        &self,
319        item: ItemStack,
320        y_offset: f64,
321    ) -> Option<Arc<entities::ItemEntity>> {
322        let world = self.level()?;
323        let pos = self.position();
324        world.spawn_item(Vector3::new(pos.x, pos.y + y_offset, pos.z), item)
325    }

327    // === Persistence Methods ===
328    // These mirror vanilla's Entity.addAdditionalSaveData/readAdditionalSaveData.
329
330    /// Saves type-specific entity data to NBT.
331    ///
332    /// Called during chunk serialization. Implementors should save all data
333    /// needed to restore entity state on load. Base fields (pos, motion,
334    /// rotation, uuid, `on_ground`) are handled by the serialization layer.
335    ///
336    /// Mirrors vanilla's `Entity.addAdditionalSaveData()`.
337    fn save_additional(&self, _nbt: &mut NbtCompound) {}
338
339    /// Loads type-specific entity data from NBT.
340    ///
341    /// Called after entity creation during chunk deserialization. Base fields
342    /// are already restored; this handles type-specific data.
343    ///
344    /// Mirrors vanilla's `Entity.readAdditionalSaveData()`.
345    fn load_additional(&self, _nbt: &BaseNbtCompound<'_>) {}
346
347    // === Tick Tracking ===
348    // These methods prevent double-ticking when an entity moves between chunks
349    // during the same server tick.
350
351    /// Checks if this entity was already ticked during the given server tick.
352    ///
353    /// This prevents double-ticking when an entity moves to a different chunk
354    /// during its tick, and that chunk gets ticked later in the same server tick.
355    ///
356    /// Returns `true` if already ticked this tick, `false` otherwise.
357    fn was_ticked_this_tick(&self, server_tick: i32) -> bool {
358        self.base()
359            .is_some_and(|b| b.was_ticked_this_tick(server_tick))
360    }
361
362    /// Marks this entity as ticked for the given server tick.
363    ///
364    /// Called by `EntityStorage::tick()` before ticking an entity.
365    fn mark_ticked(&self, server_tick: i32) {
366        if let Some(base) = self.base() {
367            base.mark_ticked(server_tick);
368        }
369    }

372/// A trait for living entities that can take damage, heal, and die.
373///
374/// This trait provides the core functionality for entities that have health,
375/// can be damaged, and can die. It's based on Minecraft's `LivingEntity` class.
376pub trait LivingEntity: Entity {
377    /// Gets the current health of the entity.
378    fn get_health(&self) -> f32;
379
380    /// Sets the health of the entity, clamped between 0 and max health.
381    fn set_health(&mut self, health: f32);
382
383    /// Gets the maximum health of the entity.
384    fn get_max_health(&self) -> f32;
385
386    /// Heals the entity by the specified amount.
387    fn heal(&mut self, amount: f32) {
388        let current_health = self.get_health();
389        if current_health > 0.0 {
390            self.set_health(current_health + amount);
391        }
392    }
393
394    /// Returns true if the entity is dead or dying (health <= 0).
395    fn is_dead_or_dying(&self) -> bool {
396        self.get_health() <= 0.0
397    }
398
399    /// Returns true if the entity is alive (health > 0).
400    fn is_alive(&self) -> bool {
401        !self.is_dead_or_dying()
402    }
403
404    /// Gets the entity's position.
405    fn get_position(&self) -> Vector3<f64>;
406
407    /// Gets the absorption amount (extra health from effects like absorption).
408    fn get_absorption_amount(&self) -> f32;
409
410    /// Sets the absorption amount.
411    fn set_absorption_amount(&mut self, amount: f32);
412
413    /// Gets the entity's armor value.
414    fn get_armor_value(&self) -> i32;
415
416    /// Checks if the entity can be affected by potions.
417    fn is_affected_by_potions(&self) -> bool {
418        true
419    }
420
421    /// Checks if the entity is attackable.
422    fn attackable(&self) -> bool {
423        true
424    }
425
426    /// Checks if the entity is currently using an item.
427    fn is_using_item(&self) -> bool {
428        false
429    }
430
431    /// Checks if the entity is blocking with a shield or similar item.
432    fn is_blocking(&self) -> bool {
433        false
434    }
435
436    /// Checks if the entity is fall flying (using elytra).
437    fn is_fall_flying(&self) -> bool {
438        false
439    }
440
441    /// Checks if the entity is sleeping.
442    fn is_sleeping(&self) -> bool {
443        false
444    }
445
446    /// Stops the entity from sleeping.
447    fn stop_sleeping(&mut self) {}
448
449    /// Checks if the entity is sprinting.
450    fn is_sprinting(&self) -> bool {
451        false
452    }
453
454    /// Sets whether the entity is sprinting.
455    fn set_sprinting(&mut self, sprinting: bool);
456
457    /// Gets the entity's speed attribute value.
458    fn get_speed(&self) -> f32;
459
460    /// Sets the entity's speed.
461    fn set_speed(&mut self, speed: f32);
462
463    // Equipment methods
464
465    /// Gets a clone of the item in the specified equipment slot.
466    ///
467    /// Default implementation returns an empty stack.
468    fn get_item_by_slot(&self, _slot: EquipmentSlot) -> ItemStack {
469        ItemStack::empty()
470    }
471
472    /// Gets the main hand item.
473    fn get_main_hand_item(&self) -> ItemStack {
474        self.get_item_by_slot(EquipmentSlot::MainHand)
475    }
476
477    /// Gets the off hand item.
478    fn get_off_hand_item(&self) -> ItemStack {
479        self.get_item_by_slot(EquipmentSlot::OffHand)
480    }
481
482    /// Checks if the main hand slot is empty.
483    fn is_main_hand_empty(&self) -> bool {
484        self.get_item_by_slot(EquipmentSlot::MainHand).is_empty()
485    }
486
487    /// Checks if the off hand slot is empty.
488    fn is_off_hand_empty(&self) -> bool {
489        self.get_item_by_slot(EquipmentSlot::OffHand).is_empty()
490    }
