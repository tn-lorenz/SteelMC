1//! Common base functionality shared by all entities.
2//!
3//! `EntityBase` contains the core fields and methods that every entity needs.
4//! Entities embed this struct and delegate common `Entity` trait methods to it.
5
6use std::sync::atomic::{AtomicBool, AtomicI32, Ordering};
7use std::sync::{Arc, Weak};
8
9use steel_utils::locks::SyncMutex;
10use steel_utils::math::Vector3;
11use uuid::Uuid;
12
13use crate::entity::{EntityLevelCallback, NullEntityCallback, RemovalReason};
14use crate::world::World;
15
16/// Common fields and methods shared by all entities.
17///
18/// Entities embed this struct to avoid duplicating core identity, position,
19/// and lifecycle management code. The `Entity` trait implementation can then
20/// delegate to `EntityBase` methods for common functionality.
21///
22/// # Example
23///
24/// ```ignore
25/// pub struct MyEntity {
26///     base: EntityBase,
27///     // Entity-specific fields...
28/// }
29///
30/// impl Entity for MyEntity {
31///     fn id(&self) -> i32 { self.base.id() }
32///     fn uuid(&self) -> Uuid { self.base.uuid() }
33///     fn position(&self) -> Vector3<f64> { self.base.position() }
34///     // ... delegate other common methods ...
35///
36///     // Entity-specific implementations:
37///     fn entity_type(&self) -> EntityTypeRef { vanilla_entities::MY_ENTITY }
38///     fn tick(&self) { /* custom tick logic */ }
39/// }
40/// ```
41pub struct EntityBase {
42    /// Unique network ID for this entity (session-local).
43    id: i32,
44    /// Persistent UUID for this entity.
45    uuid: Uuid,
46    /// The world this entity is in.
47    world: Weak<World>,
48    /// Current position in the world.
49    position: SyncMutex<Vector3<f64>>,
50    /// Whether this entity has been removed.
51    removed: AtomicBool,
52    /// Callback for entity lifecycle events.
53    level_callback: SyncMutex<Arc<dyn EntityLevelCallback>>,
54    /// The server tick count when this entity was last ticked.
55    /// Used to prevent double-ticking when moving between chunks.
56    last_world_tick: AtomicI32,
57}
58
59impl EntityBase {
60    /// Creates a new `EntityBase` with a randomly generated UUID.
61    #[must_use]
62    pub fn new(id: i32, position: Vector3<f64>, world: Weak<World>) -> Self {
63        Self::with_uuid(id, Uuid::new_v4(), position, world)
64    }
65
66    /// Creates a new `EntityBase` with the specified UUID.
67    ///
68    /// Use this when loading entities from disk or when the UUID is known.
69    #[must_use]
70    pub fn with_uuid(id: i32, uuid: Uuid, position: Vector3<f64>, world: Weak<World>) -> Self {
71        Self {
72            id,
73            uuid,
74            world,
75            position: SyncMutex::new(position),
76            removed: AtomicBool::new(false),
77            level_callback: SyncMutex::new(Arc::new(NullEntityCallback)),
78            last_world_tick: AtomicI32::new(-1),
79        }
80    }
81
82    // === Accessors for Entity trait delegation ===
83
84    /// Gets the entity's unique network ID.
85    #[inline]
86    pub const fn id(&self) -> i32 {
87        self.id
88    }
89
90    /// Gets the entity's UUID.
91    #[inline]
92    pub const fn uuid(&self) -> Uuid {
93        self.uuid
94    }
95
96    /// Gets the entity's current position.
97    #[inline]
98    pub fn position(&self) -> Vector3<f64> {
99        *self.position.lock()
100    }
101
102    /// Gets the world this entity is in.
103    ///
104    /// Returns `None` if the world has been dropped.
105    #[inline]
106    pub fn level(&self) -> Option<Arc<World>> {
107        self.world.upgrade()
108    }
109
110    /// Returns true if the entity has been marked for removal.
111    #[inline]
112    pub fn is_removed(&self) -> bool {
113        self.removed.load(Ordering::Relaxed)
114    }
115
116    /// Marks the entity as removed with the given reason.
117    ///
118    /// Notifies the level callback on first removal.
119    pub fn set_removed(&self, reason: RemovalReason) {
120        if !self.removed.swap(true, Ordering::AcqRel) {
121            self.level_callback.lock().on_remove(reason);
122        }
123    }
124
125    /// Sets the level callback for lifecycle events.
126    pub fn set_level_callback(&self, callback: Arc<dyn EntityLevelCallback>) {
127        *self.level_callback.lock() = callback;
128    }
129
130    /// Sets the entity's position and notifies the callback.
131    pub fn set_position(&self, pos: Vector3<f64>) {
132        let old_pos = {
133            let mut position = self.position.lock();
134            let old = *position;
135            *position = pos;
136            old
137        };
138        self.level_callback.lock().on_move(old_pos, pos);
139    }
140
141    /// Checks if this entity was already ticked during the given server tick.
142    #[inline]
143    pub fn was_ticked_this_tick(&self, server_tick: i32) -> bool {
144        self.last_world_tick.load(Ordering::Acquire) == server_tick
145    }
146
147    /// Marks this entity as ticked for the given server tick.
148    #[inline]
149    pub fn mark_ticked(&self, server_tick: i32) {
150        self.last_world_tick.store(server_tick, Ordering::Release);
151    }
152}
