1//! World-level entity cache using Weak references.
2//!
3//! Provides O(1) lookup by entity ID and UUID, plus spatial queries by section.
4//! The cache uses `Weak` references - when a chunk unloads and drops its `Arc`,
5//! the weak references become invalid and queries return `None`.
6
7use std::sync::Arc;
8
9use rustc_hash::FxHashSet;
10use steel_registry::blocks::shapes::AABBd;
11use steel_utils::SectionPos;
12use uuid::Uuid;
13
14use super::{SharedEntity, WeakEntity};
15
16/// World-level entity cache for fast lookups.
17///
18/// Stores `Weak` references to entities owned by chunks.
19/// When a chunk unloads, its entities' weak refs become invalid.
20#[allow(clippy::struct_field_names)] // `by_` prefix is intentional for clarity
21pub struct EntityCache {
22    /// Fast lookup by entity ID (network identifier).
23    by_id: scc::HashMap<i32, WeakEntity>,
24    /// Fast lookup by UUID (persistent identifier).
25    by_uuid: scc::HashMap<Uuid, WeakEntity>,
26    /// Spatial index by section position - stores entity IDs.
27    by_section: scc::HashMap<SectionPos, FxHashSet<i32>>,
28}
29
30impl EntityCache {
31    /// Creates a new empty entity cache.
32    #[must_use]
33    pub fn new() -> Self {
34        Self {
35            by_id: scc::HashMap::new(),
36            by_uuid: scc::HashMap::new(),
37            by_section: scc::HashMap::new(),
38        }
39    }
40
41    /// Registers an entity in the cache.
42    ///
43    /// Called when an entity is added to a chunk.
44    pub fn register(&self, entity: &SharedEntity) {
45        let id = entity.id();
46        let uuid = entity.uuid();
47        let weak = Arc::downgrade(entity);
48        let pos = entity.position();
49        let section = SectionPos::new(
50            (pos.x as i32) >> 4,
51            (pos.y as i32) >> 4,
52            (pos.z as i32) >> 4,
53        );
54
55        // Add to ID lookup
56        let _ = self.by_id.insert_sync(id, weak.clone());
57
58        // Add to UUID lookup
59        let _ = self.by_uuid.insert_sync(uuid, weak);
60
61        // Add to section index
62        self.add_to_section(section, id);
63    }
64
65    /// Unregisters an entity from the cache.
66    ///
67    /// Called when an entity is removed from the world.
68    pub fn unregister(&self, entity_id: i32, uuid: Uuid, section: SectionPos) {
69        // Remove from ID lookup
70        let _ = self.by_id.remove_sync(&entity_id);
71
72        // Remove from UUID lookup
73        let _ = self.by_uuid.remove_sync(&uuid);
74
75        // Remove from section index
76        self.remove_from_section(section, entity_id);
77    }
78
79    /// Updates the section index when an entity moves between sections.
80    pub fn on_section_change(
81        &self,
82        entity_id: i32,
83        old_section: SectionPos,
84        new_section: SectionPos,
85    ) {
86        if old_section == new_section {
87            return;
88        }
89
90        // Remove from old section
91        self.remove_from_section(old_section, entity_id);
92
93        // Add to new section
94        self.add_to_section(new_section, entity_id);
95    }
96
97    /// Gets an entity by its network ID.
98    ///
99    /// Returns `None` if the entity doesn't exist or its chunk was unloaded.
100    #[must_use]
101    pub fn get_by_id(&self, id: i32) -> Option<SharedEntity> {
102        self.by_id
103            .read_sync(&id, |_, weak| weak.upgrade())
104            .flatten()
105    }
106
107    /// Gets an entity by its UUID.
108    ///
109    /// Returns `None` if the entity doesn't exist or its chunk was unloaded.
110    #[must_use]
111    pub fn get_by_uuid(&self, uuid: &Uuid) -> Option<SharedEntity> {
112        self.by_uuid
113            .read_sync(uuid, |_, weak| weak.upgrade())
114            .flatten()
115    }
116
117    /// Gets all entities intersecting the given bounding box.
118    ///
119    /// Only returns entities in loaded chunks (where weak refs are valid).
120    #[must_use]
121    pub fn get_entities_in_aabb(&self, aabb: &AABBd) -> Vec<SharedEntity> {
122        let mut result = Vec::new();
123
124        // Determine section range (with 2 block grace like vanilla)
125        let min_x = ((aabb.min_x - 2.0) as i32) >> 4;
126        let min_y = ((aabb.min_y - 2.0) as i32) >> 4;
127        let min_z = ((aabb.min_z - 2.0) as i32) >> 4;
128        let max_x = ((aabb.max_x + 2.0) as i32) >> 4;
129        let max_y = ((aabb.max_y + 2.0) as i32) >> 4;
130        let max_z = ((aabb.max_z + 2.0) as i32) >> 4;
131
132        for sy in min_y..=max_y {
133            for sz in min_z..=max_z {
134                for sx in min_x..=max_x {
135                    let section_pos = SectionPos::new(sx, sy, sz);
136
137                    let entity_ids: Option<Vec<i32>> = self
138                        .by_section
139                        .read_sync(&section_pos, |_, set| set.iter().copied().collect());
140
141                    if let Some(ids) = entity_ids {
142                        for entity_id in ids {
143                            if let Some(entity) = self.get_by_id(entity_id)
144                                && entity.bounding_box().intersects(aabb)
145                            {
146                                result.push(entity);
147                            }
148                        }
149                    }
150                }
151            }
152        }
153
154        result
155    }
156
157    /// Gets all entities in a specific section.
158    #[must_use]
159    pub fn get_entities_in_section(&self, section: SectionPos) -> Vec<SharedEntity> {
160        let entity_ids: Option<Vec<i32>> = self
161            .by_section
162            .read_sync(&section, |_, set| set.iter().copied().collect());
163
164        entity_ids
165            .map(|ids| {
166                ids.into_iter()
167                    .filter_map(|id| self.get_by_id(id))
168                    .collect()
169            })
170            .unwrap_or_default()
171    }
172
173    /// Returns the number of registered entities (includes potentially stale weak refs).
174    #[must_use]
175    pub fn count(&self) -> usize {
176        self.by_id.len()
177    }
178
179    /// Periodic cleanup of dead weak refs.
180    ///
181    /// Call occasionally to remove stale entries where chunks have unloaded.
182    pub fn cleanup(&self) {
183        // Clean by_id - remove entries where weak ref is dead
184        self.by_id.retain_sync(|_, weak| weak.strong_count() > 0);
185
186        // Clean by_uuid
187        self.by_uuid.retain_sync(|_, weak| weak.strong_count() > 0);
188
189        // Clean sections - remove empty sections
190        // Note: scc::HashMap doesn't have a scan method that allows collecting,
191        // so we use retain_sync which handles this case
192        self.by_section.retain_sync(|_, set| !set.is_empty());
193    }
194
195    fn add_to_section(&self, section: SectionPos, entity_id: i32) {
196        // Try to update existing entry
197        if self
198            .by_section
199            .update_sync(&section, |_, set| {
200                set.insert(entity_id);
201            })
202            .is_none()
203        {
204            // Entry didn't exist, create new
205            let mut set = FxHashSet::default();
206            set.insert(entity_id);
207            let _ = self.by_section.insert_sync(section, set);
208        }
209    }
210
211    fn remove_from_section(&self, section: SectionPos, entity_id: i32) {
212        let should_remove = self
213            .by_section
214            .update_sync(&section, |_, set| {
215                set.remove(&entity_id);
216                set.is_empty()
217            })
218            .unwrap_or(false);
219
220        if should_remove {
221            let _ = self
222                .by_section
223                .remove_if_sync(&section, |set| set.is_empty());
224        }
225    }
226}
227
228impl Default for EntityCache {
229    fn default() -> Self {
230        Self::new()
231    }
