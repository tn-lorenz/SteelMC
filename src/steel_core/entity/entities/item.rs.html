59/// A dropped item entity.
60///
61/// Mirrors vanilla's `ItemEntity` behavior:
62/// - Falls with gravity (0.04 per tick)
63/// - Applies friction when on ground (0.98)
64/// - Despawns after 5 minutes (6000 ticks)
65/// - Has pickup delay before players can collect it
66pub struct ItemEntity {
67    /// Common entity fields (id, uuid, position, etc.).
68    base: EntityBase,
69
70    // === Position & Physics ===
71    /// Velocity in blocks per tick.
72    velocity: SyncMutex<Vector3<f64>>,
73    /// Rotation as (yaw, pitch) in degrees. Items have random yaw on spawn.
74    rotation: AtomicCell<(f32, f32)>,
75    /// Whether the entity is on the ground.
76    on_ground: AtomicBool,
77
78    // === Synced Entity Data ===
79    /// Entity data containing the `ItemStack`.
80    entity_data: SyncMutex<ItemEntityData>,
81
82    // === Timers ===
83    /// Age in ticks. Despawns at `LIFETIME` (6000). Special value -32768 = infinite.
84    age: AtomicI32,
85    /// Per-entity tick counter used for vanilla timing logic.
86    ///
87    /// Vanilla uses `Entity.tickCount` (always increments) for things like the
88    /// `(tickCount + id) % 4 == 0` movement fallback and periodic position sync.
89    /// This must not be tied to `age` because `age` can be set to `INFINITE_LIFETIME`.
90    tick_count: AtomicI32,
91    /// Ticks until pickupable. 0 = can pickup, 32767 = never.
92    pickup_delay: AtomicI32,
93    /// Health (damage resistance). Item is destroyed when this reaches 0.
94    health: AtomicI32,
95
96    // === Item-specific ===
97    /// UUID of the entity that threw/dropped this item.
98    thrower: SyncMutex<Option<Uuid>>,
99    /// UUID of the only entity that can pick up this item.
100    /// If `None`, any player can pick it up.
101    /// Vanilla calls this `target`.
102    owner: SyncMutex<Option<Uuid>>,
103
104    // === Network Sync ===
105    /// Last velocity sent to clients (for delta detection).
106    /// Mirrors vanilla's `ServerEntity.lastSentMovement`.
107    last_sent_velocity: SyncMutex<Vector3<f64>>,
108    /// Last position sent to clients (for delta detection).
109    /// Mirrors vanilla's `ServerEntity.lastSentXyz` fields.
110    last_sent_position: SyncMutex<Vector3<f64>>,
111    /// Last `on_ground` state sent to clients.
112    last_sent_on_ground: AtomicBool,
113    /// Whether position/velocity needs to be synced to clients.
114    /// Set when velocity changes significantly, like vanilla's `Entity.needsSync`.
115    needs_sync: AtomicBool,
116}

332    /// Attempts to have a player pick up this item.
333    ///
334    /// Returns `true` if the item was fully picked up (and the entity should be removed),
335    /// `false` if pickup failed or was only partial.
336    ///
337    /// Mirrors vanilla's `ItemEntity.playerTouch(Player)`.
338    pub fn try_pickup(&self, player: &Arc<Player>) -> bool {
339        // Check pickup delay
340        if self.has_pickup_delay() {
341            return false;
342        }
343
344        // Check owner restriction
345        if let Some(owner_uuid) = self.get_owner()
346            && owner_uuid != player.gameprofile.id
347        {
348            return false;
349        }
350
351        // Get the item and try to add to inventory
352        let mut item = self.get_item();
353        let original_count = item.count();
354
355        // Try to add to player's inventory
356        let added = player.inventory.lock().add(&mut item);
357
358        // If nothing was added, bail out
359        if item.count() == original_count {
360            return false;
361        }
362
363        // Calculate how many items were picked up
364        let picked_up_count = original_count - item.count();
365
366        // Send the take animation packet to nearby players
367        if let Some(world) = self.level() {
368            let pos = self.position();
369            let chunk_pos = steel_utils::ChunkPos::new((pos.x as i32) >> 4, (pos.z as i32) >> 4);
370
371            let take_packet = CTakeItemEntity::new(self.id(), player.id, picked_up_count);
372            world.broadcast_to_nearby(chunk_pos, take_packet, None);
373        }
374
375        // Update or remove the item entity
376        if added {
377            // Fully picked up - mark for removal
378            self.set_removed(RemovalReason::Discarded);
379            true
380        } else {
381            // Partial pickup - update the remaining item
382            self.set_item(item);
383            false
384        }
385    }

389    /// Returns true if this item entity can be merged with others.
390    ///
391    /// Mirrors vanilla's `ItemEntity.isMergeable()`.
392    /// An item is mergeable if:
393    /// - It's not removed
394    /// - It doesn't have infinite pickup delay (32767)
395    /// - It doesn't have infinite lifetime (-32768)
396    /// - Its age is less than the despawn threshold (6000)
397    /// - Its count is less than max stack size
398    #[must_use]
399    pub fn is_mergeable(&self) -> bool {
400        let item = self.get_item();
401        !self.is_removed()
402            && self.pickup_delay.load(Ordering::Relaxed) != INFINITE_PICKUP_DELAY
403            && self.age.load(Ordering::Relaxed) != INFINITE_LIFETIME
404            && self.age.load(Ordering::Relaxed) < LIFETIME
405            && item.count() < item.max_stack_size()
406    }

408    /// Checks if two item stacks can be merged together.
409    ///
410    /// Mirrors vanilla's `ItemEntity.areMergeable()`.
411    /// Returns true if the items are the same type with the same components,
412    /// and their combined count wouldn't exceed max stack size.
413    #[must_use]
414    pub fn are_mergeable(this_stack: &ItemStack, other_stack: &ItemStack) -> bool {
415        // Combined count must not exceed max stack size
416        if other_stack.count() + this_stack.count() > other_stack.max_stack_size() {
417            return false;
418        }
419        // Must be the same item with the same components
420        ItemStack::is_same_item_same_components(this_stack, other_stack)
421    }

423    /// Attempts to merge with another item entity.
424    ///
425    /// Mirrors vanilla's `ItemEntity.tryToMerge()`.
426    /// The item with fewer items is merged into the one with more.
427    fn try_to_merge(&self, other: &ItemEntity) {
428        let this_stack = self.get_item();
429        let other_stack = other.get_item();
430
431        // Both items must have the same owner (target)
432        if self.get_owner() != other.get_owner() {
433            return;
434        }
435
436        if !Self::are_mergeable(&this_stack, &other_stack) {
437            return;
438        }
439
440        // Merge smaller stack into larger stack
441        if other_stack.count() < this_stack.count() {
442            Self::merge_stacks(self, &this_stack, other, &other_stack);
443        } else {
444            Self::merge_stacks(other, &other_stack, self, &this_stack);
445        }
446    }

448    /// Merges the `from_item`'s stack into the `to_item`'s stack.
449    ///
450    /// Mirrors vanilla's `ItemEntity.merge(ItemEntity, ItemStack, ItemEntity, ItemStack)`.
451    fn merge_stacks(
452        to_item: &ItemEntity,
453        to_stack: &ItemStack,
454        from_item: &ItemEntity,
455        from_stack: &ItemStack,
456    ) {
457        // Calculate how many items to transfer
458        let max_count = to_stack.max_stack_size();
459        let space_available = max_count - to_stack.count();
460        let transfer_count = space_available.min(from_stack.count());
461
462        // Create new stacks
463        let new_to_stack = to_stack.copy_with_count(to_stack.count() + transfer_count);
464        let mut new_from_stack = from_stack.clone();
465        new_from_stack.shrink(transfer_count);
466
467        // Update the destination item
468        to_item.set_item(new_to_stack);
469
470        // Pickup delay is the max of both (so merged items don't become instantly pickable)
471        let new_pickup_delay = to_item
472            .pickup_delay
473            .load(Ordering::Relaxed)
474            .max(from_item.pickup_delay.load(Ordering::Relaxed));
475        to_item
476            .pickup_delay
477            .store(new_pickup_delay, Ordering::Relaxed);
478
479        // Age is the min of both (so merged items don't despawn prematurely)
480        let new_age = to_item
481            .age
482            .load(Ordering::Relaxed)
483            .min(from_item.age.load(Ordering::Relaxed));
484        to_item.age.store(new_age, Ordering::Relaxed);
485
486        // Update or remove the source item
487        if new_from_stack.is_empty() {
488            from_item.set_removed(RemovalReason::Discarded);
489        } else {
490            from_item.set_item(new_from_stack);
491        }
492    }

494    /// Attempts to merge this item with nearby item entities.
495    ///
496    /// Mirrors vanilla's `ItemEntity.mergeWithNeighbours()`.
497    /// Searches for other mergeable item entities within 0.5 blocks horizontally
498    /// and attempts to merge with them.
499    pub fn merge_with_neighbours(&self, world: &World) {
500        if !self.is_mergeable() {
501            return;
502        }
503
504        // Search area: 0.5 blocks horizontal, 0 vertical (vanilla uses inflate(0.5, 0.0, 0.5))
505        let search_box = self.bounding_box().inflate_xyz(0.5, 0.0, 0.5);
506
507        // Get all entities in the search area
508        for entity in world.get_entities_in_aabb(&search_box) {
509            // Skip self
510            if entity.id() == self.id() {
511                continue;
512            }
513
514            // Try to get as ItemEntity
515            if let Some(other_item) = entity.as_item_entity() {
516                // Double-check mergability (might have changed)
517                if other_item.is_mergeable() {
518                    self.try_to_merge(&other_item);
519
520                    // If we've been removed (merged into other), stop
521                    if self.is_removed() {
522                        break;
523                    }
524                }
525            }
526        }
527    }

531    /// Checks if velocity should be synced and returns the packet if needed.
532    ///
533    /// Vanilla syncs velocity when:
534    /// - Velocity changed by more than 1e-7 squared distance
535    /// - OR velocity became zero (to stop client-side prediction)
536    fn check_velocity_sync(&self) -> Option<CSetEntityMotion> {
537        let current = self.velocity();
538        let last_sent = *self.last_sent_velocity.lock();
539
540        let diff_sq = (current.x - last_sent.x).powi(2)
541            + (current.y - last_sent.y).powi(2)
542            + (current.z - last_sent.z).powi(2);
543
544        // Sync if velocity changed significantly, or if it went to zero
545        // (vanilla: ServerEntity.sendChanges lines 170-172)
546        let should_sync = diff_sq > 1.0e-7
547            || (diff_sq > 0.0 && current.x == 0.0 && current.y == 0.0 && current.z == 0.0);
548
549        if should_sync {
550            *self.last_sent_velocity.lock() = current;
551            Some(CSetEntityMotion::new(
552                self.id(),
553                current.x,
554                current.y,
555                current.z,
556            ))
557        } else {
558            None
559        }
560    }

562    /// Checks if position should be synced and returns the appropriate packet.
563    ///
564    /// Uses delta encoding (`CMoveEntityPos`) for small movements, and falls back
565    /// to absolute position sync (`CEntityPositionSync`) when:
566    /// - Delta is too large for i16 encoding
567    /// - On-ground state changed
568    /// - Periodic full sync (every 60 ticks based on `tick_count`)
569    fn check_position_sync(&self, tick_count: i32) -> Option<PositionSyncPacket> {
570        let current_pos = self.position();
571        let last_sent = *self.last_sent_position.lock();
572        let current_on_ground = self.on_ground();
573        let last_on_ground = self.last_sent_on_ground.load(Ordering::Relaxed);
574
575        // Check if position changed enough to warrant sync
576        // Vanilla threshold: 7.6293945E-6 (TOLERANCE_LEVEL_POSITION)
577        let diff_sq = (current_pos.x - last_sent.x).powi(2)
578            + (current_pos.y - last_sent.y).powi(2)
579            + (current_pos.z - last_sent.z).powi(2);
580
581        let position_changed = diff_sq >= 7.629_394_5e-6;
582        let on_ground_changed = current_on_ground != last_on_ground;
583        // Vanilla uses tickCount % 60 for periodic full position sync (FORCED_POS_UPDATE_PERIOD)
584        let force_periodic_sync = tick_count % 60 == 0;
585
586        // Vanilla: boolean pos = positionChanged || this.tickCount % 60 == 0;
587        // We sync if position changed, on_ground changed, or periodic
588        if !position_changed && !on_ground_changed && !force_periodic_sync {
589            return None;
590        }
591
592        // Try delta encoding first
593        let dx = calc_delta(current_pos.x, last_sent.x);
594        let dy = calc_delta(current_pos.y, last_sent.y);
595        let dz = calc_delta(current_pos.z, last_sent.z);
596
597        // Use full sync if delta overflow or on-ground changed or periodic
598        // (vanilla: ServerEntity.sendChanges line 123)
599        let use_full_sync = on_ground_changed
600            || force_periodic_sync
601            || dx.is_none()
602            || dy.is_none()
603            || dz.is_none();
604
605        self.last_sent_on_ground
606            .store(current_on_ground, Ordering::Relaxed);
607
608        if use_full_sync {
609            // Full sync: client sets position directly, so store current_pos
610            *self.last_sent_position.lock() = current_pos;
611
612            let vel = self.velocity();
613            // NOTE: We do NOT update last_sent_velocity here because the client
614            // ignores the velocity field in CEntityPositionSync for non-authoritative
615            // entities (like items). The velocity sync is handled separately by
616            // check_velocity_sync() which sends CSetEntityMotion.
617
618            let (yaw, pitch) = self.rotation.load();
619            Some(PositionSyncPacket::Full(CEntityPositionSync {
620                entity_id: self.id(),
621                x: current_pos.x,
622                y: current_pos.y,
623                z: current_pos.z,
624                velocity_x: vel.x,
625                velocity_y: vel.y,
626                velocity_z: vel.z,
627                yaw,
628                pitch,
629                on_ground: current_on_ground,
630            }))
631        } else {
632            // Delta sync: store the actual current position as base.
633            // Vanilla stores the actual position, not the decoded position.
634            // This works because encode() is deterministic - both server and client
635            // compute the same encoded values.
636            let dx = dx.expect("delta dx missing in delta position sync");
637            let dy = dy.expect("delta dy missing in delta position sync");
638            let dz = dz.expect("delta dz missing in delta position sync");
639
640            *self.last_sent_position.lock() = current_pos;
641
642            Some(PositionSyncPacket::Delta(CMoveEntityPos {
643                entity_id: self.id(),
644                dx,
645                dy,
646                dz,
647                on_ground: current_on_ground,
648            }))
649        }
650    }
651}

653/// Position sync packet variants.
654enum PositionSyncPacket {
655    /// Delta-encoded position update (for small movements).
656    Delta(CMoveEntityPos),
657    /// Full absolute position sync (for large movements or periodic sync).
658    Full(CEntityPositionSync),
659}

670    fn bounding_box(&self) -> AABBd {
671        let pos = self.position();
672        let dims = self.entity_type().dimensions;
673        let half_width = f64::from(dims.width) / 2.0;
674        let height = f64::from(dims.height);
675        AABBd {
676            min_x: pos.x - half_width,
677            min_y: pos.y,
678            min_z: pos.z - half_width,
679            max_x: pos.x + half_width,
680            max_y: pos.y + height,
681            max_z: pos.z + half_width,
682        }
683    }

685    fn tick(&self) {
686        // Vanilla: `Entity.tickCount` increments every tick regardless of item age/lifetime.
687        let tick_count = self.tick_count.fetch_add(1, Ordering::Relaxed) + 1;
688
689        // Check if item is empty
690        if self.get_item().is_empty() {
691            self.set_removed(RemovalReason::Discarded);
692            return;
693        }
694
695        // Decrement pickup delay
696        let pickup_delay = self.pickup_delay.load(Ordering::Relaxed);
697        if pickup_delay > 0 && pickup_delay != INFINITE_PICKUP_DELAY {
698            self.pickup_delay.fetch_sub(1, Ordering::Relaxed);
699        }
700
701        // Increment age and check for despawn
702        let age = self.age.load(Ordering::Relaxed);
703        if age != INFINITE_LIFETIME {
704            let new_age = self.age.fetch_add(1, Ordering::Relaxed) + 1;
705            if new_age >= LIFETIME {
706                self.set_removed(RemovalReason::Discarded);
707                return;
708            }
709        }
710
711        // Store old position for merge rate calculation (vanilla: xo, yo, zo)
712        let old_pos = self.position();
713        // Store old movement for needsSync check (vanilla: ItemEntity.tick line 98)
714        let old_movement = self.velocity();
715        // Store old on_ground to detect changes (triggers immediate sync)
716        let old_on_ground = self.on_ground();
717
718        // TODO: Handle water/lava movement (setUnderwaterMovement, setUnderLavaMovement)
719        // For now, just apply gravity
720        self.apply_gravity();
721
722        // Vanilla optimization: skip physics when at rest on ground.
723        // Only process physics if:
724        // 1. Not on ground, OR
725        // 2. Has significant horizontal movement, OR
726        // 3. Every 4th tick (for items that might need to fall through opened trapdoors, etc.)
727        // (vanilla: ItemEntity.tick line 121)
728        let vel = self.velocity();
729        let horizontal_movement_sq = vel.x * vel.x + vel.z * vel.z;
730        let should_move = !self.on_ground()
731            || horizontal_movement_sq > 1.0e-5
732            || (tick_count + self.id()) % 4 == 0;
733
734        if should_move {
735            // Move with collision detection (do_move handles velocity zeroing on collision)
736            if let Some(result) = self.do_move(MoverType::SelfMovement) {
737                // Get world for block queries
738                if let Some(world) = self.level() {
739                    // Apply friction (vanilla: ItemEntity.tick line 125-128)
740                    let friction = if result.on_ground {
741                        // Block below that affects movement (0.999999F offset like vanilla)
742                        let pos = self.position();
743                        let block_pos = BlockPos::new(
744                            pos.x.floor() as i32,
745                            (pos.y - 0.999_999).floor() as i32,
746                            pos.z.floor() as i32,
747                        );
748                        let block_state = world.get_block_state(&block_pos);
749                        f64::from(block_state.get_block().config.friction) * 0.98
750                    } else {
751                        0.98 // Air friction
752                    };
753
754                    let mut velocity = self.velocity();
755                    velocity.x *= friction;
756                    velocity.z *= friction;
757                    velocity.y *= AIR_DRAG;
758
759                    // Bounce when landing on ground (vanilla: ItemEntity.tick lines 145-149)
760                    if result.on_ground && velocity.y < 0.0 {
761                        velocity.y *= -0.5;
762                    }
763
764                    self.set_velocity(velocity);
765                }
766            }
767        }
768
769        // Item merging (vanilla: ItemEntity.tick lines 152-156)
770        // Merge rate depends on whether the item moved to a different block
771        let current_pos = self.position();
772        let moved_block = old_pos.x.floor() as i32 != current_pos.x.floor() as i32
773            || old_pos.y.floor() as i32 != current_pos.y.floor() as i32
774            || old_pos.z.floor() as i32 != current_pos.z.floor() as i32;
775        let merge_rate = if moved_block { 2 } else { 40 };
776
777        if tick_count % merge_rate == 0
778            && self.is_mergeable()
779            && let Some(world) = self.level()
780        {
781            self.merge_with_neighbours(&world);
782        }
783
784        // Check if velocity changed significantly -> set needsSync (vanilla: ItemEntity.tick lines 160-164)
785        // Vanilla: if (getDeltaMovement().subtract(oldMovement).lengthSqr() > 0.01) needsSync = true
786        let new_movement = self.velocity();
787        let diff = Vector3::new(
788            new_movement.x - old_movement.x,
789            new_movement.y - old_movement.y,
790            new_movement.z - old_movement.z,
791        );
792        let diff_sq = diff.x * diff.x + diff.y * diff.y + diff.z * diff.z;
793        if diff_sq > 0.01 {
794            self.needs_sync.store(true, Ordering::Relaxed);
795        }
796
797        // Also set needsSync when on_ground changes - this ensures immediate sync
798        // when the item lands or becomes airborne, preventing client desync
799        if self.on_ground() != old_on_ground {
800            self.needs_sync.store(true, Ordering::Relaxed);
801        }
802    }
803
804    fn send_changes(&self, tick_count: i32) {
805        let Some(world) = self.level() else {
806            return;
807        };
808
809        let update_interval = self.entity_type().update_interval; // 20 for items
810        let needs_sync = self.needs_sync.load(Ordering::Relaxed);
811
812        // Only send updates on the update interval OR when needsSync is set
813        // (vanilla: ServerEntity.sendChanges line 97)
814        if tick_count % update_interval != 0 && !needs_sync {
815            return;
816        }
817
818        let current_pos = self.position();
819
820        // Determine chunk for broadcasting
821        let chunk_pos =
822            steel_utils::ChunkPos::new((current_pos.x as i32) >> 4, (current_pos.z as i32) >> 4);
823
824        // Vanilla sends velocity BEFORE position (ServerEntity.sendChanges lines 168-182).
825        // Items have trackDelta=true, so we ALWAYS check velocity when in the update window.
826        //
827        // CRITICAL: The client ignores velocity in CEntityPositionSync for non-authoritative
828        // entities (like items). The client runs its own physics simulation and accumulates
829        // gravity in deltaMovement. We MUST send CSetEntityMotion to override the client's
830        // deltaMovement, otherwise the client's accumulated gravity causes visual desync.
831        if let Some(vel_packet) = self.check_velocity_sync() {
832            world.broadcast_to_nearby(chunk_pos, vel_packet, None);
833        }
834
835        // Send position update if needed (vanilla: ServerEntity.sendChanges line 182)
836        if let Some(packet) = self.check_position_sync(tick_count) {
837            match &packet {
838                PositionSyncPacket::Delta(p) => {
839                    world.broadcast_to_nearby(chunk_pos, p.clone(), None);
840                }
841                PositionSyncPacket::Full(p) => {
842                    world.broadcast_to_nearby(chunk_pos, p.clone(), None);
843                }
844            }
845        }
846
847        // Clear needsSync after processing (vanilla: ServerEntity.sendChanges line 193)
848        self.needs_sync.store(false, Ordering::Relaxed);
849    }

913    fn load_additional(&self, nbt: &BorrowedNbtCompound<'_>) {
914        // Convert to view type to access accessor methods
915        let nbt: NbtCompoundView<'_, '_> = nbt.into();
916
917        // Match vanilla's ItemEntity.readAdditionalSaveData
918        if let Some(health) = nbt.short("Health") {
919            self.health.store(i32::from(health), Ordering::Relaxed);
920        }
921        if let Some(age) = nbt.short("Age") {
922            self.age.store(i32::from(age), Ordering::Relaxed);
923        }
924        if let Some(pickup_delay) = nbt.short("PickupDelay") {
925            self.pickup_delay
926                .store(i32::from(pickup_delay), Ordering::Relaxed);
927        }
928
929        if let Some(thrower_arr) = nbt.int_array("Thrower")
930            && let Some(uuid) = Uuid::from_int_array(&thrower_arr)
931        {
932            *self.thrower.lock() = Some(uuid);
933        }
934        if let Some(owner_arr) = nbt.int_array("Owner")
935            && let Some(uuid) = Uuid::from_int_array(&owner_arr)
936        {
937            *self.owner.lock() = Some(uuid);
938        }
939
940        if let Some(item_tag) = nbt.compound("Item")
941            && let Some(item) = ItemStack::from_borrowed_compound(&item_tag)
942        {
943            self.entity_data.lock().item.set(item);
944        }
945
946        // Vanilla behavior: discard if item is empty after load
947        if self.get_item().is_empty() {
948            self.set_removed(RemovalReason::Discarded);
949        }
950    }
