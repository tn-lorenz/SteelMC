1//! Block display entity implementation.
2//!
3//! Display entities render a block, item, or text without collision.
4//! They're commonly used for visual effects, holograms, and decorations.
5

23/// A block display entity that renders a block state at its position.
24///
25/// Block displays are purely visual entities with no collision.
26/// They support transformation (translation, rotation, scale) and
27/// interpolation for smooth animations.
28pub struct BlockDisplayEntity {
29    /// Common entity fields (id, uuid, position, etc.).
30    base: EntityBase,
31    /// Synced entity data for network serialization.
32    entity_data: SyncMutex<BlockDisplayEntityData>,
33}
34
35impl BlockDisplayEntity {
36    /// Creates a new block display entity.
37    ///
38    /// The `id` should be obtained from `next_entity_id()`.
39    #[must_use]
40    pub fn new(id: i32, position: Vector3<f64>, world: Weak<World>) -> Self {
41        Self {
42            base: EntityBase::new(id, position, world),
43            entity_data: SyncMutex::new(BlockDisplayEntityData::new()),
44        }
45    }
46
47    /// Creates a new block display entity with a specific UUID.
48    ///
49    /// The `id` should be obtained from `next_entity_id()`.
50    #[must_use]
51    pub fn with_uuid(id: i32, position: Vector3<f64>, uuid: Uuid, world: Weak<World>) -> Self {
52        Self {
53            base: EntityBase::with_uuid(id, uuid, position, world),
54            entity_data: SyncMutex::new(BlockDisplayEntityData::new()),
55        }
56    }
57
58    /// Creates a block display entity from saved data.
59    ///
60    /// Display entities don't use velocity, rotation, or `on_ground`, so this is
61    /// essentially an alias for `with_uuid`. Type-specific data is restored
62    /// via `load_additional()` after construction.
63    #[must_use]
64    pub fn from_saved(id: i32, position: Vector3<f64>, uuid: Uuid, world: Weak<World>) -> Self {
65        Self::with_uuid(id, position, uuid, world)
66    }
67
68    /// Gets a reference to the entity data for reading/modifying synced state.
69    pub const fn entity_data(&self) -> &SyncMutex<BlockDisplayEntityData> {
70        &self.entity_data
71    }
72
73    /// Sets the block state ID of this entity.
74    pub fn set_block_state_id(&self, id: BlockStateId) {
75        self.entity_data.lock().block_state.set(id);
76    }
77}
78
79impl Entity for BlockDisplayEntity {
80    fn base(&self) -> Option<&EntityBase> {
81        Some(&self.base)
82    }
83
84    fn entity_type(&self) -> EntityTypeRef {
85        vanilla_entities::BLOCK_DISPLAY
86    }
87
88    fn bounding_box(&self) -> AABBd {
89        // Display entities have zero-size bounding boxes (no collision)
90        let pos = self.position();
91        AABBd {
92            min_x: pos.x,
93            min_y: pos.y,
94            min_z: pos.z,
95            max_x: pos.x,
96            max_y: pos.y,
97            max_z: pos.z,
98        }
99    }
100
101    fn pack_dirty_entity_data(&self) -> Option<Vec<DataValue>> {
102        self.entity_data.lock().pack_dirty()
103    }
104
105    fn pack_all_entity_data(&self) -> Vec<DataValue> {
106        self.entity_data.lock().pack_all()
107    }
108
109    fn save_additional(&self, nbt: &mut NbtCompound) {
110        // Save block state ID directly - these are deterministic in Minecraft
111        let block_state_id = *self.entity_data.lock().block_state.get();
112        nbt.insert("block_state", i32::from(block_state_id.0));
113    }
114
115    fn load_additional(&self, nbt: &BorrowedNbtCompound<'_>) {
116        // Convert to view type to access accessor methods
117        let nbt: NbtCompoundView<'_, '_> = nbt.into();
118
119        // Load block state ID
120        if let Some(state_id) = nbt.int("block_state") {
121            self.entity_data
122                .lock()
123                .block_state
124                .set(BlockStateId(state_id as u16));
125        }
126    }
127}
