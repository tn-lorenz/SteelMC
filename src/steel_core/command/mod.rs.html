41/// One command completion and its replacement range in UTF-16 code units.
42#[derive(Clone, Debug, PartialEq, Eq)]
43pub struct CommandCompletion {
44    replacement_start: usize,
45    replacement_length: usize,
46    text: String,
47}
48
49impl CommandCompletion {
50    pub(crate) const fn new(
51        replacement_start: usize,
52        replacement_length: usize,
53        text: String,
54    ) -> Self {
55        Self {
56            replacement_start,
57            replacement_length,
58            text,
59        }
60    }
61
62    /// Returns the inclusive replacement start in UTF-16 code units.
63    #[must_use]
64    pub const fn replacement_start(&self) -> usize {
65        self.replacement_start
66    }
67
68    /// Returns the replacement length in UTF-16 code units.
69    #[must_use]
70    pub const fn replacement_length(&self) -> usize {
71        self.replacement_length
72    }
73
74    /// Returns the replacement text.
75    #[must_use]
76    pub fn text(&self) -> &str {
77        &self.text
78    }
79}

81/// Projects Steel capabilities onto vanilla's shared gamemaster client affordance.
82/// Packet handlers still authorize game-mode and difficulty requests independently.
83pub(crate) fn client_permission_event(player: &Player, world: &World) -> EntityStatus {
84    client_permission_event_for_capabilities(
85        builtins::gamemode::player_can_use_client_switcher(player, world),
86        builtins::player_can_change_difficulty(player, world),
87    )
88}
89
90const fn client_permission_event_for_capabilities(
91    can_change_game_mode: bool,
92    can_change_difficulty: bool,
93) -> EntityStatus {
94    if can_change_game_mode || can_change_difficulty {
95        EntityStatus::PermissionLevelGamemasters
96    } else {
97        EntityStatus::PermissionLevelAll
98    }
99}
