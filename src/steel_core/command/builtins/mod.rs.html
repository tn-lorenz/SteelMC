44pub(crate) fn create_dispatcher()
45-> Result<CommandDispatcher<CommandSource, SteelCommandRuntime>, CommandRegistrationError> {
46    create_registered_dispatcher(CommandRegistry::new()).map(|registered| registered.dispatcher)
47}

49pub(crate) fn create_registered_dispatcher(
50    extension_commands: CommandRegistry,
51) -> Result<RegisteredCommandDispatcher<CommandSource>, CommandRegistrationError> {
52    let mut builder = CommandDispatcherBuilder::new();
53    builder.declare_permission(ENTITY_SELECTOR_PERMISSION_KEY)?;
54    builder.declare_permission(ENTITY_SELECTOR_ADVANCED_PERMISSION_KEY)?;
55    builder.declare_permission(perms::MANAGE_ALL_PERMISSION)?;
56    builder.declare_permission(perms::GROUP_ALL_PERMISSION)?;
57    builder.declare_permission(perms::METADATA_PERMISSION)?;
58    builder.register(clear::registration())?;
59    builder.register(operator::deop_registration())?;
60    builder.register(difficulty::registration())?;
61    builder.register(domain::registration())?;
62    builder.register(enchant::registration())?;
63    builder.register(execute::registration())?;
64    builder.register(experience::registration())?;
65    builder.register(fly::registration())?;
66    builder.register(gamemode::registration()?)?;
67    builder.register(gamerule::registration())?;
68    builder.register(give::registration())?;
69    builder.register(kill::registration())?;
70    builder.register(list::registration())?;
71    builder.register(locate::registration())?;
72    builder.register(operator::op_registration())?;
73    builder.register(perms::registration())?;
74    builder.register(return_command::registration())?;
75    builder.register(seed::registration())?;
76    builder.register(setworldspawn::registration())?;
77    builder.register(stop::registration())?;
78    builder.register(summon::registration())?;
79    builder.register(teleport::registration())?;
80    builder.register(tellraw::registration())?;
81    builder.register(tick::registration())?;
82    builder.register(time::registration())?;
83    builder.register(weather::registration())?;
84    builder.extend(extension_commands.into_inner())?;
85    builder.build_with_permissions()
86}

203            .collect::<Vec<_>>();
204        assert_eq!(weather_names, ["clear", "rain", "thunder"]);
205    }

208    fn startup_extensions_merge_after_builtins_with_namespaced_collision_fallbacks() {
209        init_test_registry();
210        let mut extensions = CommandRegistry::new();
211        let registration =
212            ExtensionCommandRegistration::new(Identifier::new("steel_test", "stop"), || {
213                extension_literal("stop").executes(|_| Ok(1))
214            });
215        assert!(extensions.register(registration).is_ok());
216
217        let Ok(registered) = create_registered_dispatcher(extensions) else {
218            panic!("built-ins and extension should register atomically");
219        };
220        let dispatcher = registered.dispatcher;
221        let Some(roots) = dispatcher.children(dispatcher.root()) else {
222            panic!("dispatcher root should exist");
223        };
224        let names = roots
225            .iter()
226            .filter_map(|root| {
227                let node = dispatcher.node(*root)?;
228                Some(node.name())
229            })
230            .collect::<Vec<_>>();
231
232        assert!(names.contains(&"stop"));
233        assert!(names.contains(&"steel_test:stop"));
234        assert!(
235            registered
236                .permissions
237                .iter()
238                .any(|permission| permission.as_str() == "steel_test.command.stop")
239        );
240    }

252        let Some(execute) = dispatcher.children(dispatcher.root()).and_then(|children| {
253            children.iter().copied().find(|child| {
254                dispatcher
255                    .node(*child)
256                    .is_some_and(|node| node.name() == "execute")
257            })
258        }) else {
259            panic!("execute root should exist");
260        };
261        let Some(execute_node) = dispatcher.node(execute) else {
262            panic!("execute root should exist");
263        };
264        assert!(!execute_node.is_executable());
265
266        let child = |parent, name| {
267            let Some(node) = dispatcher.children(parent).and_then(|children| {
268                children.iter().copied().find(|child| {
269                    dispatcher
270                        .node(*child)
271                        .is_some_and(|node| node.name() == name)
272                })
273            }) else {
274                panic!("{name} should exist below {parent:?}");
275            };
276            node
277        };
278
279        let run = child(execute, "run");
