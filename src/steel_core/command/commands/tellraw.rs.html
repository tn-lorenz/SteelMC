1//! Handler for the "tellraw" command.
2use crate::command::arguments::player::PlayerArgument;
3use crate::command::arguments::text_component::TextComponentArgument;
4use crate::command::commands::{
5    CommandExecutor, CommandHandlerBuilder, CommandHandlerDyn, argument,
6};
7use crate::command::context::CommandContext;
8use crate::command::error::CommandError;
9use crate::command::sender::CommandSender;
10use crate::player::Player;
11use std::sync::Arc;
12use text_components::TextComponent;
13
14/// Handler for the "tellraw" command.
15#[must_use]
16pub fn command_handler() -> impl CommandHandlerDyn {
17    CommandHandlerBuilder::new(
18        &["tellraw"],
19        "Sends a JSON message to players.",
20        "minecraft:command.tellraw",
21    )
22    .then(
23        argument("targets", PlayerArgument::new())
24            .then(argument("message", TextComponentArgument).executes(TellrawCommandExecutor)),
25    )
26}
27
28struct TellrawCommandExecutor;
29
30impl CommandExecutor<(((), Vec<Arc<Player>>), TextComponent)> for TellrawCommandExecutor {
31    fn execute(
32        &self,
33        args: (((), Vec<Arc<Player>>), TextComponent),
34        context: &mut CommandContext,
35    ) -> Result<(), CommandError> {
36        let sender = match &context.sender {
37            CommandSender::Player(player) => &player.gameprofile.name,
38            CommandSender::Console => "Console",
39            CommandSender::Rcon => "Rcon",
40        };
41        log::info!("{}'s tellraw: {:p}", sender, args.1);
42        for player in args.0.1 {
43            player.send_message(&args.1);
44        }
45        Ok(())
46    }
47}
