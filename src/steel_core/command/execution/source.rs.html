32type CommandResultCallbackFn = dyn Fn(bool, i32) + Send + Sync;

34/// A callback invoked after a terminal command returns or fails.
35#[derive(Clone, Default)]
36pub(crate) struct CommandResultCallback {
37    callback: Option<Arc<CommandResultCallbackFn>>,
38}
39
40impl CommandResultCallback {
41    pub(crate) fn new(callback: impl Fn(bool, i32) + Send + Sync + 'static) -> Self {
42        Self {
43            callback: Some(Arc::new(callback)),
44        }
45    }
46
47    pub(crate) const fn empty() -> Self {
48        Self { callback: None }
49    }
50
51    pub(crate) fn chain(first: Self, second: Self) -> Self {
52        match (first.callback, second.callback) {
53            (None, None) => Self::empty(),
54            (Some(callback), None) | (None, Some(callback)) => Self {
55                callback: Some(callback),
56            },
57            (Some(first), Some(second)) => Self::new(move |success, result| {
58                first(success, result);
59                second(success, result);
60            }),
61        }
62    }
63
64    pub(crate) fn on_result(&self, success: bool, result: i32) {
65        if let Some(callback) = &self.callback {
66            callback(success, result);
67        }
68    }
69}

178/// Authorization state captured when command execution starts.
179///
180/// Vanilla keeps the initiating permission state when `/execute` changes the
181/// execution entity, position, or world. Keeping this separate from the
182/// mutable execution fields prevents source transforms from changing which
183/// contextual Steel permissions apply. Player permissions come from the
184/// persistence-published server state so delayed client refreshes cannot extend
185/// revoked command access.
186#[derive(Clone, Debug, PartialEq, Eq)]
187struct CommandAuthorizationContext {
188    permission_context: PermissionContext,
189    player_permissions: Option<PermissionSet>,
190}
191
192impl CommandAuthorizationContext {
193    fn for_player(world: steel_utils::Identifier, permissions: PermissionSet) -> Self {
194        Self {
195            permission_context: PermissionContext::for_world(world),
196            player_permissions: Some(permissions),
197        }
198    }
199
200    fn unrestricted(world: steel_utils::Identifier) -> Self {
201        Self {
202            permission_context: PermissionContext::for_world(world),
203            player_permissions: None,
204        }
205    }
206
207    const fn permission_context(&self) -> &PermissionContext {
208        &self.permission_context
209    }
210
211    fn permission_state(&self, permission: &PermissionExpr) -> Option<PermissionState> {
212        let Some(permissions) = &self.player_permissions else {
213            return Some(PermissionState::Allow);
214        };
215        permissions.resolve_in(permission, self.permission_context())
216    }
217}

236    pub(crate) fn new(sender: CommandSender, server: Arc<Server>) -> Self {
237        let player = sender.get_player().map(Arc::clone);
238        let world = player.as_ref().map_or_else(
239            || Arc::clone(server.overworld()),
240            |player| player.get_world(),
241        );
242        let entity = player
243            .as_ref()
244            .map(|player| Arc::clone(player) as SharedEntity);
245        let position = entity.as_ref().map_or_else(
246            || {
247                let level_data = world.level_data.read();
248                let spawn = &level_data.data().spawn;
249                DVec3::new(f64::from(spawn.x), f64::from(spawn.y), f64::from(spawn.z))
250            },
251            |entity| entity.position(),
252        );
253        let rotation = entity
254            .as_ref()
255            .map_or((0.0, 0.0), |entity| entity.rotation());
256        let authorization = match &player {
257            Some(player) => CommandAuthorizationContext::for_player(
258                world.key.clone(),
259                server.command_permission_snapshot(player.gameprofile.id),
260            ),
261            None => CommandAuthorizationContext::unrestricted(world.key.clone()),
262        };
263
264        Self {
265            sender,
266            player,
267            entity,
268            world,
269            server,
270            position,
271            rotation,
272            anchor: EntityAnchor::default(),
273            authorization,
274            callback: CommandResultCallback::empty(),
275            silent: false,
276        }
277    }

357    pub(crate) fn facing_position(&self, target: DVec3) -> Self {
358        let delta = target - self.anchor_position();
359        let horizontal = delta.x.hypot(delta.z);
360        let pitch = -delta.y.atan2(horizontal).to_degrees() as f32;
361        let yaw = delta.z.atan2(delta.x).to_degrees() as f32 - 90.0;
362        self.with_rotation((yaw, pitch))
363    }

396    pub(crate) fn send_success(&self, message: &TextComponent, broadcast_to_admins: bool) {
397        if self.silent {
398            return;
399        }
400
401        let accepts_success = self
402            .sender
403            .get_player()
404            .is_none_or(|player| player.get_world().get_game_rule(&SEND_COMMAND_FEEDBACK));
405        if accepts_success {
406            self.sender.send_message(message);
407        }
408        if broadcast_to_admins {
409            self.broadcast_to_admins(message);
410        }
411    }

413    pub(crate) fn send_failure(&self, message: TextComponent) {
414        if !self.silent {
415            self.sender.send_message(&message.color(Color::Red));
416        }
417    }

429    fn broadcast_to_admins(&self, message: &TextComponent) {
430        let sender_name = admin_broadcast_source_name(self.entity.as_deref(), &self.sender);
431        let broadcast = translations::CHAT_TYPE_ADMIN
432            .message([sender_name, message.clone()])
433            .component()
434            .color(Color::Gray)
435            .italic(true);
436
437        if self.world.get_game_rule(&SEND_COMMAND_FEEDBACK) {
438            let sender_uuid = self.sender.get_player().map(|player| player.gameprofile.id);
439            for player in self.server.get_players() {
440                if Some(player.gameprofile.id) != sender_uuid
441                    && self.server.is_operator(player.gameprofile.id)
442                {
443                    player.send_message(&broadcast);
444                }
445            }
446        }
447
448        if !matches!(self.sender, CommandSender::Console)
449            && self.world.get_game_rule(&LOG_ADMIN_COMMANDS)
450        {
451            CommandSender::Console.send_message(&broadcast);
452        }
453    }

661fn profile_argument_uuids(
662    source: &CommandSource,
663    argument: &GameProfileArgument,
664) -> BTreeSet<uuid::Uuid> {
665    match argument {
666        GameProfileArgument::Selector(selector) => selector.find_players(source).map_or_else(
667            |_| BTreeSet::new(),
668            |players| {
669                players
670                    .into_iter()
671                    .map(|player| player.gameprofile.id)
672                    .collect()
673            },
674        ),
675        GameProfileArgument::Direct(value) => {
676            let known = source.server.known_players();
677            let uuid = uuid::Uuid::parse_str(value)
678                .ok()
679                .or_else(|| known.by_name(value).map(KnownPlayer::uuid))
680                .or_else(|| {
681                    source
682                        .server
683                        .get_players()
684                        .into_iter()
685                        .find(|player| player.gameprofile.name.eq_ignore_ascii_case(value))
686                        .map(|player| player.gameprofile.id)
687                });
688            uuid.into_iter().collect()
689        }
690    }
691}

709fn profile_names(server: &Server, operator: bool, include_known: bool) -> Vec<String> {
710    let players = server.get_players();
711    let mut names = Vec::new();
712    for player in &players {
713        if operator == server.is_operator(player.gameprofile.id) {
714            names.push(player.gameprofile.name.clone());
715        }
716    }
717
718    if !include_known {
719        return names;
720    }
721
722    for known in server.known_players().entries() {
723        if names
724            .iter()
725            .any(|name| name.eq_ignore_ascii_case(known.last_known_name()))
726            || players.iter().any(|player| {
727                player
728                    .gameprofile
729                    .name
730                    .eq_ignore_ascii_case(known.last_known_name())
731            })
732        {
733            continue;
734        }
735        let is_operator = server.is_operator(known.uuid());
736        if operator == is_operator {
737            names.push(known.last_known_name().to_owned());
738        }
739    }
740    names
741}

743fn all_profile_names(server: &Server) -> Vec<String> {
744    let players = server.get_players();
745    let mut names = players
746        .iter()
747        .map(|player| player.gameprofile.name.clone())
748        .collect::<Vec<_>>();
749    for known in server.known_players().entries() {
750        if names
751            .iter()
752            .any(|name| name.eq_ignore_ascii_case(known.last_known_name()))
753        {
754            continue;
755        }
756        names.push(known.last_known_name().to_owned());
757    }
758    names
759}

764    }

774    entity: Option<&dyn Entity>,
775    sender: &CommandSender,
776) -> TextComponent {
777    entity.map_or_else(
778        || TextComponent::plain(sender.to_string()),
779        |entity| TextComponent::plain(entity.plain_text_name()),
780    )
781}

783fn normalize_rotation((mut yaw, mut pitch): (f32, f32)) -> (f32, f32) {
784    yaw = yaw.rem_euclid(360.0);
785    if yaw >= 180.0 {
786        yaw -= 360.0;
787    }
788    pitch = pitch.rem_euclid(360.0);
789    if pitch >= 180.0 {
790        pitch -= 360.0;
791    }
792    (yaw, pitch)
793}
