43/// Runtime model interpreted by Steel's tick-owned command scheduler.
44pub(crate) struct SteelCommandRuntime;

46pub(crate) type SteelCommandContext<S> = CommandContext<S, SteelCommandRuntime>;
47pub(crate) type SteelContextChain<S> = ContextChain<S, SteelCommandRuntime>;

49type StandardExecutor<S> =
50    dyn Fn(&SteelCommandContext<S>) -> Result<i32, CommandSyntaxError> + Send + Sync;
51type SuspendedExecutor<S> = dyn Fn(&SteelCommandContext<S>) -> Result<Box<dyn CommandResultSuspension>, CommandSyntaxError>
52    + Send
53    + Sync;
54type StandardModifier<S> =
55    dyn Fn(&SteelCommandContext<S>) -> Result<Vec<S>, CommandSyntaxError> + Send + Sync;

57/// A terminal executor stored in a Steel command graph.
58pub(crate) enum SteelExecutor<S>
59where
60    S: ExecutionCommandSource,
61{
62    Standard(Box<StandardExecutor<S>>),
63    Suspended(Box<SuspendedExecutor<S>>),
64    Custom(Arc<dyn CustomCommandExecutor<S>>),
65}

67/// A redirect modifier stored in a Steel command graph.
68pub(crate) enum SteelModifier<S>
69where
70    S: ExecutionCommandSource,
71{
72    Standard(Box<StandardModifier<S>>),
73    Custom(Arc<dyn CustomModifierExecutor<S>>),
74}

134impl<S> CommandNodeBuilder<S, SteelCommandRuntime>
135where
136    S: ExecutionCommandSource,
137{
138    /// Attaches an ordinary synchronous executor.
139    #[must_use]
140    pub(crate) fn executes(
141        self,
142        executor: impl Fn(&SteelCommandContext<S>) -> Result<i32, CommandSyntaxError>
143        + Send
144        + Sync
145        + 'static,
146    ) -> Self {
147        self.executes_with_executor(Arc::new(SteelExecutor::Standard(Box::new(executor))))
148    }
149
150    /// Attaches an ordinary executor whose command result is produced across ticks.
151    #[must_use]
152    pub(crate) fn executes_suspended<T>(
153        self,
154        executor: impl Fn(&SteelCommandContext<S>) -> Result<T, CommandSyntaxError>
155        + Send
156        + Sync
157        + 'static,
158    ) -> Self
159    where
160        T: CommandResultSuspension,
161    {
162        let executor = move |context: &SteelCommandContext<S>| {
163            executor(context)
164                .map(|suspension| Box::new(suspension) as Box<dyn CommandResultSuspension>)
165        };
166        self.executes_with_executor(Arc::new(SteelExecutor::Suspended(Box::new(executor))))
167    }
168
169    /// Attaches an internal executor with frame and queue control.
170    #[must_use]
171    pub(crate) fn executes_custom(self, executor: impl CustomCommandExecutor<S> + 'static) -> Self {
172        self.executes_with_executor(Arc::new(SteelExecutor::Custom(Arc::new(executor))))
173    }
174
175    /// Redirects parsing and transforms the source once before continuing.
176    #[must_use]
177    pub(crate) fn redirects_with(
178        self,
179        target: impl Into<CommandRedirectTarget>,
180        modifier: impl Fn(&SteelCommandContext<S>) -> Result<S, CommandSyntaxError>
181        + Send
182        + Sync
183        + 'static,
184    ) -> Self {
185        let modifier = SteelModifier::Standard(Box::new(move |context| {
186            modifier(context).map(|source| vec![source])
187        }));
188        self.redirects_with_modifier(target, Arc::new(modifier), false)
189    }
190
191    /// Redirects parsing and expands one source into zero or more sources.
192    #[must_use]
193    pub(crate) fn forks(
194        self,
195        target: impl Into<CommandRedirectTarget>,
196        modifier: impl Fn(&SteelCommandContext<S>) -> Result<Vec<S>, CommandSyntaxError>
197        + Send
198        + Sync
199        + 'static,
200    ) -> Self {
201        self.redirects_with_modifier(
202            target,
203            Arc::new(SteelModifier::Standard(Box::new(modifier))),
204            true,
205        )
206    }
207
208    /// Redirects with an internal modifier that controls frames or queued work.
209    #[must_use]
210    pub(crate) fn redirects_custom(
211        self,
212        target: impl Into<CommandRedirectTarget>,
213        modifier: impl CustomModifierExecutor<S> + 'static,
214        forks: bool,
215    ) -> Self {
216        self.redirects_with_modifier(
217            target,
218            Arc::new(SteelModifier::Custom(Arc::new(modifier))),
219            forks,
220        )
221    }
222}

367impl SteelCommandContext<CommandSource> {
368    pub(crate) fn score_holders(
369        &self,
370        name: &str,
371        wildcard: ScoreHolderWildcard,
372    ) -> Result<Vec<ScoreHolder>, CommandSyntaxError> {
373        let holders = self
374            .score_holder_argument(name)
375            .ok_or_else(|| missing_score_holder_argument(name))?
376            .resolve(self.source(), wildcard)?;
377        if holders.is_empty() {
378            Err(CommandSyntaxError::dynamic(TextComponent::from(
379                &translations::ARGUMENT_SCORE_HOLDER_EMPTY,
380            )))
381        } else {
382            Ok(holders)
383        }
384    }
385
386    pub(crate) fn score_holder(&self, name: &str) -> Result<ScoreHolder, CommandSyntaxError> {
387        let mut holders = self.score_holders(name, ScoreHolderWildcard::Empty)?;
388        Ok(holders.remove(0))
389    }
390
391    pub(crate) fn optional_entities(
392        &self,
393        name: &str,
394    ) -> Result<Vec<SharedEntity>, CommandSyntaxError> {
395        self.entity_selector(name)
396            .ok_or_else(|| missing_selector_argument(name))?
397            .find_entities(self.source())
398    }
399
400    pub(crate) fn entities(&self, name: &str) -> Result<Vec<SharedEntity>, CommandSyntaxError> {
401        let entities = self.optional_entities(name)?;
402        if entities.is_empty() {
403            Err(CommandSyntaxError::dynamic(TextComponent::from(
404                &translations::ARGUMENT_ENTITY_NOTFOUND_ENTITY,
405            )))
406        } else {
407            Ok(entities)
408        }
409    }
410
411    pub(crate) fn entity(&self, name: &str) -> Result<SharedEntity, CommandSyntaxError> {
412        let mut entities = self.entities(name)?;
413        if entities.len() != 1 {
414            return Err(CommandSyntaxError::dynamic(TextComponent::from(
415                &translations::ARGUMENT_ENTITY_TOOMANY,
416            )));
417        }
418        Ok(entities.remove(0))
419    }
420
421    pub(crate) fn optional_players(
422        &self,
423        name: &str,
424    ) -> Result<Vec<Arc<Player>>, CommandSyntaxError> {
425        self.entity_selector(name)
426            .ok_or_else(|| missing_selector_argument(name))?
427            .find_players(self.source())
428    }
429
430    pub(crate) fn players(&self, name: &str) -> Result<Vec<Arc<Player>>, CommandSyntaxError> {
431        let players = self.optional_players(name)?;
432        if players.is_empty() {
433            Err(CommandSyntaxError::dynamic(TextComponent::from(
434                &translations::ARGUMENT_ENTITY_NOTFOUND_PLAYER,
435            )))
436        } else {
437            Ok(players)
438        }
439    }
440
441    pub(crate) fn player(&self, name: &str) -> Result<Arc<Player>, CommandSyntaxError> {
442        let mut players = self.players(name)?;
443        if players.len() != 1 {
444            return Err(CommandSyntaxError::dynamic(TextComponent::from(
445                &translations::ARGUMENT_PLAYER_TOOMANY,
446            )));
447        }
448        Ok(players.remove(0))
449    }
450}

452fn missing_selector_argument(name: &str) -> CommandSyntaxError {
