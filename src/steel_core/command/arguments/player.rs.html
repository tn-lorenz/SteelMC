16/// A player argument.
17#[derive(Default)]
18pub struct PlayerArgument {
19    /// If only accepts one player
20    one: bool,
21}
22impl PlayerArgument {
23    /// Creates a selector for multiple players
24    #[must_use]
25    pub const fn multiple() -> Self {
26        PlayerArgument { one: false }
27    }
28    /// Creates a selector for one player
29    #[must_use]
30    pub const fn one() -> Self {
31        PlayerArgument { one: true }
32    }
33}
34
35impl CommandArgument for PlayerArgument {
36    type Output = Vec<Arc<Player>>;
37
38    fn parse<'a>(
39        &self,
40        arg: &'a [&'a str],
41        context: &mut CommandContext,
42    ) -> Option<(&'a [&'a str], Self::Output)> {
43        let players = context.server.get_players();
44        if arg.is_empty() {
45            return None;
46        }
47        if players.is_empty() {
48            return Some((&arg[1..], vec![]));
49        }
50        let entities = match arg[0] {
51            "@a" => players,
52            "@p" => {
53                let position = context.position;
54                let mut near_dist = (f64::MAX, players[0].clone());
55                for player in players {
56                    let dist = player.get_position().squared_distance_to_vec(position);
57                    if dist < near_dist.0 {
58                        near_dist = (dist, player);
59                    }
60                }
61                vec![near_dist.1]
62            }
63            "@r" => {
64                vec![players.into_iter().choose(&mut rand::rng())?]
65            }
66            "@s" => {
67                if let Some(player) = &context.player {
68                    vec![player.clone()]
69                } else {
70                    vec![]
71                }
72            }
73            name => {
74                let uuid = if let Ok(uuid) = Uuid::parse_str(name) {
75                    uuid
76                } else {
77                    Uuid::nil()
78                };
79                let player = players.into_iter().find_map(|p| {
80                    if p.gameprofile.name == name || p.uuid() == uuid {
81                        Some(p)
82                    } else {
83                        None
84                    }
85                })?;
86                vec![player]
87            }
88        };
89        // TODO: Add entity argiments. (e.g. @e[limit=1])
90        Some((&arg[1..], entities))
91    }
92
93    fn usage(&self) -> (ArgumentType, Option<SuggestionType>) {
94        (
95            ArgumentType::Entity {
96                flags: 2 | u8::from(self.one),
97            },
98            Some(SuggestionType::AskServer),
99        )
100    }
101
102    fn suggest(&self, prefix: &str, suggestion_ctx: &SuggestionContext) -> Vec<SuggestionEntry> {
103        let mut suggestions = vec![
104            SuggestionEntry::with_tooltip("@a", &ARGUMENT_ENTITY_SELECTOR_ALL_PLAYERS),
105            SuggestionEntry::with_tooltip("@p", &ARGUMENT_ENTITY_SELECTOR_NEAREST_PLAYER),
106            SuggestionEntry::with_tooltip("@r", &ARGUMENT_ENTITY_SELECTOR_RANDOM_PLAYER),
107            SuggestionEntry::with_tooltip("@s", &ARGUMENT_ENTITY_SELECTOR_SELF),
108        ];
109        suggestions.append(
110            &mut suggestion_ctx
111                .server
112                .get_players()
113                .iter()
114                .map(|p| SuggestionEntry::new(p.gameprofile.name.clone()))
115                .collect(),
116        );
117        suggestions.retain(|s| s.text.starts_with(prefix));
118        suggestions
119    }
120}
