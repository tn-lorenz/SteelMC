1//! A vector3 argument.
2use steel_protocol::packets::game::{ArgumentType, SuggestionType};
3use steel_utils::math::Vector3;
4
5use crate::command::arguments::{CommandArgument, Helper};
6use crate::command::context::CommandContext;
7
8/// A vector3 argument.
9pub struct Vector3Argument;
10
11impl CommandArgument for Vector3Argument {
12    type Output = Vector3<f64>;
13
14    fn parse<'a>(
15        &self,
16        arg: &'a [&'a str],
17        context: &mut CommandContext,
18    ) -> Option<(&'a [&'a str], Self::Output)> {
19        let x = Helper::parse_relative_coordinate::<false>(
20            arg.first()?,
21            context.position.map(|o| o.x),
22        )?;
23        let y =
24            Helper::parse_relative_coordinate::<true>(arg.get(1)?, context.position.map(|o| o.y))?;
25        let z =
26            Helper::parse_relative_coordinate::<false>(arg.get(2)?, context.position.map(|o| o.z))?;
27
28        Some((&arg[3..], Vector3::new(x, y, z)))
29    }
30
31    fn usage(&self) -> (ArgumentType, Option<SuggestionType>) {
32        (ArgumentType::Vec3, None)
33    }
34}
