1//! A rotation argument.
2use steel_protocol::packets::game::{ArgumentType, SuggestionType};
3
4use crate::command::arguments::CommandArgument;
5use crate::command::context::CommandContext;
6
7/// A rotation argument.
8pub struct RotationArgument;
9
10impl CommandArgument for RotationArgument {
11    type Output = (f32, f32);
12
13    fn parse<'a>(
14        &self,
15        arg: &'a [&'a str],
16        _context: &mut CommandContext,
17    ) -> Option<(&'a [&'a str], Self::Output)> {
18        let mut yaw = arg.first()?.parse::<f32>().ok()?;
19        let mut pitch = arg.get(1)?.parse::<f32>().ok()?;
20
21        yaw = yaw.rem_euclid(360.0);
22        if yaw >= 180.0 {
23            yaw -= 360.0;
24        }
25        pitch = pitch.rem_euclid(360.0);
26        if pitch >= 180.0 {
27            pitch -= 360.0;
28        }
29
30        Some((&arg[2..], (yaw, pitch)))
31    }
32
33    fn usage(&self) -> (ArgumentType, Option<SuggestionType>) {
34        (ArgumentType::Rotation, None)
35    }
36}
