1//! An integer argument.
2use steel_protocol::packets::game::{ArgumentType, SuggestionType};
3
4use crate::command::arguments::CommandArgument;
5use crate::command::context::CommandContext;
6
7/// An integer argument that parses a 32-bit signed integer.
8///
9/// Can optionally have minimum and maximum bounds.
10pub struct IntegerArgument {
11    min: Option<i32>,
12    max: Option<i32>,
13}
14
15impl IntegerArgument {
16    /// Creates a new unbounded integer argument.
17    #[must_use]
18    pub const fn new() -> Self {
19        Self {
20            min: None,
21            max: None,
22        }
23    }
24
25    /// Creates a new integer argument with bounds.
26    #[must_use]
27    pub const fn bounded(min: Option<i32>, max: Option<i32>) -> Self {
28        Self { min, max }
29    }
30}
31
32impl Default for IntegerArgument {
33    fn default() -> Self {
34        Self::new()
35    }
36}
37
38impl CommandArgument for IntegerArgument {
39    type Output = i32;
40
41    fn parse<'a>(
42        &self,
43        arg: &'a [&'a str],
44        _context: &mut CommandContext,
45    ) -> Option<(&'a [&'a str], Self::Output)> {
46        let s = arg.first()?;
47        let value: i32 = s.parse().ok()?;
48
49        // Check bounds
50        if let Some(min) = self.min
51            && value < min
52        {
53            return None;
54        }
55        if let Some(max) = self.max
56            && value > max
57        {
58            return None;
59        }
60
61        Some((&arg[1..], value))
62    }
63
64    fn usage(&self) -> (ArgumentType, Option<SuggestionType>) {
65        (
66            ArgumentType::Integer {
67                min: self.min,
68                max: self.max,
69            },
70            None,
71        )
72    }
73}
