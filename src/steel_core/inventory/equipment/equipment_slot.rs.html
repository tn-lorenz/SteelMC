3/// Equipment slot types for categorization.
4#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
5pub enum EquipmentSlotType {
6    /// Hand slots (main hand, off hand).
7    Hand,
8    /// Humanoid armor slots (head, chest, legs, feet).
9    HumanoidArmor,
10    /// Animal armor slot (body).
11    AnimalArmor,
12    /// Saddle slot.
13    Saddle,
14}
15
16/// Equipment slots for entities.
17///
18/// Based on Minecraft's `EquipmentSlot` enum.
19#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
20pub enum EquipmentSlot {
21    /// The main hand slot.
22    MainHand,
23    /// The off hand slot.
24    OffHand,
25    /// The feet armor slot (boots).
26    Feet,
27    /// The legs armor slot (leggings).
28    Legs,
29    /// The chest armor slot (chestplate).
30    Chest,
31    /// The head armor slot (helmet).
32    Head,
33    /// The body armor slot (for animals like horses).
34    Body,
35    /// The saddle slot (for rideable animals).
36    Saddle,
37}
38
39impl EquipmentSlot {
40    /// All equipment slots in order.
41    pub const ALL: [EquipmentSlot; 8] = [
42        EquipmentSlot::MainHand,
43        EquipmentSlot::OffHand,
44        EquipmentSlot::Feet,
45        EquipmentSlot::Legs,
46        EquipmentSlot::Chest,
47        EquipmentSlot::Head,
48        EquipmentSlot::Body,
49        EquipmentSlot::Saddle,
50    ];
51
52    /// Humanoid armor slots (head, chest, legs, feet).
53    pub const ARMOR_SLOTS: [EquipmentSlot; 4] = [
54        EquipmentSlot::Head,
55        EquipmentSlot::Chest,
56        EquipmentSlot::Legs,
57        EquipmentSlot::Feet,
58    ];
59
60    /// Returns the slot type for this equipment slot.
61    #[must_use]
62    pub const fn slot_type(self) -> EquipmentSlotType {
63        match self {
64            EquipmentSlot::MainHand | EquipmentSlot::OffHand => EquipmentSlotType::Hand,
65            EquipmentSlot::Feet
66            | EquipmentSlot::Legs
67            | EquipmentSlot::Chest
68            | EquipmentSlot::Head => EquipmentSlotType::HumanoidArmor,
69            EquipmentSlot::Body => EquipmentSlotType::AnimalArmor,
70            EquipmentSlot::Saddle => EquipmentSlotType::Saddle,
71        }
72    }
73
74    /// Returns the index of this slot for array storage (0-7).
75    #[must_use]
76    pub const fn index(self) -> usize {
77        match self {
78            EquipmentSlot::MainHand => 0,
79            EquipmentSlot::OffHand => 1,
80            EquipmentSlot::Feet => 2,
81            EquipmentSlot::Legs => 3,
82            EquipmentSlot::Chest => 4,
83            EquipmentSlot::Head => 5,
84            EquipmentSlot::Body => 6,
85            EquipmentSlot::Saddle => 7,
86        }
87    }
88
89    /// Returns true if this is an armor slot (humanoid or animal).
90    #[must_use]
91    pub const fn is_armor(self) -> bool {
92        matches!(
93            self.slot_type(),
94            EquipmentSlotType::HumanoidArmor | EquipmentSlotType::AnimalArmor
95        )
96    }
97
98    /// Returns the equipment slot with the given name, or None if not found.
99    #[must_use]
100    pub fn by_name(name: &str) -> Option<Self> {
101        match name {
102            "mainhand" => Some(EquipmentSlot::MainHand),
103            "offhand" => Some(EquipmentSlot::OffHand),
104            "feet" => Some(EquipmentSlot::Feet),
105            "legs" => Some(EquipmentSlot::Legs),
106            "chest" => Some(EquipmentSlot::Chest),
107            "head" => Some(EquipmentSlot::Head),
108            "body" => Some(EquipmentSlot::Body),
109            "saddle" => Some(EquipmentSlot::Saddle),
110            _ => None,
111        }
112    }
113
114    /// Returns the name of this equipment slot.
115    #[must_use]
116    pub const fn name(self) -> &'static str {
117        match self {
118            EquipmentSlot::MainHand => "mainhand",
119            EquipmentSlot::OffHand => "offhand",
120            EquipmentSlot::Feet => "feet",
121            EquipmentSlot::Legs => "legs",
122            EquipmentSlot::Chest => "chest",
123            EquipmentSlot::Head => "head",
124            EquipmentSlot::Body => "body",
125            EquipmentSlot::Saddle => "saddle",
126        }
127    }
128}
