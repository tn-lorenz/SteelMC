1//! Entity equipment storage.
2
3use std::mem;
4
5use steel_registry::item_stack::ItemStack;
6
7use super::EquipmentSlot;
8
9/// Equipment storage for entities (armor, hands, etc.)
10pub struct EntityEquipment {
11    slots: [ItemStack; 8],
12}
13
14impl Default for EntityEquipment {
15    fn default() -> Self {
16        Self::new()
17    }
18}
19
20impl EntityEquipment {
21    /// Creates a new empty equipment storage.
22    #[must_use]
23    pub fn new() -> Self {
24        Self {
25            slots: [
26                ItemStack::empty(),
27                ItemStack::empty(),
28                ItemStack::empty(),
29                ItemStack::empty(),
30                ItemStack::empty(),
31                ItemStack::empty(),
32                ItemStack::empty(),
33                ItemStack::empty(),
34            ],
35        }
36    }
37
38    /// Gets a reference to the item in a slot.
39    #[must_use]
40    pub fn get_ref(&self, slot: EquipmentSlot) -> &ItemStack {
41        &self.slots[slot.index()]
42    }
43
44    /// Gets a mutable reference to the item in a slot.
45    pub fn get_mut(&mut self, slot: EquipmentSlot) -> &mut ItemStack {
46        &mut self.slots[slot.index()]
47    }
48
49    /// Takes the item from a slot, leaving an empty stack in its place.
50    pub fn take(&mut self, slot: EquipmentSlot) -> ItemStack {
51        mem::take(&mut self.slots[slot.index()])
52    }
53
54    /// Sets the item in a slot, returning the old item.
55    pub fn set(&mut self, slot: EquipmentSlot, stack: ItemStack) -> ItemStack {
56        mem::replace(&mut self.slots[slot.index()], stack)
57    }
58
59    /// Checks if a specific slot is empty.
60    #[must_use]
61    pub fn is_slot_empty(&self, slot: EquipmentSlot) -> bool {
62        self.slots[slot.index()].is_empty()
63    }
64
65    /// Checks if all slots are empty.
66    #[must_use]
67    pub fn is_empty(&self) -> bool {
68        self.slots.iter().all(ItemStack::is_empty)
69    }
70
71    /// Clears all slots, replacing them with empty stacks.
72    pub fn clear(&mut self) {
73        for slot in &mut self.slots {
74            *slot = ItemStack::empty();
75        }
76    }
77}
