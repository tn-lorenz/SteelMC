1//! Menu provider for opening menus.
2
3use steel_registry::menu_type::MenuTypeRef;
4use text_components::TextComponent;
5
6use crate::inventory::menu::Menu;
7
8/// Trait for menu instances that can be opened by players.
9///
10/// This extends `Menu` with the additional information needed to send
11/// the open screen packet: menu type and container ID.
12pub trait MenuInstance: Menu + Send + Sync {
13    /// Returns the menu type for the open screen packet.
14    fn menu_type(&self) -> MenuTypeRef;
15
16    /// Returns the container ID for this menu.
17    fn container_id(&self) -> u8;
18}
19
20/// Trait for types that can create menus.
21///
22/// Each menu type implements this with a struct that holds the necessary data.
23pub trait MenuProvider {
24    /// Returns the display title for this menu.
25    fn title(&self) -> TextComponent;
26
27    /// Creates a menu with the given container ID.
28    fn create(&self, container_id: u8) -> Box<dyn MenuInstance>;
29}
