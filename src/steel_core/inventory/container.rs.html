14/// Something that contains items.
15/// I also use container interchangeably with inventory as they mean approximately the same thing.
16/// But inventory could also refer to the player's inventory.
17/// Example: `PlayerInventory`, Chest, Temporary Crafting Table
18#[enum_dispatch]
19pub trait Container {
20    /// Returns the number of slots in this container.
21    fn get_container_size(&self) -> usize;
22
23    /// Returns true if all slots in this container are empty.
24    fn is_empty(&self) -> bool {
25        for i in 0..self.get_container_size() {
26            if !self.get_item(i).is_empty() {
27                return false;
28            }
29        }
30        true
31    }
32
33    /// Returns a reference to the item in the specified slot.
34    fn get_item(&self, slot: usize) -> &ItemStack;
35
36    /// Returns a mutable reference to the item in the specified slot.
37    fn get_item_mut(&mut self, slot: usize) -> &mut ItemStack;
38
39    /// Sets the item in the specified slot.
40    fn set_item(&mut self, slot: usize, stack: ItemStack);
41
42    /// Removes up to `count` items from the specified slot and returns them.
43    fn remove_item(&mut self, slot: usize, count: i32) -> ItemStack {
44        let item = self.get_item_mut(slot);
45
46        if item.is_empty() || count <= 0 {
47            return ItemStack::empty();
48        }
49
50        let take_count = count.min(item.count());
51        let mut taken = item.clone();
52        taken.set_count(take_count);
53
54        let remaining = item.count() - take_count;
55        if remaining <= 0 {
56            *item = ItemStack::empty();
57        } else {
58            item.set_count(remaining);
59        }
60
61        taken
62    }
63
64    /// Removes the item from the specified slot without triggering updates.
65    fn remove_item_no_update(&mut self, slot: usize) -> ItemStack {
66        mem::take(self.get_item_mut(slot))
67    }
68
69    /// Returns the maximum stack size for this container.
70    fn get_max_stack_size(&self) -> i32 {
71        99
72    }
73
74    /// Returns the maximum stack size for a specific item in this container.
75    ///
76    /// Takes the minimum of the container's max stack size and the item's max stack size.
77    /// Based on Java's `Container.getMaxStackSize(ItemStack)`.
78    fn get_max_stack_size_for_item(&self, item: &ItemStack) -> i32 {
79        self.get_max_stack_size().min(item.max_stack_size())
80    }
81
82    /// Marks this container as changed (dirty) for saving/syncing.
83    fn set_changed(&mut self);
84
85    /// Returns true if the player can still interact with this container.
86    ///
87    /// This is used to validate that:
88    /// - The container still exists (e.g., chest block hasn't been destroyed)
89    /// - The player is within interaction range
90    /// - Any other conditions for valid interaction
91    ///
92    /// The default implementation always returns true (e.g., for player inventory).
93    /// Block-based containers should override this to check block existence and distance.
94    ///
95    /// Based on Java's `Container.stillValid(Player)`.
96    fn still_valid(&self) -> bool {
97        true
98    }
99
100    /// Returns true if the specified item can be placed in the specified slot.
101    fn can_place_item(&self, _slot: usize, _stack: &ItemStack) -> bool {
102        true
103    }
104
105    /// Returns true if the specified item can be taken from the specified slot.
106    fn can_take_item(&self, _slot: usize, _stack: &ItemStack) -> bool {
107        true
108    }
109
110    /// Clears all items from this container.
111    fn clear_content(&mut self) {
112        for i in 0..self.get_container_size() {
113            self.set_item(i, ItemStack::empty());
114        }
115    }
116
117    /// Tries to add an item to the container.
118    ///
119    /// First tries to stack with existing matching items, then tries empty slots.
120    /// Returns true if the entire stack was added, false if some or all couldn't fit.
121    /// The passed stack is modified to contain any remaining items.
122    ///
123    /// Based on Java's `Inventory.add(ItemStack)`.
124    fn add(&mut self, stack: &mut ItemStack) -> bool {
125        if stack.is_empty() {
126            return true;
127        }
128
129        let size = self.get_container_size();
130        let max_size = self.get_max_stack_size_for_item(stack);
131
132        // First pass: try to stack with existing items
133        if stack.is_stackable() {
134            for slot in 0..size {
135                if stack.is_empty() {
136                    return true;
137                }
138                let existing = self.get_item_mut(slot);
139                if !existing.is_empty() && ItemStack::is_same_item_same_components(existing, stack)
140                {
141                    let space = max_size - existing.count();
142                    if space > 0 {
143                        let to_add = stack.count().min(space);
144                        existing.grow(to_add);
145                        stack.shrink(to_add);
146                    }
147                }
148            }
149        }
150
151        // Second pass: try empty slots
152        for slot in 0..size {
153            if stack.is_empty() {
154                return true;
155            }
156            if self.get_item(slot).is_empty() && self.can_place_item(slot, stack) {
157                let to_place = stack.count().min(max_size);
158                let mut placed = stack.clone();
159                placed.set_count(to_place);
160                self.set_item(slot, placed);
161                stack.shrink(to_place);
162            }
163        }
164
165        stack.is_empty()
166    }
167}

169/// Calculates the redstone comparator signal strength (0-15) from a container.
170///
171/// Based on Java's `AbstractContainerMenu.getRedstoneSignalFromContainer`.
172/// The signal is proportional to how full the container is:
173/// - 0 = empty
174/// - 1-14 = partially filled (linear interpolation)
175/// - 15 = completely full
176///
177/// # Arguments
178/// * `container` - The container to calculate the signal for
179///
180/// # Returns
181/// Signal strength from 0 to 15
182#[must_use]
183pub fn calculate_redstone_signal_from_container(container: &dyn Container) -> i32 {
184    let size = container.get_container_size();
185    if size == 0 {
186        return 0;
187    }
188
189    let mut total_percent: f32 = 0.0;
190
191    for i in 0..size {
192        let item = container.get_item(i);
193        if !item.is_empty() {
194            let max_stack = container.get_max_stack_size_for_item(item);
195            total_percent += item.count() as f32 / max_stack as f32;
196        }
197    }
198
199    total_percent /= size as f32;
200
201    // Lerp from 0 to 15 based on fullness
202    // Equivalent to Java's Mth.lerpDiscrete(totalPercent, 0, 15)
203    (total_percent * 15.0).round() as i32
204}
