16/// Manages the block breaking state for a player.
17///
18/// Based on Java's `ServerPlayerGameMode` fields and logic.
19pub struct BlockBreakingManager {
20    /// Whether the player is currently breaking a block.
21    is_destroying_block: bool,
22    /// The tick when destruction started.
23    destroy_progress_start: u64,
24    /// The position of the block being destroyed.
25    destroy_pos: BlockPos,
26    /// The current game tick counter.
27    game_ticks: u64,
28    /// Whether there's a delayed destroy pending (for slow mining).
29    has_delayed_destroy: bool,
30    /// Position of the delayed destroy.
31    delayed_destroy_pos: BlockPos,
32    /// The tick when delayed destroy started.
33    delayed_tick_start: u64,
34    /// The last sent destruction progress state (0-9, or -1 for none).
35    last_sent_state: i32,
36}
37
38impl Default for BlockBreakingManager {
39    fn default() -> Self {
40        Self::new()
41    }
42}
43
44impl BlockBreakingManager {
45    /// Creates a new block breaking manager.
46    #[must_use]
47    pub fn new() -> Self {
48        Self {
49            is_destroying_block: false,
50            destroy_progress_start: 0,
51            destroy_pos: BlockPos::new(0, 0, 0),
52            game_ticks: 0,
53            has_delayed_destroy: false,
54            delayed_destroy_pos: BlockPos::new(0, 0, 0),
55            delayed_tick_start: 0,
56            last_sent_state: -1,
57        }
58    }
59
60    /// Ticks the block breaking manager.
61    ///
62    /// This handles delayed destruction and updates break progress.
63    pub fn tick(&mut self, player: &Player, world: &World) {
64        self.game_ticks += 1;
65
66        if self.has_delayed_destroy {
67            let state = world.get_block_state(&self.delayed_destroy_pos);
68            if is_air(state) {
69                self.has_delayed_destroy = false;
70            } else {
71                let progress = self.increment_destroy_progress(
72                    player,
73                    world,
74                    state,
75                    self.delayed_destroy_pos,
76                    self.delayed_tick_start,
77                );
78                if progress >= 1.0 {
79                    self.has_delayed_destroy = false;
80                    self.destroy_block(player, world, self.delayed_destroy_pos);
81                }
82            }
83        } else if self.is_destroying_block {
84            let state = world.get_block_state(&self.destroy_pos);
85            if is_air(state) {
86                // Block was broken by something else
87                world.broadcast_block_destruction(player.id, self.destroy_pos, -1);
88                self.last_sent_state = -1;
89                self.is_destroying_block = false;
90            } else {
91                self.increment_destroy_progress(
92                    player,
93                    world,
94                    state,
95                    self.destroy_pos,
96                    self.destroy_progress_start,
97                );
98            }
99        }
100    }
101
102    /// Calculates and updates destruction progress, broadcasting to clients.
103    fn increment_destroy_progress(
104        &mut self,
105        player: &Player,
106        world: &World,
107        block_state: BlockStateId,
108        pos: BlockPos,
109        destroy_start_tick: u64,
110    ) -> f32 {
111        let ticks_spent = self.game_ticks.saturating_sub(destroy_start_tick);
112        let destroy_speed = get_destroy_progress(player, block_state);
113        let progress = destroy_speed * (ticks_spent + 1) as f32;
114        let state = (progress * 10.0) as i32;
115
116        if state != self.last_sent_state {
117            world.broadcast_block_destruction(player.id, pos, state);
118            self.last_sent_state = state;
119        }
120
121        progress
122    }

124    /// Handles a block break action from the client.
125    ///
126    /// Note: The caller (packet handler) is responsible for calling `ack_block_changes_up_to`
127    /// after this method returns, matching vanilla behavior.
128    #[allow(clippy::too_many_lines)]
129    pub fn handle_block_break_action(
130        &mut self,
131        player: &Player,
132        world: &World,
133        pos: BlockPos,
134        action: BlockBreakAction,
135        _direction: Direction,
136    ) {
137        // Validate interaction range
138        if !player.is_within_block_interaction_range(&pos) {
139            return;
140        }
141
142        // Validate Y coordinate
143        if pos.y() >= world.max_build_height() {
144            player.connection.send_packet(CBlockUpdate {
145                pos,
146                block_state: world.get_block_state(&pos),
147            });
148            return;
149        }
150
151        match action {
152            BlockBreakAction::Start => {
153                // Check may_interact permission
154                if !world.may_interact(player, &pos) {
155                    player.connection.send_packet(CBlockUpdate {
156                        pos,
157                        block_state: world.get_block_state(&pos),
158                    });
159                    return;
160                }
161
162                // Creative mode: instant break
163                if player.game_mode.load() == GameType::Creative {
164                    self.destroy_and_ack(player, world, pos);
165                    return;
166                }
167
168                // Check if player can break this block (adventure mode restrictions, etc.)
169                // TODO: Implement blockActionRestricted check
170
171                self.destroy_progress_start = self.game_ticks;
172                let block_state = world.get_block_state(&pos);
173
174                if !is_air(block_state) {
175                    // TODO: Call EnchantmentHelper.onHitBlock and blockState.attack
176
177                    let progress = get_destroy_progress(player, block_state);
178
179                    if progress >= 1.0 {
180                        // Insta-mine
181                        self.destroy_and_ack(player, world, pos);
182                    } else {
183                        // Start breaking
184                        if self.is_destroying_block {
185                            // Send block update for the old position to cancel client prediction
186                            player.connection.send_packet(CBlockUpdate {
187                                pos: self.destroy_pos,
188                                block_state: world.get_block_state(&self.destroy_pos),
189                            });
190                        }
191
192                        self.is_destroying_block = true;
193                        self.destroy_pos = pos;
194                        let state = (progress * 10.0) as i32;
195                        world.broadcast_block_destruction(player.id, pos, state);
196                        self.last_sent_state = state;
197                    }
198                }
199            }
200
201            BlockBreakAction::Stop => {
202                if pos == self.destroy_pos {
203                    let ticks_spent = self.game_ticks.saturating_sub(self.destroy_progress_start);
204                    let block_state = world.get_block_state(&pos);
205
206                    if !is_air(block_state) {
207                        let destroy_speed = get_destroy_progress(player, block_state);
208                        let progress = destroy_speed * (ticks_spent + 1) as f32;
209
210                        if progress >= 0.7 {
211                            // Complete the break
212                            self.is_destroying_block = false;
213                            world.broadcast_block_destruction(player.id, pos, -1);
214                            self.destroy_and_ack(player, world, pos);
215                            return;
216                        }
217
218                        if !self.has_delayed_destroy {
219                            // Set up delayed destroy
220                            self.is_destroying_block = false;
221                            self.has_delayed_destroy = true;
222                            self.delayed_destroy_pos = pos;
223                            self.delayed_tick_start = self.destroy_progress_start;
224                        }
225                    }
226                }
227            }
228
229            BlockBreakAction::Abort => {
230                self.is_destroying_block = false;
231
232                if self.destroy_pos != pos {
233                    log::warn!(
234                        "Mismatch in destroy block pos: {:?} vs {:?}",
235                        self.destroy_pos,
236                        pos
237                    );
238                    world.broadcast_block_destruction(player.id, self.destroy_pos, -1);
239                }
240
241                world.broadcast_block_destruction(player.id, pos, -1);
242            }
243        }
244    }

246    /// Destroys a block and sends appropriate response.
247    fn destroy_and_ack(&mut self, player: &Player, world: &World, pos: BlockPos) {
248        if !self.destroy_block(player, world, pos) {
249            // Send block update to resync client
250            player.connection.send_packet(CBlockUpdate {
251                pos,
252                block_state: world.get_block_state(&pos),
253            });
254        }
255    }

257    /// Destroys a block at the given position.
258    ///
259    /// Returns true if the block was successfully destroyed.
260    #[allow(clippy::unused_self)]
261    fn destroy_block(&self, player: &Player, world: &World, pos: BlockPos) -> bool {
262        let state = world.get_block_state(&pos);
263
264        // Check if player's tool can destroy this block
265        // TODO: Implement canDestroyBlock check for adventure mode
266
267        // Get block info
268        let Some(_block) = REGISTRY.blocks.by_state_id(state) else {
269            return false;
270        };
271
272        // TODO: Check for GameMasterBlock (command blocks, etc.)
273        // TODO: Check blockActionRestricted
274
275        // Remove the block
276        let air_state = REGISTRY.blocks.get_base_state_id(vanilla_blocks::AIR);
277        let changed = world.set_block(pos, air_state, UpdateFlags::UPDATE_ALL);
278
279        if changed {
280            // Play block destruction particles and sound (skip for fire blocks like vanilla)
281            // Exclude the breaking player as they see the effect client-side
282            let block = REGISTRY.blocks.by_state_id(state);
283            let is_fire = block.is_some_and(|b| {
284                b.key == vanilla_blocks::FIRE.key || b.key == vanilla_blocks::SOUL_FIRE.key
285            });
286            if !is_fire {
287                world.destroy_block_effect(pos, u32::from(state.0), Some(player.id));
288            }
289
290            // Check if player has correct tool for drops
291            let has_correct_tool = {
292                let inv = player.inventory.lock();
293                let main_hand = inv.get_item_in_hand(InteractionHand::MainHand);
294                main_hand.is_correct_tool_for_drops(state) || !requires_correct_tool(state)
295            };
296
297            // Damage the tool if the block has non-zero destroy time
298            // This is done before playerDestroy, matching vanilla's Item.mineBlock
299            let block_destroy_time = REGISTRY
300                .blocks
301                .by_state_id(state)
302                .map_or(0.0, |b| b.config.destroy_time);
303
304            if block_destroy_time != 0.0 {
305                let mut inv = player.inventory.lock();
306                let damage_per_block = inv.get_selected_item().get_tool_damage_per_block();
307
308                if damage_per_block > 0 {
309                    // Use with_selected_item_mut to ensure set_changed() is called
310                    // Skip damage if player has infinite materials (creative mode)
311                    let has_infinite_materials = player.has_infinite_materials();
312                    let broke = inv.with_selected_item_mut(|main_hand| {
313                        main_hand.hurt_and_break(damage_per_block, has_infinite_materials)
314                    });
315                    if broke {
316                        // TODO: Play item break sound/particles
317                        log::debug!("Tool broke while mining block at {pos:?}");
318                    }
319                }
320            }
321
322            // Handle drops (skip for creative/spectator)
323            let game_mode = player.game_mode.load();
324            if game_mode != GameType::Spectator
325                && game_mode != GameType::Creative
326                && has_correct_tool
327            {
328                // TODO: Call playerDestroy to spawn drops
329                drop_block_loot(player, world, pos, state);
330            }
331        }
332
333        changed
334    }

337/// Block break action types.
338#[derive(Debug, Clone, Copy, PartialEq, Eq)]
339pub enum BlockBreakAction {
340    /// Player started breaking a block.
341    Start,
342    /// Player stopped breaking a block (finished or released).
343    Stop,
344    /// Player aborted breaking a block.

346}

348/// Checks if a block state is air.
349fn is_air(state: BlockStateId) -> bool {
350    let Some(block) = REGISTRY.blocks.by_state_id(state) else {
351        return true;
352    };
353    block.config.is_air
354}
355
356/// Checks if a block requires the correct tool for drops.
357fn requires_correct_tool(state: BlockStateId) -> bool {
358    let Some(block) = REGISTRY.blocks.by_state_id(state) else {
359        return false;
360    };
361    block.config.requires_correct_tool_for_drops
362}

364/// Gets the destroy progress per tick for a block.
365///
366/// This is based on the vanilla formula:
367/// `1.0 / (destroy_time * 30.0)` for survival with correct tool
368/// `1.0 / (destroy_time * 100.0)` for survival with wrong tool
369/// Instant break for creative mode.
370fn get_destroy_progress(player: &Player, block_state: BlockStateId) -> f32 {
371    let Some(block) = REGISTRY.blocks.by_state_id(block_state) else {
372        return 0.0;
373    };
374
375    let destroy_time = block.config.destroy_time;
376
377    // Instant break for creative
378    if player.game_mode.load() == GameType::Creative {
379        return 1.0;
380    }
381
382    // Unbreakable block
383    if destroy_time < 0.0 {
384        return 0.0;
385    }
386
387    // Instant break for destroy_time == 0
388    if destroy_time == 0.0 {
389        return 1.0;
390    }
391
392    // Get player's mining speed
393    let mining_speed = {
394        let inv = player.inventory.lock();
395        let main_hand = inv.get_item_in_hand(InteractionHand::MainHand);
396        main_hand.get_destroy_speed(block_state)
397    };
398
399    // Check if player has the correct tool
400    let has_correct_tool = {
401        let inv = player.inventory.lock();
402        let main_hand = inv.get_item_in_hand(InteractionHand::MainHand);
403        main_hand.is_correct_tool_for_drops(block_state)
404    };
405
406    // Apply speed modifiers
407    let speed = mining_speed;
408
409    // TODO: Apply efficiency enchantment
410    // TODO: Apply haste/mining fatigue effects
411    // TODO: Apply underwater/in-air penalties
412
413    // Calculate destroy progress per tick
414    // Vanilla formula: speed / hardness / (hasCorrectTool ? 30 : 100)
415    let divisor = if has_correct_tool || !block.config.requires_correct_tool_for_drops {
416        30.0
417    } else {
418        100.0
419    };
420
421    speed / destroy_time / divisor
422}
