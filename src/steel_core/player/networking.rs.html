33#[allow(clippy::struct_field_names)]
34struct KeepAliveTracker {
35    alive_time: u64,
36    alive_pending: bool,
37    alive_id: u64,
38}

40/// A connection to a Java client.
41pub struct JavaConnection {
42    outgoing_packets: UnboundedSender<EnqueuedPacket>,
43    cancel_token: CancellationToken,
44    compression: Option<CompressionInfo>,
45    network_writer: Arc<AsyncMutex<TCPNetworkEncoder<BufWriter<OwnedWriteHalf>>>>,
46    id: u64,
47
48    player: Weak<Player>,
49    keep_alive_tracker: SyncMutex<KeepAliveTracker>,
50    latency: SyncMutex<u32>,
51}

53impl JavaConnection {
54    /// Creates a new `JavaConnection`.
55    pub fn new(
56        outgoing_packets: UnboundedSender<EnqueuedPacket>,
57        cancel_token: CancellationToken,
58        compression: Option<CompressionInfo>,
59        network_writer: Arc<AsyncMutex<TCPNetworkEncoder<BufWriter<OwnedWriteHalf>>>>,
60        id: u64,
61        player: Weak<Player>,
62    ) -> Self {
63        Self {
64            outgoing_packets,
65            cancel_token,
66            compression,
67            network_writer,
68            id,
69            player,
70            keep_alive_tracker: SyncMutex::new(KeepAliveTracker {
71                alive_time: 0,
72                alive_pending: false,
73                alive_id: 0,
74            }),
75            latency: SyncMutex::new(0),
76        }
77    }
78
79    /// Ticks the connection.
80    pub fn tick(&self) {
81        self.keep_connection_alive();
82    }
83
84    #[allow(clippy::unwrap_used)]
85    fn keep_connection_alive(&self) {
86        let mut tracker = self.keep_alive_tracker.lock();
87        let now = SystemTime::now()
88            .duration_since(UNIX_EPOCH)
89            .expect("System time before UNIX EPOCH")
90            .as_millis() as u64;
91
92        if now - tracker.alive_time >= 15000 {
93            if tracker.alive_pending {
94                self.disconnect(translations::DISCONNECT_TIMEOUT.msg());
95            } else {
96                tracker.alive_pending = true;
97                tracker.alive_id = now;
98                tracker.alive_time = now;
99                self.send_packet(CKeepAlive::new(tracker.alive_id as i64));
100            }
101        }
102    }
103
104    /// Handles a keep alive packet.
105    #[allow(clippy::cast_possible_truncation)]
106    fn handle_keep_alive(&self, packet: SKeepAlive) {
107        let mut tracker = self.keep_alive_tracker.lock();
108        if tracker.alive_pending && packet.id as u64 == tracker.alive_id {
109            let now = SystemTime::now()
110                .duration_since(UNIX_EPOCH)
111                .expect("System time before UNIX EPOCH")
112                .as_millis() as u64;
113
114            let time = now.saturating_sub(tracker.alive_time) as u32;
115            tracker.alive_pending = false;
116            drop(tracker);
117            let mut latency = self.latency.lock();
118            *latency = (*latency * 3 + time) / 4;
119        } else {
120            self.disconnect(translations::DISCONNECT_TIMEOUT.msg());
121        }
122    }
123
124    /// Disconnects the client.
125    pub fn disconnect(&self, reason: impl Into<TextComponent>) {
126        self.send_packet(CDisconnect::new(reason.into()));
127        self.close();
128    }

130    /// Sends a packet to the client.
131    ///
132    /// # Panics
133    /// - If the packet fails to be written to the buffer.
134    /// - If the packet fails to be sent through the channel.
135    pub fn send_packet<P: ClientPacket>(&self, packet: P) {
136        let packet = EncodedPacket::write_vec(packet, ConnectionProtocol::Play)
137            .expect("Failed to write packet");
138        if self
139            .outgoing_packets
140            .send(EnqueuedPacket::RawData(packet))
141            .is_err()
142        {
143            self.close();
144        }
145    }
146
147    /// Closes the connection.
148    pub fn close(&self) {
149        self.cancel_token.cancel();
150    }
151
152    /// Returns whether the connection is closed.
153    #[must_use]
154    pub fn closed(&self) -> bool {
155        self.cancel_token.is_cancelled()
156    }
157
158    /// Waits for the connection to be closed.
159    pub async fn wait_for_close(&self) {
160        self.cancel_token.cancelled().await;
161    }

163    /// Processes a packet from the client.
164    pub fn process_packet(
165        self: &Arc<Self>,
166        packet: RawPacket,
167        player: Arc<Player>,
168    ) -> Result<(), PacketError> {
169        let data = &mut Cursor::new(packet.payload);
170
171        match packet.id {
172            play::C_CUSTOM_PAYLOAD => {
173                player.handle_custom_payload(SCustomPayload::read_packet(data)?);
174            }
175            play::S_CHAT => {
176                player.handle_chat(SChat::read_packet(data)?, Arc::clone(&player));
177            }
178            play::S_CHAT_SESSION_UPDATE => {
179                player.handle_chat_session_update(SChatSessionUpdate::read_packet(data)?);
180            }
181            play::S_CHAT_ACK => {
182                player.handle_chat_ack(SChatAck::read_packet(data)?);
183            }
184            play::S_CLIENT_TICK_END => {
185                let _ = SClientTickEnd::read_packet(data)?;
186                player.handle_client_tick_end();
187            }
188            play::S_CHUNK_BATCH_RECEIVED => {
189                let packet = SChunkBatchReceived::read_packet(data)?;
190                player
191                    .chunk_sender
192                    .lock()
193                    .on_chunk_batch_received_by_client(packet.desired_chunks_per_tick);
194            }
195            play::S_KEEP_ALIVE => {
196                self.handle_keep_alive(SKeepAlive::read_packet(data)?);
197            }
198            play::S_MOVE_PLAYER_POS => {
199                player.handle_move_player(SMovePlayerPos::read_packet(data)?.into());
200            }
201            play::S_MOVE_PLAYER_POS_ROT => {
202                player.handle_move_player(SMovePlayerPosRot::read_packet(data)?.into());
203            }
204            play::S_MOVE_PLAYER_ROT => {
205                player.handle_move_player(SMovePlayerRot::read_packet(data)?.into());
206            }
207            play::S_PLAYER_LOADED => {
208                let _ = SPlayerLoad::read_packet(data)?;
209                player.client_loaded.store(true, Ordering::Relaxed);
210            }
211            id => log::info!("play packet id {id} is not known"),
212        }
213        Ok(())
214    }

216    /// Listens for packets from the client.
217    pub async fn listener(
218        self: Arc<Self>,
219        mut reader: TCPNetworkDecoder<BufReader<OwnedReadHalf>>,
220    ) {
221        loop {
222            select! {
223                () = self.wait_for_close() => {
224                    break;
225                }
226                packet = reader.get_raw_packet() => {
227                    match packet {
228                        Ok(packet) => {
229                            if let Some(player) = self.player.upgrade()
230                                && let Err(err) = self.process_packet(packet, player) {
231                                    log::warn!(
232                                        "Failed to get packet from client {}: {err}",
233                                        self.id
234                                    );
235                                }
236                        }
237                        Err(err) => {
238                            log::debug!("Failed to get raw packet from client {}: {err}", self.id);
239                            self.close();
240                        }
241                    }
242                }
243            }
244        }
245    }

247    /// Sends packets to the client.
248    ///
249    /// # Panics
250    /// - If the player is not available.
251    pub async fn sender(self: Arc<Self>, mut sender_recv: UnboundedReceiver<EnqueuedPacket>) {
252        loop {
253            select! {
254                () = self.wait_for_close() => {
255                    break;
256                }
257                packet = sender_recv.recv() => {
258                    if let Some(packet) = packet {
259
260                        let Some(encoded_packet) = (match packet {
261                            EnqueuedPacket::EncodedPacket(packet) => Some(packet),
262                            EnqueuedPacket::RawData(packet) => {
263                                EncodedPacket::from_data(packet, self.compression)
264                                    .await
265                                    .ok()
266                            }
267                        }) else {
268                            log::warn!("Failed to convert packet to encoded packet for client {}", self.id);
269                            continue;
270                        };
271
272                        if let Err(err) = self.network_writer.lock_async().await.write_packet(&encoded_packet).await
273                        {
274                            log::warn!("Failed to send packet to client {}: {err}", self.id);
275                            self.close();
276                        }
277
278                    } else {
279                        //log::warn!(
280                        //    "Internal packet_sender_recv channel closed for client {}",
281                        //    self.id
282                        //);
283                        self.close();
284                    }
285                }
286            }
287        }
288
289        let player = self.player.upgrade().expect("Player is not available");
290        let world = player.world.clone();
291        world.remove_player(player).await;
292    }
