1//! This module contains the `GameProfile` struct, which is used to store information about a player's profile.
2use serde::{Deserialize, Serialize};
3use steel_protocol::packets::login::{GameProfileProperty, LoginGameProfile};
4use uuid::{Builder, Uuid, Variant, Version};
5
6/// An enum representing a profile action.
7#[derive(Deserialize, Serialize, Clone, Debug, PartialEq, Eq)]
8#[serde(rename_all = "SCREAMING_SNAKE_CASE")]
9pub enum GameProfileAction {
10    /// The player has been forced to change their name.
11    ForcedNameChange,
12    /// The player is using a banned skin.
13    UsingBannedSkin,
14}
15
16/// A struct representing a player's game profile.
17#[derive(Deserialize, Clone, Debug)]
18pub struct GameProfile {
19    /// The player's UUID.
20    pub id: Uuid,
21    /// The player's name.
22    pub name: String,
23    /// A list of properties for the player's profile.
24    pub properties: Vec<GameProfileProperty>,
25    /// A list of profile actions for the player.
26    #[serde(rename = "profileActions")]
27    pub profile_actions: Option<Vec<GameProfileAction>>,
28}
29
30impl<'a> From<&'a GameProfile> for LoginGameProfile<'a> {
31    fn from(profile: &'a GameProfile) -> Self {
32        LoginGameProfile {
33            id: profile.id,
34            name: &profile.name,
35            properties: &profile.properties,
36        }
37    }
38}

40/// Returns whether a name is valid for online login and profile lookup.
41#[must_use]
42pub fn is_valid_player_name(name: &str) -> bool {
43    (3..=16).contains(&name.len())
44        && name
45            .chars()
46            .all(|character| character.is_ascii_alphanumeric() || character == '_')
47}
48
49/// Generates vanilla's deterministic offline-mode UUID for a player name.
50#[must_use]
51pub fn offline_uuid(username: &str) -> Uuid {
52    Builder::from_md5_bytes(md5::compute(format!("OfflinePlayer:{username}")).0)
53        .with_version(Version::Md5)
54        .with_variant(Variant::RFC4122)
55        .into_uuid()
56}
