1//! Server-side validation of client message acknowledgements.
2
3use std::collections::VecDeque;
4
5/// Maximum number of tracked messages for acknowledgement validation (Vanilla: 20)
6const MAX_TRACKED_MESSAGES: usize = 20;
7
8/// Entry tracking a sent message signature
9#[derive(Debug, Clone)]
10struct TrackedEntry {
11    signature: Option<Box<[u8]>>,
12    pending: bool, // true if not yet acknowledged by client
13}
14
15impl TrackedEntry {
16    fn acknowledge(self) -> Self {
17        Self {
18            signature: self.signature,
19            pending: false,
20        }
21    }
22}
23
24/// Validates that the client is properly acknowledging messages sent by the server
25#[derive(Debug)]
26pub struct LastSeenMessagesValidator {
27    tracked_messages: VecDeque<Option<TrackedEntry>>,
28    last_pending_signature: Option<Box<[u8]>>,
29}
30
31impl Default for LastSeenMessagesValidator {
32    fn default() -> Self {
33        Self::new()
34    }
35}
36
37impl LastSeenMessagesValidator {
38    /// Creates a new validator
39    pub fn new() -> Self {
40        let mut tracked_messages = VecDeque::with_capacity(MAX_TRACKED_MESSAGES);
41        for _ in 0..MAX_TRACKED_MESSAGES {
42            tracked_messages.push_back(None);
43        }
44        Self {
45            tracked_messages,
46            last_pending_signature: None,
47        }
48    }
49
50    /// Adds a pending message signature that the client should acknowledge
51    /// Matches vanilla's deduplication: only adds if different from last pending message
52    pub fn add_pending(&mut self, signature: Option<Box<[u8]>>) {
53        // Only add if this signature is different from the last one we added
54        // This prevents duplicates when the same message is processed multiple times
55        if signature.as_ref() != self.last_pending_signature.as_ref() {
56            let entry = TrackedEntry {
57                signature: signature.clone(),
58                pending: true,
59            };
60            self.tracked_messages.push_back(Some(entry));
61            self.last_pending_signature = signature;
62        }
63    }
64
65    /// Gets the number of tracked messages
66    pub fn tracked_count(&self) -> usize {
67        self.tracked_messages.len()
68    }
69
70    /// Applies an offset (removes old acknowledged messages from tracking)
71    pub fn apply_offset(&mut self, offset: i32) -> Result<(), String> {
72        let removable = self
73            .tracked_messages
74            .len()
75            .saturating_sub(MAX_TRACKED_MESSAGES);
76        if offset < 0 || offset as usize > removable {
77            return Err(format!(
78                "Advanced last seen window by {offset} messages, but expected at most {removable}"
79            ));
80        }
81        for _ in 0..offset {
82            self.tracked_messages.pop_front();
83        }
84        Ok(())
85    }
86
87    /// Applies an acknowledgement update from the client
88    /// acknowledged: `BitSet` of 20 bits indicating which messages in the window are acknowledged
89    /// offset: How many old messages to remove from the window
90    /// checksum: Optional checksum for validation (0 = skip checksum)
91    pub fn apply_update(
92        &mut self,
93        acknowledged: [u8; 3], // 3 bytes = 24 bits, using 20
94        offset: i32,
95        checksum: u8,
96    ) -> Result<Vec<Box<[u8]>>, String> {
97        log::debug!(
98            "apply_update: offset={}, checksum={}, tracked_messages.len()={}, acknowledged={:?}",
99            offset,
100            checksum,
101            self.tracked_messages.len(),
102            acknowledged
103        );
104
105        // First apply the offset to remove old messages
106        self.apply_offset(offset)?;
107
108        let mut acknowledged_signatures = Vec::new();
109
110        // Process acknowledgements for the tracked window
111        for i in 0..MAX_TRACKED_MESSAGES.min(self.tracked_messages.len()) {
112            let bit_index = i;
113            let byte_index = bit_index / 8;
114            let bit_offset = bit_index % 8;
115            let is_acknowledged = (acknowledged[byte_index] & (1 << bit_offset)) != 0;
116
117            if let Some(entry_opt) = self.tracked_messages.get_mut(i) {
118                if is_acknowledged {
119                    // Client acknowledged this message
120                    if let Some(entry) = entry_opt {
121                        log::debug!(
122                            "Index {}: Client acknowledged message (pending={})",
123                            i,
124                            entry.pending
125                        );
126                        let acknowledged_entry = entry.clone().acknowledge();
127                        if let Some(sig) = &entry.signature {
128                            acknowledged_signatures.push(sig.clone());
129                        }
130                        *entry_opt = Some(acknowledged_entry);
131                    } else {
132                        log::error!(
133                            "Index {i}: Client acknowledged unknown/ignored message! tracked_messages[{i}] = None"
134                        );
135                        return Err(format!(
136                            "Last seen update acknowledged unknown or previously ignored message at index {i}"
137                        ));
138                    }
139                } else {
140                    // Client did not acknowledge this message
141                    if let Some(entry) = entry_opt
142                        && !entry.pending
143                    {
144                        log::error!("Index {i}: Client ignored previously acknowledged message!");
145                        return Err(format!(
146                            "Last seen update ignored previously acknowledged message at index {i}"
147                        ));
148                    }
149                    log::debug!("Index {i}: Client did not acknowledge (setting to None)");
150                    // Set to None if not acknowledged
151                    *entry_opt = None;
152                }
153            }
154        }
155
156        log::debug!(
157            "apply_update: Successfully acknowledged {} signatures",
158            acknowledged_signatures.len()
159        );
160
161        // TODO: Verify checksum if needed (checksum == 0 means skip validation)
162        if checksum != 0 {
163            // For now, we skip checksum validation
164            // In a full implementation, compute checksum of acknowledged_signatures and compare
165        }
166
167        Ok(acknowledged_signatures)
168    }
169}
