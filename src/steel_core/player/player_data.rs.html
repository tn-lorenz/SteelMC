24/// Persistent player data that can be serialized to/from NBT.
25///
26/// This structure mirrors vanilla Minecraft's player data format where possible,
27/// allowing for potential compatibility with vanilla tools.
28///
29/// # TODO: Missing vanilla fields
30/// The following fields should be added once their systems are implemented:
31/// - Food data: `foodLevel`, `foodSaturationLevel`, `foodExhaustionLevel`, `foodTickTimer`
32/// - Experience: `XpP` (progress), `XpLevel`, `XpTotal`, `XpSeed`
33/// - Active potion effects: `active_effects` (List)
34/// - Score: `Score` (Int)
35/// - Ender chest inventory: `EnderItems` (List)
36/// - Last death location: `LastDeathLocation` (`GlobalPos`)
37/// - Respawn position: `SpawnX`, `SpawnY`, `SpawnZ`, `SpawnDimension`, `SpawnForced`, `SpawnAngle`
38#[derive(Debug, Clone)]
39pub struct PersistentPlayerData {
40    /// Position (x, y, z) in absolute world coordinates.
41    /// NBT tag: `Pos` (`DoubleList`)
42    pub pos: [f64; 3],
43
44    /// Velocity (x, y, z) in blocks per tick.
45    /// NBT tag: `Motion` (`DoubleList`)
46    pub motion: [f64; 3],
47
48    /// Rotation (yaw, pitch) in degrees.
49    /// NBT tag: `Rotation` (`FloatList`)
50    pub rotation: [f32; 2],
51
52    /// Whether the player is on the ground.
53    /// NBT tag: `OnGround` (Byte)
54    pub on_ground: bool,
55
56    /// Whether the player is elytra gliding.
57    /// NBT tag: `FallFlying` (Byte)
58    pub fall_flying: bool,
59
60    /// Current health points.
61    /// NBT tag: `Health` (Float)
62    pub health: f32,
63
64    /// Current game mode (0=survival, 1=creative, 2=adventure, 3=spectator).
65    /// NBT tag: `playerGameType` (Int)
66    pub game_mode: i32,
67
68    /// Player abilities (flight, invulnerability, etc.).
69    /// NBT tag: `abilities` (Compound)
70    pub abilities: PersistentAbilities,
71
72    /// Inventory items with slot indices.
73    /// NBT tag: `Inventory` (List of Compounds)
74    pub inventory: Vec<PersistentSlot>,
75
76    /// Currently selected hotbar slot (0-8).
77    /// NBT tag: `SelectedItemSlot` (Int)
78    pub selected_slot: i32,
79
80    /// Dimension identifier (e.g., "minecraft:overworld").
81    /// NBT tag: `Dimension` (String)
82    pub dimension: String,
83
84    /// Data version for format migrations.
85    /// NBT tag: `DataVersion` (Int)
86    pub data_version: i32,
87}

117impl PersistentPlayerData {
118    /// Extracts persistent data from a live player.
119    #[must_use]
120    pub fn from_player(player: &Player) -> Self {
121        let pos = *player.position.lock();
122        let (yaw, pitch) = player.rotation.load();
123        let delta = *player.delta_movement.lock();
124        let abilities = player.abilities.lock();
125        let inventory = player.inventory.lock();
126        let entity_data = player.entity_data.lock();
127
128        // Collect non-empty inventory slots
129        let mut slots = Vec::new();
130        // Main inventory (0-35) and equipment (36-42)
131        for slot in 0..43 {
132            let item = inventory.get_item(slot);
133            if !item.is_empty() {
134                slots.push(PersistentSlot {
135                    slot: slot as i8,
136                    item: item.clone(),
137                });
138            }
139        }
140
141        Self {
142            pos: [pos.x, pos.y, pos.z],
143            motion: [delta.x, delta.y, delta.z],
144            rotation: [yaw, pitch],
145            on_ground: player.on_ground.load(Ordering::Relaxed),
146            fall_flying: player.fall_flying.load(Ordering::Relaxed),
147            health: *entity_data.health.get(),
148            game_mode: player.game_mode.load() as i32,
149            abilities: PersistentAbilities {
150                invulnerable: abilities.invulnerable,
151                flying: abilities.flying,
152                may_fly: abilities.may_fly,
153                instabuild: abilities.instabuild,
154                may_build: abilities.may_build,
155                flying_speed: abilities.flying_speed,
156                walking_speed: abilities.walking_speed,
157            },
158            inventory: slots,
159            selected_slot: i32::from(inventory.get_selected_slot()),
160            dimension: player.world.dimension.key.to_string(),
161            data_version: PLAYER_DATA_VERSION,
162        }
163    }

165    /// Serializes the player data to an NBT compound.
166    #[must_use]
167    pub fn to_nbt(&self) -> NbtCompound {
168        let mut compound = NbtCompound::new();
169
170        // Position
171        let pos_list = NbtList::from(vec![
172            NbtTag::Double(self.pos[0]),
173            NbtTag::Double(self.pos[1]),
174            NbtTag::Double(self.pos[2]),
175        ]);
176        compound.insert("Pos", pos_list);
177
178        // Motion
179        let motion_list = NbtList::from(vec![
180            NbtTag::Double(self.motion[0]),
181            NbtTag::Double(self.motion[1]),
182            NbtTag::Double(self.motion[2]),
183        ]);
184        compound.insert("Motion", motion_list);
185
186        // Rotation
187        let rotation_list = NbtList::from(vec![
188            NbtTag::Float(self.rotation[0]),
189            NbtTag::Float(self.rotation[1]),
190        ]);
191        compound.insert("Rotation", rotation_list);
192
193        // Simple fields
194        compound.insert("OnGround", i8::from(self.on_ground));
195        compound.insert("FallFlying", i8::from(self.fall_flying));
196        compound.insert("Health", self.health);
197        compound.insert("playerGameType", self.game_mode);
198        compound.insert("SelectedItemSlot", self.selected_slot);
199        compound.insert("Dimension", self.dimension.clone());
200        compound.insert("DataVersion", self.data_version);
201
202        // Abilities compound
203        compound.insert("abilities", self.abilities.to_nbt());
204
205        // Inventory list
206        let inventory_list: Vec<NbtTag> = self
207            .inventory
208            .iter()
209            .map(|slot| {
210                let mut item_compound = match slot.item.clone().to_nbt_tag() {
211                    NbtTag::Compound(c) => c,
212                    _ => NbtCompound::new(),
213                };
214                item_compound.insert("Slot", slot.slot);
215                NbtTag::Compound(item_compound)
216            })
217            .collect();
218        compound.insert("Inventory", NbtList::from(inventory_list));
219
220        compound
221    }

223    /// Deserializes player data from an NBT compound.
224    ///
225    /// Returns `None` if required fields are missing or invalid.
226    #[must_use]
227    pub fn from_nbt(nbt: &BorrowedNbtCompound<'_>) -> Option<Self> {
228        // Convert to view type to access accessor methods
229        let nbt: NbtCompoundView<'_, '_> = nbt.into();
230        // Position
231        let pos_list = nbt.list("Pos")?;
232        let pos = [
233            pos_list.doubles()?.first().copied()?,
234            pos_list.doubles()?.get(1).copied()?,
235            pos_list.doubles()?.get(2).copied()?,
236        ];
237
238        // Motion (optional, default to zero)
239        let motion =
240            nbt.list("Motion")
241                .and_then(|l| l.doubles())
242                .map_or([0.0, 0.0, 0.0], |doubles| {
243                    [
244                        doubles.first().copied().unwrap_or(0.0),
245                        doubles.get(1).copied().unwrap_or(0.0),
246                        doubles.get(2).copied().unwrap_or(0.0),
247                    ]
248                });
249
250        // Rotation (optional, default to zero)
251        let rotation = nbt
252            .list("Rotation")
253            .and_then(|l| l.floats())
254            .map_or([0.0, 0.0], |floats| {
255                [
256                    floats.first().copied().unwrap_or(0.0),
257                    floats.get(1).copied().unwrap_or(0.0),
258                ]
259            });
260
261        // Simple fields with defaults
262        let on_ground = nbt.byte("OnGround") != Some(0);
263        let fall_flying = nbt.byte("FallFlying").is_some_and(|b| b != 0);
264        let health = nbt.float("Health").unwrap_or(20.0);
265        let game_mode = nbt.int("playerGameType").unwrap_or(0);
266        let selected_slot = nbt.int("SelectedItemSlot").unwrap_or(0);
267        let dimension = nbt.string("Dimension").map_or_else(
268            || "minecraft:overworld".to_string(),
269            |s| s.to_str().to_string(),
270        );
271        let data_version = nbt.int("DataVersion").unwrap_or(0);
272
273        // Abilities
274        let abilities = nbt
275            .compound("abilities")
276            .map(|c| PersistentAbilities::from_nbt(&c))
277            .unwrap_or_default();
278
279        // Inventory
280        let mut inventory = Vec::new();
281        if let Some(inv_list) = nbt.list("Inventory")
282            && let Some(compounds) = inv_list.compounds()
283        {
284            for item_compound in compounds {
285                let slot = item_compound.byte("Slot").unwrap_or(0);
286                if let Some(item) = ItemStack::from_borrowed_compound(&item_compound) {
287                    inventory.push(PersistentSlot { slot, item });
288                }
289            }
290        }
291
292        Some(Self {
293            pos,
294            motion,
295            rotation,
296            on_ground,
297            fall_flying,
298            health,
299            game_mode,
300            abilities,
301            inventory,
302            selected_slot,
303            dimension,
304            data_version,
305        })
306    }
307}

309impl PersistentAbilities {
310    /// Serializes abilities to an NBT compound.
311    #[must_use]
312    pub fn to_nbt(&self) -> NbtCompound {
313        let mut compound = NbtCompound::new();
314        compound.insert("invulnerable", i8::from(self.invulnerable));
315        compound.insert("flying", i8::from(self.flying));
316        compound.insert("mayfly", i8::from(self.may_fly));
317        compound.insert("instabuild", i8::from(self.instabuild));
318        compound.insert("mayBuild", i8::from(self.may_build));
319        compound.insert("flySpeed", self.flying_speed);
320        compound.insert("walkSpeed", self.walking_speed);
321        compound
322    }
323
324    /// Deserializes abilities from an NBT compound.
325    #[must_use]
326    pub fn from_nbt(nbt: &NbtCompoundView<'_, '_>) -> Self {
327        Self {
328            invulnerable: nbt.byte("invulnerable").is_some_and(|b| b != 0),
329            flying: nbt.byte("flying").is_some_and(|b| b != 0),
330            may_fly: nbt.byte("mayfly").is_some_and(|b| b != 0),
331            instabuild: nbt.byte("instabuild").is_some_and(|b| b != 0),
332            may_build: nbt.byte("mayBuild") != Some(0),
333            flying_speed: nbt.float("flySpeed").unwrap_or(0.05),
334            walking_speed: nbt.float("walkSpeed").unwrap_or(0.1),
335        }
336    }

381impl PersistentPlayerData {
382    /// Applies the saved data to a player.
383    ///
384    /// This restores position, rotation, inventory, abilities, etc.
385    pub fn apply_to_player(&self, player: &Player) {
386        use steel_utils::math::Vector3;
387
388        // Position
389        *player.position.lock() = Vector3::new(self.pos[0], self.pos[1], self.pos[2]);
390
391        // Rotation
392        player.rotation.store((self.rotation[0], self.rotation[1]));
393
394        // Motion/velocity
395        *player.delta_movement.lock() =
396            Vector3::new(self.motion[0], self.motion[1], self.motion[2]);
397
398        // Ground state
399        player.on_ground.store(self.on_ground, Ordering::Relaxed);
400        player
401            .fall_flying
402            .store(self.fall_flying, Ordering::Relaxed);
403
404        // Health
405        player.entity_data.lock().health.set(self.health);
406
407        // Game mode
408        let game_mode = match self.game_mode {
409            1 => GameType::Creative,
410            2 => GameType::Adventure,
411            3 => GameType::Spectator,
412            _ => GameType::Survival,
413        };
414        player.game_mode.store(game_mode);
415
416        // Abilities
417        *player.abilities.lock() = self.abilities.clone().into();
418
419        // Inventory
420        {
421            let mut inventory = player.inventory.lock();
422            // Clear existing inventory first
423            for slot in 0..43 {
424                inventory.set_item(slot, ItemStack::empty());
425            }
426            // Restore saved items
427            for slot_data in &self.inventory {
428                let slot_index = slot_data.slot as usize;
429                if slot_index < 43 {
430                    inventory.set_item(slot_index, slot_data.item.clone());
431                }
432            }
433            // Restore selected slot
434            let selected = self.selected_slot.clamp(0, 8) as u8;
435            inventory.set_selected_slot(selected);
436        }
437    }
