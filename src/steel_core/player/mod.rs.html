409    /// Sends multiple packets as an atomic bundle.
410    ///
411    /// The closure receives a [`BundleBuilder`](networking::BundleBuilder) to add packets to.
412    /// All packets are encoded, then sent wrapped in bundle delimiters so the
413    /// client processes them together in a single game tick.
414    pub fn send_bundle<F>(&self, f: F)
415    where
416        F: FnOnce(&mut networking::BundleBuilder),
417    {
418        let mut builder = networking::BundleBuilder::new(self.connection.compression());
419        f(&mut builder);
420        let packets = builder.into_packets();
421        if !packets.is_empty() {
422            self.connection.send_encoded_bundle(packets);

431    /// Ticks the player.
432    #[allow(clippy::cast_possible_truncation)]
433    pub fn tick(&self) {
434        // Increment local tick counter
435        self.tick_count.fetch_add(1, Ordering::Relaxed);
436
437        // Reset first_good_position to current position at start of tick (vanilla: resetPosition)
438        *self.first_good_position.lock() = *self.position.lock();
439
440        // Apply gravity to delta_movement (vanilla: applyGravity in Entity.tick/LivingEntity.travel)
441        // This must happen after resetPosition so the speed check has the correct expected velocity
442        self.apply_gravity();
443
444        // Sync packet counts for rate limiting (vanilla: knownMovePacketCount = receivedMovePacketCount)
445        self.known_move_packet_count.store(
446            self.received_move_packet_count.load(Ordering::Relaxed),
447            Ordering::Relaxed,
448        );
449
450        // Send pending block change acks (batched, once per tick like vanilla)
451        self.tick_ack_block_changes();
452
453        if !self.client_loaded.load(Ordering::Relaxed) {
454            //return;
455        }
456
457        let current_pos = *self.position.lock();
458        let chunk_x = (current_pos.x as i32) >> 4;
459        let chunk_z = (current_pos.z as i32) >> 4;
460        let chunk_pos = ChunkPos::new(chunk_x, chunk_z);
461
462        *self.last_chunk_pos.lock() = chunk_pos;
463
464        self.world.chunk_map.update_player_status(self);
465
466        self.chunk_sender
467            .lock()
468            .send_next_chunks(self.connection.clone(), &self.world, chunk_pos);
469
470        // Try to pick up nearby items (vanilla: Player.aiStep)
471        self.touch_nearby_items();
472
473        // Broadcast inventory changes to client
474        self.broadcast_inventory_changes();
475
476        // Tick block breaking
477        self.block_breaking.lock().tick(self, &self.world);
478
479        // Update pose based on current state
480        self.update_pose();
481
482        // Sync dirty entity data to nearby players
483        self.sync_entity_data();
484
485        self.connection.tick();
486
487        // TODO: Implement player ticking logic here
488        // This will include:
489        // - Checking if the player is alive
490        // - Handling movement
491        // - Updating inventory
492        // - Handling food/health regeneration
493        // - Managing game mode specific logic
494        // - Updating advancements
495        // - Handling falling
496    }

507    /// Attempts to pick up nearby item entities.
508    ///
509    /// Mirrors vanilla's `Player.aiStep()` item pickup logic:
510    /// - Calculates pickup area as bounding box inflated by (1.0, 0.5, 1.0)
511    /// - Calls `playerTouch()` on each entity in range
512    fn touch_nearby_items(&self) {
513        // Spectators can't pick up items
514        if self.game_mode.load() == GameType::Spectator {
515            return;
516        }
517
518        // Calculate pickup area (vanilla: Player.aiStep lines 454-458)
519        let pickup_area = self.bounding_box().inflate_xyz(1.0, 0.5, 1.0);
520
521        // Get all entities in the pickup area
522        let entities = self.world.get_entities_in_aabb(&pickup_area);
523
524        // Get player Arc for try_pickup (needed because try_pickup takes &Arc<Player>)
525        let Some(player_arc) = self.world.players.get_by_entity_id(self.id) else {
526            return;
527        };
528
529        for entity in entities {
530            // Skip self
531            if entity.id() == self.id {
532                continue;
533            }
534
535            // Skip removed entities
536            if entity.is_removed() {
537                continue;
538            }
539
540            // Try to pick up item entities
541            if let Some(item_entity) = entity.as_item_entity() {
542                item_entity.try_pickup(&player_arc);
543            }
544
545            // TODO: Handle other entity types (experience orbs, arrows)
546        }
547    }

564    fn verify_chat_signature(
565        &self,
566        packet: &SChat,
567    ) -> Result<(message_chain::SignedMessageLink, LastSeen), String> {
568        const MESSAGE_EXPIRES_AFTER: Duration = Duration::from_mins(5);
569
570        let session = self.chat_session.lock().clone().ok_or("No chat session")?;
571        let signature = packet.signature.as_ref().ok_or("No signature present")?;
572
573        if session
574            .profile_public_key
575            .data()
576            .has_expired_with_grace(profile_key::EXPIRY_GRACE_PERIOD)
577        {
578            return Err("Profile key has expired".to_string());
579        }
580
581        let mut chain_guard = self.message_chain.lock();
582        let chain = chain_guard.as_mut().ok_or("No message chain")?;
583
584        if chain.is_broken() {
585            return Err("Message chain is broken".to_string());
586        }
587
588        let timestamp =
589            UNIX_EPOCH + Duration::from_millis(packet.timestamp.try_into().unwrap_or(0));
590
591        let now = SystemTime::now();
592        let message_age = now
593            .duration_since(timestamp)
594            .unwrap_or(Duration::from_secs(0));
595
596        if message_age > MESSAGE_EXPIRES_AFTER {
597            return Err(format!(
598                "Message expired (age: {}s, max: 300s)",
599                message_age.as_secs()
600            ));
601        }
602
603        let last_seen_signatures = self
604            .message_validator
605            .lock()
606            .apply_update(packet.acknowledged, packet.offset, packet.checksum)
607            .map_err(|e| {
608                log::error!("Message acknowledgment validation failed: {e}");
609                e
610            })?;
611
612        let last_seen = LastSeen::new(last_seen_signatures);
613
614        let body = message_chain::SignedMessageBody::new(
615            packet.message.clone(),
616            timestamp,
617            packet.salt,
618            last_seen,
619        );
620
621        let link = chain
622            .validate_and_advance(&body)
623            .map_err(|e| format!("Chain validation failed: {e}"))?;
624
625        let updater = message_chain::MessageSignatureUpdater::new(&link, &body);
626        let validator = session.profile_public_key.create_signature_validator();
627
628        let is_valid = SignatureValidator::validate(&validator, &updater, signature)
629            .map_err(|e| format!("Signature validation error: {e}"))?;
630
631        if is_valid {
632            Ok((link, body.last_seen.clone()))
633        } else {
634            Err("Invalid signature".to_string())
635        }
636    }

638    /// Handles a chat message from the player.
639    #[allow(clippy::too_many_lines)]
640    pub fn handle_chat(&self, packet: SChat, player: Arc<Player>) {
641        let chat_message = packet.message.clone();
642
643        let verification_result = if let Some(_signature) = &packet.signature {
644            match self.verify_chat_signature(&packet) {
645                Ok((link, last_seen)) => {
646                    // Don't add to cache here - will be added during broadcast
647                    // to avoid cache state mismatch with client
648                    Some(Ok((link, last_seen)))
649                }
650                Err(err) => {
651                    log::warn!(
652                        "Player {} sent message with invalid signature: {err}",
653                        self.gameprofile.name
654                    );
655                    Some(Err(err))
656                }
657            }
658        } else {
659            None
660        };
661
662        if STEEL_CONFIG.enforce_secure_chat {
663            match &verification_result {
664                Some(Ok(_)) => {}
665                Some(Err(err)) => {
666                    self.disconnect(format!("Chat message validation failed: {err}"));
667                    return;
668                }
669                None => {
670                    self.disconnect(
671                        "Secure chat is enforced on this server, but your message was not signed",
672                    );
673                    return;
674                }
675            }
676        }
677
678        let signature = if matches!(verification_result, Some(Ok(_))) {
679            packet.signature.map(|sig| Box::new(sig) as Box<[u8]>)
680        } else {
681            None
682        };
683
684        let sender_index = player.messages_sent.fetch_add(1, Ordering::SeqCst);
685
686        let registry_id = *REGISTRY.chat_types.get_id(vanilla_chat_types::CHAT) as i32;
687
688        let chat_packet = CPlayerChat::new(
689            0,
690            player.gameprofile.id,
691            sender_index,
692            signature.clone(),
693            chat_message.clone(),
694            packet.timestamp,
695            packet.salt,
696            Box::new([]),
697            Some(TextComponent::plain(chat_message.clone())),
698            FilterType::PassThrough,
699            ChatTypeBound {
700                registry_id,
701                sender_name: TextComponent::plain(player.gameprofile.name.clone())
702                    .insertion(player.gameprofile.name.clone())
703                    .click_event(ClickEvent::suggest_command(format!(
704                        "/tell {} ",
705                        player.gameprofile.name
706                    )))
707                    .hover_event(HoverEvent::show_entity(
708                        "minecraft:player",
709                        self.uuid(),
710                        Some(player.gameprofile.name.clone()),
711                    )),
712                target_name: None,
713            },
714        );
715
716        if let Some(sig_box) = &signature {
717            if sig_box.len() == 256 {
718                let mut sig_array = [0u8; 256];
719                sig_array.copy_from_slice(&sig_box[..]);
720
721                let last_seen = if let Some(Ok((_, ref last_seen))) = verification_result {
722                    last_seen.clone()
723                } else {
724                    LastSeen::default()
725                };
726
727                steel_utils::chat!(player.gameprofile.name.clone(), "{}", chat_message);
728                self.world.broadcast_chat(
729                    chat_packet,
730                    Arc::clone(&player),
731                    last_seen,
732                    Some(sig_array),
733                );
734            } else {
735                self.world.broadcast_unsigned_chat(
736                    chat_packet,
737                    &player.gameprofile.name,
738                    &chat_message,
739                );
740            }
741        } else {
742            self.world.broadcast_unsigned_chat(
743                chat_packet,
744                &player.gameprofile.name,
745                &chat_message,
746            );
747        }
748    }

767    /// Checks if we're awaiting a teleport confirmation and handles timeout/resend.
768    ///
769    /// Returns `true` if awaiting teleport (movement should be rejected),
770    /// `false` if normal movement processing should continue.
771    fn update_awaiting_teleport(&self) -> bool {
772        let awaiting = self.awaiting_position_from_client.lock();
773        if let Some(pos) = *awaiting {
774            let current_tick = self.tick_count.load(Ordering::Relaxed);
775            let last_time = self.awaiting_teleport_time.load(Ordering::Relaxed);
776
777            // Resend teleport after 20 ticks (~1 second) timeout
778            if current_tick.wrapping_sub(last_time) > 20 {
779                self.awaiting_teleport_time
780                    .store(current_tick, Ordering::Relaxed);
781                drop(awaiting);
782
783                // Resend the teleport packet
784                let (yaw, pitch) = self.rotation.load();
785                let teleport_id = self.awaiting_teleport_id.load(Ordering::Relaxed);
786                self.send_packet(CPlayerPosition::absolute(
787                    teleport_id,
788                    pos.x,
789                    pos.y,
790                    pos.z,
791                    yaw,
792                    pitch,
793                ));
794            }
795            return true; // Still awaiting, reject movement
796        }
797
798        self.awaiting_teleport_time
799            .store(self.tick_count.load(Ordering::Relaxed), Ordering::Relaxed);
800        false
801    }

844    /// Handles a move player packet.
845    ///
846    /// Matches vanilla `ServerGamePacketListenerImpl.handleMovePlayer()`.
847    #[allow(clippy::cast_lossless, clippy::too_many_lines, clippy::similar_names)]
848    pub fn handle_move_player(&self, packet: SMovePlayer) {
849        if Self::is_invalid_position(
850            packet.get_x(0.0),
851            packet.get_y(0.0),
852            packet.get_z(0.0),
853            packet.get_x_rot(0.0),
854            packet.get_y_rot(0.0),
855        ) {
856            self.disconnect(translations::MULTIPLAYER_DISCONNECT_INVALID_PLAYER_MOVEMENT.msg());
857            return;
858        }
859
860        // Check awaiting teleport - if so, only update rotation (vanilla: absSnapRotationTo)
861        if self.update_awaiting_teleport() {
862            // While awaiting teleport, still allow rotation updates
863            if packet.has_rot {
864                self.rotation.store((packet.y_rot, packet.x_rot));
865            }
866            return;
867        }
868
869        if !self.client_loaded.load(Ordering::Relaxed) {
870            return;
871        }
872
873        let prev_pos = *self.prev_position.lock();
874        let prev_rot = self.prev_rotation.load();
875        let start_pos = *self.position.lock();
876        let game_mode = self.game_mode.load();
877        let is_spectator = game_mode == GameType::Spectator;
878        let is_creative = game_mode == GameType::Creative;
879        let is_sleeping = self.sleeping.load(Ordering::Relaxed);
880        let is_fall_flying = self.fall_flying.load(Ordering::Relaxed);
881        let was_on_ground = self.on_ground.load(Ordering::Relaxed);
882        // Skip movement checks when tick rate is frozen (vanilla: tickRateManager().runsNormally())
883        let tick_frozen = !self.world.tick_runs_normally();
884
885        // Handle position updates
886        if packet.has_pos {
887            // Clamp position to vanilla limits
888            let target_pos = Vector3::new(
889                movement::clamp_horizontal(packet.position.x),
890                movement::clamp_vertical(packet.position.y),
891                movement::clamp_horizontal(packet.position.z),
892            );
893            let first_good = *self.first_good_position.lock();
894            let last_good = *self.last_good_position.lock();
895
896            // Sleeping check - only allow small movements when sleeping
897            if is_sleeping {
898                let dx = target_pos.x - first_good.x;
899                let dy = target_pos.y - first_good.y;
900                let dz = target_pos.z - first_good.z;
901                let moved_dist_sq = dx * dx + dy * dy + dz * dz;
902
903                if moved_dist_sq > 1.0 {
904                    let (yaw, pitch) = self.rotation.load();
905                    self.teleport(start_pos.x, start_pos.y, start_pos.z, yaw, pitch);
906                    return;
907                }
908            } else {
909                // Increment received packet count
910                self.received_move_packet_count
911                    .fetch_add(1, Ordering::Relaxed);
912
913                // Calculate delta packets since last tick (for rate limiting)
914                let received = self.received_move_packet_count.load(Ordering::Relaxed);
915                let known = self.known_move_packet_count.load(Ordering::Relaxed);
916                let mut delta_packets = received - known;
917
918                // Cap delta packets to prevent abuse (vanilla caps at 5)
919                if delta_packets > 5 {
920                    delta_packets = 1;
921                }
922
923                // Skip checks for spectators, creative mode, tick frozen, or gamerules disabled
924                // Vanilla: shouldValidateMovement() checks playerMovementCheck and elytraMovementCheck
925                let gamerule_skip = !self.should_validate_movement(is_fall_flying);
926                let skip_checks = is_spectator || is_creative || tick_frozen || gamerule_skip;
927
928                // Validate movement using physics simulation
929                let mut validation = movement::validate_movement(
930                    &self.world,
931                    &movement::MovementInput {
932                        target_pos,
933                        first_good_pos: first_good,
934                        last_good_pos: last_good,
935                        expected_velocity_sq: self.get_delta_movement_length_sq(),
936                        delta_packets,
937                        is_fall_flying,
938                        skip_checks,
939                        in_impulse_grace: self.is_in_post_impulse_grace_time(),
940                        is_crouching: self.shift_key_down.load(Ordering::Relaxed),
941                        on_ground: was_on_ground,
942                    },
943                );
944
945                if !validation.is_valid {
946                    // Teleport back to start position
947                    let (yaw, pitch) = prev_rot;
948                    self.teleport(start_pos.x, start_pos.y, start_pos.z, yaw, pitch);
949                    return;
950                }
951
952                // Movement accepted - update last good position
953                *self.last_good_position.lock() = target_pos;
954
955                // Zero Y velocity when landing (vanilla: Block.updateEntityMovementAfterFallOn)
956                // This prevents gravity from accumulating while on the ground
957                if !was_on_ground && packet.on_ground {
958                    validation.move_delta.y = 0.0;
959                }
960                // Update velocity based on actual movement (vanilla: handlePlayerKnownMovement)
961                self.set_delta_movement(validation.move_delta);
962
963                // Jump detection (vanilla: jumpFromGround)
964                let moved_upwards = validation.move_delta.y > 0.0;
965                if was_on_ground && !packet.on_ground && moved_upwards {
966                    // Player jumped - could trigger jump-related mechanics here
967                    // For now, this is a placeholder for future jump handling
968                }
969            }
970        }
971
972        // Update on_ground state from packet
973        self.on_ground.store(packet.on_ground, Ordering::Relaxed);
974
975        // Update current state
976        if packet.has_pos {
977            let old_pos = *self.position.lock();
978            *self.position.lock() = packet.position;
979
980            // Notify callback of position change (updates entity cache section index)
981            self.level_callback.lock().on_move(old_pos, packet.position);
982        }
983        if packet.has_rot {
984            self.rotation.store((packet.y_rot, packet.x_rot));
985        }
986
987        // Broadcast movement to other players
988        let pos = if packet.has_pos {
989            packet.position
990        } else {
991            prev_pos
992        };
993        let (yaw, pitch) = if packet.has_rot {
994            (packet.y_rot, packet.x_rot)
995        } else {
996            prev_rot
997        };
998
999        if packet.has_pos || packet.has_rot {
1000            let new_chunk = ChunkPos::new((pos.x as i32) >> 4, (pos.z as i32) >> 4);
1001
1002            // Note: player_area_map is updated in chunk_map.update_player_status
1003            // which is called every tick and computes view diffs efficiently
1004
1005            if packet.has_pos {
1006                let dx = calc_delta(pos.x, prev_pos.x);
1007                let dy = calc_delta(pos.y, prev_pos.y);
1008                let dz = calc_delta(pos.z, prev_pos.z);
1009
1010                // Vanilla sync conditions (ServerEntity.java:148)
1011                let sync_delay = self.position_sync_delay.fetch_add(1, Ordering::Relaxed);
1012                let last_on_ground = self.last_sent_on_ground.load(Ordering::Relaxed);
1013                let on_ground_changed = last_on_ground != packet.on_ground;
1014                let force_sync = sync_delay > 400 || on_ground_changed;
1015
1016                if let (Some(dx), Some(dy), Some(dz)) = (dx, dy, dz) {
1017                    if force_sync {
1018                        // Send absolute position sync (forced by timer or on_ground change)
1019                        self.position_sync_delay.store(0, Ordering::Relaxed);
1020                        self.last_sent_on_ground
1021                            .store(packet.on_ground, Ordering::Relaxed);
1022
1023                        let delta = self.get_delta_movement();
1024                        let sync_packet = CEntityPositionSync {
1025                            entity_id: self.id,
1026                            x: pos.x,
1027                            y: pos.y,
1028                            z: pos.z,
1029                            velocity_x: delta.x,
1030                            velocity_y: delta.y,
1031                            velocity_z: delta.z,
1032                            yaw,
1033                            pitch,
1034                            on_ground: packet.on_ground,
1035                        };
1036                        self.world
1037                            .broadcast_to_nearby(new_chunk, sync_packet, Some(self.id));
1038                    } else {
1039                        let move_packet = CMoveEntityPosRot {
1040                            entity_id: self.id,
1041                            dx,
1042                            dy,
1043                            dz,
1044                            y_rot: to_angle_byte(yaw),
1045                            x_rot: to_angle_byte(pitch),
1046                            on_ground: packet.on_ground,
1047                        };
1048                        self.world
1049                            .broadcast_to_nearby(new_chunk, move_packet, Some(self.id));
1050                    }
1051                } else {
1052                    // Send absolute position sync (delta too big)
1053                    self.position_sync_delay.store(0, Ordering::Relaxed);
1054                    self.last_sent_on_ground
1055                        .store(packet.on_ground, Ordering::Relaxed);
1056
1057                    let delta = self.get_delta_movement();
1058                    let sync_packet = CEntityPositionSync {
1059                        entity_id: self.id,
1060                        x: pos.x,
1061                        y: pos.y,
1062                        z: pos.z,
1063                        velocity_x: delta.x,
1064                        velocity_y: delta.y,
1065                        velocity_z: delta.z,
1066                        yaw,
1067                        pitch,
1068                        on_ground: packet.on_ground,
1069                    };
1070                    self.world
1071                        .broadcast_to_nearby(new_chunk, sync_packet, Some(self.id));
1072                }
1073            } else {
1074                let rot_packet = CMoveEntityRot {
1075                    entity_id: self.id,
1076                    y_rot: to_angle_byte(yaw),
1077                    x_rot: to_angle_byte(pitch),
1078                    on_ground: packet.on_ground,
1079                };
1080                self.world
1081                    .broadcast_to_nearby(new_chunk, rot_packet, Some(self.id));
1082            }
1083
1084            if packet.has_rot {
1085                let head_packet = CRotateHead {
1086                    entity_id: self.id,
1087                    head_y_rot: to_angle_byte(yaw),
1088                };
1089                self.world
1090                    .broadcast_to_nearby(new_chunk, head_packet, Some(self.id));
1091            }
1092
1093            *self.prev_position.lock() = pos;
1094            self.prev_rotation.store((yaw, pitch));
1095        }
1096    }

1316    /// Processes a container click on any menu implementing the Menu trait.
1317    ///
1318    /// This is the common implementation shared between inventory menu and
1319    /// external menus (crafting table, chest, etc.).
1320    fn process_container_click(&self, menu: &mut dyn Menu, packet: SContainerClick) {
1321        // Handle spectator mode - just resync
1322        if self.game_mode.load() == GameType::Spectator {
1323            menu.behavior_mut()
1324                .send_all_data_to_remote(&self.connection);
1325            return;
1326        }
1327
1328        // Validate slot index
1329        if !menu.behavior().is_valid_slot_index(packet.slot_num) {
1330            log::debug!(
1331                "Player {} clicked invalid slot index: {}, available: {}",
1332                self.gameprofile.name,
1333                packet.slot_num,
1334                menu.behavior().slot_count()
1335            );
1336            return;
1337        }
1338
1339        // Check if we need a full resync (state ID mismatch)
1340        let full_resync_needed = packet.state_id as u32 != menu.behavior().get_state_id();
1341
1342        // Suppress remote updates during click handling
1343        menu.behavior_mut().suppress_remote_updates();
1344
1345        // Handle the click using the Menu trait method
1346        let has_infinite_materials = self.game_mode.load() == GameType::Creative;
1347        menu.clicked(
1348            packet.slot_num,
1349            packet.button_num,
1350            packet.click_type,
1351            has_infinite_materials,
1352            self,
1353        );
1354
1355        // Update remote slots from the client's perception
1356        for (slot, hash) in packet.changed_slots {
1357            menu.behavior_mut().set_remote_slot(slot as usize, hash);
1358        }
1359
1360        // Update remote carried from the client's perception
1361        menu.behavior_mut().set_remote_carried(packet.carried_item);
1362
1363        // Resume remote updates
1364        menu.behavior_mut().resume_remote_updates();
1365
1366        // Broadcast changes or full state depending on whether we had a state mismatch
1367        if full_resync_needed {
1368            menu.behavior_mut().broadcast_full_state(&self.connection);
1369        } else {
1370            menu.behavior_mut().broadcast_changes(&self.connection);
1371        }
1372    }

1479    /// Returns true if player is within block interaction range.
1480    /// Base range is ~4.5 blocks, plus 1.0 tolerance.
1481    #[must_use]
1482    pub fn is_within_block_interaction_range(&self, pos: &BlockPos) -> bool {
1483        let player_pos = *self.position.lock();
1484        let block_center_x = f64::from(pos.x()) + 0.5;
1485        let block_center_y = f64::from(pos.y()) + 0.5;
1486        let block_center_z = f64::from(pos.z()) + 0.5;
1487
1488        // Base range is ~4.5 blocks, plus 1.0 tolerance
1489        let max_range = 4.5 + 1.0;
1490        let dx = player_pos.x - block_center_x;
1491        let dy = player_pos.y - block_center_y;
1492        let dz = player_pos.z - block_center_z;
1493        (dx * dx + dy * dy + dz * dz).sqrt() <= max_range
1494    }

1768    /// Handles the use of an item on a block.
1769    ///
1770    /// Implements the logic from Java's `ServerGamePacketListenerImpl.handleUseItemOn()`.
1771    pub fn handle_use_item_on(&self, packet: SUseItemOn) {
1772        // 1. Check client loaded
1773        if !self.client_loaded.load(Ordering::Relaxed) {
1774            return;
1775        }
1776
1777        // 2. Ack block changes
1778        self.ack_block_changes_up_to(packet.sequence);
1779
1780        let pos = &packet.block_hit.block_pos;
1781        let direction = packet.block_hit.direction;
1782
1783        // 3. Validate interaction range
1784        if !self.is_within_block_interaction_range(pos) {
1785            self.send_block_updates(pos, direction);
1786            return;
1787        }
1788
1789        // 4. Validate hit location precision (must be within 1.0000001 of block center)
1790        let center_x = f64::from(pos.x()) + 0.5;
1791        let center_y = f64::from(pos.y()) + 0.5;
1792        let center_z = f64::from(pos.z()) + 0.5;
1793        let location = &packet.block_hit.location;
1794        let limit = 1.000_000_1;
1795
1796        if (location.x - center_x).abs() >= limit
1797            || (location.y - center_y).abs() >= limit
1798            || (location.z - center_z).abs() >= limit
1799        {
1800            log::warn!(
1801                "Rejecting UseItemOnPacket from {}: location {:?} too far from block {:?}",
1802                self.gameprofile.name,
1803                location,
1804                pos
1805            );
1806            self.send_block_updates(pos, direction);
1807            return;
1808        }
1809
1810        // 5. Validate Y height
1811        if pos.y() >= self.world.max_build_height() {
1812            // TODO: Send "build.tooHigh" message to player
1813            self.send_block_updates(pos, direction);
1814            return;
1815        }
1816
1817        // 6. Check awaiting teleport
1818        if self.is_awaiting_teleport() {
1819            self.send_block_updates(pos, direction);
1820            return;
1821        }
1822
1823        // 7. Check may_interact permission
1824        if !self.world.may_interact(self, pos) {
1825            self.send_block_updates(pos, direction);
1826            return;
1827        }
1828
1829        // 8. Call use_item_on
1830        let result = game_mode::use_item_on(self, &self.world, packet.hand, &packet.block_hit);
1831
1832        // 9. Handle result
1833        if let InteractionResult::Success = result {
1834            // TODO: Trigger arm swing animation if needed
1835            self.swing(packet.hand, true);
1836        }
1837
1838        // 10. Always send block updates to resync client
1839        self.send_block_updates(pos, direction);
1840
1841        // 11. Broadcast inventory changes
1842        self.broadcast_inventory_changes();
1843    }
