301    /// Computes the start (eye position) and end positions for a raytrace.
302    pub fn get_ray_endpoints(&self) -> (DVec3, DVec3) {
303        let pos = self.position();
304        let start_pos = DVec3::new(pos.x, self.get_eye_y(), pos.z);
305        let (yaw, pitch) = self.rotation();
306        let (yaw_rad, pitch_rad) = (f64::from(yaw.to_radians()), f64::from(pitch.to_radians()));
307        // Vanilla: Attributes.BLOCK_INTERACTION_RANGE defaults to 4.5,
308        // creative mode adds +0.5 via CREATIVE_BLOCK_INTERACTION_RANGE_MODIFIER.
309        let block_interaction_range = if self.has_infinite_materials() {
310            5.0
311        } else {
312            4.5
313        };
314        let direction = DVec3::new(
315            -yaw_rad.sin() * pitch_rad.cos() * block_interaction_range,
316            -pitch_rad.sin() * block_interaction_range,
317            pitch_rad.cos() * yaw_rad.cos() * block_interaction_range,
318        );
319
320        let end_pos = start_pos + direction;
321        (start_pos, end_pos)
322    }

399    /// Sends multiple packets as an atomic bundle.
400    ///
401    /// The closure receives a [`BundleBuilder`](networking::BundleBuilder) to add packets to.
402    /// All packets are encoded, then sent wrapped in bundle delimiters so the
403    /// client processes them together in a single game tick.
404    pub fn send_bundle<F>(&self, f: F)
405    where
406        F: FnOnce(&mut networking::BundleBuilder),
407    {
408        let mut builder = networking::BundleBuilder::new(self.connection.compression());
409        f(&mut builder);
410        let packets = builder.into_packets();
411        if !packets.is_empty() {
412            self.connection.send_encoded_bundle(packets);
413        }
414    }

421    /// Ticks the player.
422    #[expect(
423        clippy::cast_possible_truncation,
424        reason = "world coordinates are always within i32 range in a valid Minecraft world"
425    )]
426    pub fn tick(&self) {
427        // Increment local tick counter
428        self.tick_count.fetch_add(1, Ordering::Relaxed);
429
430        // Reset first_good_position to current position at start of tick (vanilla: resetPosition)
431        {
432            let mut mv = self.movement.lock();
433            mv.first_good_position = *self.position.lock();
434            // Sync packet counts for rate limiting (vanilla: knownMovePacketCount = receivedMovePacketCount)
435            mv.known_move_packet_count = mv.received_move_packet_count;
436        }
437
438        // Apply gravity to delta_movement (vanilla: applyGravity in Entity.tick/LivingEntity.travel)
439        // This must happen after resetPosition so the speed check has the correct expected velocity
440        self.apply_gravity();
441
442        // Send pending block change acks (batched, once per tick like vanilla)
443        self.tick_ack_block_changes();
444
445        if !self.client_loaded.load(Ordering::Relaxed) {
446            //return;
447        }
448
449        let current_pos = *self.position.lock();
450        let chunk_x = (current_pos.x as i32) >> 4;
451        let chunk_z = (current_pos.z as i32) >> 4;
452        let chunk_pos = ChunkPos::new(chunk_x, chunk_z);
453
454        *self.last_chunk_pos.lock() = chunk_pos;
455
456        self.world.chunk_map.update_player_status(self);
457
458        self.chunk_sender
459            .lock()
460            .send_next_chunks(self.connection.clone(), &self.world, chunk_pos);
461
462        {
463            let mut living_base = self.living_base.lock();
464            if living_base.invulnerable_time > 0 {
465                living_base.invulnerable_time -= 1;
466            }
467        }
468
469        if *self.entity_data.lock().health.get() <= 0.0 {
470            self.tick_death();
471        } else {
472            self.touch_nearby_items();
473            self.block_breaking.lock().tick(self, &self.world);
474            self.check_inside_blocks();
475            self.check_below_world();
476
477            // TODO: Implement remaining player ticking logic here
478            // - Handling food/health regeneration
479            // - Managing game mode specific logic
480            // - Updating advancements
481            // - Handling falling
482        }
483
484        // --- Post-tick (always runs, vanilla does not gate these behind isAlive) ---
485        self.broadcast_inventory_changes();
486        self.update_pose();
487        self.sync_entity_data();
488
489        // Only send CSetHealth when a value actually changed, matching vanilla's
490        // `lastSentHealth` / `lastSentFood` / `lastFoodSaturationZero` pattern.
491        {
492            let health = *self.entity_data.lock().health.get();
493            let food: i32 = 20; // TODO: use actual food level once hunger is implemented
494            let saturation: f32 = 5.0; // TODO: use actual saturation once hunger is implemented
495            let saturation_zero = saturation == 0.0;
496
497            let mut sync = self.health_sync.lock();
498            if sync.needs_update(health, food, saturation_zero) {
499                self.send_packet(CSetHealth {
500                    health,
501                    food,
502                    food_saturation: saturation,
503                });
504                sync.record_sent(health, food, saturation_zero);
505            }
506        }
507
508        {
509            let mut experience = self.experience.lock();
510
511            if experience.dirty {
512                self.send_packet(CSetExperience {
513                    progress: experience.progress() as f32,
514                    level: experience.level(),
515                    total_experience: experience.total_points(),
516                });
517                experience.dirty = false;
518            }
519        }
520
521        self.connection.tick();
522    }

558    /// Attempts to pick up nearby item entities.
559    ///
560    /// Mirrors vanilla's `Player.aiStep()` item pickup logic:
561    /// - Calculates pickup area as bounding box inflated by (1.0, 0.5, 1.0)
562    /// - Calls `playerTouch()` on each entity in range
563    fn touch_nearby_items(&self) {
564        // Spectators can't pick up items
565        if self.game_mode.load() == GameType::Spectator {
566            return;
567        }
568
569        // Calculate pickup area (vanilla: Player.aiStep lines 454-458)
570        let pickup_area = self.bounding_box().inflate_xyz(1.0, 0.5, 1.0);
571
572        // Get all entities in the pickup area
573        let entities = self.world.get_entities_in_aabb(&pickup_area);
574
575        // Get player Arc for try_pickup (needed because try_pickup takes &Arc<Player>)
576        let Some(player_arc) = self.world.players.get_by_entity_id(self.id) else {
577            return;
578        };
579
580        for entity in entities {
581            // Skip self
582            if entity.id() == self.id {
583                continue;
584            }
585
586            // Skip removed entities
587            if entity.is_removed() {
588                continue;
589            }
590
591            // Try to pick up item entities
592            if let Some(item_entity) = entity.as_item_entity() {
593                item_entity.try_pickup(&player_arc);
594            }
595
596            // TODO: Handle other entity types (experience orbs, arrows)
597        }
598    }

620    fn verify_chat_signature(
621        &self,
622        packet: &SChat,
623    ) -> Result<(message_chain::SignedMessageLink, LastSeen), String> {
624        const MESSAGE_EXPIRES_AFTER: Duration = Duration::from_mins(5);
625
626        let mut chat = self.chat.lock();
627        let session = chat.chat_session.clone().ok_or("No chat session")?;
628        let signature = packet.signature.as_ref().ok_or("No signature present")?;
629
630        if session
631            .profile_public_key
632            .data()
633            .has_expired_with_grace(profile_key::EXPIRY_GRACE_PERIOD)
634        {
635            return Err("Profile key has expired".to_string());
636        }
637
638        let chain = chat.message_chain.as_mut().ok_or("No message chain")?;
639
640        if chain.is_broken() {
641            return Err("Message chain is broken".to_string());
642        }
643
644        let timestamp =
645            UNIX_EPOCH + Duration::from_millis(packet.timestamp.try_into().unwrap_or(0));
646
647        let now = SystemTime::now();
648        let message_age = now
649            .duration_since(timestamp)
650            .unwrap_or(Duration::from_secs(0));
651
652        if message_age > MESSAGE_EXPIRES_AFTER {
653            return Err(format!(
654                "Message expired (age: {}s, max: 300s)",
655                message_age.as_secs()
656            ));
657        }
658
659        let last_seen_signatures = chat
660            .message_validator
661            .apply_update(packet.acknowledged, packet.offset, packet.checksum)
662            .map_err(|e| {
663                log::error!("Message acknowledgment validation failed: {e}");
664                e
665            })?;
666
667        let last_seen = LastSeen::new(last_seen_signatures);
668
669        let body = message_chain::SignedMessageBody::new(
670            packet.message.clone(),
671            timestamp,
672            packet.salt,
673            last_seen,
674        );
675
676        let chain = chat.message_chain.as_mut().ok_or("No message chain")?;
677        let link = chain
678            .validate_and_advance(&body)
679            .map_err(|e| format!("Chain validation failed: {e}"))?;
680
681        let updater = message_chain::MessageSignatureUpdater::new(&link, &body);
682        let validator = session.profile_public_key.create_signature_validator();
683
684        let is_valid = SignatureValidator::validate(&validator, &updater, signature)
685            .map_err(|e| format!("Signature validation error: {e}"))?;
686
687        if is_valid {
688            Ok((link, body.last_seen.clone()))
689        } else {
690            Err("Invalid signature".to_string())
691        }
692    }

805                        chat_packet.clone(),

807                        &chat_message,
808                    );
809                }
810            }
811        } else if let Some(server) = self.server.upgrade() {
812            for world in server.worlds.values() {
813                world.broadcast_unsigned_chat(
814                    chat_packet.clone(),
815                    &player.gameprofile.name,

910            packet.get_x_rot(0.0),
911            packet.get_y_rot(0.0),

914            return;
915        }
916
917        // Check awaiting teleport - if so, only update rotation (vanilla: absSnapRotationTo)
918        if self.update_awaiting_teleport() {
919            // While awaiting teleport, still allow rotation updates
920            if packet.has_rot {
921                self.rotation.store((packet.y_rot, packet.x_rot));
922            }
923            return;
924        }
925
926        if !self.client_loaded.load(Ordering::Relaxed) {
927            return;
928        }
929
930        let (prev_pos, prev_rot) = {
931            let mv = self.movement.lock();
932            (mv.prev_position, mv.prev_rotation)
933        };

936        let is_spectator = game_mode == GameType::Spectator;

979                if delta_packets > 5 {
980                    delta_packets = 1;
981                }
982
983                // Skip checks for spectators, creative mode, tick frozen, or gamerules disabled

1004                        first_good_pos: first_good,

1007                        delta_packets,

1060            packet.position
1061        } else {

1063        };

1437    ///
1438    /// This is the common implementation shared between inventory menu and
1439    /// external menus (crafting table, chest, etc.).
1440    fn process_container_click(&self, menu: &mut dyn Menu, packet: SContainerClick) {
1441        // Handle spectator mode - just resync
1442        if self.game_mode.load() == GameType::Spectator {
1443            menu.behavior_mut()
1444                .send_all_data_to_remote(&self.connection);
1445            return;
1446        }
1447
1448        // Validate slot index
1449        if !menu.behavior().is_valid_slot_index(packet.slot_num) {
1450            log::debug!(
1451                "Player {} clicked invalid slot index: {}, available: {}",
1452                self.gameprofile.name,
1453                packet.slot_num,
1454                menu.behavior().slot_count()
1455            );
1456            return;
1457        }
1458
1459        // Check if we need a full resync (state ID mismatch)
1460        let full_resync_needed = packet.state_id as u32 != menu.behavior().get_state_id();
1461
1462        // Suppress remote updates during click handling
1463        menu.behavior_mut().suppress_remote_updates();
1464
1465        // Handle the click using the Menu trait method
1466        let has_infinite_materials = self.game_mode.load() == GameType::Creative;
1467        menu.clicked(
1468            packet.slot_num,
1469            packet.button_num,
1470            packet.click_type,
1471            has_infinite_materials,
1472            self,
1473        );
1474
1475        // Update remote slots from the client's perception
1476        for (slot, hash) in packet.changed_slots {
1477            menu.behavior_mut().set_remote_slot(slot as usize, hash);
1478        }
1479
1480        // Update remote carried from the client's perception
1481        menu.behavior_mut().set_remote_carried(packet.carried_item);
1482
1483        // Resume remote updates
1484        menu.behavior_mut().resume_remote_updates();
1485
1486        // Broadcast changes or full state depending on whether we had a state mismatch
1487        if full_resync_needed {
1488            menu.behavior_mut().broadcast_full_state(&self.connection);
1489        } else {
1490            menu.behavior_mut().broadcast_changes(&self.connection);
1491        }
1492    }

1599    /// Returns true if player is within block interaction range.
1600    ///
1601    /// Uses eye position and AABB distance (nearest point on block surface),
1602    /// matching vanilla's `Player.isWithinBlockInteractionRange(pos, 1.0)`.
1603    #[must_use]
1604    pub fn is_within_block_interaction_range(&self, pos: BlockPos) -> bool {
1605        let player_pos = *self.position.lock();
1606        let eye_y = player_pos.y + self.get_eye_height();
1607
1608        // Block AABB is [x, y, z] to [x+1, y+1, z+1]
1609        let min_x = f64::from(pos.x());
1610        let min_y = f64::from(pos.y());
1611        let min_z = f64::from(pos.z());
1612        let max_x = min_x + 1.0;
1613        let max_y = min_y + 1.0;
1614        let max_z = min_z + 1.0;
1615
1616        // Distance from eye to nearest point on block AABB (0 if inside on that axis)
1617        let dx = f64::max(f64::max(min_x - player_pos.x, player_pos.x - max_x), 0.0);
1618        let dy = f64::max(f64::max(min_y - eye_y, eye_y - max_y), 0.0);
1619        let dz = f64::max(f64::max(min_z - player_pos.z, player_pos.z - max_z), 0.0);
1620        let dist_sq = dx * dx + dy * dy + dz * dz;
1621
1622        // Base range is 4.5 blocks + 1.0 buffer
1623        let max_range = 4.5 + 1.0;
1624        dist_sq < max_range * max_range
1625    }
