20/// Minimum chunks per tick (vanilla: 0.01)
21const MIN_CHUNKS_PER_TICK: f32 = 8f32;
22/// Maximum chunks per tick (vanilla: 64.0, we use 500.0 for faster loading)
23const MAX_CHUNKS_PER_TICK: f32 = 500.0;
24/// Starting chunks per tick (vanilla: 9.0)
25const START_CHUNKS_PER_TICK: f32 = 9.0;
26/// Maximum unacknowledged batches after first ack (vanilla: 10)
27const MAX_UNACKNOWLEDGED_BATCHES: u16 = 10;
28
29/// This struct is responsible for sending chunks to the client.
30#[derive(Debug)]
31pub struct ChunkSender {
32    /// A list of chunks that are waiting to be sent to the client.
33    pub pending_chunks: FxHashSet<ChunkPos>,
34    /// The number of batches that have been sent to the client but have not been acknowledged yet.
35    pub unacknowledged_batches: u16,
36    /// The number of chunks that should be sent to the client per tick.
37    /// This is dynamically adjusted based on client feedback.
38    pub desired_chunks_per_tick: f32,
39    /// The number of chunks that can be sent to the client in the current batch.
40    pub batch_quota: f32,
41    /// The maximum number of unacknowledged batches allowed.
42    /// Starts at 1 and increases to `MAX_UNACKNOWLEDGED_BATCHES` after first ack.
43    pub max_unacknowledged_batches: u16,
44}
45
46impl ChunkSender {
47    /// Marks a chunk as pending to be sent to the client.
48    pub fn mark_chunk_pending_to_send(&mut self, pos: ChunkPos) {
49        self.pending_chunks.insert(pos);
50    }
51
52    /// Drops a chunk from the client's view.
53    pub fn drop_chunk(&mut self, connection: &JavaConnection, pos: ChunkPos) {
54        if !self.pending_chunks.remove(&pos) && !connection.closed() {
55            connection.send_packet(CForgetLevelChunk { pos });
56        }
57    }
58
59    /// Sends the next batch of chunks to the client.
60    ///
61    /// # Panics
62    /// Panics if a chunk is not at Full status when it should be.
63    pub fn send_next_chunks(
64        &mut self,
65        connection: Arc<JavaConnection>,
66        world: &World,
67        player_chunk_pos: ChunkPos,
68    ) {
69        if self.unacknowledged_batches < self.max_unacknowledged_batches {
70            let max_batch_size = self.desired_chunks_per_tick.max(1.0);
71            self.batch_quota =
72                (self.batch_quota + self.desired_chunks_per_tick).min(max_batch_size);
73
74            if self.batch_quota >= 1.0 && !self.pending_chunks.is_empty() {
75                let chunks_to_process = self.collect_candidates(world, player_chunk_pos);
76                if !chunks_to_process.is_empty() {
77                    self.unacknowledged_batches += 1;
78                    self.batch_quota -= chunks_to_process.len() as f32;
79
80                    #[allow(clippy::let_underscore_future)]
81                    let _ = spawn_blocking(move || {
82                        let mut chunks_to_send = Vec::new();
83                        for holder in chunks_to_process {
84                            if let Some(chunk_guard) = holder.try_chunk(ChunkStatus::Full) {
85                                if let ChunkAccess::Full(chunk) = &*chunk_guard {
86                                    chunks_to_send.push(CLevelChunkWithLight {
87                                        pos: holder.get_pos(),
88                                        chunk_data: chunk.extract_chunk_data(),
89                                        light_data: chunk.extract_light_data(),
90                                    });
91                                } else {
92                                    panic!("Chunk must be at Full status to be sent to the client");
93                                }
94                            }
95                        }
96
97                        connection.send_packet(CChunkBatchStart {});
98                        let batch_size = chunks_to_send.len();
99
100                        for chunk in chunks_to_send {
101                            connection.send_packet(chunk);
102                        }
103
104                        connection.send_packet(CChunkBatchFinished {
105                            batch_size: batch_size as i32,
106                        });
107                    });
108                }
109            }
110        }
111    }
112
113    fn collect_candidates(
114        &mut self,
115        world: &World,
116        player_chunk_pos: ChunkPos,
117    ) -> Vec<Arc<ChunkHolder>> {
118        let max_batch_size = self.batch_quota.floor() as usize;
119        let mut candidates: Vec<ChunkPos> = self.pending_chunks.iter().copied().collect();
120
121        // Sort by distance to player
122        candidates.sort_by_key(|pos| {
123            let dx = pos.0.x - player_chunk_pos.0.x;
124            let dz = pos.0.y - player_chunk_pos.0.y;
125            dx * dx + dz * dz
126        });
127
128        let mut chunks_to_send = Vec::new();
129
130        for pos in candidates {
131            if chunks_to_send.len() >= max_batch_size {
132                break;
133            }
134
135            if let Some(holder) = world
136                .chunk_map
137                .chunks
138                .read_sync(&pos, |_, chunk| chunk.clone())
139                && holder.persisted_status() == Some(ChunkStatus::Full)
140            {
141                chunks_to_send.push(holder);
142                self.pending_chunks.remove(&pos);
143            }
144        }
145        chunks_to_send
146    }
147
148    /// Handles the acknowledgement of a chunk batch from the client.
149    ///
150    /// The client sends back its desired chunks per tick based on how fast it can
151    /// process chunks. We clamp this value and use it to adjust our sending rate.
152    pub fn on_chunk_batch_received_by_client(&mut self, desired_chunks_per_tick: f32) {
153        self.unacknowledged_batches = self.unacknowledged_batches.saturating_sub(1);
154
155        // Handle NaN and clamp to valid range (vanilla uses 0.01-64, we use 0.01-500)
156        self.desired_chunks_per_tick = if desired_chunks_per_tick.is_nan() {
157            MIN_CHUNKS_PER_TICK
158        } else {
159            desired_chunks_per_tick.clamp(MIN_CHUNKS_PER_TICK, MAX_CHUNKS_PER_TICK)
160        };
161
162        // Reset batch quota when all batches are acknowledged
163        if self.unacknowledged_batches == 0 {
164            self.batch_quota = 1.0;
165        }
166
167        // After receiving the first acknowledgement, allow more unacknowledged batches
168        // for better pipelining (vanilla behavior)
169        self.max_unacknowledged_batches = MAX_UNACKNOWLEDGED_BATCHES;
170    }
171}
172
173impl Default for ChunkSender {
174    fn default() -> Self {
175        Self {
176            pending_chunks: FxHashSet::default(),
177            unacknowledged_batches: 0,
178            desired_chunks_per_tick: START_CHUNKS_PER_TICK,
179            batch_quota: 0.0,
180            max_unacknowledged_batches: 1,
181        }
182    }
183}
