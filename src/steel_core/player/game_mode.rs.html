18/// Handles using an item on a block.
19///
20/// This implements the logic from Java's `ServerPlayerGameMode.useItemOn()`.
21///
22/// # Flow
23/// 1. Spectator mode: Only allow opening menus (currently returns Pass)
24/// 2. Check if block interaction should be suppressed (sneaking + holding items)
25/// 3. If not suppressed: Call block's `use_item_on` method
26/// 4. If block returns `TryEmptyHandInteraction` and main hand: Call block's `use_without_item`
27/// 5. If item not empty: Call item behavior's `use_on` for placement
28/// 6. Handle creative mode infinite materials
29pub fn use_item_on(
30    player: &Player,
31    world: &Arc<World>,
32    hand: InteractionHand,
33    hit_result: &BlockHitResult,
34) -> InteractionResult {
35    let pos = hit_result.block_pos;
36    let state = world.get_block_state(pos);
37
38    // Spectator mode: can only open menus
39    // TODO: Implement menu providers for blocks like chests
40    if player.game_mode.load() == GameType::Spectator {
41        return InteractionResult::Pass;
42    }
43
44    // Check if block interaction should be suppressed (sneaking + holding items in either hand)
45    let have_something = {
46        let inv = player.inventory.lock();
47        !inv.get_item_in_hand(InteractionHand::MainHand).is_empty()
48            || !inv.get_item_in_hand(InteractionHand::OffHand).is_empty()
49    };
50
51    let suppress_block_use = player.is_secondary_use_active() && have_something;
52
53    // Get behavior registries
54    let block_behaviors = &*BLOCK_BEHAVIORS;
55    let item_behaviors = &*ITEM_BEHAVIORS;
56
57    // Try block interaction first (if not suppressed).
58    // No inventory lock held — block behaviors may need inventory access (e.g. opening chests).
59    if !suppress_block_use {
60        let Some(block) = REGISTRY.blocks.by_state_id(state) else {
61            return InteractionResult::Pass;
62        };
63        let behavior = block_behaviors.get_behavior(block);
64
65        // Brief lock for an immutable snapshot used during block interaction check
66        let item_snapshot = player.inventory.lock().get_item_in_hand(hand).clone();
67
68        let block_result =
69            behavior.use_item_on(&item_snapshot, state, world, pos, player, hand, hit_result);
70
71        if block_result.consumes_action() {
72            return block_result;
73        }
74
75        if matches!(block_result, InteractionResult::TryEmptyHandInteraction)
76            && hand == InteractionHand::MainHand
77        {
78            let empty_result = behavior.use_without_item(state, world, pos, player, hit_result);
79
80            if empty_result.consumes_action() {
81                return empty_result;
82            }
83        }
84    }
85
86    // Item use (block placement, etc.) — acquire inventory lock via ContainerLockGuard
87    let inv_ref = ContainerRef::PlayerInventory(player.inventory.clone());
88    let mut guard = ContainerLockGuard::lock_all(&[&inv_ref]);
89
90    let inv_id = inv_ref.container_id();
91
92    let is_empty = {
93        let Some(inv) = guard.get_player_inventory_mut(inv_id) else {
94            return InteractionResult::Pass;
95        };
96        inv.get_item_in_hand(hand).is_empty()
97    };
98
99    if !is_empty {
100        // TODO: Check item cooldowns
101        // if player.getCooldowns().isOnCooldown(item_stack.item) { return Pass }
102
103        let mut context =
104            UseOnContext::new(player, hand, hit_result.clone(), world, &mut guard, inv_id);
105
106        let original_count = context.inv.item().count;
107        let item_ref = context.inv.item().item;
108        let item_behavior = item_behaviors.get_behavior(item_ref);
109        let result = item_behavior.use_on(&mut context);
110
111        // Restore count for creative mode (infinite materials)
112        if player.has_infinite_materials() && context.inv.item().count < original_count {
113            context.inv.item().count = original_count;
114        }
115
116        return result;
117    }
118
119    InteractionResult::Pass
120}

122/// Handles using an item (general usage like right-clicking air).
123///
124/// This implements logic similar to `ServerPlayerGameMode.useItem()`.
125pub fn use_item(player: &Player, world: &Arc<World>, hand: InteractionHand) -> InteractionResult {
126    // Spectator mode: can only open menus
127    if player.game_mode.load() == GameType::Spectator {
128        return InteractionResult::Pass;
129    }
130
131    // TODO: Check item cooldowns
132    // if player.getCooldowns().isOnCooldown(item_stack) { return InteractionResult::Pass }
133
134    let inv_ref = ContainerRef::PlayerInventory(player.inventory.clone());
135    let mut guard = ContainerLockGuard::lock_all(&[&inv_ref]);
136    let inv_id = inv_ref.container_id();
137
138    let is_empty = {
139        let Some(inv) = guard.get_player_inventory_mut(inv_id) else {
140            return InteractionResult::Pass;
141        };
142        inv.get_item_in_hand(hand).is_empty()
143    };
144
145    if !is_empty {
146        let mut context =
147            crate::behavior::UseItemContext::new(player, hand, world, &mut guard, inv_id);
148
149        let original_count = context.inv.item().count;
150
151        // Get behavior registries
152        let item_behaviors = &*ITEM_BEHAVIORS;
153        let item_ref = context.inv.item().item;
154        let item_behavior = item_behaviors.get_behavior(item_ref);
155
156        let result = item_behavior.use_item(&mut context);
157
158        // Restore count for creative mode (infinite materials)
159        if player.has_infinite_materials() && context.inv.item().count < original_count {
160            context.inv.item().count = original_count;
161        }
162
163        return result;
164    }
165
166    InteractionResult::Pass
