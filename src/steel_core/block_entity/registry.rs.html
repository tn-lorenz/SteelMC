17/// Factory function type for creating block entities.
18///
19/// Takes the world, position and block state, returns a new block entity instance.
20pub type BlockEntityFactory = fn(Weak<World>, BlockPos, BlockStateId) -> SharedBlockEntity;
21
22/// Registry entry for a block entity type.
23struct BlockEntityEntry {
24    /// Factory function to create instances.
25    factory: Option<BlockEntityFactory>,
26}
27
28/// Registry for block entity factories.
29///
30/// Maps `BlockEntityType` to factory functions that can create block entity instances.
31/// This is used when loading block entities from disk or when blocks with entities
32/// are placed.
33pub struct BlockEntityRegistry {
34    entries: Vec<BlockEntityEntry>,
35}
36
37impl BlockEntityRegistry {
38    /// Creates a new empty registry with entries for all block entity types.
39    #[must_use]
40    pub fn new() -> Self {
41        let count = REGISTRY.block_entity_types.len();
42        let entries = (0..count)
43            .map(|_| BlockEntityEntry { factory: None })
44            .collect();
45
46        Self { entries }
47    }
48
49    /// Registers a factory function for a block entity type.
50    pub fn register(&mut self, block_entity_type: BlockEntityTypeRef, factory: BlockEntityFactory) {
51        let id = *REGISTRY.block_entity_types.get_id(block_entity_type);
52        self.entries[id].factory = Some(factory);
53    }
54
55    /// Creates a new block entity instance.
56    ///
57    /// Returns `None` if no factory is registered for the given type.
58    #[must_use]
59    pub fn create(
60        &self,
61        block_entity_type: BlockEntityTypeRef,
62        level: Weak<World>,
63        pos: BlockPos,
64        state: BlockStateId,
65    ) -> Option<SharedBlockEntity> {
66        let id = *REGISTRY.block_entity_types.get_id(block_entity_type);
67        self.entries.get(id)?.factory.map(|f| f(level, pos, state))
68    }
69
70    /// Creates a new block entity and loads NBT data into it.
71    ///
72    /// Returns `None` if no factory is registered for the given type.
73    #[must_use]
74    pub fn create_and_load(
75        &self,
76        block_entity_type: BlockEntityTypeRef,
77        level: Weak<World>,
78        pos: BlockPos,
79        state: BlockStateId,
80        nbt: &BorrowedNbtCompound<'_>,
81    ) -> Option<SharedBlockEntity> {
82        let entity = self.create(block_entity_type, level, pos, state)?;
83        entity.lock().load_additional(nbt);
84        Some(entity)
85    }
86
87    /// Returns whether a factory is registered for the given type.
88    #[must_use]
89    pub fn has_factory(&self, block_entity_type: BlockEntityTypeRef) -> bool {
90        let id = *REGISTRY.block_entity_types.get_id(block_entity_type);
91        self.entries.get(id).is_some_and(|e| e.factory.is_some())
92    }
93}
94
95impl Default for BlockEntityRegistry {
96    fn default() -> Self {
97        Self::new()
98    }
99}

101/// Wrapper for the global block entity registry that implements `Deref`.
102pub struct BlockEntityRegistryLock(OnceLock<BlockEntityRegistry>);
103
104impl Deref for BlockEntityRegistryLock {
105    type Target = BlockEntityRegistry;
106
107    fn deref(&self) -> &Self::Target {
108        self.0.get().expect("Block entity registry not initialized")
109    }
110}
111
112impl BlockEntityRegistryLock {
113    /// Sets the registry. Returns `Err` if already initialized.
114    pub fn set(&self, registry: BlockEntityRegistry) -> Result<(), BlockEntityRegistry> {
115        self.0.set(registry)
116    }
117}
118
119/// Global block entity registry.
120///
121/// Access via deref: `BLOCK_ENTITIES.create(type, pos, state)`
122pub static BLOCK_ENTITIES: BlockEntityRegistryLock = BlockEntityRegistryLock(OnceLock::new());
123
124/// Initializes the global block entity registry.
125///
126/// This should be called once after the main registry is frozen.
127///
128/// # Panics
129///
130/// Panics if called more than once.
131pub fn init_block_entities() {
132    let mut registry = BlockEntityRegistry::new();
133
134    // Register sign block entity factory
135    registry.register(vanilla_block_entity_types::SIGN, |level, pos, state| {
136        Arc::new(SyncMutex::new(SignBlockEntity::new(level, pos, state)))
137    });
138
139    // Register hanging sign block entity factory
140    registry.register(
141        vanilla_block_entity_types::HANGING_SIGN,
142        |level, pos, state| {
143            Arc::new(SyncMutex::new(SignBlockEntity::new_hanging(
144                level, pos, state,
145            )))
146        },
147    );
148
149    // Register barrel block entity factory
150    registry.register(vanilla_block_entity_types::BARREL, |level, pos, state| {
151        Arc::new(SyncMutex::new(BarrelBlockEntity::new(level, pos, state)))
152    });
153
154    assert!(
155        BLOCK_ENTITIES.set(registry).is_ok(),
156        "Block entity registry already initialized"
157    );
158}
