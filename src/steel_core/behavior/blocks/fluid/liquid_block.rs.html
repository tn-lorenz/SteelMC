37/// Behavior for liquid blocks (water and lava).
38///
39/// Liquid blocks have a LEVEL property (0-15) that determines the fluid state:
40/// - LEVEL 0 = source block (full fluid)
41/// - LEVEL 1-7 = flowing fluid with decreasing height
42/// - LEVEL 8-15 = falling fluid
43#[block_behavior]
44pub struct LiquidBlock {
45    block: BlockRef,
46    #[json_arg(vanilla_fluids, ref)]
47    fluid: FluidRef,
48}
49
50impl LiquidBlock {
51    /// Creates a new liquid block behavior.
52    #[must_use]
53    pub const fn new(block: BlockRef, fluid: FluidRef) -> Self {
54        Self { block, fluid }
55    }
56
57    /// Checks if this liquid should spread and handles lava-water interactions.
58    /// Based on vanilla's `LiquidBlock.shouldSpreadLiquid()`.
59    ///
60    /// Returns `true` if the liquid should spread (schedule tick),
61    /// Returns `false` if the liquid was converted to a block (obsidian/cobblestone/basalt).
62    fn should_spread_liquid(&self, world: &Arc<World>, pos: BlockPos) -> bool {
63        // Only lava has special interactions with water and blue ice
64        if !is_lava_fluid(self.fluid) {
65            return true;
66        }
67        // Check if there's soul soil below (for basalt generation)
68        let below_pos = pos.offset(0, -1, 0);
69        let below_state = world.get_block_state(below_pos);
70        let has_soul_soil_below = below_state.get_block() == &vanilla_blocks::SOUL_SOIL;
71
72        // Get fluid state to check if this is a source
73        let fluid_state = world.get_block_state(pos).get_fluid_state();
74
75        for direction in Direction::FLOW_NEIGHBOR_CHECK {
76            let neighbor_pos = direction.relative(pos);
77            let neighbor_fluid = world.get_block_state(neighbor_pos).get_fluid_state();
78
79            // Check for water (including flowing_water and waterlogged blocks)
80            // Using fluid tag check to support modded fluids registered in the water tag
81            if neighbor_fluid.is_water() {
82                // Lava + Water = Obsidian (if source) or Cobblestone (if flowing)
83                let new_block = if fluid_state.is_source() {
84                    &vanilla_blocks::OBSIDIAN
85                } else {
86                    &vanilla_blocks::COBBLESTONE
87                };
88
89                let new_state = REGISTRY.blocks.get_default_state_id(new_block);
90                world.set_block(pos, new_state, UpdateFlags::UPDATE_ALL);
91                world.level_event(level_events::LAVA_FIZZ, pos, 0, None);
92                return false; // Don't schedule fluid tick - block was converted
93            }
94
95            // Check for basalt generation: soul soil below + blue ice adjacent
96            if has_soul_soil_below {
97                let neighbor_state = world.get_block_state(neighbor_pos);
98                if neighbor_state.get_block() == &vanilla_blocks::BLUE_ICE {
99                    let new_state = REGISTRY
100                        .blocks
101                        .get_default_state_id(&vanilla_blocks::BASALT);
102                    world.set_block(pos, new_state, UpdateFlags::UPDATE_ALL);
103                    world.level_event(level_events::LAVA_FIZZ, pos, 0, None);
104                    return false; // Don't schedule fluid tick - block was converted
105                }
106            }
107        }
108
109        true // No interaction occurred, proceed with normal fluid tick
110    }
111
112    fn should_bubble_column_occupy(state: BlockStateId) -> bool {
113        let fluid_state = state.get_fluid_state();
114        fluid_state
115            .fluid_id
116            .has_tag(&FluidTag::BUBBLE_COLUMN_CAN_OCCUPY)
117            && fluid_state.is_source()
118            && fluid_state.amount >= 8
119    }
120
121    fn try_schedule_bubble_block_column(
122        &self,
123        ticks: &dyn ScheduledTickAccess,
124        pos: BlockPos,
125        state_below: BlockStateId,
126    ) {
127        let block_below = state_below.get_block();
128        if block_below.has_tag(&BlockTag::ENABLES_BUBBLE_COLUMN_DRAG_DOWN)
129            || block_below.has_tag(&BlockTag::ENABLES_BUBBLE_COLUMN_PUSH_UP)
130        {
131            let _ = ticks.schedule_block_tick_default(pos, self.block, 20);
132        }
133    }
134}
135
136impl BlockBehavior for LiquidBlock {
137    fn get_state_for_placement(&self, _context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
138        Some(self.block.default_state())
139    }
140
141    fn get_fluid_state(&self, state: BlockStateId) -> FluidState {
142        let level = state.get_value(&BlockStateProperties::LEVEL);
143        FluidState::from_block_level(self.fluid, level)
144    }
145
146    fn is_pathfindable(
147        &self,
148        _state: BlockStateId,
149        _computation_type: PathComputationType,
150    ) -> bool {
151        !is_lava_fluid(self.fluid)
152    }
153
154    /// Called when the block is placed.
155    fn on_place(
156        &self,
157        state: BlockStateId,
158        world: &Arc<World>,
159        pos: BlockPos,
160        _old_state: BlockStateId,
161        _moved_by_piston: bool,
162    ) {
163        if self.should_spread_liquid(world, pos) {
164            let fluid = state.get_fluid_state().fluid_id;
165            let delay = FLUID_BEHAVIORS.get_behavior(fluid).tick_delay(world);
166            world.schedule_fluid_tick_default(pos, fluid, delay);
167        }
168
169        if Self::should_bubble_column_occupy(state) {
170            self.try_schedule_bubble_block_column(world, pos, world.get_block_state(pos.below()));
171        }
172    }
173
174    fn tick(&self, state: BlockStateId, world: &Arc<World>, pos: BlockPos) {
175        if Self::should_bubble_column_occupy(state) {
176            BubbleColumnBlock::update_column(
177                &vanilla_blocks::BUBBLE_COLUMN,
178                world,
179                pos,
180                world.get_block_state(pos.below()),
181            );
182        }
183    }
184
185    /// Called when a neighboring block changes.
186    fn handle_neighbor_changed(
187        &self,
188        state: BlockStateId,
189        world: &Arc<World>,
190        pos: BlockPos,
191        _source_block: BlockRef,
192        _moved_by_piston: bool,
193    ) {
194        if self.should_spread_liquid(world, pos) {
195            let fluid = world.get_block_state(pos).get_fluid_state().fluid_id;
196            let delay = FLUID_BEHAVIORS.get_behavior(fluid).tick_delay(world);
197            world.schedule_fluid_tick_default(pos, fluid, delay);
198        }
199
200        if Self::should_bubble_column_occupy(state) {
201            self.try_schedule_bubble_block_column(world, pos, world.get_block_state(pos.below()));
202        }
203    }
204
205    /// Called when a neighbor's shape changes.
206    ///
207    /// Vanilla parity: `LiquidBlock.updateShape` schedules a tick only when
208    /// either the current block or the neighbor contains a source fluid.
209    fn update_shape(
210        &self,
211        state: BlockStateId,
212        world: &dyn ScheduledTickAccess,
213        pos: BlockPos,
214        direction: Direction,
215        _neighbor_pos: BlockPos,
216        neighbor_state: BlockStateId,
217    ) -> BlockStateId {
218        let fluid_state =
219            FluidState::from_block_level(self.fluid, state.get_value(&BlockStateProperties::LEVEL));
220        let neighbor_fluid = neighbor_state.get_fluid_state();
221
222        if fluid_state.is_source() || neighbor_fluid.is_source() {
223            let delay = world.fluid_tick_delay(fluid_state.fluid_id);
224            world.schedule_fluid_tick_default(pos, fluid_state.fluid_id, delay);
225        }
226
227        if direction == Direction::Down && Self::should_bubble_column_occupy(state) {
228            self.try_schedule_bubble_block_column(world, pos, neighbor_state);
229        }
230
231        state
232    }
233
234    /// Vanilla parity: `LiquidBlock.isRandomlyTicking` delegates to the fluid.
235    fn is_randomly_ticking(&self, _state: BlockStateId) -> bool {
236        FLUID_BEHAVIORS
237            .get_behavior(self.fluid)
238            .is_randomly_ticking()
239    }
240
241    /// Vanilla parity: `LiquidBlock.randomTick` delegates to the fluid.
242    fn random_tick(&self, _state: BlockStateId, world: &Arc<World>, pos: BlockPos) {
243        FLUID_BEHAVIORS
244            .get_behavior(self.fluid)
245            .random_tick(world, pos);
246    }
247
248    fn pickup_block(
249        &self,
250        world: &Arc<World>,
251        pos: BlockPos,
252        state: BlockStateId,
253        _player: Option<&Player>,
254    ) -> Option<PickupResult> {
255        if state.try_get_value(&BlockStateProperties::LEVEL) != Some(0) {
256            return None;
257        }
258
259        let air = REGISTRY.blocks.get_default_state_id(&vanilla_blocks::AIR);
260        world.set_block(pos, air, UpdateFlags::UPDATE_ALL_IMMEDIATE);
261
262        let bucket = if is_water_fluid(self.fluid) {
263            &vanilla_items::WATER_BUCKET
264        } else {
265            &vanilla_items::LAVA_BUCKET
266        };
267
268        let sound = if is_water_fluid(self.fluid) {
269            &sound_events::ITEM_BUCKET_FILL
270        } else {
271            &sound_events::ITEM_BUCKET_FILL_LAVA
272        };
273
274        Some(PickupResult {
275            filled_bucket: ItemStack::new(bucket),
276            sound: Some(sound),
277        })
278    }
279}

290    #[test]
291    fn update_shape_schedules_actual_flowing_fluid_variant() {
292        init_test_registry();
293        init_behaviors();
294
295        let block = LiquidBlock::new(&vanilla_blocks::WATER, &vanilla_fluids::WATER);
296        let state = vanilla_blocks::WATER
297            .default_state()
298            .set_value(&BlockStateProperties::LEVEL, 1);
299        let neighbor_state = vanilla_blocks::WATER.default_state();
300        let level = TestLevel::default();
301
302        let updated = block.update_shape(
303            state,
304            &level,
305            BlockPos::ZERO,
306            Direction::North,
307            Direction::North.relative(BlockPos::ZERO),
308            neighbor_state,
309        );
310
311        assert_eq!(updated, state);
312        assert_eq!(
313            level
314                .scheduled_fluid_ticks
315                .borrow()
316                .iter()
317                .map(|tick| (tick.fluid, tick.delay))
318                .collect::<Vec<_>>(),
319            vec![(&vanilla_fluids::FLOWING_WATER, 5)]
320        );
321    }
