1//! Button block behavior.
2//!
3//! Buttons are face-attached blocks that emit a redstone signal when pressed.
4//! They automatically unpress after a delay via the scheduled tick system.
5//!
6//! Vanilla equivalent: `ButtonBlock` + `FaceAttachedHorizontalDirectionalBlock`.
7

9use steel_registry::blocks::BlockRef;

12use steel_registry::vanilla_blocks;

15use steel_utils::{BlockPos, BlockStateId};

22/// Behavior for all button block variants.
23///
24/// Stone buttons stay pressed for 20 ticks, wood buttons for 30 ticks.
25/// Each variant has its own click on/off sounds determined by the block set type.
26pub struct ButtonBlock {
27    block: BlockRef,
28    ticks_to_stay_pressed: i32,
29    sound_click_on: i32,
30    sound_click_off: i32,
31}
32
33impl ButtonBlock {
34    /// Creates a new button block behavior.
35    ///
36    /// Parameters are provided by the build system from `classes.json`.
37    #[must_use]
38    pub const fn new(
39        block: BlockRef,
40        ticks_to_stay_pressed: i32,
41        sound_click_on: i32,
42        sound_click_off: i32,
43    ) -> Self {
44        Self {
45            block,
46            ticks_to_stay_pressed,
47            sound_click_on,
48            sound_click_off,
49        }
50    }
51
52    /// Returns the outward direction the button faces (away from the support block).
53    ///
54    /// Vanilla equivalent: `FaceAttachedHorizontalDirectionalBlock.getConnectedDirection()`.
55    fn get_connected_direction(state: BlockStateId) -> Direction {
56        let face: AttachFace = state.get_value(&BlockStateProperties::ATTACH_FACE);
57        match face {
58            AttachFace::Floor => Direction::Up,
59            AttachFace::Ceiling => Direction::Down,
60            AttachFace::Wall => state.get_value(&BlockStateProperties::HORIZONTAL_FACING),
61        }
62    }
63
64    /// Checks if a button with the given state can survive at the given position.
65    fn can_survive(world: &World, pos: BlockPos, state: BlockStateId) -> bool {
66        let support_dir = Self::get_connected_direction(state).opposite();
67        let support_pos = support_dir.relative(&pos);
68        let support_state = world.get_block_state(&support_pos);
69        support_state.is_face_sturdy(support_dir.opposite())
70    }
71
72    /// Updates neighbors at both the button position and the support block position.
73    ///
74    /// Vanilla equivalent: `ButtonBlock.updateNeighbours()`.
75    fn update_button_neighbors(&self, state: BlockStateId, world: &World, pos: BlockPos) {
76        world.update_neighbors_at(&pos, self.block);
77        let support_dir = Self::get_connected_direction(state).opposite();
78        let support_pos = support_dir.relative(&pos);
79        world.update_neighbors_at(&support_pos, self.block);
80    }
81
82    /// Presses the button: sets POWERED=true, updates neighbors, schedules unpress tick,
83    /// and plays the click sound.
84    fn press(&self, state: BlockStateId, world: &World, pos: BlockPos, player: &Player) {
85        let powered_state = state.set_value(&BlockStateProperties::POWERED, true);
86        world.set_block(pos, powered_state, UpdateFlags::UPDATE_ALL);
87        self.update_button_neighbors(powered_state, world, pos);
88        world.schedule_block_tick_default(pos, self.block, self.ticks_to_stay_pressed);
89        world.play_block_sound(self.sound_click_on, pos, 1.0, 1.0, Some(player.id));
90        // TODO: GameEvent.BLOCK_ACTIVATE when game event system exists
91    }
92}

94impl BlockBehaviour for ButtonBlock {
95    fn get_state_for_placement(&self, context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
96        for direction in context.get_nearest_looking_directions() {
97            let state = if direction.get_axis() == Axis::Y {
98                let face = if direction == Direction::Up {
99                    AttachFace::Ceiling
100                } else {
101                    AttachFace::Floor
102                };
103                self.block
104                    .default_state()
105                    .set_value(&BlockStateProperties::ATTACH_FACE, face)
106                    .set_value(
107                        &BlockStateProperties::HORIZONTAL_FACING,
108                        context.horizontal_direction,
109                    )
110            } else {
111                self.block
112                    .default_state()
113                    .set_value(&BlockStateProperties::ATTACH_FACE, AttachFace::Wall)
114                    .set_value(
115                        &BlockStateProperties::HORIZONTAL_FACING,
116                        direction.opposite(),
117                    )
118            };
119
120            if Self::can_survive(context.world, context.relative_pos, state) {
121                return Some(state);
122            }
123        }
124        None
125    }
126
127    fn update_shape(
128        &self,
129        state: BlockStateId,
130        world: &World,
131        pos: BlockPos,
132        direction: Direction,
133        _neighbor_pos: BlockPos,
134        _neighbor_state: BlockStateId,
135    ) -> BlockStateId {
136        let support_dir = Self::get_connected_direction(state).opposite();
137        if direction == support_dir && !Self::can_survive(world, pos, state) {
138            return REGISTRY.blocks.get_default_state_id(vanilla_blocks::AIR);
139        }
140        state
141    }
142
143    fn use_without_item(
144        &self,
145        state: BlockStateId,
146        world: &World,
147        pos: BlockPos,
148        player: &Player,
149        _hit_result: &BlockHitResult,
150    ) -> InteractionResult {
151        let powered: bool = state.get_value(&BlockStateProperties::POWERED);
152        if powered {
153            return InteractionResult::Fail;
154        }
155        self.press(state, world, pos, player);
156        InteractionResult::Success
157    }
158
159    fn tick(&self, state: BlockStateId, world: &World, pos: BlockPos) {
160        let powered: bool = state.get_value(&BlockStateProperties::POWERED);
161        if !powered {
162            return;
163        }
164        // TODO: Check for arrows via checkPressed() â€” wooden buttons should stay
165        // pressed while an arrow is touching them and reschedule the tick.
166        // Also needs entity_inside() on BlockBehaviour trait for arrows pressing
167        // unpowered wooden buttons. Blocked on entity collision system.
168
169        // Unpress the button
170        let unpowered_state = state.set_value(&BlockStateProperties::POWERED, false);
171        world.set_block(pos, unpowered_state, UpdateFlags::UPDATE_ALL);
172        self.update_button_neighbors(state, world, pos);
173        world.play_block_sound(self.sound_click_off, pos, 1.0, 1.0, None);
174        // TODO: GameEvent.BLOCK_DEACTIVATE when game event system exists
175    }
176
177    fn affect_neighbors_after_removal(
178        &self,
179        state: BlockStateId,
180        world: &World,
181        pos: BlockPos,
182        moved_by_piston: bool,
183    ) {
184        if moved_by_piston {
185            return;
186        }
187        let powered: bool = state.get_value(&BlockStateProperties::POWERED);
188        if !powered {
189            return;
190        }
191        self.update_button_neighbors(state, world, pos);
192    }
193}
