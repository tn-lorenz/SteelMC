23/// Behavior for barrel blocks.
24///
25/// Barrels are container block entities with 27 slots (3x9 grid).
26/// They use the same menu as chests but cannot form double containers.
27pub struct BarrelBlock {
28    block: BlockRef,
29}
30
31impl BarrelBlock {
32    /// Creates a new barrel block behavior.
33    #[must_use]
34    pub const fn new(block: BlockRef) -> Self {
35        Self { block }
36    }
37}
38
39impl BlockBehaviour for BarrelBlock {
40    fn get_state_for_placement(&self, context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
41        // Barrel faces opposite to the player's look direction (all 6 directions).
42        let facing = context.get_nearest_looking_direction().opposite();
43
44        Some(
45            self.block
46                .default_state()
47                .set_value(&BlockStateProperties::FACING, facing),
48        )
49    }
50
51    fn use_without_item(
52        &self,
53        _state: BlockStateId,
54        world: &World,
55        pos: BlockPos,
56        player: &Player,
57        _hit_result: &BlockHitResult,
58    ) -> InteractionResult {
59        // Get the block entity
60        let Some(block_entity) = world.get_block_entity(&pos) else {
61            return InteractionResult::Pass;
62        };
63
64        // Create a container reference from the block entity
65        let Some(container_ref) = ContainerRef::from_block_entity(block_entity) else {
66            return InteractionResult::Pass;
67        };
68
69        // Open the chest menu (3 rows for barrel)
70        player.open_menu(&ChestMenuProvider::three_rows(
71            player.inventory.clone(),
72            container_ref,
73            TextComponent::translated(translations::CONTAINER_BARREL.msg()),
74        ));
75
76        // TODO: Award stat OPEN_BARREL
77        // TODO: Anger nearby piglins (PiglinAi.angerNearbyPiglins)
78        // TODO: Implement ContainerOpenersCounter to track open state, play sounds,
79        //       and update OPEN block property. Requires scheduled block ticks (scheduleTick)
80        //       for recheck functionality. See vanilla BarrelBlockEntity and ContainerOpenersCounter.
81
82        InteractionResult::Success
83    }
84
85    fn has_block_entity(&self) -> bool {
86        true
87    }
88
89    fn new_block_entity(
90        &self,
91        level: Weak<World>,
92        pos: BlockPos,
93        state: BlockStateId,
94    ) -> Option<SharedBlockEntity> {
95        BLOCK_ENTITIES.create(vanilla_block_entity_types::BARREL, level, pos, state)
96    }
97
98    fn has_analog_output_signal(&self, _state: BlockStateId) -> bool {
99        true
100    }
101
102    fn get_analog_output_signal(&self, _state: BlockStateId, world: &World, pos: BlockPos) -> i32 {
103        // Get the block entity and calculate signal from container contents
104        world.get_block_entity(&pos).map_or(0, |be| {
105            let guard = be.lock();
106            if let Some(container) = guard.as_container() {
107                calculate_redstone_signal_from_container(container)
108            } else {
109                0
110            }
111        })
112    }
113}
