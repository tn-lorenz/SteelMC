25/// Converts a rotation in degrees to a 16-segment rotation value (0-15).
26///
27/// This is equivalent to vanilla's `RotationSegment.convertToSegment(float)`.
28/// Each segment is 22.5 degrees, and rotation is measured clockwise from south.
29fn convert_to_rotation_segment(degrees: f32) -> u8 {
30    // Normalize to 0-360
31    let normalized = degrees.rem_euclid(360.0);
32    // Convert to segment (each segment is 22.5 degrees)
33    // Round to nearest segment
34    (((normalized / 22.5) + 0.5) as u8) & 15
35}

37/// Gets the nearest looking directions from the player's rotation.
38///
39/// Returns horizontal directions in order of how closely they match the player's look direction.
40fn get_nearest_looking_directions(rotation: f32, clicked_face: Direction) -> Vec<Direction> {
41    // Build list of directions in order of preference
42    // Start with the opposite of the clicked face (most natural for wall signs)
43    // Then add directions based on player facing
44    let mut directions = Vec::with_capacity(4);
45
46    // Add horizontal directions in order of how closely they match player's look
47    let all_horizontal = [
48        Direction::North,
49        Direction::East,
50        Direction::South,
51        Direction::West,
52    ];
53
54    // Calculate angle for each direction and sort by distance to player's rotation
55    let mut scored: Vec<(Direction, f32)> = all_horizontal
56        .iter()
57        .map(|&dir| {
58            let dir_angle = dir.to_yaw();
59            let diff = (rotation - dir_angle + 180.0).rem_euclid(360.0) - 180.0;
60            (dir, diff.abs())
61        })
62        .collect();
63
64    scored.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap_or(Ordering::Equal));
65
66    for (dir, _) in scored {
67        directions.push(dir);
68    }
69
70    // If clicked face is horizontal, prefer placing on the opposite side
71    if clicked_face.is_horizontal() {
72        let opposite = clicked_face.opposite();
73        if let Some(pos) = directions.iter().position(|&d| d == opposite) {
74            directions.remove(pos);
75            directions.insert(0, opposite);
76        }
77    }
78
79    directions
80}

82/// Calculates whether the player is facing the front of a sign.
83///
84/// Uses the sign's rotation (from block state) and the player's position
85/// relative to the sign to determine which side they're looking at.
86fn is_facing_front_text(state: BlockStateId, pos: BlockPos, player: &Player) -> bool {
87    // Get the sign's Y rotation in degrees from the block state
88    let sign_y_rot = get_sign_rotation_degrees(state);
89
90    // Calculate player's angle relative to the sign center
91    let player_pos = *player.position.lock();
92    let dx = player_pos.x - (f64::from(pos.0.x) + 0.5);
93    let dz = player_pos.z - (f64::from(pos.0.z) + 0.5);
94
95    // Calculate angle from sign to player (in degrees, -90 to account for Minecraft's coordinate system)
96    let player_angle = (dz.atan2(dx) * 180.0 / PI) as f32 - 90.0;
97
98    // Front text if the angle difference is <= 90 degrees
99    let diff = (sign_y_rot - player_angle + 180.0).rem_euclid(360.0) - 180.0;
100    diff.abs() <= 90.0
101}

103/// Gets the Y rotation of a sign in degrees from its block state.
104fn get_sign_rotation_degrees(state: BlockStateId) -> f32 {
105    // Standing signs use "rotation" property (0-15, each step is 22.5 degrees)
106    if let Some(rotation) = state.try_get_value(&BlockStateProperties::ROTATION_16) {
107        return f32::from(rotation) * 22.5;
108    }
109
110    // Wall signs use "facing" property
111    if let Some(facing) = state.try_get_value(&BlockStateProperties::HORIZONTAL_FACING) {
112        return facing.to_yaw();
113    }
114
115    0.0
116}

118/// Checks if a block state can support a standing sign.
119///
120/// Vanilla uses `isSolid()` which checks if the collision shape is a full cube.
121/// This means signs cannot be placed on other signs, fences, walls, etc.
122fn can_support_standing_sign(world: &World, pos: &BlockPos) -> bool {
123    let below_pos = BlockPos::new(pos.x(), pos.y() - 1, pos.z());
124    let below_state = world.get_block_state(&below_pos);
125    below_state.is_solid()
126}

128/// Checks if a wall sign can survive at the given position with the given facing.
129///
130/// Vanilla uses `isSolid()` which allows wall signs to be placed on other signs
131/// (since signs have `forceSolidOn`).
132fn can_wall_sign_survive(world: &World, pos: &BlockPos, facing: Direction) -> bool {
133    // Wall sign needs a solid block behind it
134    let behind_pos = facing.opposite().relative(pos);
135    let behind_state = world.get_block_state(&behind_pos);
136    behind_state.is_solid()
137}

139/// Checks if a ceiling hanging sign can survive at the given position.
140fn can_ceiling_hanging_sign_survive(world: &World, pos: &BlockPos) -> bool {
141    let above_pos = BlockPos::new(pos.x(), pos.y() + 1, pos.z());
142    let above_state = world.get_block_state(&above_pos);
143    above_state.is_face_sturdy_for(Direction::Down, SupportType::Center)
144}

146/// Checks if a wall hanging sign can attach to a neighboring block.
147///
148/// Vanilla's `WallHangingSignBlock.canAttachTo` checks:
149/// 1. If the neighbor is a wall hanging sign on the same axis, allow attachment
150/// 2. Otherwise, check if the face is sturdy with FULL support type
151fn can_attach_to(
152    world: &World,
153    sign_facing: Direction,
154    attach_pos: &BlockPos,
155    attach_face: Direction,
156) -> bool {
157    let attach_state = world.get_block_state(attach_pos);
158    let attach_block = REGISTRY.blocks.by_state_id(attach_state);
159
160    // Check if it's another wall hanging sign (vanilla uses BlockTags.WALL_HANGING_SIGNS)
161    if let Some(block) = attach_block
162        && block.key.path.contains("wall_hanging_sign")
163    {
164        // Wall hanging signs can chain if they're on the same axis
165        if let Some(neighbor_facing) =
166            attach_state.try_get_value(&BlockStateProperties::HORIZONTAL_FACING)
167        {
168            return neighbor_facing.axis() == sign_facing.axis();
169        }
170    }
171
172    // Otherwise, check for sturdy face with FULL support
173    attach_state.is_face_sturdy_for(attach_face, SupportType::Full)
174}

176/// Checks if a wall hanging sign can survive at the given position.
177///
178/// Wall hanging signs need support on at least one side perpendicular to facing.
179/// This matches vanilla's `WallHangingSignBlock.canPlace`.
180fn can_wall_hanging_sign_survive(world: &World, pos: &BlockPos, facing: Direction) -> bool {
181    let clockwise = facing.rotate_y_clockwise();
182    let counter_clockwise = facing.rotate_y_counter_clockwise();
183
184    let can_attach_clockwise = {
185        let attach_pos = clockwise.relative(pos);
186        can_attach_to(world, facing, &attach_pos, counter_clockwise)
187    };
188
189    let can_attach_counter = {
190        let attach_pos = counter_clockwise.relative(pos);
191        can_attach_to(world, facing, &attach_pos, clockwise)
192    };
193
194    can_attach_clockwise || can_attach_counter
195}

219/// Attempts to open the sign editor for a player.
220///
221/// Checks all conditions required by vanilla:
222/// 1. Block entity exists and is a sign
223/// 2. Sign is not waxed
224/// 3. No other player is currently editing
225/// 4. Player has build permission (`may_build`)
226///
227/// Returns `Success` if the editor was opened, `Pass` otherwise.
228fn try_open_sign_editor(
229    state: BlockStateId,
230    world: &World,
231    pos: BlockPos,
232    player: &Player,
233) -> InteractionResult {
234    // Get the block entity
235    let Some(block_entity) = world.get_block_entity(&pos) else {
236        return InteractionResult::Pass;
237    };
238
239    let mut guard = block_entity.lock();
240    let Some(sign) = guard.as_any_mut().downcast_mut::<SignBlockEntity>() else {
241        return InteractionResult::Pass;
242    };
243
244    // Check 1: Is the sign waxed?
245    if sign.is_waxed {
246        // TODO: Play waxed sign interaction fail sound
247        return InteractionResult::Success; // Vanilla returns SUCCESS even when waxed
248    }
249
250    // Check 2: Is another player editing?
251    if sign.is_other_player_editing(player.gameprofile.id) {
252        return InteractionResult::Pass;
253    }
254
255    // Check 3: Player must have build permission
256    // TODO: Implement may_build check properly
257    // if !player.may_build() {
258    //     return InteractionResult::Pass;
259    // }
260
261    // Determine which side the player is facing
262    let is_front_text = is_facing_front_text(state, pos, player);
263
264    // Set the editing player lock
265    sign.set_player_who_may_edit(Some(player.gameprofile.id));
266
267    // Release lock before calling player method
268    drop(guard);
269
270    // Open the editor
271    player.open_sign_editor(pos, is_front_text);
272    InteractionResult::Success
273}

278}

280impl StandingSignBlock {

282    #[must_use]

288impl BlockBehaviour for StandingSignBlock {
289    fn update_shape(
290        &self,
291        state: BlockStateId,
292        world: &World,
293        pos: BlockPos,
294        direction: Direction,
295        _neighbor_pos: BlockPos,
296        _neighbor_state: BlockStateId,
297    ) -> BlockStateId {
298        // Standing signs break when the block below is removed
299        if direction == Direction::Down && !can_support_standing_sign(world, &pos) {
300            return REGISTRY.blocks.get_default_state_id(vanilla_blocks::AIR);
301        }
302        state
303    }
304
305    fn get_state_for_placement(&self, context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
306        // Check if we can place on the block below
307        if !can_support_standing_sign(context.world, &context.relative_pos) {
308            return None;
309        }
310
311        // Calculate rotation from player's yaw
312        // Vanilla: RotationSegment.convertToSegment(context.getRotation() + 180.0F)
313        let rotation = convert_to_rotation_segment(context.rotation + 180.0);
314
315        Some(
316            self.block
317                .default_state()
318                .set_value(&BlockStateProperties::ROTATION_16, rotation),
319        )
320    }
321
322    fn has_block_entity(&self) -> bool {
323        true
324    }
325
326    fn new_block_entity(
327        &self,
328        level: Weak<World>,
329        pos: BlockPos,
330        state: BlockStateId,
331    ) -> Option<SharedBlockEntity> {
332        Some(Arc::new(SyncMutex::new(SignBlockEntity::new(
333            level, pos, state,
334        ))))
335    }
336
337    fn should_keep_block_entity(&self, _old_state: BlockStateId, _new_state: BlockStateId) -> bool {
338        // Signs don't keep their block entity when changing to a different block
339        false
340    }
341
342    fn use_without_item(
343        &self,
344        state: BlockStateId,
345        world: &World,
346        pos: BlockPos,
347        player: &Player,
348        _hit_result: &BlockHitResult,
349    ) -> InteractionResult {
350        try_open_sign_editor(state, world, pos, player)
351    }
352}

373        direction: Direction,
374        _neighbor_pos: BlockPos,
375        _neighbor_state: BlockStateId,
376    ) -> BlockStateId {
377        // Wall signs break when the block they're attached to is removed
378        // The sign is attached to the block opposite of its facing direction
379        if let Some(facing) = state.try_get_value(&BlockStateProperties::HORIZONTAL_FACING)
380            && direction.opposite() == facing
381            && !can_wall_sign_survive(world, &pos, facing)
382        {
383            return REGISTRY.blocks.get_default_state_id(vanilla_blocks::AIR);
384        }
385        state
386    }

389        // Try each horizontal direction based on player's look direction
390        let directions = get_nearest_looking_directions(context.rotation, context.clicked_face);
391
392        for direction in directions {
393            // The sign faces the opposite direction of where it's attached
394            let facing = direction.opposite();
395
396            // Check if sign can survive with this facing
397            if can_wall_sign_survive(context.world, &context.relative_pos, facing) {
398                return Some(
399                    self.block
400                        .default_state()
401                        .set_value(&BlockStateProperties::HORIZONTAL_FACING, facing),
402                );
403            }
404        }
405
406        // No valid placement found
407        None
408    }

411        true
412    }

415        &self,
416        level: Weak<World>,

441/// Behavior for ceiling hanging sign blocks.
442pub struct CeilingHangingSignBlock {
443    block: BlockRef,
444}
445
446impl CeilingHangingSignBlock {
447    /// Creates a new ceiling hanging sign block behavior.
448    #[must_use]
449    pub const fn new(block: BlockRef) -> Self {
450        Self { block }
451    }
452}
453
454impl BlockBehaviour for CeilingHangingSignBlock {
455    fn update_shape(
456        &self,
457        state: BlockStateId,
458        world: &World,
459        pos: BlockPos,
460        direction: Direction,
461        _neighbor_pos: BlockPos,
462        _neighbor_state: BlockStateId,
463    ) -> BlockStateId {
464        // Ceiling hanging signs break when the block above is removed
465        if direction == Direction::Up && !can_ceiling_hanging_sign_survive(world, &pos) {
466            return REGISTRY.blocks.get_default_state_id(vanilla_blocks::AIR);
467        }
468        state
469    }
470
471    fn get_state_for_placement(&self, context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
472        // Check if we can hang from the block above
473        if !can_ceiling_hanging_sign_survive(context.world, &context.relative_pos) {
474            return None;
475        }
476
477        let above_pos = BlockPos::new(
478            context.relative_pos.x(),
479            context.relative_pos.y() + 1,
480            context.relative_pos.z(),
481        );
482        let above_state = context.world.get_block_state(&above_pos);
483
484        // Determine if we should attach to the middle or not based on block above
485        let direction = Direction::from_yaw(context.rotation);
486        let is_above_full = above_state.is_face_sturdy_for(Direction::Down, SupportType::Full);
487
488        // Check if block above is also a hanging sign
489        let above_block = REGISTRY.blocks.by_state_id(above_state);
490        let is_below_hanging_sign =
491            above_block.is_some_and(|b| b.key.path.contains("hanging_sign"));
492
493        // Determine if attached to middle based on vanilla logic
494        let attached_to_middle = if is_below_hanging_sign {
495            // When below another hanging sign, check if we can chain
496            if let Some(above_facing) =
497                above_state.try_get_value(&BlockStateProperties::HORIZONTAL_FACING)
498            {
499                // Wall hanging sign above - check axis alignment
500                above_facing.axis() != direction.axis()
501            } else if let Some(above_rotation) =
502                above_state.try_get_value(&BlockStateProperties::ROTATION_16)
503            {
504                // Ceiling hanging sign above - check if we can align
505                let above_direction = rotation_to_direction(above_rotation);
506                above_direction.is_none_or(|d| d.axis() != direction.axis())
507            } else {
508                !is_above_full
509            }
510        } else {
511            !is_above_full
512        };
513
514        // Calculate rotation
515        let rotation = if attached_to_middle {
516            // Attached to middle - use player rotation
517            convert_to_rotation_segment(context.rotation + 180.0)
518        } else {
519            // Attached to chains - align with direction
520            convert_to_rotation_segment(direction.opposite().to_yaw())
521        };
522
523        Some(
524            self.block
525                .default_state()
526                .set_value(&BlockStateProperties::ROTATION_16, rotation)
527                .set_value(&BlockStateProperties::ATTACHED, attached_to_middle),
528        )
529    }
530
531    fn has_block_entity(&self) -> bool {
532        true
533    }
534
535    fn new_block_entity(
536        &self,
537        level: Weak<World>,
538        pos: BlockPos,
539        state: BlockStateId,
540    ) -> Option<SharedBlockEntity> {
541        Some(Arc::new(SyncMutex::new(SignBlockEntity::new_hanging(
542            level, pos, state,
543        ))))
544    }
545
546    fn should_keep_block_entity(&self, _old_state: BlockStateId, _new_state: BlockStateId) -> bool {
547        false
548    }
549
550    fn use_without_item(
551        &self,
552        state: BlockStateId,
553        world: &World,
554        pos: BlockPos,
555        player: &Player,
556        _hit_result: &BlockHitResult,
557    ) -> InteractionResult {
558        try_open_sign_editor(state, world, pos, player)
559    }
560}

566        4 => Some(Direction::West),
567        8 => Some(Direction::North),
568        12 => Some(Direction::East),
569        _ => None,
