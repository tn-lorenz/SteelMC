1//! Fence gate block behavior.
2//!
3//! Vanilla equivalent: `FenceGateBlock` + `HorizontalDirectionalBlock`.
4//!
5//! Fence gates open/close on use, sit flush in walls (`IN_WALL`), and in vanilla
6//! also react to redstone (`POWERED`/`OPEN` driven by `Level.hasNeighborSignal`).
7//! Steel has no neighbor-signal/redstone foundation yet, so the redstone-driven
8//! paths are left as `// TODO:` rather than faked — every other behavior is 1:1.
9

32/// Behavior for all fence gate variants.
33#[block_behavior]
34pub struct FenceGateBlock {
35    block: BlockRef,
36    #[json_arg(sound_events, json = "type_fence_gate_open")]
37    sound_open: SoundEventRef,
38    #[json_arg(sound_events, json = "type_fence_gate_close")]
39    sound_close: SoundEventRef,
40}
41
42/// Horizontal facing of the gate.
43const FACING: EnumProperty<Direction> = BlockStateProperties::HORIZONTAL_FACING;
44/// Whether the gate is open.
45const OPEN: BoolProperty = BlockStateProperties::OPEN;
46/// Whether the gate is powered by redstone.
47const POWERED: BoolProperty = BlockStateProperties::POWERED;
48/// Whether the gate is lowered to sit flush inside a wall.
49const IN_WALL: BoolProperty = BlockStateProperties::IN_WALL;
50
51impl FenceGateBlock {
52    /// Creates a new fence gate behavior.
53    ///
54    /// Sound events are provided by the build system from `classes.json`.
55    #[must_use]
56    pub const fn new(
57        block: BlockRef,
58        sound_open: SoundEventRef,
59        sound_close: SoundEventRef,
60    ) -> Self {
61        Self {
62            block,
63            sound_open,
64            sound_close,
65        }
66    }
67
68    /// Vanilla `FenceGateBlock.connectsToDirection`.
69    ///
70    /// A gate connects perpendicular to its facing, i.e. to a wall/fence whose
71    /// connecting axis matches the gate's clockwise-rotated facing axis.
72    #[must_use]
73    pub fn connects_to_direction(state: BlockStateId, direction: Direction) -> bool {
74        state.get_value(&FACING).axis() == direction.rotate_y_clockwise().axis()
75    }
76
77    /// Vanilla `FenceGateBlock.isWall`.
78    fn is_wall(state: BlockStateId) -> bool {
79        state.get_block().has_tag(&BlockTag::WALLS)
80    }
81}
82
83impl BlockBehavior for FenceGateBlock {
84    fn get_state_for_placement(&self, context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
85        let world = context.world;
86        let pos = context.place_pos();
87        let direction = context.horizontal_direction();
88        let axis = direction.axis();
89
90        let in_wall = match axis {
91            Axis::Z => {
92                Self::is_wall(world.get_block_state(Direction::West.relative(pos)))
93                    || Self::is_wall(world.get_block_state(Direction::East.relative(pos)))
94            }
95            Axis::X => {
96                Self::is_wall(world.get_block_state(Direction::North.relative(pos)))
97                    || Self::is_wall(world.get_block_state(Direction::South.relative(pos)))
98            }
99            Axis::Y => false,
100        };
101
102        // TODO: vanilla sets OPEN/POWERED from level.hasNeighborSignal(pos);
103        // blocked on the redstone neighbor-signal foundation. Default closed.
104        let is_open = false;
105
106        Some(
107            self.block
108                .default_state()
109                .set_value(&FACING, direction)
110                .set_value(&OPEN, is_open)
111                .set_value(&POWERED, is_open)
112                .set_value(&IN_WALL, in_wall),
113        )
114    }
115
116    fn update_shape(
117        &self,
118        state: BlockStateId,
119        world: &dyn ScheduledTickAccess,
120        pos: BlockPos,
121        direction: Direction,
122        _neighbor_pos: BlockPos,
123        neighbor_state: BlockStateId,
124    ) -> BlockStateId {
125        // Only the axis perpendicular to the gate (its clockwise facing axis)
126        // can change whether it sits in a wall.
127        if state.get_value(&FACING).rotate_y_clockwise().axis() != direction.axis() {
128            return state;
129        }
130        let opposite_neighbor = world.get_block_state(direction.opposite().relative(pos));
131        let in_wall = Self::is_wall(neighbor_state) || Self::is_wall(opposite_neighbor);
132        state.set_value(&IN_WALL, in_wall)
133    }
134
135    fn use_without_item(
136        &self,
137        state: BlockStateId,
138        world: &Arc<World>,
139        pos: BlockPos,
140        player: &Player,
141        _hit_result: &BlockHitResult,
142        _inv: &mut InventoryAccess,
143    ) -> InteractionResult {
144        let mut new_state = state;
145        if new_state.get_value(&OPEN) {
146            new_state = new_state.set_value(&OPEN, false);
147        } else {
148            let (yaw, _) = player.rotation();
149            let player_direction = Direction::from_yaw(yaw);
150            // Re-face the gate toward the player if they opened it from behind.
151            if new_state.get_value(&FACING) == player_direction.opposite() {
152                new_state = new_state.set_value(&FACING, player_direction);
153            }
154            new_state = new_state.set_value(&OPEN, true);
155        }
156
157        // Vanilla flag 10 = UPDATE_CLIENTS | UPDATE_IMMEDIATE.
158        world.set_block(
159            pos,
160            new_state,
161            UpdateFlags::UPDATE_CLIENTS | UpdateFlags::UPDATE_IMMEDIATE,
162        );
163
164        let opens = new_state.get_value(&OPEN);
165        let sound = if opens {
166            self.sound_open
167        } else {
168            self.sound_close
169        };
170        let pitch = rand::random::<f32>() * 0.1 + 0.9;
171        world.play_block_sound(sound, pos, 1.0, pitch, Some(player.id()));
172        let event = if opens {
173            &vanilla_game_events::BLOCK_OPEN
174        } else {
175            &vanilla_game_events::BLOCK_CLOSE
176        };
177        world.game_event(event, pos, &GameEventContext::new(Some(player), None));
178        InteractionResult::Success
179    }
180
181    fn is_pathfindable(&self, state: BlockStateId, computation_type: PathComputationType) -> bool {
182        match computation_type {
183            PathComputationType::Land | PathComputationType::Air => state.get_value(&OPEN),
184            PathComputationType::Water => false,
185        }
186    }
187}
