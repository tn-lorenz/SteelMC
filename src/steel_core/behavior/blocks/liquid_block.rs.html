1//! Liquid block behavior implementation for water and lava.
2
3use steel_registry::blocks::BlockRef;
4use steel_registry::blocks::block_state_ext::BlockStateExt;
5use steel_registry::blocks::properties::BlockStateProperties;
6use steel_registry::fluid::{FluidRef, FluidState};
7use steel_utils::BlockStateId;
8
9use crate::behavior::block::BlockBehaviour;
10use crate::behavior::context::BlockPlaceContext;
11
12/// Behavior for liquid blocks (water and lava).
13///
14/// Liquid blocks have a LEVEL property (0-15) that determines the fluid state:
15/// - LEVEL 0 = source block (full fluid)
16/// - LEVEL 1-7 = flowing fluid with decreasing height
17/// - LEVEL 8-15 = falling fluid
18pub struct LiquidBlock {
19    block: BlockRef,
20    fluid: FluidRef,
21}
22
23impl LiquidBlock {
24    /// Creates a new liquid block behavior.
25    #[must_use]
26    pub const fn new(block: BlockRef, fluid: FluidRef) -> Self {
27        Self { block, fluid }
28    }
29}
30
31impl BlockBehaviour for LiquidBlock {
32    fn get_state_for_placement(&self, _context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
33        Some(self.block.default_state())
34    }
35
36    fn get_fluid_state(&self, state: BlockStateId) -> FluidState {
37        let level = state.get_value(&BlockStateProperties::LEVEL);
38        FluidState::from_block_level(self.fluid, level)
39    }
40}
