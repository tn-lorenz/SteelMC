26const TILT: EnumProperty<Tilt> = BlockStateProperties::TILT;
27const WATERLOGGED: BoolProperty = BlockStateProperties::WATERLOGGED;
28const FACING: EnumProperty<Direction> = BlockStateProperties::FACING;

30/// Vanilla `BigDripleafBlock` survival.
31///
32/// Survives if the block below is big dripleaf (self), big dripleaf stem, or
33/// in the `SUPPORTS_BIG_DRIPLEAF` tag.
34#[block_behavior]
35pub struct BigDripleafBlock {
36    block: BlockRef,
37}
38
39impl BigDripleafBlock {
40    /// Creates a new big dripleaf block behavior.
41    #[must_use]
42    pub const fn new(block: BlockRef) -> Self {
43        Self { block }
44    }
45    fn can_entity_tilt(pos: &BlockPos, entity: &dyn Entity) -> bool {
46        entity.on_ground() && entity.position().y > f64::from(pos.y()) + 0.6875_f64
47    }
48    fn set_tilt_and_schedule_tick(
49        &self,
50        state_id: BlockStateId,
51        world: &Arc<World>,
52        pos: &BlockPos,
53        tilt: Tilt,
54        sound_wrapper: Option<SoundEventRef>,
55    ) {
56        Self::set_tilt(state_id, world, pos, tilt.clone());
57        if let Some(tilt_sound) = sound_wrapper {
58            Self::play_tilt_sound(world, pos, tilt_sound);
59        }
60        let tick_delay = match tilt {
61            Tilt::None => None,
62            Tilt::Unstable | Tilt::Partial => Some(10),
63            Tilt::Full => Some(100),
64        };
65        if let Some(tick_delay) = tick_delay {
66            world.schedule_block_tick(*pos, self.block, tick_delay, TickPriority::Normal);
67        }
68    }
69    const fn tilt_causes_vibration(tilt: &Tilt) -> bool {
70        matches!(tilt, Tilt::None | Tilt::Partial | Tilt::Full)
71    }
72
73    fn set_tilt(state_id: BlockStateId, world: &Arc<World>, pos: &BlockPos, new_tilt: Tilt) {
74        let previous_tilt = state_id.get_value(&TILT);
75        let new_state = state_id.set_value(&TILT, new_tilt.clone());
76
77        world.set_block(*pos, new_state, UpdateFlags::UPDATE_CLIENTS);
78
79        if Self::tilt_causes_vibration(&new_tilt) && new_tilt != previous_tilt {
80            world.game_event(
81                &vanilla_game_events::BLOCK_CHANGE,
82                *pos,
83                &GameEventContext::default(),
84            );
85        }
86    }
87    fn play_tilt_sound(world: &Arc<World>, pos: &BlockPos, tilt_sound: SoundEventRef) {
88        let pitch = rand::rng().random_range(0.8f32..1.2f32);
89        world.play_block_sound(tilt_sound, *pos, 1f32, pitch, None);
90    }
91    fn reset_tilt(state_id: BlockStateId, world: &Arc<World>, pos: &BlockPos) {
92        Self::set_tilt(state_id, world, pos, Tilt::None);
93        let tilt = state_id.get_value(&TILT);
94
95        if tilt != Tilt::None {
96            Self::play_tilt_sound(world, pos, &BLOCK_BIG_DRIPLEAF_TILT_UP);
97        }
98    }
99    fn can_replace(old_state: BlockStateId) -> bool {
100        old_state.is_air()
101            || old_state.get_block() == &vanilla_blocks::WATER
102            || old_state.get_block() == &vanilla_blocks::SMALL_DRIPLEAF
103    }
104    /// Determines whether big dripleaf can grow into target position
105    pub fn can_grow_into(world: &dyn LevelReader, pos: BlockPos) -> bool {
106        let state = world.get_block_state(pos);
107        !world.is_outside_build_height(pos.y()) && Self::can_replace(state)
108    }
109    /// Places big dripleaf block on target position with properties
110    pub fn place(
111        world: &Arc<World>,
112        pos: BlockPos,
113        fluid_state: FluidState,
114        facing: Direction,
115    ) -> bool {
116        let new_state = vanilla_blocks::BIG_DRIPLEAF
117            .default_state()
118            .set_value(
119                &WATERLOGGED,
120                fluid_state.is_source() && fluid_state.is_water(),
121            )
122            .set_value(&FACING, facing);
123        world.set_block(pos, new_state, UpdateFlags::UPDATE_ALL)
124    }
125}

127impl BlockBehavior for BigDripleafBlock {
128    fn can_survive(&self, _state: BlockStateId, world: &dyn LevelReader, pos: BlockPos) -> bool {
129        let below = world.get_block_state(pos.below());
130        let below_block = below.get_block();
131        below_block == self.block
132            || below_block == &vanilla_blocks::BIG_DRIPLEAF_STEM
133            || below_block.has_tag(&BlockTag::SUPPORTS_BIG_DRIPLEAF)
134    }
135    fn update_shape(
136        &self,
137        state: BlockStateId,
138        world: &dyn ScheduledTickAccess,
139        pos: BlockPos,
140        direction: Direction,
141        _neighbor_pos: BlockPos,
142        neighbor_state: BlockStateId,
143    ) -> BlockStateId {
144        if direction == Direction::Down && !self.can_survive(state, world, pos) {
145            return vanilla_blocks::AIR.default_state();
146        }
147        if state.get_value(&WATERLOGGED) {
148            world.schedule_fluid_tick_default(
149                pos,
150                &vanilla_fluids::WATER,
151                vanilla_fluids::WATER.tick_delay as i32,
152            );
153        }
154
155        if direction == Direction::Up && neighbor_state.get_block() == self.block {
156            vanilla_blocks::BIG_DRIPLEAF_STEM
157                .default_state()
158                .with_properties_of(state)
159        } else {
160            state
161        }
162    }
163    fn get_state_for_placement(&self, context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
164        let below_state = context.world.get_block_state(context.place_pos().below());
165        let below_is_dripleaf_part = below_state.get_block() == &vanilla_blocks::BIG_DRIPLEAF
166            || below_state.get_block() == &vanilla_blocks::BIG_DRIPLEAF_STEM;
167        let facing = {
168            if below_is_dripleaf_part {
169                below_state.get_value(&FACING)
170            } else {
171                context.horizontal_direction().opposite()
172            }
173        };
174        Some(
175            self.block
176                .default_state()
177                .set_value(&WATERLOGGED, context.is_water_source())
178                .set_value(&FACING, facing),
179        )
180    }
181    fn entity_inside(
182        &self,
183        state: BlockStateId,
184        world: &Arc<World>,
185        pos: BlockPos,
186        entity: &dyn Entity,
187        _effect_collector: &mut InsideBlockEffectCollector,
188        _is_precise: bool,
189    ) {
190        let tilt = state.get_value(&TILT);
191        //TODO: also check !level.hasNeighborSignal(pos)) once steel implements redstone
192        if tilt == Tilt::None && BigDripleafBlock::can_entity_tilt(&pos, entity) {
193            Self::set_tilt_and_schedule_tick(self, state, world, &pos, Tilt::Unstable, None);
194        }
195    }
196    fn on_projectile_hit(
197        &self,
198        state: BlockStateId,
199        world: &Arc<World>,
200        hit: &ClipHitResult,
201        _projectile: &dyn Projectile,
202    ) {
203        self.set_tilt_and_schedule_tick(
204            state,
205            world,
206            &hit.block_pos,
207            Tilt::Full,
208            Some(&BLOCK_BIG_DRIPLEAF_TILT_DOWN),
209        );
210    }
211    fn tick(&self, state: BlockStateId, world: &Arc<World>, pos: BlockPos) {
212        //if block_receives_redstone_power(world, pos) {
213        //    reset_tilt(state.id, world, pos);
214        //} else {
215        let tilt = state.get_value(&TILT);
216
217        if tilt == Tilt::Unstable {
218            Self::set_tilt_and_schedule_tick(
219                self,
220                state,
221                world,
222                &pos,
223                Tilt::Partial,
224                Some(&BLOCK_BIG_DRIPLEAF_TILT_DOWN),
225            );
226        } else if tilt == Tilt::Partial {
227            Self::set_tilt_and_schedule_tick(
228                self,
229                state,
230                world,
231                &pos,
232                Tilt::Full,
233                Some(&BLOCK_BIG_DRIPLEAF_TILT_DOWN),
234            );
235        } else if tilt == Tilt::Full {
236            Self::reset_tilt(state, world, &pos);
237        }
238        //}
239    }
240    fn as_bonemealable(&self) -> Option<&dyn Bonemealable> {
241        Some(self)
242    }
243}
244impl Bonemealable for BigDripleafBlock {
245    fn is_valid_bonemeal_target(
246        &self,
247        _state: BlockStateId,
248        world: &dyn LevelReader,
249        pos: BlockPos,
250    ) -> bool {
251        let grow_pos = pos.above();
252        Self::can_grow_into(world, grow_pos)
253    }
254
255    fn is_bonemeal_success(
256        &self,
257        _state: BlockStateId,
258        _world: &Arc<World>,
259        _rng: &mut dyn Rng,
260        _pos: BlockPos,
261    ) -> bool {
262        true
263    }
264
265    fn perform_bonemeal(
266        &self,
267        state: BlockStateId,
268        world: &Arc<World>,
269        _rng: &mut dyn Rng,
270        pos: BlockPos,
271    ) {
272        let above_pos = pos.above();
273        if Self::can_grow_into(world, above_pos) {
274            let facing = state.get_value(&FACING);
275            BigDripleafStemBlock::place(
276                world,
277                pos,
278                world.get_block_state(pos).get_fluid_state(),
279                facing,
280            );
281            Self::place(
282                world,
283                above_pos,
284                world.get_block_state(above_pos).get_fluid_state(),
285                facing,
286            );
287        }
288    }
289
290    fn bonemeal_action_type(&self) -> BonemealAction {
291        BonemealAction::Grower
292    }
