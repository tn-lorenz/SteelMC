15/// Behavior for crop blocks (wheat, carrots, potatoes, beetroot).
16///
17/// Crops grow through random ticks when placed on farmland with sufficient light.
18/// Growth speed is affected by nearby farmland moisture and crop arrangement.
19pub struct CropBlock {
20    block: BlockRef,
21    age_property: IntProperty,
22    max_age: u8,
23}
24
25impl CropBlock {
26    /// Creates a new crop block behavior with the default age property (0-7).
27    #[must_use]
28    pub const fn new(block: BlockRef) -> Self {
29        Self {
30            block,
31            age_property: BlockStateProperties::AGE_7,
32            max_age: 7,
33        }
34    }
35
36    /// Creates a new crop block behavior with a custom age property.
37    #[must_use]
38    pub const fn with_age(block: BlockRef, age_property: IntProperty, max_age: u8) -> Self {
39        Self {
40            block,
41            age_property,
42            max_age,
43        }
44    }
45
46    /// Gets the age of the crop from its block state.
47    fn get_age(&self, state: BlockStateId) -> u8 {
48        state.get_value(&self.age_property)
49    }
50
51    /// Returns the block state for the given age.
52    fn get_state_for_age(&self, age: u8) -> BlockStateId {
53        self.block
54            .default_state()
55            .set_value(&self.age_property, age)
56    }
57
58    /// Returns true if the crop is fully grown.
59    fn is_max_age(&self, state: BlockStateId) -> bool {
60        self.get_age(state) >= self.max_age
61    }
62
63    /// Helper to check if a block matches this crop type using pointer equality.
64    fn is_same_block(&self, other: BlockRef) -> bool {
65        ptr::eq(self.block, other)
66    }
67
68    /// Calculates the growth speed based on surrounding farmland.
69    ///
70    /// Factors affecting growth speed:
71    /// - Farmland below: +1.0 (dry) or +3.0 (hydrated)
72    /// - Adjacent farmland: +0.25 (dry) or +0.75 (hydrated)
73    /// - Same crop in row: /2.0 speed penalty
74    fn get_growth_speed(&self, world: &World, pos: BlockPos) -> f32 {
75        let mut speed = 1.0f32;
76        let below = pos.below();
77
78        // Check 3x3 area of farmland below
79        for dx in -1..=1 {
80            for dz in -1..=1 {
81                let check_pos = below.offset(dx, 0, dz);
82                let block_state = world.get_block_state(&check_pos);
83                let mut block_speed = 0.0f32;
84
85                if ptr::eq(block_state.get_block(), vanilla_blocks::FARMLAND) {
86                    block_speed = 1.0;
87                    // Check moisture level
88                    let moisture = block_state.get_value(&BlockStateProperties::MOISTURE);
89                    if moisture > 0 {
90                        block_speed = 3.0;
91                    }
92                }
93
94                // Diagonal/adjacent farmland contributes less
95                if dx != 0 || dz != 0 {
96                    block_speed /= 4.0;
97                }
98
99                speed += block_speed;
100            }
101        }
102
103        // Check for same crop in adjacent positions (reduces growth speed)
104        let north = world.get_block_state(&pos.north());
105        let south = world.get_block_state(&pos.south());
106        let west = world.get_block_state(&pos.west());
107        let east = world.get_block_state(&pos.east());
108
109        let horizontal_row =
110            self.is_same_block(west.get_block()) || self.is_same_block(east.get_block());
111        let vertical_row =
112            self.is_same_block(north.get_block()) || self.is_same_block(south.get_block());
113
114        if horizontal_row && vertical_row {
115            // Crops in both directions - penalty
116            speed /= 2.0;
117        } else {
118            // Check diagonals
119            let nw = world.get_block_state(&pos.north().west());
120            let ne = world.get_block_state(&pos.north().east());
121            let sw = world.get_block_state(&pos.south().west());
122            let se = world.get_block_state(&pos.south().east());
123
124            let has_diagonal = self.is_same_block(nw.get_block())
125                || self.is_same_block(ne.get_block())
126                || self.is_same_block(sw.get_block())
127                || self.is_same_block(se.get_block());
128
129            if has_diagonal {
130                speed /= 2.0;
131            }
132        }
133
134        speed
135    }
136}
137
138impl BlockBehaviour for CropBlock {
139    fn get_state_for_placement(&self, _context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
140        // Crops are placed at age 0
141        Some(self.get_state_for_age(0))
142    }
143
144    fn is_randomly_ticking(&self, state: BlockStateId) -> bool {
145        // Only tick if not fully grown
146        !self.is_max_age(state)
147    }
148
149    fn random_tick(&self, state: BlockStateId, world: &World, pos: BlockPos) {
150        // TODO: Check light level >= 9 when light engine is implemented
151        // For now, assume sufficient light
152
153        let age = self.get_age(state);
154        if age < self.max_age {
155            let growth_speed = self.get_growth_speed(world, pos);
156
157            // Random chance to grow based on growth speed
158            // Vanilla formula: random.nextInt((int)(25.0F / growthSpeed) + 1) == 0
159            let growth_chance = (25.0 / growth_speed) as u32 + 1;
160
161            if rand::random::<u32>().is_multiple_of(growth_chance) {
162                let new_state = self.get_state_for_age(age + 1);
163                world.set_block(pos, new_state, UpdateFlags::UPDATE_CLIENTS);
164            }
165        }
166    }
167}
