1//! Block and item behavior system.
2//!
3//! This module contains the behavior traits and registries that define how
4//! blocks and items behave dynamically. This is separate from the static data
5//! in steel-registry to maintain a clean separation between constant data and
6//! functional/dynamic behavior.
7//!
8//! # Architecture
9//!
10//! After the main registry (`steel-registry`) is frozen, behavior registries
11//! are created:
12//! - `BlockBehaviorRegistry` - assigns default or custom behaviors to each block
13//! - `ItemBehaviorRegistry` - assigns default or custom behaviors to each item
14//!
15//! # Usage
16//!
17//! ```ignore
18//! use steel_core::behavior::{init_behaviors, BLOCK_BEHAVIORS, ITEM_BEHAVIORS};
19//!
20//! // After registry is frozen, call once at startup:
21//! init_behaviors();
22//!
23//! // Then access behaviors via the global registries:
24//! let behavior = BLOCK_BEHAVIORS.get_behavior(block);
25//! ```

93/// Wrapper for the global block behavior registry that implements `Deref`.

115/// Extension trait for `BlockStateId` that provides access to behavior-dependent methods.
116///
117/// This is separate from `BlockStateExt` (in steel-registry) because these methods
118/// require access to the behavior registry which lives in steel-core.
119pub trait BlockStateBehaviorExt {
120    /// Returns the fluid state for this block state.
121    ///
122    /// Delegates to the block's `BlockBehavior::get_fluid_state` implementation.
123    fn get_fluid_state(&self) -> FluidState;
124
125    /// Returns true when vanilla `BlockState.getFluidState()` is non-empty.
126    fn has_fluid(&self) -> bool;
127
128    /// Returns whether this block state belongs to a vanilla `LiquidBlockContainer`.
129    fn is_liquid_container(&self) -> bool;
130
131    /// Returns whether this block state can be replaced by the given fluid block.
132    fn can_be_replaced_by_fluid(&self, fluid_block: BlockRef) -> bool;
133
134    /// Returns whether this block state can be replaced in this placement context.
135    fn can_be_replaced(&self, context: &BlockPlaceContext<'_>) -> bool;
136
137    /// Returns whether this block state is pathfindable for the supplied vanilla computation type.
138    fn is_pathfindable(&self, computation_type: PathComputationType) -> bool;
139}
140
141impl BlockStateBehaviorExt for BlockStateId {
142    fn get_fluid_state(&self) -> FluidState {
143        let block = self.get_block();
144        let behavior = BLOCK_BEHAVIORS.get_behavior(block);
145        behavior.get_fluid_state(*self)
146    }
147
148    fn has_fluid(&self) -> bool {
149        !self.get_fluid_state().is_empty()
150    }
151
152    fn is_liquid_container(&self) -> bool {
153        let block = self.get_block();
154        let behavior = BLOCK_BEHAVIORS.get_behavior(block);
155        behavior.is_liquid_container(*self)
156    }
157
158    fn can_be_replaced_by_fluid(&self, fluid_block: BlockRef) -> bool {
159        let block = self.get_block();
160        let behavior = BLOCK_BEHAVIORS.get_behavior(block);
161        behavior.can_be_replaced_by_fluid(*self, fluid_block)
162    }
163
164    fn can_be_replaced(&self, context: &BlockPlaceContext<'_>) -> bool {
165        let block = self.get_block();
166        let behavior = BLOCK_BEHAVIORS.get_behavior(block);
167        behavior.can_be_replaced(*self, context)
168    }
169
170    fn is_pathfindable(&self, computation_type: PathComputationType) -> bool {
171        let block = self.get_block();
172        let behavior = BLOCK_BEHAVIORS.get_behavior(block);
173        behavior.is_pathfindable(*self, computation_type)
174    }
175}

177/// Global block behavior registry.
178///
179/// Access behaviors directly via deref: `BLOCK_BEHAVIORS.get_behavior(block)`
180pub static BLOCK_BEHAVIORS: BlockBehaviorLock = BlockBehaviorLock(OnceLock::new());

182/// Global item behavior registry.
183///
184/// Access behaviors directly via deref: `ITEM_BEHAVIORS.get_behavior(item)`
185pub static ITEM_BEHAVIORS: ItemBehaviorLock = ItemBehaviorLock(OnceLock::new());

187/// Initializes the global behavior registries.
188///
189/// This should be called after the main registry is frozen. Repeated calls are a no-op.
190pub fn init_behaviors() {
191    BLOCK_BEHAVIORS.0.get_or_init(|| {
192        let mut block_behaviors = BlockBehaviorRegistry::new();
193        register_block_behaviors(&mut block_behaviors);
194        block_behaviors
195    });
196
197    FLUID_BEHAVIORS.0.get_or_init(|| {
198        let mut fluid_behaviors = FluidBehaviorRegistry::new();
199
200        // Water: WaterFluid implements FluidBehavior directly
201        let water_behavior: Box<dyn FluidBehavior> = Box::new(WaterFluid);
202        // Both WATER and FLOWING_WATER share the same behavior
203        fluid_behaviors.set_behavior(&vanilla_fluids::WATER, water_behavior);
204        fluid_behaviors.set_behavior(&vanilla_fluids::FLOWING_WATER, Box::new(WaterFluid));
205
206        // Lava: LavaFluid implements FluidBehavior directly
207        let lava_behavior: Box<dyn FluidBehavior> = Box::new(LavaFluid);
208        fluid_behaviors.set_behavior(&vanilla_fluids::LAVA, lava_behavior);
209        fluid_behaviors.set_behavior(&vanilla_fluids::FLOWING_LAVA, Box::new(LavaFluid));
210
211        fluid_behaviors
212    });
213
214    ITEM_BEHAVIORS.0.get_or_init(|| {
215        let mut item_behaviors = ItemBehaviorRegistry::new();
216        register_item_behaviors(&mut item_behaviors);
217        item_behaviors
218    });
