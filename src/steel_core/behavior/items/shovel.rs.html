17const FLATTENABLES: [&Block; 6] = [
18    vanilla_blocks::GRASS_BLOCK,
19    vanilla_blocks::DIRT,
20    vanilla_blocks::PODZOL,
21    vanilla_blocks::COARSE_DIRT,
22    vanilla_blocks::MYCELIUM,
23    vanilla_blocks::ROOTED_DIRT,
24];
25
26const LIT_PROPERTY: BoolProperty = BlockStateProperties::LIT;
27
28/// Behaviour for Shovels, extinguises campfires and turns grass blocks into paths
29pub struct ShovelBehaviour;
30
31impl ItemBehavior for ShovelBehaviour {
32    fn use_on(&self, context: &mut UseOnContext) -> InteractionResult {
33        if context.hit_result.direction == Direction::Down {
34            return InteractionResult::Pass;
35        }
36
37        let block_state = context.world.get_block_state(&context.hit_result.block_pos);
38        let block = block_state.get_block();
39
40        if REGISTRY
41            .blocks
42            .is_in_tag(block, &vanilla_block_tags::CAMPFIRES_TAG)
43        {
44            if !block_state.get_value(&LIT_PROPERTY) {
45                return InteractionResult::Pass;
46            }
47            context.world.set_block(
48                context.hit_result.block_pos,
49                block_state.set_value(&LIT_PROPERTY, false),
50                UpdateFlags::UPDATE_ALL_IMMEDIATE,
51            );
52            return InteractionResult::Success;
53        }
54
55        if !context
56            .world
57            .get_block_state(&context.hit_result.block_pos.above())
58            .is_air()
59        {
60            return InteractionResult::Pass;
61        }
62
63        if FLATTENABLES.iter().any(|it| ptr::eq(*it, block)) {
64            context
65                .item_stack
66                .hurt_and_break(1, context.player.has_infinite_materials());
67            context.world.set_block(
68                context.hit_result.block_pos,
69                vanilla_blocks::DIRT_PATH.default_state(),
70                UpdateFlags::UPDATE_ALL_IMMEDIATE,
71            );
72            return InteractionResult::Success;
73        }
74        InteractionResult::Pass
75    }
76}
