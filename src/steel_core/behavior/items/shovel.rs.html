16const FLATTENABLES: [&Block; 6] = [
17    vanilla_blocks::GRASS_BLOCK,
18    vanilla_blocks::DIRT,
19    vanilla_blocks::PODZOL,
20    vanilla_blocks::COARSE_DIRT,
21    vanilla_blocks::MYCELIUM,

25const LIT_PROPERTY: BoolProperty = BlockStateProperties::LIT;

27/// Behavior for Shovels, extinguishes campfires and turns grass blocks into paths
28#[item_behavior]
29pub struct ShovelItem;
30
31impl ItemBehavior for ShovelItem {
32    fn use_on(&self, context: &mut UseOnContext) -> InteractionResult {
33        if context.hit_result.direction == Direction::Down {
34            return InteractionResult::Pass;
35        }
36
37        let block_state = context.world.get_block_state(context.hit_result.block_pos);
38        let block = block_state.get_block();
39
40        // Flattenables — vanilla checks these first
41        if FLATTENABLES.contains(&block) {
42            if !context
43                .world
44                .get_block_state(context.hit_result.block_pos.above())
45                .is_air()
46            {
47                return InteractionResult::Pass;
48            }
49            // TODO: Play SoundEvents.SHOVEL_FLATTEN
50            let infinite_materials = context.player.has_infinite_materials();
51            context.inv.item().hurt_and_break(1, infinite_materials);
52            context.world.set_block(
53                context.hit_result.block_pos,
54                vanilla_blocks::DIRT_PATH.default_state(),
55                UpdateFlags::UPDATE_ALL_IMMEDIATE,
56            );
57            // TODO: Emit GameEvent::BLOCK_CHANGE
58            return InteractionResult::Success;
59        }
60
61        // Campfire extinguishing
62        if REGISTRY
63            .blocks
64            .is_in_tag(block, &vanilla_block_tags::CAMPFIRES_TAG)
65        {
66            if !block_state.get_value(&LIT_PROPERTY) {
67                return InteractionResult::Pass;
68            }
69            // TODO: level_event(1009, pos, 0) — extinguish particle/sound
70            // TODO: CampfireBlock::dowse() — eject cooking items
71            context.world.set_block(
72                context.hit_result.block_pos,
73                block_state.set_value(&LIT_PROPERTY, false),
74                UpdateFlags::UPDATE_ALL_IMMEDIATE,
75            );
76            // TODO: hurt_and_break(1, ...) — shovels take durability damage
77            // TODO: Emit GameEvent::BLOCK_CHANGE
78            return InteractionResult::Success;
79        }
80
81        InteractionResult::Pass
82    }
83}
