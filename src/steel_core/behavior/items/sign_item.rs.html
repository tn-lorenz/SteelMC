18/// Behavior for sign items that place sign blocks and open the editor.
19///
20/// Sign items can place either standing signs (on ground) or wall signs (on walls),
21/// depending on where the player clicks.
22pub struct SignItemBehavior {
23    /// The standing sign block (placed on ground).
24    pub standing_block: BlockRef,
25    /// The wall sign block (attached to walls).
26    pub wall_block: BlockRef,
27}
28
29impl SignItemBehavior {
30    /// Creates a new sign item behavior for the given sign blocks.
31    #[must_use]
32    pub const fn new(standing_block: BlockRef, wall_block: BlockRef) -> Self {
33        Self {
34            standing_block,
35            wall_block,
36        }
37    }
38
39    /// Tries to get a valid placement state, following vanilla's `StandingAndWallBlockItem` logic.
40    ///
41    /// Iterates through directions ordered by player look direction (with clicked face opposite
42    /// prioritized when not replacing), trying standing sign for DOWN and wall sign for horizontals.
43    fn get_placement_state(
44        &self,
45        context: &BlockPlaceContext<'_>,
46        place_pos: &BlockPos,
47        pitch: f32,
48    ) -> Option<steel_utils::BlockStateId> {
49        let block_behaviors = &*BLOCK_BEHAVIORS;
50
51        // Get nearest looking directions - this matches vanilla's getNearestLookingDirections()
52        let directions = get_nearest_looking_directions(
53            context.rotation,
54            pitch,
55            context.clicked_face,
56            context.replace_clicked,
57        );
58
59        for direction in directions {
60            // Skip UP - signs don't attach to ceilings (attachmentDirection.getOpposite() == UP)
61            if direction == Direction::Up {
62                continue;
63            }
64
65            // Try standing sign for DOWN, wall sign for horizontal directions
66            let (_block, state) = if direction == Direction::Down {
67                let behavior = block_behaviors.get_behavior(self.standing_block);
68                if let Some(state) = behavior.get_state_for_placement(context) {
69                    (self.standing_block, state)
70                } else {
71                    continue;
72                }
73            } else {
74                let behavior = block_behaviors.get_behavior(self.wall_block);
75                if let Some(state) = behavior.get_state_for_placement(context) {
76                    (self.wall_block, state)
77                } else {
78                    continue;
79                }
80            };
81
82            // Check canSurvive (canPlace in vanilla's StandingAndWallBlockItem)
83            let can_survive = if direction == Direction::Down {
84                can_survive_standing(context.world, place_pos)
85            } else {
86                true // Wall sign's get_state_for_placement already checks survival
87            };
88
89            if can_survive {
90                // Check collision (isUnobstructed)
91                let collision_shape = state.get_collision_shape();
92                if context.world.is_unobstructed(collision_shape, place_pos) {
93                    return Some(state);
94                }
95            }
96        }
97
98        None
99    }
100}

102/// Checks if a standing sign can survive at the given position.
103///
104/// Vanilla uses `isSolid()` which checks if the collision shape is a full cube.
105/// This means signs cannot be placed on other signs, fences, walls, etc.
106fn can_survive_standing(world: &World, pos: &BlockPos) -> bool {
107    let below_pos = BlockPos::new(pos.x(), pos.y() - 1, pos.z());
108    let below_state = world.get_block_state(&below_pos);
109    below_state.is_solid()
110}

112/// Gets the nearest looking directions for sign placement.
113///
114/// This matches vanilla's `BlockPlaceContext.getNearestLookingDirections()` behavior:
115/// - When not replacing the clicked block, the opposite of the clicked face comes first
116/// - Otherwise, directions are ordered by player look direction
117fn get_nearest_looking_directions(
118    rotation: f32,
119    pitch: f32,
120    clicked_face: Direction,
121    replace_clicked: bool,
122) -> Vec<Direction> {
123    // Get all directions ordered by how closely they match player's look direction
124    let mut directions = Direction::ordered_by_nearest(rotation, pitch);
125
126    // If not replacing the clicked block, put the opposite of clicked face first
127    // This is how vanilla prioritizes wall placement when clicking on a block's side
128    if !replace_clicked {
129        let opposite = clicked_face.opposite();
130        if let Some(pos) = directions.iter().position(|&d| d == opposite) {
131            directions.remove(pos);
132            directions.insert(0, opposite);
133        }
134    }
135
136    directions
137}

139impl ItemBehavior for SignItemBehavior {
140    fn use_on(&self, context: &mut UseOnContext) -> InteractionResult {
141        let clicked_pos = context.hit_result.block_pos;
142        let clicked_state = context.world.get_block_state(&clicked_pos);
143
144        // Get the clicked block to check if it's replaceable
145        let clicked_block = REGISTRY.blocks.by_state_id(clicked_state);
146        let clicked_replaceable = clicked_block.is_some_and(|b| b.config.replaceable);
147
148        // Determine placement position: replace clicked block if replaceable,
149        // otherwise place adjacent to the clicked face
150        let (place_pos, replace_clicked) = if clicked_replaceable {
151            (clicked_pos, true)
152        } else {
153            (context.hit_result.direction.relative(&clicked_pos), false)
154        };
155
156        // Check if placement position is within world bounds
157        if !context.world.is_in_valid_bounds(&place_pos) {
158            return InteractionResult::Fail;
159        }
160
161        // Check if the placement position already has a non-replaceable block
162        let existing_state = context.world.get_block_state(&place_pos);
163        let existing_block = REGISTRY.blocks.by_state_id(existing_state);
164        let existing_replaceable = existing_block.is_some_and(|b| b.config.replaceable);
165
166        if !existing_replaceable {
167            return InteractionResult::Fail;
168        }
169
170        // Get player rotation for placement context
171        let (yaw, pitch) = context.player.rotation.load();
172
173        let place_context = BlockPlaceContext {
174            clicked_pos,
175            clicked_face: context.hit_result.direction,
176            click_location: context.hit_result.location,
177            inside: context.hit_result.inside,
178            relative_pos: place_pos,
179            replace_clicked,
180            horizontal_direction: Direction::from_yaw(yaw),
181            rotation: yaw,
182            pitch,
183            world: context.world,
184        };
185
186        // Try to get a valid placement state (standing or wall)
187        let Some(new_state) =
188            self.get_placement_state(&place_context, &place_pos, place_context.pitch)
189        else {
190            return InteractionResult::Fail;
191        };
192
193        // Place the block
194        if !context
195            .world
196            .set_block(place_pos, new_state, UpdateFlags::UPDATE_ALL_IMMEDIATE)
197        {
198            return InteractionResult::Fail;
199        }
200
201        // Consume one item from the stack
202        context.item_stack.shrink(1);
203
204        // Open the sign editor for the player (front text by default)
205        context.player.open_sign_editor(place_pos, true);
206
207        // TODO: Play place sound
208
209        InteractionResult::Success
210    }
211}

234/// Checks if a wall hanging sign can attach to a neighboring block.
235///
236/// This matches vanilla's `WallHangingSignBlock.canAttachTo`.
237fn can_attach_to(
238    world: &World,
239    sign_facing: Direction,
240    attach_pos: &BlockPos,
241    attach_face: Direction,
242) -> bool {
243    let attach_state = world.get_block_state(attach_pos);
244    let attach_block = REGISTRY.blocks.by_state_id(attach_state);
245
246    // Check if it's another wall hanging sign (vanilla uses BlockTags.WALL_HANGING_SIGNS)
247    if let Some(block) = attach_block
248        && block.key.path.contains("wall_hanging_sign")
249    {
250        // Wall hanging signs can chain if they're on the same axis
251        if let Some(neighbor_facing) =
252            attach_state.try_get_value(&BlockStateProperties::HORIZONTAL_FACING)
253        {
254            return neighbor_facing.axis() == sign_facing.axis();
255        }
256    }
257
258    // Otherwise, check for sturdy face with FULL support
259    attach_state.is_face_sturdy_for(attach_face, SupportType::Full)
260}

262/// Checks if a wall hanging sign can be placed at the given position.
263///
264/// This matches vanilla's `WallHangingSignBlock.canPlace` which is called
265/// from `HangingSignItem.canPlace` in addition to `canSurvive`.
266fn can_wall_hanging_sign_place(world: &World, state: BlockStateId, pos: &BlockPos) -> bool {
267    let Some(facing) = state.try_get_value(&BlockStateProperties::HORIZONTAL_FACING) else {
268        return false;
269    };
270
271    let clockwise = facing.rotate_y_clockwise();
272    let counter_clockwise = facing.rotate_y_counter_clockwise();
273
274    let can_attach_clockwise = {
275        let attach_pos = clockwise.relative(pos);
276        can_attach_to(world, facing, &attach_pos, counter_clockwise)
277    };
278
279    let can_attach_counter = {
280        let attach_pos = counter_clockwise.relative(pos);
281        can_attach_to(world, facing, &attach_pos, clockwise)
282    };
283
284    can_attach_clockwise || can_attach_counter
285}

287/// Checks if a wall hanging sign block state can be placed.
288///
289/// This matches vanilla's `HangingSignItem.canPlace` override which adds
290/// an additional check for `WallHangingSignBlock.canPlace`.
291fn can_place_hanging_sign(world: &World, state: BlockStateId, pos: &BlockPos) -> bool {
292    let block = REGISTRY.blocks.by_state_id(state);
293
294    // If it's a wall hanging sign, we need the additional canPlace check
295    if let Some(b) = block
296        && b.key.path.contains("wall_hanging_sign")
297        && !can_wall_hanging_sign_place(world, state, pos)
298    {
299        return false;
300    }
301
302    // All hanging signs need canSurvive check (handled by get_state_for_placement)
303    true
304}

306impl ItemBehavior for HangingSignItemBehavior {
307    fn use_on(&self, context: &mut UseOnContext) -> InteractionResult {
308        let clicked_pos = context.hit_result.block_pos;
309        let clicked_state = context.world.get_block_state(&clicked_pos);
310
311        // Get the clicked block to check if it's replaceable
312        let clicked_block = REGISTRY.blocks.by_state_id(clicked_state);
313        let clicked_replaceable = clicked_block.is_some_and(|b| b.config.replaceable);
314
315        // Determine placement position
316        let (place_pos, replace_clicked) = if clicked_replaceable {
317            (clicked_pos, true)
318        } else {
319            (context.hit_result.direction.relative(&clicked_pos), false)
320        };
321
322        // Check if placement position is within world bounds
323        if !context.world.is_in_valid_bounds(&place_pos) {
324            return InteractionResult::Fail;
325        }
326
327        // Check if the placement position already has a non-replaceable block
328        let existing_state = context.world.get_block_state(&place_pos);
329        let existing_block = REGISTRY.blocks.by_state_id(existing_state);
330        let existing_replaceable = existing_block.is_some_and(|b| b.config.replaceable);
331
332        if !existing_replaceable {
333            return InteractionResult::Fail;
334        }
335
336        // Get player rotation for placement context
337        let (yaw, pitch) = context.player.rotation.load();
338
339        let place_context = BlockPlaceContext {
340            clicked_pos,
341            clicked_face: context.hit_result.direction,
342            click_location: context.hit_result.location,
343            inside: context.hit_result.inside,
344            relative_pos: place_pos,
345            replace_clicked,
346            horizontal_direction: Direction::from_yaw(yaw),
347            rotation: yaw,
348            pitch,
349            world: context.world,
350        };
351
352        let block_behaviors = &*BLOCK_BEHAVIORS;
353
354        // Try ceiling hanging sign first if clicked from below, otherwise try wall
355        let blocks_to_try = if context.hit_result.direction == Direction::Down {
356            vec![self.ceiling_block, self.wall_block]
357        } else {
358            vec![self.wall_block, self.ceiling_block]
359        };
360
361        let mut new_state = None;
362        for block in blocks_to_try {
363            let behavior = block_behaviors.get_behavior(block);
364            if let Some(state) = behavior.get_state_for_placement(&place_context) {
365                // Vanilla's HangingSignItem.canPlace has additional check for wall hanging signs
366                if !can_place_hanging_sign(context.world, state, &place_pos) {
367                    continue;
368                }
369
370                let collision_shape = state.get_collision_shape();
371                if context.world.is_unobstructed(collision_shape, &place_pos) {
372                    new_state = Some(state);
373                    break;
374                }
375            }
376        }
377
378        let Some(state) = new_state else {
379            return InteractionResult::Fail;
380        };
381
382        // Place the block
383        if !context
384            .world
385            .set_block(place_pos, state, UpdateFlags::UPDATE_ALL_IMMEDIATE)
386        {
387            return InteractionResult::Fail;
388        }
389
390        // Consume one item from the stack
391        context.item_stack.shrink(1);
392
393        // Open the sign editor for the player (front text by default)
394        context.player.open_sign_editor(place_pos, true);
395
396        // TODO: Play place sound
397
398        InteractionResult::Success
399    }
