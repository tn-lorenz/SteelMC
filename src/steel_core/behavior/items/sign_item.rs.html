2//!
3//! Places sign blocks and opens the sign editor after placement.
4//! Handles both standing signs (on ground) and wall signs (on walls).
5//!
6//! **Vanilla reference:** `SignItem` extends `StandingAndWallBlockItem` and only
7//! overrides `updateCustomBlockEntityTag` to open the sign editor after placement.

22/// Behavior for sign items that place sign blocks and open the editor.
23///
24/// In vanilla, `SignItem` extends `StandingAndWallBlockItem` with `attachmentDirection = DOWN`.
25/// We use composition here - wrapping `StandingAndWallBlockItem` and adding sign-specific
26/// behavior (opening the sign editor after placement).
27pub struct SignItemBehavior {
28    /// The underlying standing and wall block item behavior.
29    inner: StandingAndWallBlockItem,
30}
31
32impl SignItemBehavior {
33    /// Creates a new sign item behavior for the given sign blocks.
34    #[must_use]
35    pub const fn new(standing_block: BlockRef, wall_block: BlockRef) -> Self {
36        Self {
37            inner: StandingAndWallBlockItem::new(standing_block, wall_block, Direction::Down),
38        }
39    }
40}

42impl ItemBehavior for SignItemBehavior {
43    fn use_on(&self, context: &mut UseOnContext) -> InteractionResult {
44        let clicked_pos = context.hit_result.block_pos;
45        let clicked_state = context.world.get_block_state(&clicked_pos);
46
47        // Get the clicked block to check if it's replaceable
48        let clicked_block = REGISTRY.blocks.by_state_id(clicked_state);
49        let clicked_replaceable = clicked_block.is_some_and(|b| b.config.replaceable);
50
51        // Determine placement position: replace clicked block if replaceable,
52        // otherwise place adjacent to the clicked face
53        let (place_pos, replace_clicked) = if clicked_replaceable {
54            (clicked_pos, true)
55        } else {
56            (context.hit_result.direction.relative(&clicked_pos), false)
57        };
58
59        // Check if placement position is within world bounds
60        if !context.world.is_in_valid_bounds(&place_pos) {
61            return InteractionResult::Fail;
62        }
63
64        // Check if the placement position already has a non-replaceable block
65        let existing_state = context.world.get_block_state(&place_pos);
66        let existing_block = REGISTRY.blocks.by_state_id(existing_state);
67        let existing_replaceable = existing_block.is_some_and(|b| b.config.replaceable);
68
69        if !existing_replaceable {
70            return InteractionResult::Fail;
71        }
72
73        // Get player rotation for placement context
74        let (yaw, pitch) = context.player.rotation.load();
75
76        let place_context = BlockPlaceContext {
77            clicked_pos,
78            clicked_face: context.hit_result.direction,
79            click_location: context.hit_result.location,
80            inside: context.hit_result.inside,
81            relative_pos: place_pos,
82            replace_clicked,
83            horizontal_direction: Direction::from_yaw(yaw),
84            rotation: yaw,
85            pitch,
86            world: context.world,
87        };
88
89        // Use StandingAndWallBlockItem's placement logic
90        let Some(new_state) = self.inner.get_placement_state(&place_context) else {
91            return InteractionResult::Fail;
92        };
93
94        // Place the block
95        if !context
96            .world
97            .set_block(place_pos, new_state, UpdateFlags::UPDATE_ALL_IMMEDIATE)
98        {
99            return InteractionResult::Fail;
100        }
101
102        // Play place sound
103        let block = self.inner.get_block_for_state(new_state);
104        let sound_type = &block.config.sound_type;
105        context.world.play_block_sound(
106            sound_type.place_sound,
107            place_pos,
108            sound_type.volume,
109            sound_type.pitch,
110            Some(context.player.id),
111        );
112
113        // Consume one item from the stack
114        context.item_stack.shrink(1);
115
116        // Sign-specific: Open the sign editor for the player (front text by default)
117        context.player.open_sign_editor(place_pos, true);
118
119        InteractionResult::Success
120    }
121}

123/// Behavior for hanging sign items that place hanging sign blocks.
124///
125/// Hanging signs can be placed as ceiling hanging signs or wall hanging signs.
126pub struct HangingSignItemBehavior {

128    pub ceiling_block: BlockRef,
129    /// The wall hanging sign block.
130    pub wall_block: BlockRef,
131}

135    #[must_use]

138            ceiling_block,
139            wall_block,
140        }

144/// Checks if a wall hanging sign can attach to a neighboring block.
145///
146/// This matches vanilla's `WallHangingSignBlock.canAttachTo`.
147fn can_attach_to(
148    world: &World,
149    sign_facing: Direction,
150    attach_pos: &BlockPos,
151    attach_face: Direction,
152) -> bool {
153    let attach_state = world.get_block_state(attach_pos);
154    let attach_block = REGISTRY.blocks.by_state_id(attach_state);
155
156    // Check if it's another wall hanging sign (vanilla uses BlockTags.WALL_HANGING_SIGNS)
157    if let Some(block) = attach_block
158        && block.key.path.contains("wall_hanging_sign")
159    {
160        // Wall hanging signs can chain if they're on the same axis
161        if let Some(neighbor_facing) =
162            attach_state.try_get_value(&BlockStateProperties::HORIZONTAL_FACING)
163        {
164            return neighbor_facing.axis() == sign_facing.axis();
165        }
166    }
167
168    // Otherwise, check for sturdy face with FULL support
169    attach_state.is_face_sturdy_for(attach_face, SupportType::Full)
170}

172/// Checks if a wall hanging sign can be placed at the given position.
173///
174/// This matches vanilla's `WallHangingSignBlock.canPlace` which is called
175/// from `HangingSignItem.canPlace` in addition to `canSurvive`.
176fn can_wall_hanging_sign_place(world: &World, state: BlockStateId, pos: &BlockPos) -> bool {
177    let Some(facing) = state.try_get_value(&BlockStateProperties::HORIZONTAL_FACING) else {
178        return false;
179    };
180
181    let clockwise = facing.rotate_y_clockwise();
182    let counter_clockwise = facing.rotate_y_counter_clockwise();
183
184    let can_attach_clockwise = {
185        let attach_pos = clockwise.relative(pos);
186        can_attach_to(world, facing, &attach_pos, counter_clockwise)
187    };
188
189    let can_attach_counter = {
190        let attach_pos = counter_clockwise.relative(pos);
191        can_attach_to(world, facing, &attach_pos, clockwise)
192    };
193
194    can_attach_clockwise || can_attach_counter
195}

197/// Checks if a wall hanging sign block state can be placed.
198///
199/// This matches vanilla's `HangingSignItem.canPlace` override which adds
200/// an additional check for `WallHangingSignBlock.canPlace`.
201fn can_place_hanging_sign(world: &World, state: BlockStateId, pos: &BlockPos) -> bool {
202    let block = REGISTRY.blocks.by_state_id(state);
203
204    // If it's a wall hanging sign, we need the additional canPlace check
205    if let Some(b) = block
206        && b.key.path.contains("wall_hanging_sign")
207        && !can_wall_hanging_sign_place(world, state, pos)
208    {
209        return false;
210    }
211
212    // All hanging signs need canSurvive check (handled by get_state_for_placement)
213    true
214}

216impl ItemBehavior for HangingSignItemBehavior {
217    fn use_on(&self, context: &mut UseOnContext) -> InteractionResult {
218        let clicked_pos = context.hit_result.block_pos;
219        let clicked_state = context.world.get_block_state(&clicked_pos);
220
221        // Get the clicked block to check if it's replaceable
222        let clicked_block = REGISTRY.blocks.by_state_id(clicked_state);
223        let clicked_replaceable = clicked_block.is_some_and(|b| b.config.replaceable);
224
225        // Determine placement position
226        let (place_pos, replace_clicked) = if clicked_replaceable {
227            (clicked_pos, true)
228        } else {
229            (context.hit_result.direction.relative(&clicked_pos), false)
230        };
231
232        // Check if placement position is within world bounds
233        if !context.world.is_in_valid_bounds(&place_pos) {
234            return InteractionResult::Fail;
235        }
236
237        // Check if the placement position already has a non-replaceable block
238        let existing_state = context.world.get_block_state(&place_pos);
239        let existing_block = REGISTRY.blocks.by_state_id(existing_state);
240        let existing_replaceable = existing_block.is_some_and(|b| b.config.replaceable);
241
242        if !existing_replaceable {
243            return InteractionResult::Fail;
244        }
245
246        // Get player rotation for placement context
247        let (yaw, pitch) = context.player.rotation.load();
248
249        let place_context = BlockPlaceContext {
250            clicked_pos,
251            clicked_face: context.hit_result.direction,
252            click_location: context.hit_result.location,
253            inside: context.hit_result.inside,
254            relative_pos: place_pos,
255            replace_clicked,
256            horizontal_direction: Direction::from_yaw(yaw),
257            rotation: yaw,
258            pitch,
259            world: context.world,
260        };
261
262        let block_behaviors = &*BLOCK_BEHAVIORS;
263
264        // Try ceiling hanging sign first if clicked from below, otherwise try wall
265        let blocks_to_try = if context.hit_result.direction == Direction::Down {
266            [self.ceiling_block, self.wall_block]
267        } else {
268            [self.wall_block, self.ceiling_block]
269        };
270
271        let mut new_state = None;
272        let mut placed_block = None;
273        for block in blocks_to_try {
274            let behavior = block_behaviors.get_behavior(block);
275            if let Some(state) = behavior.get_state_for_placement(&place_context) {
276                // Vanilla's HangingSignItem.canPlace has additional check for wall hanging signs
277                if !can_place_hanging_sign(context.world, state, &place_pos) {
278                    continue;
279                }
280
281                let collision_shape = state.get_collision_shape();
282                if context.world.is_unobstructed(collision_shape, &place_pos) {
283                    new_state = Some(state);
284                    placed_block = Some(block);
285                    break;
286                }
287            }
288        }
289
290        let Some(state) = new_state else {
291            return InteractionResult::Fail;
292        };
293
294        // Place the block
295        if !context
296            .world
297            .set_block(place_pos, state, UpdateFlags::UPDATE_ALL_IMMEDIATE)
298        {
299            return InteractionResult::Fail;
300        }
301
302        // Play place sound
303        if let Some(block) = placed_block {
304            let sound_type = &block.config.sound_type;
305            context.world.play_block_sound(
306                sound_type.place_sound,
307                place_pos,
308                sound_type.volume,
309                sound_type.pitch,
310                Some(context.player.id),
311            );
312        }
313
314        // Consume one item from the stack
315        context.item_stack.shrink(1);
316
317        // Open the sign editor for the player (front text by default)
318        context.player.open_sign_editor(place_pos, true);
319
320        InteractionResult::Success
321    }
