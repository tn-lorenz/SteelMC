13/// Behavior for filled bucket items (water bucket, lava bucket, etc.)
14///
15/// When used on a block, places the fluid at the target position and
16/// replaces the bucket with an empty bucket.
17pub struct FilledBucketBehavior {
18    /// The fluid block to place.
19    fluid_block: BlockRef,
20    /// The empty bucket item to give back.
21    empty_bucket: ItemRef,
22}
23
24impl FilledBucketBehavior {
25    /// Creates a new filled bucket behavior.
26    #[must_use]
27    pub const fn new(fluid_block: BlockRef, empty_bucket: ItemRef) -> Self {
28        Self {
29            fluid_block,
30            empty_bucket,
31        }
32    }
33}
34
35impl ItemBehavior for FilledBucketBehavior {
36    fn use_on(&self, context: &mut UseOnContext) -> InteractionResult {
37        let clicked_pos = context.hit_result.block_pos;
38        let clicked_state = context.world.get_block_state(&clicked_pos);
39        let clicked_block = clicked_state.get_block();
40
41        // Determine placement position
42        // If the clicked block is replaceable, place there; otherwise place adjacent
43        let place_pos = if clicked_block.config.replaceable {
44            clicked_pos
45        } else {
46            context.hit_result.direction.relative(&clicked_pos)
47        };
48
49        // Check world bounds
50        if !context.world.is_in_valid_bounds(&place_pos) {
51            return InteractionResult::Fail;
52        }
53
54        // Check if we can place at this position
55        let existing_state = context.world.get_block_state(&place_pos);
56        let existing_block = existing_state.get_block();
57
58        // Can only place in air or replaceable blocks
59        // Also check that we're not placing water inside water (no-op)
60        if !existing_block.config.replaceable {
61            return InteractionResult::Fail;
62        }
63
64        // Don't place if it's already the same fluid
65        if ptr::eq(existing_block, self.fluid_block) {
66            return InteractionResult::Pass;
67        }
68
69        // Place the fluid block
70        let fluid_state = self.fluid_block.default_state();
71        if !context
72            .world
73            .set_block(place_pos, fluid_state, UpdateFlags::UPDATE_ALL_IMMEDIATE)
74        {
75            return InteractionResult::Fail;
76        }
77
78        // Replace the bucket with an empty bucket (unless in creative mode)
79        if !context.player.has_infinite_materials() {
80            context.item_stack.set_item(&self.empty_bucket.key);
81        }
82
83        // TODO: Play bucket empty sound
84
85        InteractionResult::Success
86    }
87}
