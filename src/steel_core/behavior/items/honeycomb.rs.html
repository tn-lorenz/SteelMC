12/// Behavior for the honeycomb item. Waxes copper blocks and signs.
13#[item_behavior]
14pub struct HoneycombItem;
15
16impl ItemBehavior for HoneycombItem {
17    fn use_on(&self, context: &mut UseOnContext) -> InteractionResult {
18        let pos = context.hit_result.block_pos;
19
20        // Try block waxing first
21        let old_block_state = context.world.get_block_state(pos);
22        if let Some(waxed_block) = get_waxed_from_normal_variant(old_block_state.get_block()) {
23            context.inv.item().shrink(1);
24            // TODO: trigger CriteriaTriggers.ITEM_USED_ON_BLOCK advancement
25            context.world.set_block(
26                pos,
27                REGISTRY
28                    .blocks
29                    .copy_matching_properties(old_block_state, waxed_block),
30                UpdateFlags::UPDATE_ALL,
31            );
32            // TODO: dispatch GameEvent::BLOCK_CHANGE
33            context.world.level_event(
34                level_events::PARTICLES_AND_SOUND_WAX_ON,
35                pos,
36                0,
37                Some(context.player.id),
38            );
39            return InteractionResult::Success;
40        }
41
42        // Fall through to sign waxing
43        let Some(block_entity) = context.world.get_block_entity(pos) else {
44            return InteractionResult::Pass;
45        };
46
47        let mut guard = block_entity.lock();
48        let Some(sign) = guard.as_any_mut().downcast_mut::<SignBlockEntity>() else {
49            return InteractionResult::Pass;
50        };
51
52        if sign.is_waxed {
53            return InteractionResult::Pass;
54        }
55
56        sign.is_waxed = true;
57        sign.set_changed();
58        context.inv.item().shrink(1);
59        context.world.level_event(
60            level_events::PARTICLES_AND_SOUND_WAX_ON,
61            pos,
62            0,
63            Some(context.player.id),
64        );
65        InteractionResult::Success
66    }
67}
