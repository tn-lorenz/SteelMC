1//! Standing and wall block item behavior implementation.
2//!
3//! This handles items like torches that place different block variants
4//! depending on whether they're placed on top of a block (standing) or
5//! on the side of a block (wall).
6//!
7//! **Vanilla differences:** None - this matches vanilla's `StandingAndWallBlockItem` exactly.
8//! The placement logic iterates through `getNearestLookingDirections()` and tries each
9//! direction (skipping the opposite of `attachmentDirection`), using the standing block
10//! when direction matches `attachmentDirection` and wall block otherwise.
11
12use steel_registry::REGISTRY;
13use steel_registry::blocks::BlockRef;
14use steel_registry::blocks::block_state_ext::BlockStateExt;
15use steel_registry::blocks::properties::Direction;
16use steel_utils::types::UpdateFlags;
17
18use crate::behavior::context::{BlockPlaceContext, InteractionResult, UseOnContext};
19use crate::behavior::{BLOCK_BEHAVIORS, ItemBehavior};
20
21/// Behavior for items that place either a standing or wall variant of a block.
22///
23/// Used for torches (`torch/wall_torch`), soul torches, copper torches, etc.
24/// When placed looking down (toward `attachmentDirection`), places the standing variant.
25/// When placed looking horizontally or up, places the wall variant.
26///
27/// The `attachmentDirection` is typically `Direction::Down` for torches, meaning:
28/// - Looking down → place standing torch on top of block below
29/// - Looking horizontally → place wall torch on side of block
30pub struct StandingAndWallBlockItem {
31    /// The block to place when looking toward `attachmentDirection` (e.g., `torch`).
32    pub standing_block: BlockRef,
33    /// The block to place otherwise (e.g., `wall_torch`).
34    pub wall_block: BlockRef,
35    /// The direction that triggers the standing block placement.
36    /// For torches this is `Direction::Down` - when looking down, place standing torch.
37    pub attachment_direction: Direction,
38}
39
40impl StandingAndWallBlockItem {
41    /// Creates a new standing and wall block item behavior.
42    ///
43    /// # Arguments
44    /// * `standing_block` - Block placed when looking toward `attachment_direction`
45    /// * `wall_block` - Block placed when looking away from `attachment_direction`
46    /// * `attachment_direction` - Direction that triggers standing block (e.g., `Down` for torches)
47    #[must_use]
48    pub const fn new(
49        standing_block: BlockRef,
50        wall_block: BlockRef,
51        attachment_direction: Direction,
52    ) -> Self {
53        Self {
54            standing_block,
55            wall_block,
56            attachment_direction,
57        }
58    }
59
60    /// Determines which block variant to use based on placement context.
61    ///
62    /// Vanilla caches the wall state once, then iterates through directions to decide
63    /// between standing and wall variants. This matches `StandingAndWallBlockItem.getPlacementState`.
64    #[must_use]
65    pub fn get_placement_state(
66        &self,
67        place_context: &BlockPlaceContext<'_>,
68    ) -> Option<steel_utils::BlockStateId> {
69        let block_behaviors = &*BLOCK_BEHAVIORS;
70
71        // Cache wall state once (vanilla does this before the loop)
72        let wall_state = block_behaviors
73            .get_behavior(self.wall_block)
74            .get_state_for_placement(place_context);
75
76        let directions = place_context.get_nearest_looking_directions();
77        let skip_direction = self.attachment_direction.opposite();
78
79        for direction in directions {
80            // Skip the opposite of attachment direction
81            // (e.g., for torches with attachment_direction=Down, skip Up)
82            if direction == skip_direction {
83                continue;
84            }
85
86            // Choose state based on direction
87            let possible_state = if direction == self.attachment_direction {
88                // Try standing block
89                block_behaviors
90                    .get_behavior(self.standing_block)
91                    .get_state_for_placement(place_context)
92            } else {
93                // Use cached wall state
94                wall_state
95            };
96
97            let Some(state) = possible_state else {
98                continue;
99            };
100
101            // Vanilla's canPlace checks canSurvive (already done in get_state_for_placement)
102            // Then checks isUnobstructed
103            let collision_shape = state.get_collision_shape();
104            if place_context
105                .world
106                .is_unobstructed(collision_shape, &place_context.relative_pos)
107            {
108                return Some(state);
109            }
110        }
111
112        None
113    }
114
115    /// Gets the block reference for a placed state (for sound lookup).
116    #[must_use]
117    pub fn get_block_for_state(&self, state: steel_utils::BlockStateId) -> BlockRef {
118        // Determine which block was placed by checking the state
119        if REGISTRY
120            .blocks
121            .by_state_id(state)
122            .is_some_and(|b| b.key == self.standing_block.key)
123        {
124            self.standing_block
125        } else {
126            self.wall_block
127        }
128    }
129}

131impl ItemBehavior for StandingAndWallBlockItem {
132    fn use_on(&self, context: &mut UseOnContext) -> InteractionResult {
133        let clicked_pos = context.hit_result.block_pos;
134        let clicked_state = context.world.get_block_state(&clicked_pos);
135
136        // Get the clicked block to check if it's replaceable
137        let clicked_block = REGISTRY.blocks.by_state_id(clicked_state);
138        let clicked_replaceable = clicked_block.is_some_and(|b| b.config.replaceable);
139
140        // Determine placement position: replace clicked block if replaceable,
141        // otherwise place adjacent to the clicked face
142        let (place_pos, replace_clicked) = if clicked_replaceable {
143            (clicked_pos, true)
144        } else {
145            (context.hit_result.direction.relative(&clicked_pos), false)
146        };
147
148        // Check if placement position is within world bounds
149        if !context.world.is_in_valid_bounds(&place_pos) {
150            return InteractionResult::Fail;
151        }
152
153        // Check if the placement position already has a non-replaceable block
154        let existing_state = context.world.get_block_state(&place_pos);
155        let existing_block = REGISTRY.blocks.by_state_id(existing_state);
156        let existing_replaceable = existing_block.is_some_and(|b| b.config.replaceable);
157
158        if !existing_replaceable {
159            return InteractionResult::Fail;
160        }
161
162        // Get player rotation for placement context
163        let (yaw, pitch) = context.player.rotation.load();
164
165        let place_context = BlockPlaceContext {
166            clicked_pos,
167            clicked_face: context.hit_result.direction,
168            click_location: context.hit_result.location,
169            inside: context.hit_result.inside,
170            relative_pos: place_pos,
171            replace_clicked,
172            horizontal_direction: Direction::from_yaw(yaw),
173            rotation: yaw,
174            pitch,
175            world: context.world,
176        };
177
178        // Get the placement state (includes canSurvive and isUnobstructed checks)
179        let Some(new_state) = self.get_placement_state(&place_context) else {
180            return InteractionResult::Fail;
181        };
182
183        // Place the block
184        if !context
185            .world
186            .set_block(place_pos, new_state, UpdateFlags::UPDATE_ALL_IMMEDIATE)
187        {
188            return InteractionResult::Fail;
189        }
190
191        // Play place sound (exclude the placing player, they hear it client-side)
192        let block = self.get_block_for_state(new_state);
193        let sound_type = &block.config.sound_type;
194        context.world.play_block_sound(
195            sound_type.place_sound,
196            place_pos,
197            sound_type.volume,
198            sound_type.pitch,
199            Some(context.player.id),
200        );
201
202        // Consume one item from the stack
203        context.item_stack.shrink(1);
204
205        InteractionResult::Success
206    }
