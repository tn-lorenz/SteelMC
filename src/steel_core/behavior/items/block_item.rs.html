3use steel_registry::REGISTRY;
4use steel_registry::blocks::BlockRef;
5use steel_registry::blocks::block_state_ext::BlockStateExt;
6use steel_registry::blocks::properties::Direction;
7use steel_utils::types::UpdateFlags;
8
9use crate::behavior::context::{BlockPlaceContext, InteractionResult, UseOnContext};
10use crate::behavior::{BLOCK_BEHAVIORS, ItemBehavior};
11
12/// Behavior for items that place blocks.
13pub struct BlockItemBehavior {
14    /// The block this item places.
15    pub block: BlockRef,
16}
17
18impl BlockItemBehavior {
19    /// Creates a new block item behavior for the given block.
20    #[must_use]
21    pub const fn new(block: BlockRef) -> Self {
22        Self { block }
23    }
24}
25
26impl ItemBehavior for BlockItemBehavior {
27    fn use_on(&self, context: &mut UseOnContext) -> InteractionResult {
28        let clicked_pos = context.hit_result.block_pos;
29        let clicked_state = context.world.get_block_state(&clicked_pos);
30
31        // Get the clicked block to check if it's replaceable
32        let clicked_block = REGISTRY.blocks.by_state_id(clicked_state);
33        let clicked_replaceable = clicked_block.is_some_and(|b| b.config.replaceable);
34
35        // Determine placement position: replace clicked block if replaceable,
36        // otherwise place adjacent to the clicked face
37        let (place_pos, replace_clicked) = if clicked_replaceable {
38            (clicked_pos, true)
39        } else {
40            (context.hit_result.direction.relative(&clicked_pos), false)
41        };
42
43        // Check if placement position is within world bounds
44        if !context.world.is_in_valid_bounds(&place_pos) {
45            return InteractionResult::Fail;
46        }
47
48        // Check if the placement position already has a non-replaceable block
49        let existing_state = context.world.get_block_state(&place_pos);
50        let existing_block = REGISTRY.blocks.by_state_id(existing_state);
51        let existing_replaceable = existing_block.is_some_and(|b| b.config.replaceable);
52
53        if !existing_replaceable {
54            return InteractionResult::Fail;
55        }
56
57        // Get player rotation for placement context
58        let (yaw, pitch) = context.player.rotation.load();
59
60        let place_context = BlockPlaceContext {
61            clicked_pos,
62            clicked_face: context.hit_result.direction,
63            click_location: context.hit_result.location,
64            inside: context.hit_result.inside,
65            relative_pos: place_pos,
66            replace_clicked,
67            horizontal_direction: Direction::from_yaw(yaw),
68            rotation: yaw,
69            pitch,
70            world: context.world,
71        };
72
73        // Get block state for placement from the block behavior
74        let block_behaviors = &*BLOCK_BEHAVIORS;
75        let behavior = block_behaviors.get_behavior(self.block);
76        let Some(new_state) = behavior.get_state_for_placement(&place_context) else {
77            return InteractionResult::Fail;
78        };
79
80        // Check if the block placement would intersect with any entity (vanilla: Level.isUnobstructed)
81        let collision_shape = new_state.get_collision_shape();
82        if !context.world.is_unobstructed(collision_shape, &place_pos) {
83            return InteractionResult::Fail;
84        }
85
86        // Place the block
87        if !context
88            .world
89            .set_block(place_pos, new_state, UpdateFlags::UPDATE_ALL_IMMEDIATE)
90        {
91            return InteractionResult::Fail;
92        }
93
94        // Play place sound (exclude the placing player, they hear it client-side)
95        let sound_type = &self.block.config.sound_type;
96        context.world.play_block_sound(
97            sound_type.place_sound,
98            place_pos,
99            sound_type.volume,
100            sound_type.pitch,
101            Some(context.player.id),
102        );
103
104        // Consume one item from the stack
105        context.item_stack.shrink(1);
106
107        // TODO: Call behavior.on_place()
108
109        InteractionResult::Success
110    }
111}
