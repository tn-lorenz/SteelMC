1//! Flint and steel item behavior with portal ignition.
2
3use crate::behavior::blocks::FireBlock;
4use crate::behavior::context::{InteractionResult, UseOnContext};
5use crate::behavior::item::ItemBehavior;
6use steel_macros::item_behavior;
7use steel_registry::sound_events;
8use steel_registry::vanilla_blocks::FIRE;
9use steel_utils::Direction;
10use steel_utils::types::UpdateFlags;
11
12/// Behavior for flint and steel items.
13#[item_behavior]
14pub struct FlintAndSteelItem;
15
16impl ItemBehavior for FlintAndSteelItem {
17    fn use_on(&self, context: &mut UseOnContext) -> InteractionResult {
18        // TODO: light campfires, candles, and candle cakes (set LIT=true) before fire placement
19
20        let click_pos = context.hit_result.block_pos;
21        let fire_pos = click_pos.relative(context.hit_result.direction);
22        let (yaw, _) = context.player.rotation.load();
23        let forward_dir = Direction::from_yaw(yaw);
24
25        if !FireBlock::can_be_placed_at(context.world, fire_pos, forward_dir) {
26            return InteractionResult::Fail;
27        }
28
29        context.world.play_block_sound(
30            sound_events::ITEM_FLINTANDSTEEL_USE,
31            fire_pos,
32            1.0,
33            rand::random::<f32>() * 0.4 + 0.8,
34            Some(context.player.id),
35        );
36
37        // TODO: use BaseFireBlock.getState() equivalent to select soul fire vs regular fire
38        context
39            .world
40            .set_block(fire_pos, FIRE.default_state(), UpdateFlags::UPDATE_ALL);
41
42        let has_infinite_materials = context.player.has_infinite_materials();
43        context.inv.item().hurt_and_break(1, has_infinite_materials);
44
45        InteractionResult::Success
46    }
47}
