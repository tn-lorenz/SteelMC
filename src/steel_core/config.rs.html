15/// Reference to the server configuration.
16///
17/// This is initialized by the `steel` crate during server startup.
18/// Access configuration via the `STEEL_CONFIG` static.
19pub struct ServerConfigRef;
20
21static CONFIG: OnceLock<&'static ServerConfig> = OnceLock::new();
22
23impl ServerConfigRef {
24    /// Initializes the configuration reference.
25    ///
26    /// # Panics
27    /// Panics if called more than once.
28    pub fn init(config: &'static ServerConfig) {
29        CONFIG
30            .set(config)
31            .expect("Server config already initialized");
32    }
33}
34
35impl Deref for ServerConfigRef {
36    type Target = ServerConfig;
37
38    fn deref(&self) -> &Self::Target {
39        CONFIG
40            .get()
41            .expect("Server config not initialized - steel crate must call ServerConfigRef::init()")
42    }
43}

45/// The server configuration.

47/// Access via `STEEL_CONFIG` static after initialization by the steel crate.
48pub static STEEL_CONFIG: ServerConfigRef = ServerConfigRef;

50/// Label type for server links - either built-in string or custom `TextComponent`
51#[derive(Debug, Clone, Deserialize)]
52#[serde(untagged)]
53#[allow(clippy::large_enum_variant)]
54pub enum ConfigLabel {
55    /// Built-in server link type (e.g., "`bug_report`", "website")
56    BuiltIn(ServerLinksType),
57    /// Custom text component with formatting
58    Custom(TextComponent),
59}
60
61/// A single server link configuration entry
62#[derive(Debug, Clone, Deserialize)]
63pub struct ConfigLink {
64    /// The label for this link (built-in type or custom `TextComponent`)
65    pub label: ConfigLabel,
66    /// The URL for this link
67    pub url: String,
68}
69
70/// Server links configuration
71#[derive(Debug, Clone, Deserialize, Default)]
72#[serde(default)]
73pub struct ServerLinks {
74    /// Enable the server links feature
75    pub enable: bool,
76    /// List of server links to display
77    #[serde(default)]
78    pub links: Vec<ConfigLink>,
79}
80
81impl ServerLinks {
82    /// Creates the server link package from the server config
83    #[must_use]
84    pub fn from_config() -> Option<CServerLinks> {
85        let server_links = STEEL_CONFIG.server_links.as_ref()?;
86
87        if !server_links.enable || server_links.links.is_empty() {
88            return None;
89        }
90
91        let links: Vec<Link> = server_links
92            .links
93            .iter()
94            .map(|config_link| {
95                let label = match &config_link.label {
96                    ConfigLabel::BuiltIn(link_type) => Or::Left(*link_type),
97                    ConfigLabel::Custom(text_component) => Or::Right(text_component.clone()),
98                };
99                Link::new(label, config_link.url.clone())
100            })
101            .collect();
102
103        Some(CServerLinks { links })
104    }
105}

107/// The different types of world generators that can be used.
108#[derive(Debug, Clone, Deserialize)]
109#[serde(rename_all = "snake_case")]
110pub enum WordGeneratorTypes {
111    /// produces a flat gras world
112    Flat,
113    /// creates an empty world which can be used for test
114    Empty,
115}

117/// Configuration for world storage.
118#[derive(Debug, Clone, Deserialize)]
119#[serde(rename_all = "snake_case")]
120pub enum WorldStorageConfig {
121    /// Standard disk persistence using region files.
122    Disk {
123        /// Path to the world directory (e.g., "world/overworld").
124        path: String,
125    },
126    /// RAM-only storage with empty chunks created on demand.
127    /// No data is persisted - useful for testing and minigames.
128    RamOnly,
129}

131/// The server configuration.
132#[derive(Debug, Clone, Deserialize)]
133pub struct ServerConfig {
134    /// The Minecraft version this server supports.
135    /// This is not loaded from config but set programmatically.
136    #[serde(skip)]
137    pub mc_version: &'static str,
138    /// The port the server will listen on.
139    pub server_port: u16,
140    /// The seed for the world generator.
141    pub seed: String,
142    /// The maximum number of players that can be on the server at once.
143    pub max_players: u32,
144    /// The view distance of the server.
145    pub view_distance: u8,
146    /// The simulation distance of the server.
147    pub simulation_distance: u8,
148    /// Whether the server is in online mode.
149    pub online_mode: bool,
150    /// Whether the server should use encryption.
151    pub encryption: bool,
152    /// The message of the day.
153    pub motd: String,
154    /// Whether to use a favicon.
155    pub use_favicon: bool,
156    /// The path to the favicon.
157    pub favicon: String,
158    /// Whether to enforce secure chat.
159    pub enforce_secure_chat: bool,
160    /// Defines which generator should be used for the world.
161    pub world_generator: WordGeneratorTypes,
162    /// Defines which storage format and storage option should be used for the world
163    pub world_storage_config: WorldStorageConfig,
164    /// The compression settings for the server.
165    pub compression: Option<CompressionInfo>,
166    /// All settings and configurations for server links
167    pub server_links: Option<ServerLinks>,
168}
