16/// Type of movement being performed.
17///
18/// Affects how the entity interacts with the world during movement.
19/// Matches vanilla's `MoverType` enum.
20#[derive(Debug, Clone, Copy, PartialEq, Eq)]
21pub enum MoverType {
22    /// Normal entity movement (walking, jumping, gravity).
23    #[allow(dead_code)]
24    SelfMovement,
25    /// Movement caused by external forces (pistons, etc).
26    #[allow(dead_code)]
27    Piston,
28    /// Movement from shulker box opening/closing.
29    #[allow(dead_code)]
30    ShulkerBox,
31    /// Movement from shulker entity teleportation.
32    #[allow(dead_code)]
33    Shulker,
34}

36/// Result of a movement operation.
37#[derive(Debug, Clone)]
38pub struct MoveResult {
39    /// The entity's final position after movement and collision resolution.
40    pub final_position: Vector3<f64>,
41
42    /// The actual movement delta applied (may differ from requested due to collisions).
43    pub actual_movement: Vector3<f64>,
44
45    /// Whether the entity is on the ground after movement.
46    pub on_ground: bool,
47
48    /// Whether horizontal collision occurred (X or Z).
49    pub horizontal_collision: bool,
50
51    /// Whether vertical collision occurred.
52    pub vertical_collision: bool,
53
54    /// Whether X-axis collision occurred (requested.x != actual.x).
55    pub x_collision: bool,
56
57    /// Whether Z-axis collision occurred (requested.z != actual.z).
58    pub z_collision: bool,
59
60    /// The entity's AABB at the final position.
61    pub final_aabb: AABBd,
62}

64/// Moves an entity through the world with collision detection and resolution.
65///
66/// This is the main physics function that implements vanilla's `Entity.move()` behavior,
67/// including step-up mechanics and sneak-edge prevention.
68///
69/// # Arguments
70/// * `state` - The entity's current physics state
71/// * `delta` - The desired movement vector (velocity * dt)
72/// * `mover_type` - Type of movement being performed
73/// * `world` - World collision provider
74///
75/// # Returns
76/// A `MoveResult` containing the final position and collision information.
77///
78/// # Vanilla Reference
79/// `net.minecraft.world.entity.Entity.move(MoverType, Vec3)`
80pub fn move_entity(
81    state: &EntityPhysicsState,
82    delta: Vector3<f64>,
83    mover_type: MoverType,
84    world: &dyn CollisionWorld,
85) -> MoveResult {
86    // Early exit for zero movement
87    if delta.x.abs() < 1.0e-7 && delta.y.abs() < 1.0e-7 && delta.z.abs() < 1.0e-7 {
88        return MoveResult {
89            final_position: state.position,
90            actual_movement: Vector3::new(0.0, 0.0, 0.0),
91            on_ground: state.on_ground,
92            horizontal_collision: false,
93            vertical_collision: false,
94            x_collision: false,
95            z_collision: false,
96            final_aabb: state.bounding_box,
97        };
98    }
99
100    // Vanilla `Entity.move()` uses the real bounding box for collision.
101    // Deflating here causes entities (especially items) to end up slightly inside blocks,
102    // which can create client-side desync where the client thinks the entity is falling.
103    let aabb = state.bounding_box;
104
105    // Apply sneak-edge prevention if crouching and on ground
106    let movement = if state.is_crouching && state.on_ground && mover_type == MoverType::SelfMovement
107    {
108        apply_sneak_edge_prevention(state, delta, &aabb, world)
109    } else {
110        delta
111    };
112
113    // Perform basic collision resolution
114    let collision_result = collide_with_world(state, movement, &aabb, world);
115
116    // Try step-up if horizontal collision occurred
117    if should_try_step_up(state, &collision_result, mover_type) {
118        try_step_up(state, movement, &aabb, &collision_result, world)
119    } else {
120        collision_result
121    }
122}

124/// Applies sneak-edge prevention to keep player from walking off blocks.
125///
126/// When crouching, checks if the movement would cause the player to fall off
127/// a block edge. If so, clips the movement to keep them on the block.
128///
129/// Matches: `Player.maybeBackOffFromEdge(Vec3, MoverType)`
130fn apply_sneak_edge_prevention(
131    _state: &EntityPhysicsState,
132    delta: Vector3<f64>,
133    aabb: &AABBd,
134    world: &dyn CollisionWorld,
135) -> Vector3<f64> {
136    // Only prevent edge falling for horizontal movement
137    if delta.x.abs() < 1.0e-7 && delta.z.abs() < 1.0e-7 {
138        return delta;
139    }
140
141    // Calculate position after movement
142    let new_aabb = AABBd {
143        min_x: aabb.min_x + delta.x,
144        min_y: aabb.min_y + delta.y,
145        min_z: aabb.min_z + delta.z,
146        max_x: aabb.max_x + delta.x,
147        max_y: aabb.max_y + delta.y,
148        max_z: aabb.max_z + delta.z,
149    };
150
151    // Check if there's ground below the new position
152    // We check down to 1 block below (vanilla checks maxUpStep + 1.0)
153    let check_down_aabb = AABBd {
154        min_x: new_aabb.min_x,
155        min_y: new_aabb.min_y - 1.0,
156        min_z: new_aabb.min_z,
157        max_x: new_aabb.max_x,
158        max_y: new_aabb.min_y,
159        max_z: new_aabb.max_z,
160    };
161
162    let ground_below = world.get_block_collisions(&check_down_aabb);
163
164    // If no ground below, prevent the movement
165    if ground_below.is_empty() {
166        return Vector3::new(0.0, delta.y, 0.0); // Allow Y movement but block X/Z
167    }
168
169    // Could add more sophisticated edge detection here (checking if we're
170    // moving away from a supporting block), but this basic version matches
171    // most of vanilla's behavior
172
173    delta
174}

176/// Returns the axis step order for collision resolution.
177///
178/// Vanilla's `Direction.axisStepOrder(Vec3)` returns:
179/// - YZX if `|x| < |z|` (move along Z before X)
180/// - YXZ otherwise (move along X before Z)
181///
182/// Y is always first because gravity/vertical movement should be resolved first.
183fn axis_step_order(movement: Vector3<f64>) -> [Axis; 3] {
184    if movement.x.abs() < movement.z.abs() {
185        [Axis::Y, Axis::Z, Axis::X]
186    } else {
187        [Axis::Y, Axis::X, Axis::Z]
188    }
189}

191/// Performs collision detection and resolution along all three axes.
192///
193/// Matches vanilla's `Entity.collideWithShapes()` behavior exactly:
194/// - Uses dynamic axis order based on movement direction (Y first, then X/Z based on magnitude)
195/// - Accumulates resolved movement and moves AABB after each axis
196#[allow(clippy::float_cmp)] // Intentional: checking if collision clipped the movement value
197fn collide_with_world(
198    state: &EntityPhysicsState,
199    movement: Vector3<f64>,
200    aabb: &AABBd,
201    world: &dyn CollisionWorld,
202) -> MoveResult {
203    // Get all collision shapes that could intersect with our movement
204    let swept_aabb = sweep_aabb(aabb, movement);
205    let collisions = world.get_block_collisions(&swept_aabb);
206
207    // Vanilla: collideWithShapes iterates in dynamic axis order
208    let axes = axis_step_order(movement);
209
210    // Track resolved movement per axis and current AABB position
211    let mut resolved = Vector3::new(0.0, 0.0, 0.0);
212    let mut current_aabb = *aabb;
213
214    for axis in axes {
215        let axis_movement = match axis {
216            Axis::X => movement.x,
217            Axis::Y => movement.y,
218            Axis::Z => movement.z,
219        };
220
221        if axis_movement != 0.0 {
222            let collision = collide(axis, &current_aabb, &collisions, axis_movement);
223
224            // Update resolved movement for this axis
225            match axis {
226                Axis::X => resolved.x = collision,
227                Axis::Y => resolved.y = collision,
228                Axis::Z => resolved.z = collision,
229            }
230
231            // Move AABB by the resolved amount (vanilla: boundingBox.move(resolvedMovement))
232            current_aabb = move_aabb(&current_aabb, axis, collision);
233        }
234    }
235
236    let final_position = state.position + resolved;
237
238    // Check if on ground (touching block below with epsilon tolerance)
239    let on_ground = resolved.y != movement.y && movement.y < 0.0;
240
241    // Detect collisions (vanilla: Entity.move lines 751-757)
242    let x_collision = resolved.x != movement.x;
243    let z_collision = resolved.z != movement.z;
244    let horizontal_collision = x_collision || z_collision;
245    let vertical_collision = resolved.y != movement.y;
246
247    MoveResult {
248        final_position,
249        actual_movement: resolved,
250        on_ground,
251        horizontal_collision,
252        vertical_collision,
253        x_collision,
254        z_collision,
255        final_aabb: current_aabb,
256    }
257}

259/// Moves an AABB along a single axis by the given amount.
260fn move_aabb(aabb: &AABBd, axis: Axis, amount: f64) -> AABBd {
261    match axis {
262        Axis::X => AABBd {
263            min_x: aabb.min_x + amount,
264            min_y: aabb.min_y,
265            min_z: aabb.min_z,
266            max_x: aabb.max_x + amount,
267            max_y: aabb.max_y,
268            max_z: aabb.max_z,
269        },
270        Axis::Y => AABBd {
271            min_x: aabb.min_x,
272            min_y: aabb.min_y + amount,
273            min_z: aabb.min_z,
274            max_x: aabb.max_x,
275            max_y: aabb.max_y + amount,
276            max_z: aabb.max_z,
277        },
278        Axis::Z => AABBd {
279            min_x: aabb.min_x,
280            min_y: aabb.min_y,
281            min_z: aabb.min_z + amount,
282            max_x: aabb.max_x,
283            max_y: aabb.max_y,
284            max_z: aabb.max_z + amount,
285        },
286    }
287}

289/// Checks if step-up should be attempted.
290fn should_try_step_up(
291    state: &EntityPhysicsState,
292    collision_result: &MoveResult,
293    mover_type: MoverType,
294) -> bool {
295    // Only try step-up for self-movement
296    if mover_type != MoverType::SelfMovement {
297        return false;
298    }
299
300    // Must have step height > 0
301    if state.max_up_step <= 0.0 {
302        return false;
303    }
304
305    // Must have horizontal collision
306    if !collision_result.horizontal_collision {
307        return false;
308    }
309
310    // Must be on ground or just landed
311    if !state.on_ground && !collision_result.on_ground {
312        return false;
313    }
314
315    true
316}

318/// Attempts to step up over an obstacle.
319///
320/// This implements vanilla's step-up algorithm from `Entity.collide()`.
321///
322/// Vanilla calls `collideWithShapes(Vec3(movement.x, stepHeight, movement.z), groundedAABB, colliders)`
323/// which uses the dynamic axis order. If step-up gives more horizontal progress, use it.
324///
325/// Matches: `Entity.collide()` lines 1077-1095
326#[allow(clippy::float_cmp)] // Intentional: checking if collision clipped the movement value
327fn try_step_up(
328    state: &EntityPhysicsState,
329    movement: Vector3<f64>,
330    aabb: &AABBd,
331    ground_result: &MoveResult,
332    world: &dyn CollisionWorld,
333) -> MoveResult {
334    let max_step = f64::from(state.max_up_step);
335
336    // Sweep for collisions during the entire step attempt
337    let step_sweep_aabb = AABBd {
338        min_x: (aabb.min_x + movement.x).min(aabb.min_x),
339        min_y: aabb.min_y,
340        min_z: (aabb.min_z + movement.z).min(aabb.min_z),
341        max_x: (aabb.max_x + movement.x).max(aabb.max_x),
342        max_y: aabb.max_y + max_step,
343        max_z: (aabb.max_z + movement.z).max(aabb.max_z),
344    };
345    let collisions = world.get_block_collisions(&step_sweep_aabb);
346
347    // Vanilla: collideWithShapes(Vec3(movement.x, maxUpStep, movement.z), groundedAABB, colliders)
348    // This uses the dynamic axis order based on movement.x vs movement.z
349    let step_movement = Vector3::new(movement.x, max_step, movement.z);
350    let axes = axis_step_order(step_movement);
351
352    let mut resolved = Vector3::new(0.0, 0.0, 0.0);
353    let mut current_aabb = *aabb;
354
355    for axis in axes {
356        let axis_movement = match axis {
357            Axis::X => step_movement.x,
358            Axis::Y => step_movement.y,
359            Axis::Z => step_movement.z,
360        };
361
362        if axis_movement != 0.0 {
363            let collision = collide(axis, &current_aabb, &collisions, axis_movement);
364
365            match axis {
366                Axis::X => resolved.x = collision,
367                Axis::Y => resolved.y = collision,
368                Axis::Z => resolved.z = collision,
369            }
370
371            current_aabb = move_aabb(&current_aabb, axis, collision);
372        }
373    }
374
375    // Check if we made more horizontal progress
376    let ground_dist_sq =
377        ground_result.actual_movement.x.powi(2) + ground_result.actual_movement.z.powi(2);
378    let step_dist_sq = resolved.x.powi(2) + resolved.z.powi(2);
379
380    if step_dist_sq <= ground_dist_sq {
381        // Step didn't help, use ground result
382        return ground_result.clone();
383    }
384
385    // Vanilla subtracts the distance to ground from Y: stepFromGround.subtract(0.0, distanceToGround, 0.0)
386    // distanceToGround = aabb.minY - groundedAABB.minY (0 for our setup), so the final movement is just resolved.
387
388    let actual_movement = resolved;
389    let final_position = state.position + actual_movement;
390
391    // After stepping, we're on ground if we ended up at a position lower than max step height
392    // (i.e., we landed on something)
393    let on_ground = resolved.y < max_step && resolved.y > 0.0;
394
395    let x_collision = resolved.x != movement.x;
396    let z_collision = resolved.z != movement.z;
397
398    MoveResult {
399        final_position,
400        actual_movement,
401        on_ground,
402        horizontal_collision: x_collision || z_collision,
403        vertical_collision: false, // Step-up resolved the vertical collision
404        x_collision,
405        z_collision,
406        final_aabb: current_aabb,
407    }
408}

410/// Creates an AABB that encompasses the start and end positions of a movement.
411fn sweep_aabb(aabb: &AABBd, movement: Vector3<f64>) -> AABBd {
412    AABBd {
413        min_x: if movement.x < 0.0 {
414            aabb.min_x + movement.x
415        } else {
416            aabb.min_x
417        },
418        min_y: if movement.y < 0.0 {
419            aabb.min_y + movement.y
420        } else {
421            aabb.min_y
422        },
423        min_z: if movement.z < 0.0 {
424            aabb.min_z + movement.z
425        } else {
426            aabb.min_z
427        },
428        max_x: if movement.x > 0.0 {
429            aabb.max_x + movement.x
430        } else {
431            aabb.max_x
432        },
433        max_y: if movement.y > 0.0 {
434            aabb.max_y + movement.y
435        } else {
436            aabb.max_y
437        },
438        max_z: if movement.z > 0.0 {
439            aabb.max_z + movement.z
440        } else {
441            aabb.max_z
442        },
443    }
