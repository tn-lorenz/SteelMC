36/// Manages region files with seek-based chunk access.
37///
38/// Only keeps region headers (8KB each) in memory, not chunk data.
39/// Chunks are loaded on-demand and converted directly to runtime format.
40pub struct RegionManager {
41    /// Base directory for region files (e.g., "world/region").
42    base_path: PathBuf,
43    /// Open region file handles with their headers.
44    regions: AsyncRwLock<FxHashMap<RegionPos, RegionHandle>>,
45    /// Registry for block state and biome conversions.
46    registry: Arc<Registry>,
47}
48
49/// An open region file with its header.
50struct RegionHandle {
51    /// File handle for reading/writing.
52    file: File,
53    /// Chunk location header (8KB).
54    header: RegionHeader,
55    /// Number of chunks currently loaded from this region.
56    loaded_chunk_count: usize,
57    /// Whether the header has been modified since last save.
58    header_dirty: bool,
59    /// Current file size in sectors.
60    file_sectors: u32,
61}

63/// Builder for creating a persistent chunk with its own palettes.
64struct ChunkBuilder<'a> {
65    block_states: Vec<PersistentBlockState>,
66    biomes: Vec<Identifier>,
67    registry: &'a Registry,
68}

79    /// Ensures a block state exists in the chunk's palette, returning its index.
80    fn ensure_block_state(&mut self, block_id: BlockStateId) -> u16 {
81        // Get block and properties from registry
82        let block = self
83            .registry
84            .blocks
85            .by_state_id(block_id)
86            .expect("Invalid block state ID");
87        let properties = self.registry.blocks.get_properties(block_id);
88
89        let persistent = PersistentBlockState {
90            name: block.key.clone(),
91            properties: properties
92                .into_iter()
93                .map(|(k, v)| (k.to_string(), v.to_string()))
94                .collect(),
95        };
96
97        // Check if already exists
98        if let Some(idx) = self.block_states.iter().position(|s| s == &persistent) {
99            return idx as u16;
100        }
101
102        // Add new entry
103        let idx = self.block_states.len();
104        self.block_states.push(persistent);
105        idx as u16
106    }
107
108    /// Ensures a biome exists in the chunk's palette, returning its index.
109    fn ensure_biome(&mut self, biome_id: u8) -> u16 {
110        // Get biome identifier from registry
111        let biome = self
112            .registry
113            .biomes
114            .by_id(biome_id as usize)
115            .expect("Invalid biome ID");
116        let identifier = biome.key.clone();
117
118        if let Some(idx) = self.biomes.iter().position(|b| b == &identifier) {
119            return idx as u16;
120        }
121
122        let idx = self.biomes.len();
123        self.biomes.push(identifier);
124        idx as u16
125    }
126}

142    /// Gets the file path for a region.
143    fn region_path(&self, pos: RegionPos) -> PathBuf {
144        self.base_path.join(pos.filename())
145    }

147    /// Opens or creates a region file, loading only the header.
148    async fn open_region(&self, pos: RegionPos) -> io::Result<RegionHandle> {
149        let path = self.region_path(pos);
150
151        if !path.exists() {
152            // Create new region file with empty header
153            return self.create_region(pos).await;
154        }
155
156        let mut file = OpenOptions::new()
157            .read(true)
158            .write(true)
159            .open(&path)
160            .await?;
161
162        // Read and verify magic + version
163        let mut header_bytes = [0u8; FILE_HEADER_SIZE];
164        file.read_exact(&mut header_bytes).await?;
165
166        let magic = &header_bytes[0..4];
167        if magic != REGION_MAGIC {
168            return Err(io::Error::new(
169                io::ErrorKind::InvalidData,
170                "Invalid region file magic",
171            ));
172        }
173
174        let version = u16::from_le_bytes([header_bytes[4], header_bytes[5]]);
175        if version > FORMAT_VERSION {
176            return Err(io::Error::new(
177                io::ErrorKind::InvalidData,
178                format!(
179                    "Region file version {version} is newer than supported version {FORMAT_VERSION}"
180                ),
181            ));
182        }
183
184        // Read chunk table
185        let mut table_bytes = vec![0u8; CHUNK_TABLE_SIZE];
186        file.read_exact(&mut table_bytes).await?;
187        let header = RegionHeader::from_bytes(&table_bytes);
188
189        // Calculate file size in sectors
190        let file_size = file.seek(io::SeekFrom::End(0)).await?;
191        let file_sectors = file_size.div_ceil(SECTOR_SIZE as u64) as u32;
192
193        Ok(RegionHandle {
194            file,
195            header,
196            loaded_chunk_count: 0,
197            header_dirty: false,
198            file_sectors,
199        })
200    }
201
202    /// Creates a new empty region file.
203    async fn create_region(&self, pos: RegionPos) -> io::Result<RegionHandle> {
204        fs::create_dir_all(&self.base_path).await?;
205
206        let path = self.region_path(pos);
207        let mut file = OpenOptions::new()
208            .read(true)
209            .write(true)
210            .create(true)
211            .truncate(true)
212            .open(&path)
213            .await?;
214
215        // Write header
216        let mut header_bytes = [0u8; FILE_HEADER_SIZE];
217        header_bytes[0..4].copy_from_slice(&REGION_MAGIC);
218        header_bytes[4..6].copy_from_slice(&FORMAT_VERSION.to_le_bytes());
219        file.write_all(&header_bytes).await?;
220
221        // Write empty chunk table
222        let header = RegionHeader::new();
223        file.write_all(&header.to_bytes()).await?;
224        file.flush().await?;
225
226        Ok(RegionHandle {
227            file,
228            header,
229            loaded_chunk_count: 0,
230            header_dirty: false,
231            file_sectors: FIRST_DATA_SECTOR,
232        })
233    }

235    /// Writes the header to disk.
236    async fn write_header(file: &mut File, header: &RegionHeader) -> io::Result<()> {
237        file.seek(io::SeekFrom::Start(FILE_HEADER_SIZE as u64))
238            .await?;
239        file.write_all(&header.to_bytes()).await?;
240        file.flush().await?;
241        Ok(())
242    }

244    /// Reads a chunk's compressed data from disk.
245    async fn read_chunk_data(
246        file: &mut File,
247        sector_offset: u32,
248        size: u32,
249    ) -> io::Result<Vec<u8>> {
250        let byte_offset = u64::from(sector_offset) * SECTOR_SIZE as u64;
251        file.seek(io::SeekFrom::Start(byte_offset)).await?;
252
253        let mut compressed = vec![0u8; size as usize];
254        file.read_exact(&mut compressed).await?;
255        Ok(compressed)
256    }
257
258    /// Writes chunk data to disk at the specified sector offset.
259    async fn write_chunk_data(
260        file: &mut File,
261        sector_offset: u32,
262        data: &[u8],
263        file_sectors: &mut u32,
264    ) -> io::Result<()> {
265        let byte_offset = u64::from(sector_offset) * SECTOR_SIZE as u64;
266        file.seek(io::SeekFrom::Start(byte_offset)).await?;
267        file.write_all(data).await?;
268
269        // Pad to sector boundary
270        let padding_needed = (SECTOR_SIZE - (data.len() % SECTOR_SIZE)) % SECTOR_SIZE;
271        if padding_needed > 0 {
272            file.write_all(&vec![0u8; padding_needed]).await?;
273        }
274
275        // Update file sectors if we wrote past the end
276        let sectors_used = data.len().div_ceil(SECTOR_SIZE) as u32;
277        let end_sector = sector_offset + sectors_used;
278        if end_sector > *file_sectors {
279            *file_sectors = end_sector;
280        }
281
282        file.flush().await?;
283        Ok(())
284    }

286    /// Saves a chunk to the appropriate region.
287    ///
288    /// The chunk is serialized, compressed, and written to disk immediately.
289    /// If the region was already open (has loaded chunks), the header update is
290    /// deferred. If this call opened the region, it will be closed after saving.
291    ///
292    /// If the chunk is not dirty, this is a no-op and returns `Ok(false)`.
293    /// Returns `Ok(true)` if the chunk was saved.
294    ///
295    /// # Panics
296    /// Panics if the chunk is not loaded.
297    #[allow(clippy::missing_panics_doc)]
298    pub async fn save_chunk(
299        &self,
300        chunk: &SyncRwLock<Option<ChunkAccess>>,
301        status: ChunkStatus,
302    ) -> io::Result<bool> {
303        // Skip saving if chunk hasn't been modified
304        if !chunk
305            .read()
306            .as_ref()
307            .expect("Chunk is not loaded")
308            .is_dirty()
309        {
310            return Ok(false);
311        }
312
313        let pos = chunk.read().as_ref().expect("Chunk is not loaded").pos();
314        let region_pos = RegionPos::from_chunk(pos.0.x, pos.0.y);
315        let (local_x, local_z) = RegionPos::local_chunk_pos(pos.0.x, pos.0.y);
316        let index = RegionHeader::chunk_index(local_x, local_z);
317
318        let mut regions = self.regions.write_async().await;
319
320        // Track if we opened the region (so we can close it after)
321        let we_opened_region = !regions.contains_key(&region_pos);
322
323        // Get or open the region
324        let handle = if let Some(handle) = regions.get_mut(&region_pos) {
325            handle
326        } else {
327            let handle = self.open_region(region_pos).await?;
328            regions.insert(region_pos, handle);
329            regions.get_mut(&region_pos).expect("just inserted")
330        };
331
332        // Convert chunk to persistent format and serialize
333        let persistent = self.sections_to_persistent(
334            chunk
335                .read()
336                .as_ref()
337                .expect("Chunk is not loaded")
338                .sections(),
339        );
340        let data = wincode::serialize(&persistent)
341            .map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e.to_string()))?;
342
343        // Compress with zstd
344        let compressed = zstd::encode_all(&data[..], 3)?;
345
346        if compressed.len() > MAX_CHUNK_SIZE {
347            // Clean up if we opened the region
348            if we_opened_region {
349                regions.remove(&region_pos);
350            }
351            return Err(io::Error::new(
352                io::ErrorKind::InvalidData,
353                format!(
354                    "Chunk too large: {} bytes (max {})",
355                    compressed.len(),
356                    MAX_CHUNK_SIZE
357                ),
358            ));
359        }
360
361        // Find space for the chunk
362        let sectors_needed = compressed.len().div_ceil(SECTOR_SIZE) as u32;
363        let old_entry = handle.header.entries[index];
364
365        // Try to reuse existing space if it fits
366        let sector_offset = if old_entry.exists() && old_entry.sector_count() >= sectors_needed {
367            old_entry.sector_offset
368        } else {
369            handle
370                .header
371                .find_free_sectors(sectors_needed, handle.file_sectors)
372        };
373
374        // Write chunk data
375        Self::write_chunk_data(
376            &mut handle.file,
377            sector_offset,
378            &compressed,
379            &mut handle.file_sectors,
380        )
381        .await?;
382
383        // Update header entry
384        handle.header.entries[index] =
385            super::format::ChunkEntry::new(sector_offset, compressed.len() as u32, status);
386
387        // If we opened this region and no chunks are loaded from it,
388        // write the header and close it immediately
389        if we_opened_region && handle.loaded_chunk_count == 0 {
390            Self::write_header(&mut handle.file, &handle.header).await?;
391            regions.remove(&region_pos);
392        } else {
393            handle.header_dirty = true;
394        }
395
396        drop(regions);
397        chunk
398            .write()
399            .as_mut()
400            .expect("Chunk is not loaded")
401            .clear_dirty();
402
403        Ok(true)
404    }

406    /// Loads a chunk from the appropriate region.
407    ///
408    /// Automatically opens the region if not already open. The region's reference
409    /// count is incremented, so you must call `release_chunk` when done with the chunk.
410    ///
411    /// Returns `Ok(None)` if the chunk doesn't exist on disk.
412    #[allow(clippy::missing_panics_doc)]
413    pub async fn load_chunk(
414        &self,
415        pos: ChunkPos,
416    ) -> io::Result<Option<(ChunkAccess, ChunkStatus)>> {
417        let region_pos = RegionPos::from_chunk(pos.0.x, pos.0.y);
418        let (local_x, local_z) = RegionPos::local_chunk_pos(pos.0.x, pos.0.y);
419        let index = RegionHeader::chunk_index(local_x, local_z);
420
421        let mut regions = self.regions.write_async().await;
422
423        // Track if we just opened this region
424        let was_already_open = regions.contains_key(&region_pos);
425
426        // Get or open the region
427        let handle = if let Some(handle) = regions.get_mut(&region_pos) {
428            handle
429        } else {
430            // Try to open region file
431            match self.open_region(region_pos).await {
432                Ok(handle) => {
433                    regions.insert(region_pos, handle);
434                    regions.get_mut(&region_pos).expect("just inserted")
435                }
436                Err(e) if e.kind() == io::ErrorKind::NotFound => return Ok(None),
437                Err(e) => return Err(e),
438            }
439        };
440
441        // Check if chunk exists
442        let entry = handle.header.entries[index];
443        if !entry.exists() {
444            // Clean up if we just opened this region for nothing
445            if !was_already_open && handle.loaded_chunk_count == 0 && !handle.header_dirty {
446                regions.remove(&region_pos);
447            }
448            return Ok(None);
449        }
450
451        // Read chunk data from disk
452        let compressed =
453            Self::read_chunk_data(&mut handle.file, entry.sector_offset, entry.size_bytes).await?;
454
455        // Decompress
456        let data = zstd::decode_all(&compressed[..])?;
457
458        // Deserialize
459        let persistent: PersistentChunk = wincode::deserialize(&data)
460            .map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e.to_string()))?;
461
462        // Convert to runtime format (persistent is dropped after this - no duplication!)
463        let status = entry.status;
464        let chunk = self.persistent_to_chunk(&persistent, pos, status);
465
466        // Increment ref count
467        handle.loaded_chunk_count += 1;
468
469        Ok(Some((chunk, status)))
470    }

472    /// Releases a loaded chunk, decrementing the region's reference count.
473    ///
474    /// When all chunks from a region are released, the header is saved (if dirty)
475    /// and the file handle is closed.
476    ///
477    /// This must be called for each chunk returned by `load_chunk`.
478    pub async fn release_chunk(&self, pos: ChunkPos) -> io::Result<()> {
479        let region_pos = RegionPos::from_chunk(pos.0.x, pos.0.y);
480
481        let mut regions = self.regions.write_async().await;
482
483        let should_close = if let Some(handle) = regions.get_mut(&region_pos) {
484            handle.loaded_chunk_count = handle.loaded_chunk_count.saturating_sub(1);
485            handle.loaded_chunk_count == 0
486        } else {
487            return Ok(());
488        };
489
490        if should_close
491            && let Some(mut handle) = regions.remove(&region_pos)
492            && handle.header_dirty
493        {
494            Self::write_header(&mut handle.file, &handle.header).await?;
495        }
496
497        Ok(())
498    }

500    /// Checks if a chunk exists on disk without loading it.
501    pub async fn chunk_exists(&self, pos: ChunkPos) -> io::Result<bool> {
502        let region_pos = RegionPos::from_chunk(pos.0.x, pos.0.y);
503        let (local_x, local_z) = RegionPos::local_chunk_pos(pos.0.x, pos.0.y);
504        let index = RegionHeader::chunk_index(local_x, local_z);
505
506        let regions = self.regions.read_async().await;
507
508        // Check cached header first
509        if let Some(handle) = regions.get(&region_pos) {
510            return Ok(handle.header.entries[index].exists());
511        }
512
513        drop(regions);
514
515        // Need to read header from disk
516        let path = self.region_path(region_pos);
517        if !path.exists() {
518            return Ok(false);
519        }
520
521        let mut file = File::open(&path).await?;
522
523        // Skip magic + version
524        file.seek(io::SeekFrom::Start(FILE_HEADER_SIZE as u64))
525            .await?;
526
527        // Read just the one entry we need (8 bytes at index * 8)
528        file.seek(io::SeekFrom::Current((index * 8) as i64)).await?;
529        let mut entry_bytes = [0u8; 8];
530        file.read_exact(&mut entry_bytes).await?;
531
532        let entry = super::format::ChunkEntry::from_bytes(entry_bytes);
533        Ok(entry.exists())
534    }

536    /// Flushes all dirty headers to disk.
537    pub async fn flush_all(&self) -> io::Result<()> {
538        let mut regions = self.regions.write_async().await;
539
540        for handle in regions.values_mut() {
541            if handle.header_dirty {
542                Self::write_header(&mut handle.file, &handle.header).await?;
543                handle.header_dirty = false;
544            }
545        }
546
547        Ok(())
548    }

550    /// Converts sections to persistent format.
551    fn sections_to_persistent(&self, sections: &Sections) -> PersistentChunk {
552        let mut builder = ChunkBuilder::new(&self.registry);
553
554        let persistent_sections = sections
555            .sections
556            .iter()
557            .map(|section| Self::section_to_persistent(section, &mut builder))
558            .collect();
559
560        PersistentChunk {
561            last_modified: std::time::SystemTime::now()
562                .duration_since(std::time::UNIX_EPOCH)
563                .map(|d| d.as_secs() as u32)
564                .unwrap_or(0),
565            block_states: builder.block_states,
566            biomes: builder.biomes,
567            sections: persistent_sections,
568            block_entities: Vec::new(), // TODO: Implement block entity serialization
569        }
570    }

572    /// Converts a runtime section to persistent format.
573    fn section_to_persistent(
574        section: &ChunkSection,
575        builder: &mut ChunkBuilder,
576    ) -> PersistentSection {
577        let biomes = Self::biomes_to_persistent(&section.biomes, builder);
578
579        match &section.states {
580            PalettedContainer::Homogeneous(block_id) => {
581                let block_idx = builder.ensure_block_state(*block_id);
582                PersistentSection::Homogeneous {
583                    block_state: block_idx,
584                    biomes,
585                }
586            }
587            PalettedContainer::Heterogeneous(data) => {
588                // Build section-local palette (indices into chunk's block_states)
589                let palette: Vec<u16> = data
590                    .palette
591                    .iter()
592                    .map(|(block_id, _)| builder.ensure_block_state(*block_id))
593                    .collect();
594
595                // Pack block indices (indices into section-local palette)
596                let bits = bits_for_palette_len(palette.len())
597                    .expect("Heterogeneous section should have palette length >= 2");
598                let indices: Vec<u32> = data
599                    .cube
600                    .iter()
601                    .flatten()
602                    .flatten()
603                    .map(|block_id| {
604                        data.palette
605                            .iter()
606                            .position(|(v, _)| v == block_id)
607                            .unwrap_or(0) as u32
608                    })
609                    .collect();
610
611                let block_data = pack_indices(&indices, bits);
612
613                PersistentSection::Heterogeneous {
614                    palette,
615                    bits_per_entry: bits,
616                    block_data,
617                    biomes,
618                }
619            }
620        }
621    }

623    /// Converts runtime biome data to persistent format.
624    fn biomes_to_persistent(
625        biomes: &PalettedContainer<u8, 4>,
626        builder: &mut ChunkBuilder,
627    ) -> PersistentBiomeData {
628        match biomes {
629            PalettedContainer::Homogeneous(biome_id) => {
630                let biome_idx = builder.ensure_biome(*biome_id);
631                PersistentBiomeData::Homogeneous { biome: biome_idx }
632            }
633            PalettedContainer::Heterogeneous(data) => {
634                // Build section-local palette (indices into chunk's biomes)
635                let palette: Vec<u16> = data
636                    .palette
637                    .iter()
638                    .map(|(biome_id, _)| builder.ensure_biome(*biome_id))
639                    .collect();
640
641                let bits = bits_for_palette_len(palette.len())
642                    .expect("Heterogeneous biome data should have palette length >= 2");
643                let indices: Vec<u32> = data
644                    .cube
645                    .iter()
646                    .flatten()
647                    .flatten()
648                    .map(|biome_id| {
649                        data.palette
650                            .iter()
651                            .position(|(v, _)| v == biome_id)
652                            .unwrap_or(0) as u32
653                    })
654                    .collect();
655
656                let biome_data = pack_indices(&indices, bits);
657
658                PersistentBiomeData::Heterogeneous {
659                    palette,
660                    bits_per_entry: bits,
661                    biome_data,
662                }
663            }
664        }
665    }

667    /// Converts a persistent chunk to runtime format.
668    /// The returned chunk is not dirty (freshly loaded from disk).
669    fn persistent_to_chunk(
670        &self,
671        persistent: &PersistentChunk,
672        pos: ChunkPos,
673        status: ChunkStatus,
674    ) -> ChunkAccess {
675        let sections: Vec<ChunkSection> = persistent
676            .sections
677            .iter()
678            .map(|section| self.persistent_to_section(section, persistent))
679            .collect();
680
681        match status {
682            ChunkStatus::Full => ChunkAccess::Full(LevelChunk::from_disk(
683                Sections {
684                    sections: sections.into_boxed_slice(),
685                },
686                pos,
687            )),
688            _ => ChunkAccess::Proto(ProtoChunk::from_disk(
689                Sections {
690                    sections: sections.into_boxed_slice(),
691                },
692                pos,
693            )),
694        }
695    }

697    /// Converts a persistent section to runtime format.
698    fn persistent_to_section(
699        &self,
700        persistent: &PersistentSection,
701        chunk: &PersistentChunk,
702    ) -> ChunkSection {
703        match persistent {
704            PersistentSection::Homogeneous {
705                block_state,
706                biomes,
707            } => {
708                let block_id = self.resolve_block_state(chunk, *block_state);
709                let biome_data = self.persistent_to_biomes(biomes, chunk);
710
711                ChunkSection {
712                    states: PalettedContainer::Homogeneous(block_id),
713                    biomes: biome_data,
714                }
715            }
716            PersistentSection::Heterogeneous {
717                palette,
718                bits_per_entry,
719                block_data,
720                biomes,
721            } => {
722                // Unpack indices (into section-local palette)
723                let indices = unpack_indices(block_data, *bits_per_entry, BLOCKS_PER_SECTION);
724
725                // Build runtime palette by resolving section-local -> chunk -> runtime
726                let runtime_palette: Vec<BlockStateId> = palette
727                    .iter()
728                    .map(|&idx| self.resolve_block_state(chunk, idx))
729                    .collect();
730
731                // Build cube
732                let mut cube = Box::new([[[BlockStateId(0); 16]; 16]; 16]);
733                for (i, &idx) in indices.iter().enumerate() {
734                    let y = i / 256;
735                    let z = (i / 16) % 16;
736                    let x = i % 16;
737                    cube[y][z][x] = runtime_palette
738                        .get(idx as usize)
739                        .copied()
740                        .unwrap_or(BlockStateId(0));
741                }
742
743                let states = PalettedContainer::from_cube(cube);
744                let biome_data = self.persistent_to_biomes(biomes, chunk);
745
746                ChunkSection {
747                    states,
748                    biomes: biome_data,
749                }
750            }
751        }
752    }
753
754    /// Converts persistent biome data to runtime format.
755    fn persistent_to_biomes(
756        &self,
757        persistent: &PersistentBiomeData,
758        chunk: &PersistentChunk,
759    ) -> PalettedContainer<u8, 4> {
760        match persistent {
761            PersistentBiomeData::Homogeneous { biome } => {
762                let biome_id = self.resolve_biome(chunk, *biome);
763                PalettedContainer::Homogeneous(biome_id)
764            }
765            PersistentBiomeData::Heterogeneous {
766                palette,
767                bits_per_entry,
768                biome_data,
769            } => {
770                let indices = unpack_indices(biome_data, *bits_per_entry, BIOMES_PER_SECTION);
771
772                // Resolve section-local palette -> chunk palette -> runtime
773                let runtime_palette: Vec<u8> = palette
774                    .iter()
775                    .map(|&idx| self.resolve_biome(chunk, idx))
776                    .collect();
777
778                let mut cube = Box::new([[[0u8; 4]; 4]; 4]);
779                for (i, &idx) in indices.iter().enumerate() {
780                    let y = i / 16;
781                    let z = (i / 4) % 4;
782                    let x = i % 4;
783                    cube[y][z][x] = runtime_palette.get(idx as usize).copied().unwrap_or(0);
784                }
785
786                PalettedContainer::from_cube(cube)
787            }
788        }
789    }

791    /// Resolves a chunk palette index to a runtime `BlockStateId`.
792    fn resolve_block_state(&self, chunk: &PersistentChunk, index: u16) -> BlockStateId {
793        if let Some(state) = chunk.block_states.get(index as usize) {
794            // Convert properties to the format expected by the registry
795            let properties: Vec<(&str, &str)> = state
796                .properties
797                .iter()
798                .map(|(k, v)| (k.as_str(), v.as_str()))
799                .collect();
800
801            if let Some(state_id) = self
802                .registry
803                .blocks
804                .state_id_from_properties(&state.name, &properties)
805            {
806                return state_id;
807            }
808        }
809        BlockStateId(0) // Air fallback
810    }

812    /// Resolves a chunk palette index to a runtime biome ID.
813    fn resolve_biome(&self, chunk: &PersistentChunk, index: u16) -> u8 {
814        if let Some(biome_key) = chunk.biomes.get(index as usize)
815            && let Some(id) = self.registry.biomes.id_from_key(biome_key)
816        {
817            return id as u8;
818        }
819        0 // Plains fallback
820    }
