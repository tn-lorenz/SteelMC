44/// Manages region files with seek-based chunk access.
45///
46/// Only keeps region headers (8KB each) in memory, not chunk data.
47/// Chunks are loaded on-demand and converted directly to runtime format.
48pub struct RegionManager {
49    /// Base directory for region files (e.g., "world/region").
50    base_path: PathBuf,
51    /// Open region file handles with their headers.
52    regions: AsyncRwLock<FxHashMap<RegionPos, RegionHandle>>,
53}

64/// An open region file with its header.
65struct RegionHandle {

94    /// Ensures a block state exists in the chunk's palette, returning its index.
95    fn ensure_block_state(&mut self, block_id: BlockStateId) -> u16 {
96        // Get block and properties from registry
97        let block = self
98            .registry
99            .blocks
100            .by_state_id(block_id)
101            .expect("Invalid block state ID");
102        let properties = self.registry.blocks.get_properties(block_id);
103
104        let persistent = PersistentBlockState {
105            name: block.key.clone(),
106            properties: properties
107                .into_iter()
108                .map(|(k, v)| (k.to_string(), v.to_string()))
109                .collect(),
110        };
111
112        // Check if already exists
113        if let Some(idx) = self.block_states.iter().position(|s| s == &persistent) {
114            return idx as u16;
115        }
116
117        // Add new entry
118        let idx = self.block_states.len();
119        self.block_states.push(persistent);
120        idx as u16
121    }
122
123    /// Ensures a biome exists in the chunk's palette, returning its index.
124    fn ensure_biome(&mut self, biome_id: u8) -> u16 {
125        // Get biome identifier from registry
126        let biome = self
127            .registry
128            .biomes
129            .by_id(biome_id as usize)
130            .expect("Invalid biome ID");
131        let identifier = biome.key.clone();
132
133        if let Some(idx) = self.biomes.iter().position(|b| b == &identifier) {
134            return idx as u16;
135        }
136
137        let idx = self.biomes.len();
138        self.biomes.push(identifier);
139        idx as u16
140    }
141}

156    /// Gets the file path for a region.
157    fn region_path(&self, pos: RegionPos) -> PathBuf {
158        self.base_path.join(pos.filename())
159    }

161    /// Opens or creates a region file, loading only the header.
162    async fn open_region(&self, pos: RegionPos) -> io::Result<RegionHandle> {
163        let path = self.region_path(pos);
164
165        if !path.exists() {
166            // Create new region file with empty header
167            return self.create_region(pos).await;
168        }
169
170        let mut file = OpenOptions::new()
171            .read(true)
172            .write(true)
173            .open(&path)
174            .await?;
175
176        // Read and verify magic + version
177        let mut header_bytes = [0u8; FILE_HEADER_SIZE];
178        file.read_exact(&mut header_bytes).await?;
179
180        let magic = &header_bytes[0..4];
181        if magic != REGION_MAGIC {
182            return Err(io::Error::new(
183                io::ErrorKind::InvalidData,
184                "Invalid region file magic",
185            ));
186        }
187
188        let version = u16::from_le_bytes([header_bytes[4], header_bytes[5]]);
189        if version > FORMAT_VERSION {
190            return Err(io::Error::new(
191                io::ErrorKind::InvalidData,
192                format!(
193                    "Region file version {version} is newer than supported version {FORMAT_VERSION}"
194                ),
195            ));
196        }
197
198        // Read chunk table
199        let mut table_bytes = vec![0u8; CHUNK_TABLE_SIZE];
200        file.read_exact(&mut table_bytes).await?;
201        let header = RegionHeader::from_bytes(&table_bytes);
202
203        // Calculate file size in sectors
204        let file_size = file.seek(io::SeekFrom::End(0)).await?;
205        let file_sectors = file_size.div_ceil(SECTOR_SIZE as u64) as u32;
206
207        Ok(RegionHandle {
208            file,
209            header,
210            loaded_chunk_count: 0,
211            header_dirty: false,
212            file_sectors,
213        })
214    }

216    /// Creates a new empty region file.
217    async fn create_region(&self, pos: RegionPos) -> io::Result<RegionHandle> {
218        fs::create_dir_all(&self.base_path).await?;
219
220        let path = self.region_path(pos);
221        let mut file = OpenOptions::new()
222            .read(true)
223            .write(true)
224            .create(true)
225            .truncate(true)
226            .open(&path)
227            .await?;
228
229        // Write header
230        let mut header_bytes = [0u8; FILE_HEADER_SIZE];
231        header_bytes[0..4].copy_from_slice(&REGION_MAGIC);
232        header_bytes[4..6].copy_from_slice(&FORMAT_VERSION.to_le_bytes());
233        file.write_all(&header_bytes).await?;
234
235        // Write empty chunk table
236        let header = RegionHeader::new();
237        file.write_all(&header.to_bytes()).await?;
238        file.flush().await?;
239
240        Ok(RegionHandle {
241            file,
242            header,
243            loaded_chunk_count: 0,
244            header_dirty: false,
245            file_sectors: FIRST_DATA_SECTOR,
246        })
247    }

249    /// Writes the header to disk.
250    async fn write_header(file: &mut File, header: &RegionHeader) -> io::Result<()> {
251        file.seek(io::SeekFrom::Start(FILE_HEADER_SIZE as u64))
252            .await?;
253        file.write_all(&header.to_bytes()).await?;
254        file.flush().await?;
255        Ok(())
256    }

258    /// Reads a chunk's compressed data from disk.
259    async fn read_chunk_data(
260        file: &mut File,
261        sector_offset: u32,
262        size: u32,
263    ) -> io::Result<Vec<u8>> {
264        let byte_offset = u64::from(sector_offset) * SECTOR_SIZE as u64;
265        file.seek(io::SeekFrom::Start(byte_offset)).await?;
266
267        let mut compressed = vec![0u8; size as usize];
268        file.read_exact(&mut compressed).await?;
269        Ok(compressed)
270    }

272    /// Writes chunk data to disk at the specified sector offset.
273    async fn write_chunk_data(
274        file: &mut File,
275        sector_offset: u32,
276        data: &[u8],
277        file_sectors: &mut u32,
278    ) -> io::Result<()> {
279        let byte_offset = u64::from(sector_offset) * SECTOR_SIZE as u64;
280        file.seek(io::SeekFrom::Start(byte_offset)).await?;
281        file.write_all(data).await?;
282
283        // Pad to sector boundary
284        let padding_needed = (SECTOR_SIZE - (data.len() % SECTOR_SIZE)) % SECTOR_SIZE;
285        if padding_needed > 0 {
286            file.write_all(&vec![0u8; padding_needed]).await?;
287        }
288
289        // Update file sectors if we wrote past the end
290        let sectors_used = data.len().div_ceil(SECTOR_SIZE) as u32;
291        let end_sector = sector_offset + sectors_used;
292        if end_sector > *file_sectors {
293            *file_sectors = end_sector;
294        }
295
296        file.flush().await?;
297        Ok(())
298    }

300    /// Saves a chunk to the appropriate region.
301    ///
302    /// The chunk is serialized, compressed, and written to disk immediately.
303    /// If the region was already open (has loaded chunks), the header update is
304    /// deferred. If this call opened the region, it will be closed after saving.
305    ///
306    /// If the chunk is not dirty, this is a no-op and returns `Ok(false)`.
307    /// Returns `Ok(true)` if the chunk was saved.
308    /// Prepares chunk data for saving. Call this while holding the chunk lock,
309    /// then pass the result to `save_chunk_data` after releasing the lock.
310    #[must_use]
311    pub fn prepare_chunk_save(chunk: &ChunkAccess) -> Option<PreparedChunkSave> {
312        if !chunk.is_dirty() {
313            return None;
314        }
315
316        let pos = chunk.pos();
317
318        // Get block entities if this is a full chunk
319        let block_entities: Vec<SharedBlockEntity> = chunk
320            .as_full()
321            .map(super::super::chunk::level_chunk::LevelChunk::get_block_entities)
322            .unwrap_or_default();
323
324        let persistent = Self::to_persistent(chunk.sections(), &block_entities, pos);
325
326        Some(PreparedChunkSave { pos, persistent })
327    }

329    /// Saves prepared chunk data to disk. This is the async part that doesn't
330    /// need to hold the chunk lock.
331    #[allow(clippy::missing_panics_doc)]
332    pub async fn save_chunk_data(
333        &self,
334        prepared: PreparedChunkSave,
335        status: ChunkStatus,
336    ) -> io::Result<bool> {
337        let pos = prepared.pos;
338        let region_pos = RegionPos::from_chunk(pos.0.x, pos.0.y);
339        let (local_x, local_z) = RegionPos::local_chunk_pos(pos.0.x, pos.0.y);
340        let index = RegionHeader::chunk_index(local_x, local_z);
341
342        let mut regions = self.regions.write().await;
343
344        // Track if we opened the region (so we can close it after)
345        let we_opened_region = !regions.contains_key(&region_pos);
346
347        // Get or open the region
348        let handle = if let Some(handle) = regions.get_mut(&region_pos) {
349            handle
350        } else {
351            let handle = self.open_region(region_pos).await?;
352            regions.insert(region_pos, handle);
353            regions.get_mut(&region_pos).expect("just inserted")
354        };
355
356        // Serialize the prepared data
357        let persistent = prepared.persistent;
358        let data = wincode::serialize(&persistent)
359            .map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e.to_string()))?;
360
361        // Compress with zstd
362        let compressed = zstd::encode_all(&data[..], 3)?;
363
364        if compressed.len() > MAX_CHUNK_SIZE {
365            // Clean up if we opened the region
366            if we_opened_region {
367                regions.remove(&region_pos);
368            }
369            return Err(io::Error::new(
370                io::ErrorKind::InvalidData,
371                format!(
372                    "Chunk too large: {} bytes (max {})",
373                    compressed.len(),
374                    MAX_CHUNK_SIZE
375                ),
376            ));
377        }
378
379        // Find space for the chunk
380        let sectors_needed = compressed.len().div_ceil(SECTOR_SIZE) as u32;
381        let old_entry = handle.header.entries[index];
382
383        // Try to reuse existing space if it fits
384        let sector_offset = if old_entry.exists() && old_entry.sector_count() >= sectors_needed {
385            old_entry.sector_offset
386        } else {
387            handle
388                .header
389                .find_free_sectors(sectors_needed, handle.file_sectors)
390        };
391
392        // Write chunk data
393        Self::write_chunk_data(
394            &mut handle.file,
395            sector_offset,
396            &compressed,
397            &mut handle.file_sectors,
398        )
399        .await?;
400
401        // Update header entry
402        handle.header.entries[index] =
403            super::format::ChunkEntry::new(sector_offset, compressed.len() as u32, status);
404
405        // If we opened this region and no chunks are loaded from it,
406        // write the header and close it immediately
407        if we_opened_region && handle.loaded_chunk_count == 0 {
408            Self::write_header(&mut handle.file, &handle.header).await?;
409            regions.remove(&region_pos);
410        } else {
411            handle.header_dirty = true;
412        }
413
414        Ok(true)
415    }

417    /// Loads a chunk from the appropriate region.
418    ///
419    /// Automatically opens the region if not already open. The region's reference
420    /// count is incremented, so you must call `release_chunk` when done with the chunk.
421    ///
422    /// Returns `Ok(None)` if the chunk doesn't exist on disk.
423    ///
424    /// # Arguments
425    /// * `pos` - The chunk position
426    /// * `min_y` - The minimum Y coordinate of the world
427    /// * `height` - The total height of the world
428    /// * `level` - Weak reference to the world for `LevelChunk`
429    ///
430    /// The region must already be acquired via `acquire_chunk` before calling this.
431    #[allow(clippy::missing_panics_doc)]
432    pub async fn load_chunk(
433        &self,
434        pos: ChunkPos,
435        min_y: i32,
436        height: i32,
437        level: Weak<World>,
438    ) -> io::Result<Option<(ChunkAccess, ChunkStatus)>> {
439        let region_pos = RegionPos::from_chunk(pos.0.x, pos.0.y);
440        let (local_x, local_z) = RegionPos::local_chunk_pos(pos.0.x, pos.0.y);
441        let index = RegionHeader::chunk_index(local_x, local_z);
442
443        let mut regions = self.regions.write().await;
444
445        // Get the region (should already be open via acquire_chunk)
446        let Some(handle) = regions.get_mut(&region_pos) else {
447            log::warn!("load_chunk called without acquire_chunk for region {region_pos:?}");
448            return Ok(None);
449        };
450
451        // Check if chunk exists
452        let entry = handle.header.entries[index];
453        if !entry.exists() {
454            return Ok(None);
455        }
456
457        // Read chunk data from disk
458        let compressed =
459            Self::read_chunk_data(&mut handle.file, entry.sector_offset, entry.size_bytes).await?;
460
461        // Decompress
462        let data = zstd::decode_all(&compressed[..])?;
463
464        // Deserialize
465        let persistent: PersistentChunk = wincode::deserialize(&data)
466            .map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e.to_string()))?;
467
468        // Convert to runtime format (persistent is dropped after this - no duplication!)
469        let status = entry.status;
470        let chunk = Self::persistent_to_chunk(&persistent, pos, status, min_y, height, level);
471
472        Ok(Some((chunk, status)))
473    }

475    /// Acquires a chunk, incrementing the region's reference count.
476    ///
477    /// This opens or creates the region file. Call this before loading or
478    /// generating a chunk, and call `release_chunk` when done with the chunk.
479    ///
480    /// Returns `Ok(true)` if the chunk exists on disk, `Ok(false)` if it doesn't.
481    #[allow(clippy::missing_panics_doc)]
482    pub async fn acquire_chunk(&self, pos: ChunkPos) -> io::Result<bool> {
483        let region_pos = RegionPos::from_chunk(pos.0.x, pos.0.y);
484        let (local_x, local_z) = RegionPos::local_chunk_pos(pos.0.x, pos.0.y);
485        let index = RegionHeader::chunk_index(local_x, local_z);
486
487        let mut regions = self.regions.write().await;
488
489        // Get or open/create the region
490        let handle = if let Some(handle) = regions.get_mut(&region_pos) {
491            handle
492        } else {
493            // open_region creates the file if it doesn't exist
494            let handle = self.open_region(region_pos).await?;
495            regions.insert(region_pos, handle);
496            regions.get_mut(&region_pos).expect("just inserted")
497        };
498
499        // Check if chunk exists
500        let exists = handle.header.entries[index].exists();
501
502        // Increment ref count
503        handle.loaded_chunk_count += 1;
504
505        Ok(exists)
506    }

508    /// Releases a loaded chunk, decrementing the region's reference count.
509    ///
510    /// When all chunks from a region are released, the header is saved (if dirty)
511    /// and the file handle is closed.
512    ///
513    /// This must be called for each chunk returned by `load_chunk`.
514    pub async fn release_chunk(&self, pos: ChunkPos) -> io::Result<()> {
515        let region_pos = RegionPos::from_chunk(pos.0.x, pos.0.y);
516
517        let mut regions = self.regions.write().await;
518
519        let should_close = if let Some(handle) = regions.get_mut(&region_pos) {
520            handle.loaded_chunk_count = handle.loaded_chunk_count.saturating_sub(1);
521            handle.loaded_chunk_count == 0
522        } else {
523            return Ok(());
524        };
525
526        if should_close
527            && let Some(mut handle) = regions.remove(&region_pos)
528            && handle.header_dirty
529        {
530            Self::write_header(&mut handle.file, &handle.header).await?;
531        }
532
533        Ok(())
534    }

536    /// Checks if a chunk exists on disk without loading it.
537    pub async fn chunk_exists(&self, pos: ChunkPos) -> io::Result<bool> {
538        let region_pos = RegionPos::from_chunk(pos.0.x, pos.0.y);
539        let (local_x, local_z) = RegionPos::local_chunk_pos(pos.0.x, pos.0.y);
540        let index = RegionHeader::chunk_index(local_x, local_z);
541
542        let regions = self.regions.write().await;
543
544        // Check cached header first
545        if let Some(handle) = regions.get(&region_pos) {
546            return Ok(handle.header.entries[index].exists());
547        }
548
549        drop(regions);
550
551        // Need to read header from disk
552        let path = self.region_path(region_pos);
553        if !path.exists() {
554            return Ok(false);
555        }
556
557        let mut file = File::open(&path).await?;
558
559        // Skip magic + version
560        file.seek(io::SeekFrom::Start(FILE_HEADER_SIZE as u64))
561            .await?;
562
563        // Read just the one entry we need (8 bytes at index * 8)
564        file.seek(io::SeekFrom::Current((index * 8) as i64)).await?;
565        let mut entry_bytes = [0u8; 8];
566        file.read_exact(&mut entry_bytes).await?;
567
568        let entry = super::format::ChunkEntry::from_bytes(entry_bytes);
569        Ok(entry.exists())
570    }

572    /// Flushes all dirty headers to disk.
573    pub async fn flush_all(&self) -> io::Result<()> {
574        let mut regions = self.regions.write().await;
575
576        for handle in regions.values_mut() {
577            if handle.header_dirty {
578                Self::write_header(&mut handle.file, &handle.header).await?;
579                handle.header_dirty = false;
580            }
581        }
582
583        Ok(())
584    }

586    /// Flushes all dirty headers and closes all region file handles.
587    ///
588    /// This should be called during graceful shutdown after all chunks have been saved.
589    /// It ensures all data is persisted and file handles are properly closed.
590    pub async fn close_all(&self) -> io::Result<()> {
591        let mut regions = self.regions.write().await;
592
593        for (_, mut handle) in regions.drain() {
594            if handle.header_dirty {
595                Self::write_header(&mut handle.file, &handle.header).await?;
596            }
597            // File handle is dropped here, closing the file
598        }
599
600        Ok(())
601    }

603    /// Converts chunk data to persistent format.
604    fn to_persistent(
605        sections: &Sections,
606        block_entities: &[SharedBlockEntity],
607        chunk_pos: ChunkPos,
608    ) -> PersistentChunk {
609        let mut builder = ChunkBuilder::new(&REGISTRY);
610
611        let persistent_sections = sections
612            .sections
613            .iter()
614            .map(|section| Self::section_to_persistent(section, &mut builder))
615            .collect();
616
617        // Serialize block entities
618        let persistent_block_entities: Vec<PersistentBlockEntity> = block_entities
619            .iter()
620            .map(|entity| {
621                let guard = entity.lock();
622                let pos = guard.get_block_pos();
623
624                // Serialize NBT data
625                let mut nbt = NbtCompound::new();
626                guard.save_additional(&mut nbt);
627                let mut nbt_bytes = Vec::new();
628                nbt.write(&mut nbt_bytes);
629
630                PersistentBlockEntity {
631                    x: (pos.0.x - chunk_pos.0.x * 16) as u8,
632                    y: pos.0.y as i16,
633                    z: (pos.0.z - chunk_pos.0.y * 16) as u8,
634                    entity_type: guard.get_type().key.clone(),
635                    nbt_data: nbt_bytes,
636                }
637            })
638            .collect();
639
640        PersistentChunk {
641            last_modified: SystemTime::now()
642                .duration_since(UNIX_EPOCH)
643                .map_or(0, |d| d.as_secs() as u32),
644            block_states: builder.block_states,
645            biomes: builder.biomes,
646            sections: persistent_sections,
647            block_entities: persistent_block_entities,
648        }
649    }

651    /// Converts a runtime section to persistent format.
652    fn section_to_persistent(
653        section: &SectionHolder,
654        builder: &mut ChunkBuilder,
655    ) -> PersistentSection {
656        let section = section.read();
657        let biomes = Self::biomes_to_persistent(&section.biomes, builder);
658
659        match &section.states {
660            PalettedContainer::Homogeneous(block_id) => {
661                let block_idx = builder.ensure_block_state(*block_id);
662                PersistentSection::Homogeneous {
663                    block_state: block_idx,
664                    biomes,
665                }
666            }
667            PalettedContainer::Heterogeneous(data) => {
668                // Build section-local palette (indices into chunk's block_states)
669                let palette: Vec<u16> = data
670                    .palette
671                    .iter()
672                    .map(|(block_id, _)| builder.ensure_block_state(*block_id))
673                    .collect();
674
675                // Pack block indices (indices into section-local palette)
676                let bits = bits_for_palette_len(palette.len())
677                    .expect("Heterogeneous section should have palette length >= 2");
678                let indices: Vec<u32> = data
679                    .cube
680                    .iter()
681                    .flatten()
682                    .flatten()
683                    .map(|block_id| {
684                        data.palette
685                            .iter()
686                            .position(|(v, _)| v == block_id)
687                            .unwrap_or(0) as u32
688                    })
689                    .collect();
690
691                let block_data = pack_indices(&indices, bits);
692
693                PersistentSection::Heterogeneous {
694                    palette,
695                    bits_per_entry: bits,
696                    block_data,
697                    biomes,
698                }
699            }
700        }
701    }

704    fn biomes_to_persistent(

706        builder: &mut ChunkBuilder,
707    ) -> PersistentBiomeData {

719                    .collect();

726                    .flatten()
727                    .flatten()
728                    .map(|biome_id| {
729                        data.palette
730                            .iter()
731                            .position(|(v, _)| v == biome_id)
732                            .unwrap_or(0) as u32
733                    })
734                    .collect();
735
736                let biome_data = pack_indices(&indices, bits);
737
738                PersistentBiomeData::Heterogeneous {
739                    palette,
740                    bits_per_entry: bits,

747    /// Converts a persistent chunk to runtime format.
748    /// The returned chunk is not dirty (freshly loaded from disk).
749    ///
750    /// # Arguments
751    /// * `persistent` - The persistent chunk data
752    /// * `pos` - The chunk position
753    /// * `status` - The chunk status
754    /// * `min_y` - The minimum Y coordinate of the world
755    /// * `height` - The total height of the world
756    /// * `level` - Weak reference to the world for `LevelChunk`
757    fn persistent_to_chunk(
758        persistent: &PersistentChunk,
759        pos: ChunkPos,
760        status: ChunkStatus,
761        min_y: i32,
762        height: i32,
763        level: Weak<World>,
764    ) -> ChunkAccess {
765        let sections: Vec<ChunkSection> = persistent
766            .sections
767            .iter()
768            .map(|section| Self::persistent_to_section(section, persistent))
769            .collect();
770
771        match status {
772            ChunkStatus::Full => {
773                let chunk = LevelChunk::from_disk(
774                    Sections::from_owned(sections.into_boxed_slice()),
775                    pos,
776                    min_y,
777                    height,
778                    level,
779                );
780
781                // Load block entities
782                for persistent_be in &persistent.block_entities {
783                    if let Some(block_entity) =
784                        Self::persistent_to_block_entity(persistent_be, pos, &chunk)
785                    {
786                        chunk.add_and_register_block_entity(block_entity);
787                    }
788                }
789
790                // Clear dirty flag since we just loaded (add_and_register marks dirty)
791                chunk.dirty.store(false, Ordering::Release);
792
793                ChunkAccess::Full(chunk)
794            }
795            _ => ChunkAccess::Proto(ProtoChunk::from_disk(
796                Sections::from_owned(sections.into_boxed_slice()),
797                pos,
798                status,
799                min_y,
800                height,
801            )),
802        }
803    }

805    /// Converts a persistent block entity to runtime format.
806    fn persistent_to_block_entity(
807        persistent: &PersistentBlockEntity,
808        chunk_pos: ChunkPos,
809        chunk: &LevelChunk,
810    ) -> Option<SharedBlockEntity> {
811        // Calculate absolute position
812        let abs_x = chunk_pos.0.x * 16 + i32::from(persistent.x);
813        let abs_z = chunk_pos.0.y * 16 + i32::from(persistent.z);
814        let pos = BlockPos::new(abs_x, i32::from(persistent.y), abs_z);
815
816        // Get the block state at this position
817        let state = chunk.get_block_state(pos);
818
819        // Look up the block entity type
820        let block_entity_type = REGISTRY
821            .block_entity_types
822            .by_key(&persistent.entity_type)?;
823
824        // Get the world reference from the chunk
825        let level = chunk.level_weak();
826
827        // Parse and load NBT data
828        if persistent.nbt_data.is_empty() {
829            // No NBT data, just create the entity without loading
830            BLOCK_ENTITIES.create(block_entity_type, level, pos, state)
831        } else {
832            // Parse NBT from bytes as borrowed
833            let Ok(nbt) = read_borrowed_compound(&mut Cursor::new(&persistent.nbt_data)) else {
834                return BLOCK_ENTITIES.create(block_entity_type, level, pos, state);
835            };
836
837            // Create the block entity and load NBT
838            BLOCK_ENTITIES.create_and_load(block_entity_type, level, pos, state, &nbt)
839        }
840    }

842    /// Converts a persistent section to runtime format.
843    fn persistent_to_section(
844        persistent: &PersistentSection,
845        chunk: &PersistentChunk,
846    ) -> ChunkSection {
847        match persistent {
848            PersistentSection::Homogeneous {
849                block_state,
850                biomes,
851            } => {
852                let block_id = Self::resolve_block_state(chunk, *block_state);
853                let biome_data = Self::persistent_to_biomes(biomes, chunk);
854
855                ChunkSection::new_with_biomes(PalettedContainer::Homogeneous(block_id), biome_data)
856            }
857            PersistentSection::Heterogeneous {
858                palette,
859                bits_per_entry,
860                block_data,
861                biomes,
862            } => {
863                // Unpack indices (into section-local palette)
864                let indices = unpack_indices(block_data, *bits_per_entry, BLOCKS_PER_SECTION);
865
866                // Build runtime palette by resolving section-local -> chunk -> runtime
867                let runtime_palette: Vec<BlockStateId> = palette
868                    .iter()
869                    .map(|&idx| Self::resolve_block_state(chunk, idx))
870                    .collect();
871
872                // Build cube
873                let mut cube = Box::new([[[BlockStateId(0); 16]; 16]; 16]);
874                for (i, &idx) in indices.iter().enumerate() {
875                    let y = i / 256;
876                    let z = (i / 16) % 16;
877                    let x = i % 16;
878                    cube[y][z][x] = runtime_palette
879                        .get(idx as usize)
880                        .copied()
881                        .unwrap_or(BlockStateId(0));
882                }
883
884                let states = PalettedContainer::from_cube(cube);
885                let biome_data = Self::persistent_to_biomes(biomes, chunk);
886
887                ChunkSection::new_with_biomes(states, biome_data)
888            }
889        }
890    }

892    /// Converts persistent biome data to runtime format.
893    fn persistent_to_biomes(
894        persistent: &PersistentBiomeData,
895        chunk: &PersistentChunk,
896    ) -> PalettedContainer<u8, 4> {
897        match persistent {
898            PersistentBiomeData::Homogeneous { biome } => {
899                let biome_id = Self::resolve_biome(chunk, *biome);
900                PalettedContainer::Homogeneous(biome_id)
901            }
902            PersistentBiomeData::Heterogeneous {
903                palette,
904                bits_per_entry,
905                biome_data,
906            } => {
907                let indices = unpack_indices(biome_data, *bits_per_entry, BIOMES_PER_SECTION);
908
909                // Resolve section-local palette -> chunk palette -> runtime
910                let runtime_palette: Vec<u8> = palette
911                    .iter()
912                    .map(|&idx| Self::resolve_biome(chunk, idx))
913                    .collect();
914
915                let mut cube = Box::new([[[0u8; 4]; 4]; 4]);
916                for (i, &idx) in indices.iter().enumerate() {
917                    let y = i / 16;
918                    let z = (i / 4) % 4;
919                    let x = i % 4;
920                    cube[y][z][x] = runtime_palette.get(idx as usize).copied().unwrap_or(0);
921                }
922
923                PalettedContainer::from_cube(cube)
924            }
925        }
926    }

928    /// Resolves a chunk palette index to a runtime `BlockStateId`.
929    fn resolve_block_state(chunk: &PersistentChunk, index: u16) -> BlockStateId {
930        if let Some(state) = chunk.block_states.get(index as usize) {
931            // Convert properties to the format expected by the registry
932            let properties: Vec<(&str, &str)> = state
933                .properties
934                .iter()
935                .map(|(k, v)| (k.as_str(), v.as_str()))
936                .collect();
937
938            if let Some(state_id) = REGISTRY
939                .blocks
940                .state_id_from_properties(&state.name, &properties)
941            {
942                return state_id;
943            }
944        }
945        BlockStateId(0) // Air fallback
946    }

948    /// Resolves a chunk palette index to a runtime biome ID.
949    fn resolve_biome(chunk: &PersistentChunk, index: u16) -> u8 {
950        if let Some(biome_key) = chunk.biomes.get(index as usize)
951            && let Some(id) = REGISTRY.biomes.id_from_key(biome_key)
952        {
953            return id as u8;
954        }
955        0 // Plains fallback
956    }
