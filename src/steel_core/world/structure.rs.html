1//! Structure start and reference types for chunk-level structure tracking.
2//!
3//! In vanilla, chunks store two maps:
4//! - `structureStarts`: structures originating in this chunk
5//! - `structuresReferences`: references to structures from nearby chunks
6//!
7//! Structure generation is not yet implemented. These types support
8//! persistence, the proto-chunk → level-chunk data flow, and future generation.
9//! The structure key is `Identifier` until a structure registry is added.
10
11use rustc_hash::FxHashMap;
12
13use steel_utils::{BoundingBox, ChunkPos, Direction, Identifier};
14
15/// A structure start placed in a chunk.
16///
17/// Corresponds to vanilla's `StructureStart`. A start is "valid" if it has
18/// at least one piece; invalid starts are not stored (they correspond to
19/// vanilla's `INVALID_START` sentinel).
20#[derive(Debug, Clone)]
21pub struct StructureStart {
22    /// The structure type identifier (e.g., `minecraft:village`).
23    pub structure: Identifier,
24    /// The chunk where this structure originates.
25    pub chunk_pos: ChunkPos,
26    /// How many neighboring chunks reference this start.
27    pub references: i32,
28    /// The pieces composing this structure.
29    pub pieces: Vec<StructurePiece>,
30}
31
32/// A single piece of a structure.
33///
34/// Corresponds to vanilla's `StructurePiece`. Type-specific data is stored
35/// as an NBT blob since there are 56+ piece types in vanilla.
36#[derive(Debug, Clone)]
37pub struct StructurePiece {
38    /// Piece type identifier (e.g., `minecraft:jigsaw`).
39    pub piece_type: Identifier,
40    /// World-space bounding box of this piece.
41    pub bounding_box: BoundingBox,
42    /// Generation depth (distance from start piece in the piece tree).
43    pub gen_depth: i32,
44    /// Horizontal orientation of this piece (`None` for unoriented pieces).
45    /// Only horizontal directions (North/South/East/West) are used.
46    pub orientation: Option<Direction>,
47    /// Type-specific NBT data (simdnbt binary format).
48    pub nbt_data: Vec<u8>,
49}
50
51/// Map of structure starts keyed by structure identifier.
52pub type StructureStartMap = FxHashMap<Identifier, StructureStart>;
53
54/// Map of structure references keyed by structure identifier.
55/// Values are the chunk positions of origin chunks that contain the structure start.
56pub type StructureReferenceMap = FxHashMap<Identifier, Vec<ChunkPos>>;
