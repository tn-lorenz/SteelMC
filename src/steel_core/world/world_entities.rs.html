1//! This module contains the implementation of the world's entity-related methods.
2use std::sync::Arc;
3
4use steel_protocol::packets::game::{CGameEvent, GameEventType};
5use tokio::time::Instant;
6
7use crate::{player::Player, world::World};
8
9impl World {
10    /// Removes a player from the world.
11    pub async fn remove_player(self: &Arc<Self>, player: Arc<Player>) {
12        let uuid = player.gameprofile.id;
13
14        if self.players.remove_async(&uuid).await.is_some() {
15            let self_clone = self.clone();
16            let start = Instant::now();
17            self_clone.chunk_map.remove_player(&player);
18            player.cleanup();
19            log::info!("Player {uuid} removed in {:?}", start.elapsed());
20        }
21    }
22
23    /// Adds a player to the world.
24    pub fn add_player(self: &Arc<Self>, player: Arc<Player>) {
25        if self
26            .players
27            .insert_sync(player.gameprofile.id, player.clone())
28            .is_err()
29        {
30            player.connection.close();
31            return;
32        }
33
34        // Send existing players to the new player (ADD_PLAYER without chat sessions yet)
35        // The chat sessions will be sent separately when they become available
36        self.players.iter_sync(|_, existing_player| {
37            if existing_player.gameprofile.id != player.gameprofile.id {
38                let add_existing = steel_protocol::packets::game::CPlayerInfoUpdate::add_player(
39                    existing_player.gameprofile.id,
40                    existing_player.gameprofile.name.clone(),
41                );
42                player.connection.send_packet(add_existing);
43
44                // If the existing player has a chat session, send it too
45                if let Some(session) = existing_player.chat_session()
46                    && let Ok(protocol_data) = session.as_data().to_protocol_data()
47                {
48                    let session_packet =
49                        steel_protocol::packets::game::CPlayerInfoUpdate::update_chat_session(
50                            existing_player.gameprofile.id,
51                            protocol_data,
52                        );
53                    player.connection.send_packet(session_packet);
54                }
55            }
56            true
57        });
58
59        // Broadcast new player to all existing players (ADD_PLAYER)
60        let player_info_packet = steel_protocol::packets::game::CPlayerInfoUpdate::add_player(
61            player.gameprofile.id,
62            player.gameprofile.name.clone(),
63        );
64
65        self.players.iter_sync(|_, p| {
66            p.connection.send_packet(player_info_packet.clone());
67            true
68        });
69
70        player.connection.send_packet(CGameEvent {
71            event: GameEventType::LevelChunksLoadStart,
72            data: 0.0,
73        });
74
75        player.connection.send_packet(CGameEvent {
76            event: GameEventType::ChangeGameMode,
77            data: 1.0,
78        });
79    }
80}
