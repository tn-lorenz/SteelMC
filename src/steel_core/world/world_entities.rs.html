18    /// Removes a player from the world.
19    pub async fn remove_player(self: &Arc<Self>, player: Arc<Player>) {
20        let uuid = player.gameprofile.id;
21        let entity_id = player.id;
22
23        if self.players.remove(&uuid).await.is_some() {
24            let start = Instant::now();
25
26            // Save player data before removal
27            if let Some(server) = player.server.upgrade()
28                && let Err(e) = server.player_data_storage.save(&player).await
29            {
30                log::error!("Failed to save player data for {uuid}: {e}");
31            }
32
33            // Unregister from entity cache
34            let pos = player.position();
35            let section = steel_utils::SectionPos::new(
36                (pos.x as i32) >> 4,
37                (pos.y as i32) >> 4,
38                (pos.z as i32) >> 4,
39            );
40            self.entity_cache.unregister(entity_id, uuid, section);
41
42            // Remove player from entity tracking (stop tracking all entities for this player)
43            self.entity_tracker().on_player_leave(entity_id);
44
45            self.player_area_map.on_player_leave(&player);
46            self.broadcast_to_all(CRemoveEntities::single(entity_id));
47            self.broadcast_to_all(CRemovePlayerInfo::single(uuid));
48
49            self.chunk_map.remove_player(&player);
50            player.cleanup();
51            log::info!("Player {uuid} removed in {:?}", start.elapsed());
52        }
53    }

55    /// Adds a player to the world.
56    pub fn add_player(self: &Arc<Self>, player: Arc<Player>) {
57        if !self.players.insert(player.clone()) {
58            player.connection.close();
59            return;
60        }
61
62        // Set up level callback for section tracking
63        let pos = player.position();
64        let callback = Arc::new(PlayerEntityCallback::new(
65            player.id,
66            pos,
67            Arc::downgrade(self),
68        ));
69        player.set_level_callback(callback);
70
71        // Register player in entity cache for unified entity lookups
72        self.entity_cache
73            .register(&(player.clone() as SharedEntity));
74
75        // Note: player_area_map.on_player_join is called in chunk_map.update_player_status
76        // when the player's view is first computed
77
78        let pos = *player.position.lock();
79        let (yaw, pitch) = player.rotation.load();
80
81        // Send existing players to the new player (tab list + entity spawn)
82        self.players.iter_players(|_, existing_player| {
83            if existing_player.gameprofile.id != player.gameprofile.id {
84                // Add to tab list with full player info
85                let add_existing = CPlayerInfoUpdate::create_player_initializing(
86                    existing_player.gameprofile.id,
87                    existing_player.gameprofile.name.clone(),
88                    existing_player.gameprofile.properties.clone(),
89                    existing_player.game_mode.load().into(),
90                    existing_player.connection.latency(),
91                    None, // display_name
92                    true, // show_hat
93                );
94                player.send_packet(add_existing);
95
96                // Send chat session if available
97                if let Some(session) = existing_player.chat_session()
98                    && let Ok(protocol_data) = session.as_data().to_protocol_data()
99                {
100                    let session_packet = CPlayerInfoUpdate::update_chat_session(
101                        existing_player.gameprofile.id,
102                        protocol_data,
103                    );
104                    player.send_packet(session_packet);
105                }
106
107                // Spawn existing player entity for new player (bundled for atomic processing)
108                let existing_pos = *existing_player.position.lock();
109                let (existing_yaw, existing_pitch) = existing_player.rotation.load();
110                let player_type_id = *REGISTRY.entity_types.get_id(vanilla_entities::PLAYER) as i32;
111                player.send_bundle(|bundle| {
112                    bundle.add(CAddEntity::player(
113                        existing_player.id,
114                        existing_player.gameprofile.id,
115                        player_type_id,
116                        existing_pos.x,
117                        existing_pos.y,
118                        existing_pos.z,
119                        existing_yaw,
120                        existing_pitch,
121                    ));
122                    // TODO: Add entity metadata and equipment packets here when implemented
123                });
124            }
125            true
126        });
127
128        // Broadcast new player to all existing players (tab list + entity spawn)
129        let player_info_packet = CPlayerInfoUpdate::create_player_initializing(
130            player.gameprofile.id,
131            player.gameprofile.name.clone(),
132            player.gameprofile.properties.clone(),
133            player.game_mode.load().into(),
134            player.connection.latency(),
135            None, // display_name
136            true, // show_hat
137        );
138        let player_type_id = *REGISTRY.entity_types.get_id(vanilla_entities::PLAYER) as i32;
139        let spawn_packet = CAddEntity::player(
140            player.id,
141            player.gameprofile.id,
142            player_type_id,
143            pos.x,
144            pos.y,
145            pos.z,
146            yaw,
147            pitch,
148        );
149
150        self.players.iter_players(|_, p| {
151            p.send_packet(player_info_packet.clone());
152            // Don't send spawn packet to self
153            if p.gameprofile.id != player.gameprofile.id {
154                // Bundle spawn packet for atomic processing
155                p.send_bundle(|bundle| {
156                    bundle.add(spawn_packet.clone());
157                    // TODO: Add entity metadata and equipment packets here when implemented
158                });
159            }
160            true
161        });
162
163        player.send_packet(CGameEvent {
164            event: GameEventType::LevelChunksLoadStart,
165            data: 0.0,
166        });
167
168        player.send_packet(CGameEvent {
169            event: GameEventType::ChangeGameMode,
170            data: player.game_mode.load().into(),
171        });
172    }
