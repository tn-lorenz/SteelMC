19/// A struct that represents a world.
20pub struct World {
21    /// The chunk map of the world.
22    pub chunk_map: Arc<ChunkMap>,
23    /// A map of all the players in the world.
24    pub players: HashMap<Uuid, Arc<Player>>,
25}
26
27impl World {
28    /// Creates a new world.
29    #[allow(clippy::new_without_default)]
30    #[must_use]
31    pub fn new(registry: &Arc<Registry>, chunk_runtime: Arc<Runtime>) -> Self {
32        Self {
33            chunk_map: Arc::new(ChunkMap::new(registry, chunk_runtime)),
34            players: HashMap::new(),
35        }
36    }
37
38    /// Ticks the world.
39    pub fn tick_b(&self, tick_count: u64) {
40        self.chunk_map.tick_b(tick_count);
41
42        // Tick players
43        let start = tokio::time::Instant::now();
44        self.players.iter_sync(|_uuid, player| {
45            player.tick();
46
47            true
48        });
49        let player_tick_elapsed = start.elapsed();
50        if player_tick_elapsed >= Duration::from_millis(100) {
51            log::warn!("Player tick slow: {player_tick_elapsed:?}");
52        }
53    }
54
55    /// Broadcasts a signed chat message to all players in the world.
56    ///
57    /// # Panics
58    /// Panics if `message_signature` is `None` after checking `is_some()` (should never happen).
59    pub fn broadcast_chat(
60        &self,
61        mut packet: CPlayerChat,
62        _sender: Arc<Player>,
63        sender_last_seen: LastSeen,
64        message_signature: Option<[u8; 256]>,
65    ) {
66        log::debug!(
67            "broadcast_chat: sender_last_seen has {} signatures, message_signature present: {}",
68            sender_last_seen.len(),
69            message_signature.is_some()
70        );
71
72        self.players.iter_sync(|_, recipient| {
73            let messages_received = recipient.get_and_increment_messages_received();
74            packet.global_index = VarInt(messages_received);
75
76            log::debug!(
77                "Broadcasting to player {} (UUID: {}), global_index={}",
78                recipient.gameprofile.name,
79                recipient.gameprofile.id,
80                messages_received
81            );
82
83            // IMPORTANT: Index previous messages BEFORE updating the cache
84            // This matches vanilla's order: pack() then push()
85            let previous_messages = {
86                let recipient_cache = recipient.signature_cache.lock();
87                recipient_cache.index_previous_messages(&sender_last_seen)
88            };
89
90            log::debug!(
91                "  Indexed {} previous messages for recipient",
92                previous_messages.len()
93            );
94
95            packet.previous_messages.clone_from(&previous_messages);
96
97            // Send the packet
98            recipient.connection.send_packet(packet.clone());
99
100            // AFTER sending, update the recipient's cache using vanilla's push algorithm
101            // This adds all lastSeen signatures + current signature to the cache
102            if let Some(signature) = message_signature {
103                recipient
104                    .signature_cache
105                    .lock()
106                    .push(&sender_last_seen, Some(&signature));
107
108                log::debug!("  Added signature to recipient's cache and pending list");
109
110                // Add to pending messages for acknowledgment tracking
111                recipient
112                    .message_validator
113                    .lock()
114                    .add_pending(Some(Box::new(signature) as Box<[u8]>));
115            } else {
116                // Even unsigned messages update the pending tracker
117                recipient.message_validator.lock().add_pending(None);
118                log::debug!("  Added unsigned message to pending list");
119            }
120
121            true
122        });
123    }
124
125    /// Broadcasts a system chat message to all players.
126    pub fn broadcast_system_chat(&self, packet: CSystemChat) {
127        self.players.iter_sync(|_, player| {
128            player.connection.send_packet(packet.clone());
129            true
130        });
131    }
132
133    /// Broadcasts an unsigned player chat message to all players.
134    pub fn broadcast_unsigned_chat(
135        &self,
136        mut packet: CPlayerChat,
137        sender_name: &str,
138        message: &str,
139    ) {
140        log::info!("<{sender_name}> {message}");
141
142        self.players.iter_sync(|_, recipient| {
143            let messages_received = recipient.get_and_increment_messages_received();
144            packet.global_index = VarInt(messages_received);
145
146            recipient.connection.send_packet(packet.clone());
147            true
148        });
149    }
