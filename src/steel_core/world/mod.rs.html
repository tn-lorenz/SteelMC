229fn portal_candidate_distance_sqr(candidate: BlockPos, center: BlockPos) -> i64 {
230    let dx = i64::from(candidate.x()) - i64::from(center.x());
231    let dy = i64::from(candidate.y()) - i64::from(center.y());
232    let dz = i64::from(candidate.z()) - i64::from(center.z());
233    dx * dx + dy * dy + dz * dz
234}

236fn dist_to_origin_center_sqr(pos: BlockPos) -> f64 {
237    let x = f64::from(pos.x()) + 0.5;
238    let y = f64::from(pos.y()) + 0.5;
239    let z = f64::from(pos.z()) + 0.5;
240    x * x + y * y + z * z
241}

243fn closest_portal_candidate(
244    candidates: impl IntoIterator<Item = BlockPos>,
245    approximate_exit_pos: BlockPos,
246    is_valid: impl Fn(BlockPos) -> bool,
247) -> Option<BlockPos> {
248    candidates
249        .into_iter()
250        .filter(|pos| is_valid(*pos))
251        .min_by_key(|pos| {
252            (
253                portal_candidate_distance_sqr(*pos, approximate_exit_pos),
254                pos.y(),
255            )
256        })
257}

267    origin: BlockPos,
268    direction: Direction,
269    width: i32,
270    height: i32,
271    offset: i32,
272) -> BlockPos {
273    let clockwise = direction.rotate_y_clockwise();
274    let (direction_x, _, direction_z) = direction.offset();
275    let (clockwise_x, _, clockwise_z) = clockwise.offset();
276    origin.offset(
277        direction_x * width + clockwise_x * offset,
278        height,
279        direction_z * width + clockwise_z * offset,
280    )

430    ///
431    /// This allows specifying storage backend (disk or RAM-only) and other options.
432    /// Uses `Arc::new_cyclic` to create a cyclic reference between
433    /// the World and its `ChunkMap`'s `WorldGenContext`.
434    ///
435    /// # Arguments
436    /// * `chunk_runtime` - The Tokio runtime for chunk operations
437    /// * `dimension_type` - Vanilla dimension type (overworld, nether, end)

441        chunk_runtime: Arc<Runtime>,
442        key: Identifier,
443        dimension_type: DimensionTypeRef,
444        seed: i64,
445        config: WorldConfig,
446        generation_pool: Arc<rayon::ThreadPool>,
447    ) -> io::Result<Arc<Self>> {
448        let view_distance = config.view_distance;
449        let simulation_distance = config.simulation_distance;
450        let compression = config.compression;
451        let is_flat = config.is_flat;
452        let sea_level = config.sea_level;
453        let default_gamemode = config.default_gamemode;
454        // Create storage backend based on config
455        let storage: Arc<ChunkStorage> = match &config.storage {
456            WorldStorageConfig::Disk { path } => {
457                Arc::new(ChunkStorage::Disk(RegionManager::new(path.clone())))
458            }
459            WorldStorageConfig::RamOnly => {
460                Arc::new(ChunkStorage::RamOnly(RamOnlyStorage::empty_world()))
461            }
462        };
463
464        // Create or skip level data based on config
465
466        let path = config.level_data_path.as_deref().map(Path::new);
467        let saved_data = SavedDataManager::new(path);
468        let mut level_data =
469            LevelDataManager::new(path, seed, config.difficulty, config.generation_settings)

471        if level_data.is_dirty() {

473        }
474        let persistent_chunk_tickets: PersistentChunkTickets = saved_data
475            .load_or_default(saved_data_names::CHUNK_TICKETS)

477        let timed_chunk_tickets = TimedChunkTickets::from_persistent(persistent_chunk_tickets);
478        let world_border = WorldBorder::new(level_data.data().world_border)

485        //     REGISTRY

492            weather.rain_level = 1.0;

494                weather.thunder_level = 1.0;
495            }
496        }
497
498        Ok(Arc::new_cyclic(|weak_self: &Weak<World>| {
499            let chunk_map = Arc::new(ChunkMap::new_with_storage_and_timed_tickets(
500                chunk_runtime,
501                weak_self.clone(),
502                dimension_type,
503                sea_level,
504                storage,
505                config.generator,
506                generation_pool,
507                timed_chunk_tickets,
508            ));
509            chunk_map.start_generation_refill_loop();
510
511            Self {
512                chunk_map,
513                players: PlayerMap::new(),
514                player_area_map: PlayerAreaMap::new(),
515                key,
516                dimension_type,
517                level_data: SyncRwLock::new(level_data),
518                saved_data,
519                world_border: SyncMutex::new(world_border),
520                view_distance,
521                simulation_distance,
522                compression,
523                is_flat,
524                sea_level,
525                default_gamemode,
526                tick_runs_normally: AtomicBool::new(true),
527                entity_manager: WorldEntityManager::new(),
528                entity_tracker: EntityTracker::new(),
529                navigating_mobs: NavigatingMobTracker::new(),
530                weather: SyncMutex::new(weather),
531                sub_tick_count: AtomicI64::new(0),
532                poi_storage: SyncMutex::new(PointOfInterestStorage::new()),

590    /// Finds the closest existing Nether portal POI using vanilla `PortalForcer` ordering.
591    ///
592    /// `to_nether` selects vanilla's 16-block Nether search radius; non-Nether targets use 128.
593    ///
594    /// # Panics
595    ///
596    /// Panics if vanilla POI registries were not initialized before portal lookup.
597    #[must_use]
598    pub fn find_closest_nether_portal_position(
599        &self,
600        approximate_exit_pos: BlockPos,
601        to_nether: bool,
602    ) -> Option<BlockPos> {
603        let radius = if to_nether { 16 } else { 128 };
604        let nether_portal_type = vanilla_poi_types::NETHER_PORTAL
605            .try_id()
606            .expect("vanilla nether portal POI type should be registered");
607        let candidates = self.poi_storage.lock().get_in_horizontal_square(
608            &|type_id| type_id == nether_portal_type,
609            approximate_exit_pos,
610            radius,
611            OccupationStatus::Any,
612        );
613
614        closest_portal_candidate(
615            candidates.into_iter().map(|(pos, _)| pos),
616            approximate_exit_pos,
617            |pos| {
618                self.is_block_within_world_border(pos)

627    /// Creates a Nether portal using vanilla `PortalForcer.createPortal` placement rules.
628    ///
629    /// The caller must keep the target search area loaded as full chunks before calling. Steel
630    /// returns `None` if any required chunk read or write is unavailable, rather than treating
631    /// unloaded chunks as replaceable air.
632    #[must_use]
633    pub fn create_nether_portal(
634        self: &Arc<Self>,
635        origin: BlockPos,
636        portal_axis: Axis,
637    ) -> Option<FoundRectangle> {
638        if portal_axis == Axis::Y {
639            return None;
640        }
641
642        let direction = Direction::positive_for_axis(portal_axis);
643        let max_placeable_y = self
644            .get_max_y()
645            .min(self.get_min_y() + self.dimension_type.logical_height - 1);
646
647        let portal_origin =
648            match self.find_nether_portal_creation_position(origin, direction, max_placeable_y) {
649                Ok(Some(pos)) => pos,
650                Ok(None) => {
651                    let fallback =
652                        self.fallback_nether_portal_position(origin, direction, max_placeable_y)?;
653                    if !self.can_write_nether_portal_fallback_box(fallback, direction) {
654                        return None;
655                    }
656                    if !self.clear_nether_portal_fallback_box(fallback, direction) {
657                        return None;
658                    }
659                    fallback
660                }
661                Err(MissingPortalCreationChunk) => return None,
662            };
663
664        if !self.can_write_nether_portal_rectangle(portal_origin, direction) {
665            return None;
666        }
667        if !self.place_nether_portal_frame_and_blocks(portal_origin, direction, portal_axis) {
668            return None;
669        }
670
671        Some(FoundRectangle {
672            min_corner: portal_origin,
673            axis1_size: 2,
674            axis2_size: 3,
675        })
676    }

683    fn find_nether_portal_creation_position(
684        &self,
685        origin: BlockPos,
686        direction: Direction,
687        max_placeable_y: i32,
688    ) -> Result<Option<BlockPos>, MissingPortalCreationChunk> {
689        let mut closest_full_position: Option<(i64, BlockPos)> = None;
690        let mut closest_partial_position: Option<(i64, BlockPos)> = None;
691        let border = self.world_border_snapshot();
692
693        for column_pos in BlockPos::spiral_around(
694            origin,
695            NETHER_PORTAL_CREATE_RADIUS,
696            Direction::East,
697            Direction::South,
698        ) {
699            let height = self
700                .height_at(
701                    HeightmapType::MotionBlocking,
702                    column_pos.x(),
703                    column_pos.z(),
704                )
705                .ok_or(MissingPortalCreationChunk)?
706                .min(max_placeable_y);
707            if !border.is_block_within_bounds(column_pos)
708                || !border.is_block_within_bounds(column_pos.relative(direction))
709            {
710                continue;
711            }
712
713            let mut column_pos = nether_portal_creation_scan_origin(column_pos, direction, height);
714            let mut y = height;
715            while y >= self.get_min_y() {
716                column_pos = column_pos.at_y(y);
717                if self.can_nether_portal_replace_block(column_pos)? {
718                    let first_empty_y = y;
719
720                    while y > self.get_min_y()
721                        && self.can_nether_portal_replace_block(column_pos.below())?
722                    {
723                        y -= 1;
724                        column_pos = column_pos.below();
725                    }
726
727                    if y + 4 <= max_placeable_y {
728                        let delta_y = first_empty_y - y;
729                        if (delta_y <= 0 || delta_y >= 3)
730                            && self.can_host_nether_portal_frame(column_pos, direction, 0)?
731                        {
732                            let distance = portal_candidate_distance_sqr(column_pos, origin);
733                            let full_frame = self
734                                .can_host_nether_portal_frame(column_pos, direction, -1)?
735                                && self.can_host_nether_portal_frame(column_pos, direction, 1)?;
736
737                            if full_frame
738                                && closest_full_position
739                                    .is_none_or(|(closest_distance, _)| closest_distance > distance)
740                            {
741                                closest_full_position = Some((distance, column_pos));
742                            }
743
744                            if closest_full_position.is_none()
745                                && closest_partial_position
746                                    .is_none_or(|(closest_distance, _)| closest_distance > distance)
747                            {
748                                closest_partial_position = Some((distance, column_pos));
749                            }
750                        }
751                    }
752                }
753
754                y -= 1;
755            }
756        }
757
758        if closest_full_position.is_none() {
759            closest_full_position = closest_partial_position;
760        }
761
762        Ok(closest_full_position.map(|(_, pos)| pos))
763    }

1418    /// Initializes this world's default spawn using vanilla's first-world spawn search.
1419    pub async fn initialize_spawn_if_needed(self: &Arc<Self>) -> Result<(), String> {
1420        if self.level_data.read().data().initialized {
1421            return Ok(());
1422        }
1423
1424        if self.dimension_type.key != vanilla_dimension_types::OVERWORLD.key {
1425            self.level_data.write().data_mut().initialized = true;
1426            return Ok(());
1427        }
1428
1429        log::info!("Selecting global world spawn for {}...", self.key);
1430
1431        let origin = self
1432            .chunk_map
1433            .world_gen_context
1434            .generator
1435            .initial_spawn_search_origin();
1436        let spawn_chunk = ChunkPos::new(
1437            SectionPos::block_to_section_coord(origin.x()),
1438            SectionPos::block_to_section_coord(origin.z()),
1439        );
1440
1441        let mut spawn_y = self
1442            .chunk_map
1443            .world_gen_context
1444            .generator
1445            .spawn_height(self.get_min_y(), self.get_height());
1446        if spawn_y < self.get_min_y() {
1447            let x = chunk_min_block_x(spawn_chunk) + 8;
1448            let z = chunk_min_block_z(spawn_chunk) + 8;
1449            spawn_y = self
1450                .height_at(HeightmapType::WorldSurface, x, z)
1451                .unwrap_or(self.get_min_y());
1452        }
1453
1454        let mut spawn_pos = BlockPos::new(
1455            chunk_min_block_x(spawn_chunk) + 8,
1456            spawn_y,
1457            chunk_min_block_z(spawn_chunk) + 8,
1458        );
1459
1460        spawn_pos = self
1461            .chunk_map
1462            .with_full_chunks_in_radius(spawn_chunk, 5, || {
1463                self.find_spawn_in_loaded_radius(spawn_chunk)
1464                    .unwrap_or(spawn_pos)
1465            })
1466            .await
1467            .unwrap_or(spawn_pos);
1468
1469        {
1470            let mut level_data = self.level_data.write();
1471            let data = level_data.data_mut();
1472            data.set_spawn_pos(spawn_pos);
1473            data.spawn.angle = 0.0;
1474            data.initialized = true;
1475        }
1476
1477        log::info!("World {} spawn initialized at {spawn_pos:?}", self.key);
1478        Ok(())
1479    }

1481    #[expect(
1482        clippy::similar_names,
1483        reason = "dx_chunk/dz_chunk mirror vanilla's dXChunk/dZChunk"
1484    )]
1485    fn find_spawn_in_loaded_radius(&self, spawn_chunk: ChunkPos) -> Option<BlockPos> {
1486        let mut x_chunk_offset = 0;
1487        let mut z_chunk_offset = 0;
1488        let mut dx_chunk = 0;
1489        let mut dz_chunk = -1;
1490
1491        for _ in 0..(11 * 11) {
1492            if (-5..=5).contains(&x_chunk_offset) && (-5..=5).contains(&z_chunk_offset) {
1493                let candidate_chunk = ChunkPos::new(
1494                    spawn_chunk.0.x + x_chunk_offset,
1495                    spawn_chunk.0.y + z_chunk_offset,
1496                );
1497                if let Some(candidate) = self.spawn_pos_in_chunk(candidate_chunk) {
1498                    return Some(candidate);
1499                }
1500            }
1501
1502            if x_chunk_offset == z_chunk_offset
1503                || (x_chunk_offset < 0 && x_chunk_offset == -z_chunk_offset)
1504                || (x_chunk_offset > 0 && x_chunk_offset == 1 - z_chunk_offset)
1505            {
1506                let old_dx = dx_chunk;
1507                dx_chunk = -dz_chunk;
1508                dz_chunk = old_dx;
1509            }
1510
1511            x_chunk_offset += dx_chunk;
1512            z_chunk_offset += dz_chunk;
1513        }
1514
1515        None
1516    }

1530    fn level_respawn_pos(&self, x: i32, z: i32) -> Option<BlockPos> {
1531        let top_y = if self.dimension_type.has_ceiling {
1532            self.chunk_map
1533                .world_gen_context
1534                .generator
1535                .spawn_height(self.get_min_y(), self.get_height())
1536        } else {
1537            self.vanilla_chunk_height_at(HeightmapType::MotionBlocking, x, z)?
1538        };
1539
1540        if top_y < self.get_min_y() {
1541            return None;
1542        }
1543
1544        let surface = self.vanilla_chunk_height_at(HeightmapType::WorldSurface, x, z)?;
1545        let ocean_floor = self.vanilla_chunk_height_at(HeightmapType::OceanFloor, x, z)?;
1546        if surface <= top_y && surface > ocean_floor {
1547            return None;
1548        }
1549
1550        for y in (self.get_min_y()..=top_y + 1).rev() {
1551            let pos = BlockPos::new(x, y, z);
1552            let state = self.get_block_state(pos);
1553            if state.has_fluid() {
1554                break;
1555            }
1556
1557            if is_offset_face_full(state.get_collision_shape_at(pos), Direction::Up) {
1558                return Some(BlockPos::new(x, y + 1, z));
1559            }
1560        }
1561
1562        None
1563    }

1892    /// Sets a block at the given position with a custom update limit.
1893    ///
1894    /// The update limit prevents infinite recursion when shape updates trigger
1895    /// further block changes. Each recursive call decrements the limit.
1896    ///
1897    /// Returns `true` if the block was successfully set, `false` otherwise.
1898    pub fn set_block_with_limit(
1899        self: &Arc<Self>,
1900        pos: BlockPos,
1901        block_state: BlockStateId,
1902        flags: UpdateFlags,
1903        update_limit: i32,
1904    ) -> bool {
1905        if update_limit <= 0 {
1906            return false;
1907        }
1908
1909        if !self.is_in_valid_bounds(pos) {
1910            return false;
1911        }
1912
1913        let chunk_pos = Self::chunk_pos_for_block(pos);
1914        let Some(old_state) = self
1915            .chunk_map
1916            .with_full_chunk(chunk_pos, |chunk| {
1917                chunk.set_block_state(pos, block_state, flags)
1918            })
1919            .flatten()
1920        else {
1921            return false;
1922        };
1923
1924        // Record the block change for broadcasting to clients
1925        self.chunk_map.block_changed(pos);
1926        self.update_navigating_mobs_after_block_collision_change(pos, old_state, block_state);
1927
1928        // Neighbor updates (when UPDATE_NEIGHBORS is set)
1929        if flags.contains(UpdateFlags::UPDATE_NEIGHBORS) {
1930            self.update_neighbors_at(pos, old_state.get_block());
1931            let behavior = BLOCK_BEHAVIORS.get_behavior(block_state.get_block());
1932            if behavior.has_analog_output_signal(block_state) {
1933                self.update_neighbor_for_output_signal(pos, block_state.get_block());
1934            }
1935        }
1936
1937        // Shape updates (unless UPDATE_KNOWN_SHAPE is set)
1938        if !flags.contains(UpdateFlags::UPDATE_KNOWN_SHAPE) && update_limit > 0 {
1939            // Clear UPDATE_NEIGHBORS and UPDATE_SUPPRESS_DROPS for propagation
1940            let neighbor_flags =
1941                flags & !(UpdateFlags::UPDATE_NEIGHBORS | UpdateFlags::UPDATE_SUPPRESS_DROPS);
1942
1943            // Notify all 6 neighbors about our shape change
1944            for direction in Direction::UPDATE_SHAPE_ORDER {
1945                let neighbor_pos = pos.relative(direction);
1946
1947                // Tell the neighbor that we (at pos) changed
1948                self.neighbor_shape_changed(
1949                    direction.opposite(), // Direction from us to neighbor
1950                    neighbor_pos,         // Neighbor's position
1951                    pos,                  // Our position (the one that changed)
1952                    block_state,          // Our new state
1953                    neighbor_flags,
1954                    update_limit - 1,
1955                );
1956            }
1957        }
1958        true
1959    }

1961    fn update_navigating_mobs_after_block_collision_change(
1962        self: &Arc<Self>,
1963        pos: BlockPos,
1964        old_state: BlockStateId,
1965        new_state: BlockStateId,
1966    ) {
1967        let collision_shape_changed = self.block_collision_shape_changed(pos, old_state, new_state);
1968        let game_time = self.game_time();
1969        for entity_id in self.navigating_mob_ids() {
1970            let Some(entity) = self.entity_manager.get_by_id(entity_id) else {
1971                self.untrack_navigating_mob(entity_id);
1972                continue;
1973            };
1974            let Some(pathfinder) = entity.as_pathfinder_mob() else {
1975                self.untrack_navigating_mob(entity_id);
1976                continue;
1977            };
1978            {
1979                let mut navigation = pathfinder.mob_base().navigation().lock();
1980                navigation.invalidate_path_type(pos);
1981            }
1982            if !collision_shape_changed {
1983                continue;
1984            }
1985            if !pathfinder.is_path_finding() {
1986                continue;
1987            }
1988
1989            let should_recompute = {
1990                let navigation = pathfinder.mob_base().navigation().lock();
1991                navigation.should_recompute_path(pos, pathfinder.position())
1992            };
1993            if !should_recompute {
1994                continue;
1995            }
1996
1997            let request = {
1998                let mut navigation = pathfinder.mob_base().navigation().lock();
1999                navigation.request_recompute_path(game_time, pathfinder.can_update_path())
2000            };
2001            if let Some(request) = request {
2002                pathfinder.recompute_path(request);
2003            }
2004        }
2005    }

2688                    if small < 0.8 {

2702            .get_value(f64::from(pos.x()) / 8.0, f64::from(pos.z()) / 8.0)

2704            * 8.0;

2817    pub fn broadcast_chat(
2818        &self,
2819        mut packet: CPlayerChat,
2820        _sender: Arc<Player>,
2821        sender_last_seen: LastSeen,
2822        message_signature: Option<&[u8; 256]>,
2823    ) {
2824        log::debug!(
2825            "broadcast_chat: sender_last_seen has {} signatures, message_signature present: {}",
2826            sender_last_seen.len(),
2827            message_signature.is_some()
2828        );
2829
2830        self.players.iter_players(|_, recipient| {
2831            let messages_received = recipient.get_and_increment_messages_received();
2832            packet.global_index = messages_received;
2833

2835                "Broadcasting to player {} (UUID: {}), global_index={}",
2836                recipient.gameprofile.name,
2837                recipient.gameprofile.id,

2842            // This matches vanilla's order: pack() then push()

2844                let chat = recipient.chat.lock();
2845                chat.signature_cache
2846                    .index_previous_messages(&sender_last_seen)

2857            recipient.send_packet(packet.clone());
2858
2859            // AFTER sending, update the recipient's cache using vanilla's push algorithm
2860            // This adds all lastSeen signatures + current signature to the cache
2861            {
2862                let mut chat = recipient.chat.lock();
2863                if let Some(signature) = message_signature {
2864                    chat.signature_cache
2865                        .push(&sender_last_seen, Some(signature));

2868

3740    /// Ray-AABB intersection returning the entry t-parameter and the hit face.
3741    ///
3742    /// Returns `Some((tmin, direction))` where `tmin` is the ray parameter at entry
3743    /// and `direction` is the face normal pointing away from the hit surface.
3744    /// Returns `None` if the AABB is missed or entirely behind the ray origin.
3745    ///
3746    /// Used internally by [`ray_outline_check`] to pick the *closest* hit across
3747    /// a multi-box voxel shape, matching vanilla's `VoxelShape.clip()` behavior.
3748    fn intersects_aabb_with_t(
3749        start: DVec3,
3750        end: DVec3,
3751        min: DVec3,
3752        max: DVec3,
3753    ) -> Option<(f64, Direction)> {
3754        let dir = end - start;
3755
3756        let mut tmin = f64::NEG_INFINITY;
3757        let mut tmax = f64::INFINITY;
3758        let mut hit_dir = None;
3759
3760        macro_rules! slab {
3761            ($start:expr, $dir:expr, $min:expr, $max:expr, $neg:expr, $pos:expr) => {{
3762                if $dir.abs() < 1e-8 {
3763                    if $start < $min || $start > $max {
3764                        return None;
3765                    }
3766                } else {
3767                    let inv = 1.0 / $dir;
3768                    let mut t1 = ($min - $start) * inv;
3769                    let mut t2 = ($max - $start) * inv;
3770
3771                    let dir_hit = if t1 > t2 {
3772                        std::mem::swap(&mut t1, &mut t2);
3773                        $pos
3774                    } else {
3775                        $neg
3776                    };
3777
3778                    if t1 > tmin {
3779                        tmin = t1;
3780                        hit_dir = Some(dir_hit);
3781                    }
3782
3783                    tmax = tmax.min(t2);
3784                    if tmin > tmax {
3785                        return None;
3786                    }
3787                }
3788            }};
3789        }
3790
3791        slab!(
3792            start.x,
3793            dir.x,
3794            min.x,
3795            max.x,
3796            Direction::West,
3797            Direction::East
3798        );
3799        slab!(start.y, dir.y, min.y, max.y, Direction::Down, Direction::Up);
3800        slab!(
3801            start.z,
3802            dir.z,
3803            min.z,
3804            max.z,
3805            Direction::North,
3806            Direction::South
3807        );
3808
3809        if tmax < 0.0 {
3810            None
3811        } else {
3812            hit_dir.map(|d| (tmin, d))
3813        }
3814    }

3816    /// Performs a raytrace in the world.
3817    ///
3818    /// Adapted from Pumpkin project.
3819    pub fn raytrace<F>(
3820        &self,
3821        start_pos: DVec3,
3822        end_pos: DVec3,
3823        hit_check: F,
3824    ) -> (Option<BlockPos>, Option<Direction>)
3825    where
3826        F: Fn(BlockPos, &Self) -> RaytraceAction,
3827    {
3828        if start_pos == end_pos {
3829            return (None, None);
3830        }
3831
3832        let adjust = -1.0e-7f64;
3833        let to = end_pos.lerp(start_pos, adjust);
3834        let from = start_pos.lerp(end_pos, adjust);
3835
3836        let mut block = BlockPos::new(
3837            from.x.floor() as i32,
3838            from.y.floor() as i32,
3839            from.z.floor() as i32,
3840        );
3841
3842        match hit_check(block, self) {
3843            RaytraceAction::ImmediateHit => return (Some(block), None),
3844            RaytraceAction::CheckShape => {
3845                let (hit, face) = self.ray_outline_check(block, start_pos, end_pos);
3846                if hit {
3847                    return (Some(block), face);
3848                }
3849            }
3850            RaytraceAction::Pass => {}
3851        }
3852
3853        let difference = to - from;
3854
3855        let step = difference.signum().as_ivec3();
3856
3857        let delta = DVec3::new(
3858            if step.x == 0 {
3859                f64::MAX
3860            } else {
3861                (f64::from(step.x)) / difference.x
3862            },
3863            if step.y == 0 {
3864                f64::MAX
3865            } else {
3866                (f64::from(step.y)) / difference.y
3867            },
3868            if step.z == 0 {
3869                f64::MAX
3870            } else {
3871                (f64::from(step.z)) / difference.z
3872            },
3873        );
3874
3875        let mut next = DVec3::new(
3876            delta.x
3877                * (if step.x > 0 {
3878                    1.0 - (from.x - from.x.floor())
3879                } else {
3880                    from.x - from.x.floor()
3881                }),
3882            delta.y
3883                * (if step.y > 0 {
3884                    1.0 - (from.y - from.y.floor())
3885                } else {
3886                    from.y - from.y.floor()
3887                }),
3888            delta.z
3889                * (if step.z > 0 {
3890                    1.0 - (from.z - from.z.floor())
3891                } else {
3892                    from.z - from.z.floor()
3893                }),
3894        );
3895
3896        while next.x <= 1.0 || next.y <= 1.0 || next.z <= 1.0 {
3897            // Vanilla parity: traverseBlocks tie-breaking — Z wins on any tie.
3898            // X wins only when strictly less than both Y and Z.
3899            // Y wins only when strictly less than both X and Z.
3900            // Everything else (including all ties) goes to Z.
3901            let block_direction = if next.x < next.y && next.x < next.z {
3902                block.0.x += step.x;
3903                next.x += delta.x;
3904                if step.x > 0 {
3905                    Direction::West
3906                } else {
3907                    Direction::East
3908                }
3909            } else if next.y < next.x && next.y < next.z {
3910                block.0.y += step.y;
3911                next.y += delta.y;
3912                if step.y > 0 {
3913                    Direction::Down
3914                } else {
3915                    Direction::Up
3916                }
3917            } else {
3918                block.0.z += step.z;
3919                next.z += delta.z;
3920                if step.z > 0 {
3921                    Direction::North
3922                } else {
3923                    Direction::South
3924                }
3925            };
3926
3927            match hit_check(block, self) {
3928                RaytraceAction::ImmediateHit => {
3929                    return (Some(block), Some(block_direction));
3930                }
3931                RaytraceAction::CheckShape => {
3932                    let (hit, face) = self.ray_outline_check(block, start_pos, end_pos);
3933                    if hit {
3934                        return (Some(block), face);
3935                    }
3936                }
3937                RaytraceAction::Pass => {}
3938            }
3939        }
3940
3941        (None, None)
3942    }

4347    /// Plays a sound at an exact world position, broadcasting to nearby players.
4348    pub fn play_sound_at(
4349        &self,
4350        sound: SoundEventRef,
4351        source: SoundSource,
4352        pos: DVec3,
4353        volume: f32,
4354        pitch: f32,
4355        exclude: Option<i32>,
4356    ) {
4357        let chunk = ChunkPos::new(
4358            SectionPos::block_to_section_coord(pos.x.floor() as i32),
4359            SectionPos::block_to_section_coord(pos.z.floor() as i32),
4360        );
4361
4362        // Generate a random seed for sound variations
4363        let seed = rand::random::<i64>();
4364        let packet = CSound::new(sound, source, pos, volume, pitch, seed);
4365        let Ok(encoded) =
4366            EncodedPacket::from_bare(packet, self.compression, ConnectionProtocol::Play)
4367        else {
4368            log::warn!("Failed to encode sound packet");
4369            return;
4370        };
4371
4372        // Get players tracking this chunk, then apply vanilla's strict range check.
4373        for entity_id in self.player_area_map.get_tracking_players(chunk) {
4374            // Skip excluded player (they hear the sound client-side)
4375            if exclude == Some(entity_id) {
4376                continue;
4377            }
4378            if let Some(player) = self.players.get_by_entity_id(entity_id) {
4379                let player_pos = player.position();
4380                let dx = player_pos.x - pos.x;
4381                let dy = player_pos.y - pos.y;
4382                let dz = player_pos.z - pos.z;
4383                let dist_sq = dx * dx + dy * dy + dz * dz;
4384
4385                if sound_is_within_range(sound, volume, dist_sq) {
4386                    player.connection.send_encoded(encoded.clone());
4387                }
4388            }
4389        }
4390    }

4920                if nearest_player_distance_in_range(distance_sqr, max_distance, max_distance_sqr)
4921                    && nearest
4922                        .as_ref()
4923                        .is_none_or(|(_, current)| distance_sqr < *current)
4924                {
4925                    nearest = Some((player.clone(), distance_sqr));
4926                }
4927            }
4928            true
4929        });
4930        nearest.map(|(player, _)| player)
4931    }

4933    /// Gets the squared distance to the nearest player, if any player is present.
4934    #[must_use]
4935    pub fn nearest_player_distance_sqr(&self, position: DVec3) -> Option<f64> {
4936        let mut nearest = None;
4937        self.players.iter_players(|_, player| {
4938            if player.is_spectator() {
4939                return true;
4940            }
4941            let distance_sqr = player.position().distance_squared(position);
4942            if nearest.is_none_or(|current| distance_sqr < current) {
4943                nearest = Some(distance_sqr);
4944            }
4945            true
4946        });
4947        nearest
4948    }
