1//! Water fluid callbacks.
2//!
3//! Based on vanilla's `WaterFluid.java`.
4//! Implements `FluidBehavior` and `FlowingFluid` for sharing base spread logic.
5
6use std::sync::Arc;
7
8use steel_registry::blocks::properties::Direction;
9use steel_registry::game_rules::GameRuleValue;
10use steel_registry::sound_events;
11use steel_registry::vanilla_game_rules::WATER_SOURCE_CONVERSION;
12use steel_utils::BlockPos;
13use steel_utils::BlockStateId;
14
15use crate::fluid::{FlowingFluid, FluidBehavior};
16use crate::fluid::{FluidRef, FluidState, is_water_fluid, water_id};
17use crate::world::World;
18
19/// Water fluid implementation.
20///
21/// Implements [`FluidBehavior`] with water-specific parameters and
22/// behaviors (ambient sounds, lava→water chemistry, game-rule source conversion).
23pub struct WaterFluid;
24
25impl FluidBehavior for WaterFluid {
26    fn fluid_type(&self) -> FluidRef {
27        water_id()
28    }
29
30    fn is_same(&self, fluid: FluidRef) -> bool {
31        is_water_fluid(fluid)
32    }
33
34    fn tick_delay(&self, _world: &Arc<World>) -> i32 {
35        5
36    }
37
38    fn drop_off(&self, _world: &Arc<World>) -> u8 {
39        1
40    }
41
42    fn slope_find_distance(&self, _world: &Arc<World>) -> u8 {
43        4
44    }
45
46    fn explosion_resistance(&self) -> f32 {
47        100.0
48    }
49
50    fn can_convert_to_source(&self, world: &Arc<World>) -> bool {
51        match world.get_game_rule(WATER_SOURCE_CONVERSION) {
52            GameRuleValue::Bool(val) => val,
53            GameRuleValue::Int(_) => true,
54        }
55    }
56
57    /// Water can only be replaced from below and only by non-water fluids.
58    fn can_be_replaced_with(
59        &self,
60        _fluid_state: FluidState,
61        _world: &Arc<World>,
62        _pos: BlockPos,
63        other_fluid: FluidRef,
64        direction: Direction,
65    ) -> bool {
66        direction == Direction::Down && !is_water_fluid(other_fluid)
67    }
68
69    /// Drops block resources and plays destruction particles when water replaces a non-air block.
70    fn before_destroying_block(&self, world: &Arc<World>, pos: BlockPos, state: BlockStateId) {
71        world.drop_resources(state, pos);
72        world.destroy_block_effect(pos, u32::from(state.0), None);
73    }
74
75    /// Flowing water: 1/64 chance for ambient sound.
76    /// Source water: 1/10 chance for underwater particles.
77    fn animate_tick(&self, world: &Arc<World>, pos: BlockPos, fluid_state: FluidState) {
78        if !fluid_state.is_source() && !fluid_state.falling {
79            // 1/64 chance for flowing water ambient sound
80            if rand::random_range(0u32..64) == 0 {
81                let volume: f32 = rand::random::<f32>() * 0.25 + 0.75;
82                let pitch: f32 = rand::random::<f32>() + 0.5;
83                world.play_block_sound(sound_events::BLOCK_WATER_AMBIENT, pos, volume, pitch, None);
84            }
85        } else {
86            // 1/10 chance for underwater particles
87            if rand::random_range(0u32..10) == 0 {
88                // TODO: Spawn UNDERWATER particles (needs CLevelParticles packet)
89            }
90        }
91    }
92
93    fn tick(&self, world: &Arc<World>, pos: BlockPos) {
94        self.base_tick(world, pos);
95    }
96
97    fn spread(&self, world: &Arc<World>, pos: BlockPos, fluid_state: FluidState) {
98        self.base_spread(world, pos, fluid_state);
99    }
100}
