1//! Level data persistence module.
2//!
3//! This module handles saving and loading world-level data like game rules,
4//! time, weather, spawn point, and seed. This data is stored in `level.json`
5//! in each world's directory.

19/// Persistent level data that gets saved to disk.
20#[derive(Debug, Clone, Serialize, Deserialize)]
21pub struct LevelData {
22    /// World seed for terrain generation.
23    pub seed: i64,
24    /// Total game time in ticks.
25    pub game_time: i64,
26    /// Time of day in ticks (0-24000).
27    pub day_time: i64,
28    /// World spawn point.
29    pub spawn: SpawnPoint,
30    /// Weather state.
31    pub weather: WeatherState,
32    /// Game rules (stored as name -> value pairs for serialization).
33    pub game_rules: FxHashMap<String, GameRuleValue>,
34    /// Runtime game rule values (not serialized, loaded from `game_rules`).
35    #[serde(skip)]
36    pub game_rules_values: GameRuleValues,
37    /// Whether the world has been initialized.
38    pub initialized: bool,
39}

41/// Spawn point data.
42#[derive(Debug, Clone, Serialize, Deserialize)]
43pub struct SpawnPoint {
44    /// X coordinate.
45    pub x: i32,
46    /// Y coordinate.
47    pub y: i32,
48    /// Z coordinate.
49    pub z: i32,
50    /// Spawn angle (yaw).
51    pub angle: f32,
52}

54impl Default for SpawnPoint {

65/// Weather state.
66#[derive(Debug, Clone, Serialize, Deserialize, Default)]
67pub struct WeatherState {
68    /// Whether it is currently raining.
69    pub raining: bool,
70    /// Ticks until rain state changes.
71    pub rain_time: i32,
72    /// Whether it is currently thundering.
73    pub thundering: bool,
74    /// Ticks until thunder state changes.
75    pub thunder_time: i32,
76    /// Ticks of clear weather remaining.
77    pub clear_weather_time: i32,

80impl Default for LevelData {

86impl LevelData {
87    /// Creates new level data with the given seed.
88    #[must_use]
89    pub fn new_with_seed(seed: i64) -> Self {
90        Self {
91            seed,
92            game_time: 0,
93            day_time: 0,
94            spawn: SpawnPoint::default(),
95            weather: WeatherState::default(),
96            game_rules: FxHashMap::default(),
97            game_rules_values: GameRuleValues::new(&REGISTRY.game_rules),
98            initialized: false,
99        }
100    }
101
102    /// Loads game rules from the serialized map into the runtime values.
103    pub fn load_game_rules(&mut self) {
104        self.game_rules_values = GameRuleValues::new(&REGISTRY.game_rules);
105        for (name, value) in &self.game_rules {
106            self.game_rules_values
107                .set_by_name(name, *value, &REGISTRY.game_rules);
108        }
109    }
110
111    /// Saves game rules from the runtime values to the serialized map.
112    pub fn save_game_rules(&mut self) {
113        self.game_rules.clear();
114        for (_, rule) in REGISTRY.game_rules.iter() {
115            let name = rule.key.path.to_string();
116            let value = self.game_rules_values.get(rule, &REGISTRY.game_rules);
117            self.game_rules.insert(name, value);
118        }
119    }
120
121    /// Gets the spawn position as a `BlockPos`.
122    #[must_use]
123    pub const fn spawn_pos(&self) -> BlockPos {
124        BlockPos::new(self.spawn.x, self.spawn.y, self.spawn.z)
125    }
126
127    /// Sets the spawn position from a `BlockPos`.
128    pub const fn set_spawn_pos(&mut self, pos: BlockPos) {
129        self.spawn.x = pos.x();
130        self.spawn.y = pos.y();
131        self.spawn.z = pos.z();
132    }
133}

135/// Manages level data persistence for a world.
136pub struct LevelDataManager {
137    /// Path to the level.json file.
138    path: Option<PathBuf>,
139    /// Cached level data.
140    data: LevelData,
141    /// Whether data has been modified since last save.
142    dirty: bool,
143}
144
145impl LevelDataManager {
146    /// Creates a new level data manager for the given world directory.
147    ///
148    /// If `level.json` exists, it will be loaded (the provided seed is ignored).
149    /// Otherwise, new data will be created with the provided seed.
150    pub async fn new(world_dir: Option<impl AsRef<Path>>, seed: i64) -> io::Result<Self> {
151        let (data, path) = match &world_dir {
152            Some(dir) => {
153                let path = dir.as_ref().join("level.json");
154
155                let data = if path.exists() {
156                    // Load existing level data (seed from file takes precedence)
157                    let content = fs::read_to_string(&path).await?;
158                    let mut loaded: LevelData = serde_json::from_str(&content).map_err(|e| {
159                        io::Error::new(
160                            io::ErrorKind::InvalidData,
161                            format!("Invalid level.json: {e}"),
162                        )
163                    })?;
164                    // Initialize runtime game rules from serialized values
165                    loaded.load_game_rules();
166                    loaded
167                } else {
168                    // Create new level data with the provided seed
169                    LevelData::new_with_seed(seed)
170                };
171                (data, Some(path))
172            }
173            None => (LevelData::new_with_seed(seed), None),
174        };
175
176        Ok(Self {
177            path,
178            data,
179            dirty: false,
180        })
181    }
182
183    /// Gets a reference to the level data.
184    #[must_use]
185    pub const fn data(&self) -> &LevelData {
186        &self.data
187    }
188
189    /// Gets a mutable reference to the level data and marks it as dirty.
190    pub const fn data_mut(&mut self) -> &mut LevelData {
191        self.dirty = true;
192        &mut self.data
193    }
194
195    /// Returns whether the data has been modified since last save.
196    #[must_use]
197    pub const fn is_dirty(&self) -> bool {
198        self.dirty
199    }
200
201    /// Marks the data as dirty (needs saving).
202    pub const fn mark_dirty(&mut self) {
203        self.dirty = true;
204    }
205
206    /// Saves the level data to disk if it has been modified.
207    pub async fn save(&mut self) -> io::Result<()> {
208        if !self.dirty {
209            return Ok(());
210        }
211
212        let Some(world_path) = &self.path else {
213            self.dirty = false;
214            return Ok(());
215        };
216        if let Some(parent) = world_path.parent() {
217            fs::create_dir_all(parent).await?;
218        }
219
220        // Export runtime game rules to serializable format before saving
221        self.data.save_game_rules();
222
223        let content = serde_json::to_string_pretty(&self.data)
224            .map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e))?;
225        fs::write(world_path, content).await?;
226        self.dirty = false;
227
228        log::debug!("Saved level data to {}", world_path.display());
229        Ok(())
230    }
