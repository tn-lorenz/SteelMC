22/// The main server struct.
23pub struct Server {
24    /// The key store for the server.
25    pub key_store: KeyStore,
26    /// The registry for the server.
27    pub registry: Arc<Registry>,
28    /// The registry cache for the server.
29    pub registry_cache: RegistryCache,
30    /// A list of all the worlds on the server.
31    pub worlds: Vec<Arc<World>>,
32    /// The tick rate manager for the server.
33    pub tick_rate_manager: SyncRwLock<TickRateManager>,
34}
35
36impl Server {
37    /// Creates a new server.
38    pub async fn new(chunk_runtime: Arc<Runtime>) -> Self {
39        let start = Instant::now();
40        let mut registry = Registry::new_vanilla();
41        registry.freeze();
42        log::info!("Vanilla registry loaded in {:?}", start.elapsed());
43
44        let registry = Arc::new(registry);
45        let registry_cache = RegistryCache::new(&registry).await;
46
47        Server {
48            key_store: KeyStore::create(),
49            worlds: vec![Arc::new(World::new(&registry, chunk_runtime))],
50            registry,
51            registry_cache,
52            tick_rate_manager: SyncRwLock::new(TickRateManager::new()),
53        }
54    }
55
56    /// Adds a player to the server.
57    pub fn add_player(&self, player: Arc<Player>) {
58        player.connection.send_packet(CLogin {
59            player_id: 0,
60            hardcore: false,
61            levels: vec![Identifier::vanilla_static("overworld")],
62            max_players: 5,
63            chunk_radius: STEEL_CONFIG.view_distance.into(),
64            simulation_distance: STEEL_CONFIG.simulation_distance.into(),
65            reduced_debug_info: false,
66            show_death_screen: false,
67            do_limited_crafting: false,
68            common_player_spawn_info: CommonPlayerSpawnInfo {
69                dimension_type: 0,
70                dimension: Identifier::vanilla_static("overworld"),
71                seed: 0,
72                game_type: GameType::Survival,
73                previous_game_type: None,
74                is_debug: false,
75                is_flat: true,
76                last_death_location: None,
77                portal_cooldown: 0,
78                sea_level: 64,
79            },
80            enforces_secure_chat: STEEL_CONFIG.enforce_secure_chat,
81        });
82        self.worlds[0].add_player(player);
83    }

85    /// Runs the server tick loop.
86    pub async fn run(self: Arc<Self>, cancel_token: CancellationToken) {
87        let mut next_tick_time = Instant::now();
88
89        loop {
90            if cancel_token.is_cancelled() {
91                break;
92            }
93
94            let (nanoseconds_per_tick, is_sprinting, should_sprint_this_tick) = {
95                let mut tick_manager = self.tick_rate_manager.write();
96                let nanoseconds_per_tick = tick_manager.nanoseconds_per_tick;
97
98                // Handle sprinting
99                let is_sprinting = tick_manager.is_sprinting();
100                let should_sprint_this_tick = if is_sprinting {
101                    tick_manager.check_should_sprint_this_tick()
102                } else {
103                    false
104                };
105                (nanoseconds_per_tick, is_sprinting, should_sprint_this_tick)
106            };
107
108            if is_sprinting && should_sprint_this_tick {
109                // If sprinting, we don't wait
110                next_tick_time = Instant::now();
111            } else {
112                // Normal wait logic
113                let now = Instant::now();
114                if now < next_tick_time {
115                    tokio::select! {
116                        () = cancel_token.cancelled() => break,
117                        () = tokio::time::sleep(next_tick_time - now) => {}
118                    }
119                }
120                next_tick_time += std::time::Duration::from_nanos(nanoseconds_per_tick);
121            }
122
123            if cancel_token.is_cancelled() {
124                break;
125            }
126
127            let tick_count = {
128                let mut tick_manager = self.tick_rate_manager.write();
129                tick_manager.tick();
130                tick_manager.tick_count
131            };
132
133            // Tick worlds
134            self.tick_worlds(tick_count).await;
135
136            if is_sprinting && should_sprint_this_tick {
137                let mut tick_manager = self.tick_rate_manager.write();
138                tick_manager.end_tick_work();
139            }
140        }
141    }
142
143    async fn tick_worlds(&self, tick_count: u64) {
144        let mut tasks = Vec::with_capacity(self.worlds.len());
145        for world in &self.worlds {
146            let world_clone = world.clone();
147            tasks.push(spawn_blocking(move || world_clone.tick_b(tick_count)));
148        }
149        let start = Instant::now();
150        for task in tasks {
151            let _ = task.await;
152        }
153        if start.elapsed().as_millis() > 1 {
154            log::warn!(
155                "Worlds ticked in {:?}, tick count: {tick_count}",
156                start.elapsed()
157            );
158        }
159    }
