70    pub async fn new(chunk_runtime: Arc<Runtime>, cancel_token: CancellationToken) -> Self {
71        let start = Instant::now();
72        let mut registry = Registry::new_vanilla();
73        registry.freeze();
74        log::info!("Vanilla registry loaded in {:?}", start.elapsed());
75
76        REGISTRY
77            .init(registry)
78            .expect("We should be the ones who init the REGISTRY");
79
80        // Initialize behavior registries after the main registry is frozen
81        init_behaviors();
82        init_block_entities();
83        init_entities();
84        log::info!("Behavior registries initialized");
85
86        let registry_cache = RegistryCache::new();
87
88        let seed: i64 = if STEEL_CONFIG.seed.is_empty() {
89            rand::random()
90        } else {
91            STEEL_CONFIG.seed.parse().unwrap_or_else(|_| {
92                let mut hash: i64 = 0;
93                for byte in STEEL_CONFIG.seed.bytes() {
94                    hash = hash.wrapping_mul(31).wrapping_add(i64::from(byte));
95                }
96                hash
97            })
98        };
99
100        let generation_pool: Arc<ThreadPool> = Arc::new({
101            let mut builder = ThreadPoolBuilder::new().thread_name(|i| format!("rayon-gen-{i}"));
102            // Debug builds have deep call chains in density functions that overflow the default 2 MB stack
103            if cfg!(debug_assertions) {
104                builder = builder.stack_size(8 * 1024 * 1024);
105            }
106            builder
107                .build()
108                .expect("Failed to create generation thread pool")
109        });
110
111        let overworld = World::new_with_config(
112            chunk_runtime.clone(),
113            OVERWORLD,
114            seed,
115            Self::make_world_config(OVERWORLD, seed),
116            generation_pool.clone(),
117        )
118        .await
119        .expect("Failed to create overworld");
120
121        let nether = World::new_with_config(
122            chunk_runtime.clone(),
123            THE_NETHER,
124            seed,
125            Self::make_world_config(THE_NETHER, seed),
126            generation_pool.clone(),
127        )
128        .await
129        .expect("Failed to create nether");
130
131        let end = World::new_with_config(
132            chunk_runtime.clone(),
133            THE_END,
134            seed,
135            Self::make_world_config(THE_END, seed),
136            generation_pool,
137        )
138        .await
139        .expect("Failed to create end");
140
141        let player_data_storage = PlayerDataStorage::new()
142            .await
143            .expect("Failed to create player data storage");
144        let mut worlds: FxSmallMap<8, Identifier, Arc<World>> = FxSmallMap::default();
145        worlds.insert(OVERWORLD.key.clone(), overworld);
146        worlds.insert(THE_NETHER.key.clone(), nether);
147        worlds.insert(THE_END.key.clone(), end);
148
149        Server {
150            cancel_token,
151            key_store: KeyStore::create(),
152            worlds,
153            registry_cache,
154            tick_rate_manager: SyncRwLock::new(TickRateManager::new()),
155            command_dispatcher: SyncRwLock::new(CommandDispatcher::new()),
156            player_data_storage,
157        }
158    }

160    /// Adds a player to the server.
161    ///
162    /// # Panics
163    /// Panics if the registry is not initialized.
164    pub async fn add_player(&self, player: Arc<Player>) {
165        // Load saved player data if it exists
166        match self.player_data_storage.load(player.gameprofile.id).await {
167            Ok(Some(saved_data)) => {
168                log::info!("Loaded saved data for player {}", player.gameprofile.name);
169                saved_data.apply_to_player(&player);
170            }
171            Ok(None) => {
172                log::debug!(
173                    "No saved data for player {}, using defaults",
174                    player.gameprofile.name
175                );
176            }
177            Err(e) => {
178                log::error!(
179                    "Failed to load player data for {}: {e}",
180                    player.gameprofile.name
181                );
182            }
183        }
184
185        player.reset_health_if_dead();
186        let world = &self.overworld();
187
188        // Get gamerule values
189        let reduced_debug_info =
190            world.get_game_rule(REDUCED_DEBUG_INFO) == GameRuleValue::Bool(true);
191        let immediate_respawn = world.get_game_rule(IMMEDIATE_RESPAWN) == GameRuleValue::Bool(true);
192        let do_limited_crafting =
193            world.get_game_rule(LIMITED_CRAFTING) == GameRuleValue::Bool(true);
194
195        // Get world data
196        let hashed_seed = world.obfuscated_seed();
197        let dimension_key = world.dimension.key.clone();
198
199        player.send_packet(CLogin {
200            player_id: player.id,
201            hardcore: false,
202            levels: REGISTRY.dimension_types.get_ids(),
203            max_players: STEEL_CONFIG.max_players as i32,
204            chunk_radius: player.view_distance().into(),
205            simulation_distance: STEEL_CONFIG.simulation_distance.into(),
206            reduced_debug_info,
207            show_death_screen: !immediate_respawn,
208            do_limited_crafting,
209            common_player_spawn_info: CommonPlayerSpawnInfo {
210                dimension_type: REGISTRY
211                    .dimension_types
212                    .by_key(&dimension_key)
213                    .expect("Should be registered")
214                    .id() as i32,
215                dimension: dimension_key,
216                seed: hashed_seed,
217                game_type: player.game_mode.load(),
218                previous_game_type: Some(player.prev_game_mode.load()),
219                is_debug: false,
220                is_flat: matches!(STEEL_CONFIG.world_generator, WorldGeneratorTypes::Flat),
221                last_death_location: None,
222                portal_cooldown: 0,
223                sea_level: 63, // Standard overworld sea level
224            },
225            enforces_secure_chat: STEEL_CONFIG.enforce_secure_chat,
226        });
227
228        // Send player abilities (flight, invulnerability, etc.)
229        player.send_abilities();
230
231        player.send_packet(CSetHeldSlot {
232            slot: i32::from(player.inventory.lock().get_selected_slot()),
233        });
234
235        if world.can_have_weather() {
236            let (rain_level, thunder_level) = {
237                let weather = world.weather.lock();
238                (weather.rain_level, weather.thunder_level)
239            };
240
241            if world.is_raining() {
242                player.send_packet(CGameEvent {
243                    event: GameEventType::StartRaining,
244                    data: 0.0,
245                });
246            }
247
248            player.send_packet(CGameEvent {
249                event: GameEventType::RainLevelChange,
250                data: rain_level,
251            });
252
253            player.send_packet(CGameEvent {
254                event: GameEventType::ThunderLevelChange,
255                data: thunder_level,
256            });
257        }
258
259        let commands = self.command_dispatcher.read().get_commands();
260        player.send_packet(commands);
261
262        // TODO: Set permissions level to match player's level
263        player.send_packet(CEntityEvent {
264            entity_id: player.id,
265            event: EntityStatus::PermissionLevelOwners,
266        });
267
268        // Send current ticking state to the joining player
269        self.send_ticking_state_to_player(&player);
270
271        // Get player position for teleport sync (must be done before add_player moves the Arc)
272        let pos = *player.position.lock();
273        let (yaw, pitch) = player.rotation.load();
274
275        // Send position sync to client (ensures client is at the correct loaded position)
276        // This must be sent after the player is added to the world
277        player.teleport(pos.x, pos.y, pos.z, yaw, pitch);
278
279        player.reset_sent_info();
280
281        world.add_player(player.clone());
282    }

286        let mut players = vec![];
287        for world in self.worlds.values() {

302    /// Returns a sample of up to 12 online players for the server list ping.
303    #[must_use]
304    pub fn player_sample(&self) -> Vec<(String, String)> {
305        const MAX_SAMPLE: usize = 12;
306
307        let players = self.get_players();
308        if players.is_empty() {
309            return vec![];
310        }
311
312        let sample_size = players.len().min(MAX_SAMPLE);
313        // Random starting offset into the player list
314        let offset = if players.len() > sample_size {
315            (rand::random::<u64>() as usize) % (players.len() - sample_size + 1)
316        } else {
317            0
318        };
319
320        let mut sample: Vec<(String, String)> = players[offset..offset + sample_size]
321            .iter()
322            .map(|p| {
323                (
324                    p.gameprofile.name.clone(),
325                    p.gameprofile.id.hyphenated().to_string(),
326                )
327            })
328            .collect();
329
330        // Shuffle using Fisher-Yates with random indices
331        for i in (1..sample.len()).rev() {
332            let j = (rand::random::<u64>() as usize) % (i + 1);
333            sample.swap(i, j);
334        }
335
336        sample
337    }

361    /// Runs the server tick loop.
362    pub async fn run(self: Arc<Self>, cancel_token: CancellationToken) {
363        let mut next_tick_time = Instant::now();
364
365        loop {
366            if cancel_token.is_cancelled() {
367                break;
368            }
369
370            let (nanoseconds_per_tick, should_sprint_this_tick) = {
371                let mut tick_manager = self.tick_rate_manager.write();
372                let nanoseconds_per_tick = tick_manager.nanoseconds_per_tick;
373
374                // Handle sprinting - returns (should_sprint, Option<sprint_report>)
375                let (should_sprint, sprint_report) = tick_manager.check_should_sprint_this_tick();
376                drop(tick_manager);
377
378                // If sprint finished, broadcast the report and state change to all players
379                if let Some(report) = sprint_report {
380                    self.broadcast_sprint_report(&report);
381                    self.broadcast_ticking_state();
382                }
383
384                (nanoseconds_per_tick, should_sprint)
385            };
386
387            if should_sprint_this_tick {
388                // If sprinting, we don't wait
389                next_tick_time = Instant::now();
390            } else {
391                // Normal wait logic
392                let now = Instant::now();
393                if now < next_tick_time {
394                    tokio::select! {
395                        () = cancel_token.cancelled() => break,
396                        () = sleep(next_tick_time - now) => {}
397                    }
398                }
399                next_tick_time += Duration::from_nanos(nanoseconds_per_tick);
400            }
401
402            if cancel_token.is_cancelled() {
403                break;
404            }
405
406            // Record tick start time for MSPT tracking
407            let tick_start = Instant::now();
408
409            let (tick_count, runs_normally) = {
410                let mut tick_manager = self.tick_rate_manager.write();
411                tick_manager.tick();
412                let runs_normally = tick_manager.runs_normally();
413                if runs_normally {
414                    tick_manager.increment_tick_count();
415                }
416                (tick_manager.tick_count, runs_normally)
417            };
418
419            // Always tick worlds (for chunk loading/gen), but pass runs_normally
420            // so game elements like random ticks only run when not frozen
421            self.tick_worlds(tick_count, runs_normally).await;
422
423            // Record tick duration for TPS/MSPT tracking
424            let (tps, mspt) = {
425                let tick_duration_nanos = tick_start.elapsed().as_nanos() as u64;
426                let mut tick_manager = self.tick_rate_manager.write();
427                tick_manager.record_tick_time(tick_duration_nanos);
428                (tick_manager.get_tps(), tick_manager.get_average_mspt())
429            };
430
431            // Update tab list with TPS/MSPT periodically
432            if tick_count % TAB_LIST_UPDATE_INTERVAL == 0 {
433                self.broadcast_tab_list(tps, mspt);
434            }
435
436            if should_sprint_this_tick {
437                let mut tick_manager = self.tick_rate_manager.write();
438                tick_manager.end_tick_work();
439            }
440        }
441    }

443    #[tracing::instrument(level = "trace", skip(self), name = "tick_worlds")]
444    async fn tick_worlds(&self, tick_count: u64, runs_normally: bool) {
445        let mut tasks = Vec::with_capacity(self.worlds.len());
446        for world in self.worlds.values() {
447            let world_clone = world.clone();
448            tasks.push(spawn_blocking(move || {
449                world_clone.tick_b(tick_count, runs_normally)
450            }));
451        }
452        let start = Instant::now();
453        let mut all_timings: Vec<WorldTickTimings> = Vec::with_capacity(tasks.len());
454        for task in tasks {
455            if let Ok(timings) = task.await {
456                all_timings.push(timings);
457            }
458        }
459        let elapsed = start.elapsed();
460        if elapsed.as_millis() >= 30 {
461            // Log detailed breakdown when tick is slow
462            for (i, timings) in all_timings.iter().enumerate() {
463                let cm = &timings.chunk_map;
464                tracing::warn!(
465                    world = i,
466                    ?elapsed,
467                    tick_count,
468                    player_tick = ?timings.player_tick,
469                    ticket_updates = ?cm.ticket_updates,
470                    holder_creation = ?cm.holder_creation,
471                    schedule_generation = ?cm.schedule_generation,
472                    scheduled_count = cm.scheduled_count,
473                    run_generation = ?cm.run_generation,
474                    broadcast_changes = ?cm.broadcast_changes,
475                    process_unloads = ?cm.process_unloads,
476                    collect_tickable = ?cm.collect_tickable,
477                    tick_chunks = ?cm.tick_chunks,
478                    tickable_count = cm.tickable_count,
479                    total_chunks = cm.total_chunks,
480                    "Worlds tick slow"
481                );
482            }
483        }
484    }

486    /// Broadcasts the tab list header/footer with current TPS and MSPT values.
487    fn broadcast_tab_list(&self, tps: f32, mspt: f32) {
488        // Color TPS based on value
489        let tps_color = if tps >= 19.5 {
490            Color::Green
491        } else if tps >= 15.0 {
492            Color::Yellow
493        } else {
494            Color::Red
495        };
496
497        // Color MSPT based on value (under 50ms is good)
498        let mspt_color = if mspt <= 50.0 {
499            Color::Aqua
500        } else {
501            Color::Red
502        };
503
504        let header = TextComponent::plain("\n").add_children(vec![
505            TextComponent::plain("Steel Dev Build").color(Color::Yellow),
506            TextComponent::plain("\n"),
507        ]);
508        let footer = TextComponent::plain("\n").add_children(vec![
509            TextComponent::plain("TPS: ").color(Color::Gray),
510            TextComponent::plain(format!("{tps:.1}")).color(tps_color),
511            TextComponent::plain(" | ").color(Color::DarkGray),
512            TextComponent::plain("MSPT: ").color(Color::Gray),
513            TextComponent::plain(format!("{mspt:.2}")).color(mspt_color),
514            TextComponent::plain("\n"),
515        ]);
516
517        // Broadcast to all players in all worlds
518        for world in self.worlds.values() {
519            world.broadcast_to_all_with(|player| CTabList::new(&header, &footer, player));
520        }
521    }

564    /// This should be called when a player joins the server.

566        let tick_manager = self.tick_rate_manager.read();
567        let state_packet = CTickingState::new(tick_manager.tick_rate(), tick_manager.is_frozen());
568        let step_packet = CTickingStep::new(tick_manager.frozen_ticks_to_run());

571        player.send_packet(state_packet);
572        player.send_packet(step_packet);

574    /// Selects the appropriate chunk generator for the given dimension type.
575    fn make_generator_for_dimension(dimension: DimensionTypeRef, seed: i64) -> ChunkGeneratorType {
576        match STEEL_CONFIG.world_generator {
577            WorldGeneratorTypes::Empty => ChunkGeneratorType::Empty(EmptyChunkGenerator::new()),
578            WorldGeneratorTypes::Vanilla => {
579                let seed_u64 = seed as u64;
580                if dimension == OVERWORLD {
581                    let source = BiomeSourceKind::overworld(seed_u64);
582                    ChunkGeneratorType::Overworld(VanillaGenerator::new(source, seed_u64))
583                } else if dimension == THE_NETHER {
584                    let source = BiomeSourceKind::nether(seed_u64);
585                    ChunkGeneratorType::Nether(VanillaGenerator::new(source, seed_u64))
586                } else {
587                    let source = BiomeSourceKind::end(seed_u64);
588                    ChunkGeneratorType::End(VanillaGenerator::new(source, seed_u64))
589                }
590            }
591            WorldGeneratorTypes::Flat => {
592                if dimension == THE_NETHER {
593                    ChunkGeneratorType::Flat(FlatChunkGenerator::new(
594                        REGISTRY
595                            .blocks
596                            .get_default_state_id(vanilla_blocks::BEDROCK),
597                        REGISTRY
598                            .blocks
599                            .get_default_state_id(vanilla_blocks::NETHER_BRICKS),
600                        REGISTRY
601                            .blocks
602                            .get_default_state_id(vanilla_blocks::NETHERRACK),
603                    ))
604                } else if dimension == THE_END {
605                    ChunkGeneratorType::Flat(FlatChunkGenerator::new(
606                        REGISTRY
607                            .blocks
608                            .get_default_state_id(vanilla_blocks::BEDROCK),
609                        REGISTRY
610                            .blocks
611                            .get_default_state_id(vanilla_blocks::END_STONE),
612                        REGISTRY
613                            .blocks
614                            .get_default_state_id(vanilla_blocks::END_STONE),
615                    ))
616                } else {
617                    ChunkGeneratorType::Flat(FlatChunkGenerator::new(
618                        REGISTRY
619                            .blocks
620                            .get_default_state_id(vanilla_blocks::BEDROCK),
621                        REGISTRY.blocks.get_default_state_id(vanilla_blocks::DIRT),
622                        REGISTRY
623                            .blocks
624                            .get_default_state_id(vanilla_blocks::GRASS_BLOCK),
625                    ))
626                }
627            }
628        }
629    }
