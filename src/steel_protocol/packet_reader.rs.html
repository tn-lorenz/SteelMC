24// decrypt -> decompress -> raw
25/// A reader that can decompress data.
26pub enum DecompressionReader<R: AsyncRead + Unpin> {
27    /// A reader that decompresses data.
28    Decompress(ZlibDecoder<BufReader<R>>),
29    /// A reader that does not decompress data.
30    None(R),
31}
32
33impl<R: AsyncRead + Unpin> AsyncRead for DecompressionReader<R> {
34    #[inline]
35    fn poll_read(
36        self: Pin<&mut Self>,
37        cx: &mut Context<'_>,
38        buf: &mut ReadBuf<'_>,
39    ) -> Poll<io::Result<()>> {
40        match self.get_mut() {
41            Self::Decompress(reader) => {
42                let reader = Pin::new(reader);
43                reader.poll_read(cx, buf)
44            }
45            Self::None(reader) => {
46                let reader = Pin::new(reader);
47                reader.poll_read(cx, buf)
48            }
49        }
50    }
51}

53/// A reader that can decrypt data.
54pub enum DecryptionReader<R: AsyncRead + Unpin> {
55    /// A reader that decrypts data.
56    Decrypt(Box<StreamDecryptor<R>>),
57    /// A reader that does not decrypt data.
58    None(R),
59}
60
61impl<R: AsyncRead + Unpin> DecryptionReader<R> {
62    /// Upgrades the reader to decrypt data.
63    ///
64    /// # Panics
65    /// - If the reader is already decrypting data.
66    #[must_use]
67    pub fn upgrade(self, cipher: Aes128Cfb8Dec) -> Self {
68        match self {
69            Self::None(stream) => Self::Decrypt(Box::new(StreamDecryptor::new(cipher, stream))),
70            Self::Decrypt(_) => panic!("Cannot upgrade a stream that already has a cipher!"),
71        }
72    }
73}
74
75impl<R: AsyncRead + Unpin> AsyncRead for DecryptionReader<R> {
76    #[inline]
77    fn poll_read(
78        self: Pin<&mut Self>,
79        cx: &mut Context<'_>,
80        buf: &mut ReadBuf<'_>,
81    ) -> Poll<io::Result<()>> {
82        match self.get_mut() {
83            Self::Decrypt(reader) => {
84                let reader = Pin::new(reader);
85                reader.poll_read(cx, buf)
86            }
87            Self::None(reader) => {
88                let reader = Pin::new(reader);
89                reader.poll_read(cx, buf)
90            }
91        }
92    }
93}

95/// Decoder: Client -> Server
96/// Supports `ZLib` decoding/decompression
97/// Supports Aes128 Encryption
98pub struct TCPNetworkDecoder<R: AsyncRead + Unpin> {
99    reader: DecryptionReader<R>,
100    compression: Option<NonZeroU32>,
101}
102
103impl<R: AsyncRead + Unpin> TCPNetworkDecoder<R> {
104    /// Creates a new `TCPNetworkDecoder`.
105    pub fn new(reader: R) -> Self {
106        Self {
107            reader: DecryptionReader::None(reader),
108            compression: None,
109        }
110    }
111
112    /// Sets the compression threshold for the decoder.
113    pub fn set_compression(&mut self, threshold: NonZeroU32) {
114        self.compression = Some(threshold);
115    }
116
117    /// NOTE: Encryption can only be set; a minecraft stream cannot go back to being unencrypted
118    ///
119    /// # Panics
120    /// - If the reader is already decrypting data.
121    /// - If the key is invalid.
122    pub fn set_encryption(&mut self, key: &[u8; 16]) {
123        if matches!(self.reader, DecryptionReader::Decrypt(_)) {
124            panic!("Cannot upgrade a stream that already has a cipher!");
125        }
126        let cipher = Aes128Cfb8Dec::new_from_slices(key, key).expect("invalid key");
127        take_mut::take(&mut self.reader, |decoder| decoder.upgrade(cipher));
128    }
129
130    /// Gets a raw packet from the stream.
131    ///
132    /// # Errors
133    /// - If the packet length is invalid.
134    /// - If the packet is too long.
135    /// - If the packet is not compressed when it should be.
136    /// - If the packet fails to decompress.
137    #[allow(clippy::cast_sign_loss)]
138    pub async fn get_raw_packet(&mut self) -> Result<RawPacket, PacketError> {
139        let packet_len = VarInt::read_async(&mut self.reader).await? as usize;
140
141        if packet_len > MAX_PACKET_SIZE {
142            Err(PacketError::OutOfBounds)?;
143        }
144
145        let mut bounded_reader = (&mut self.reader).take(packet_len as _);
146
147        let mut reader = if let Some(threshold) = self.compression {
148            let decompressed_len = VarInt::read_async(&mut bounded_reader).await?;
149            let raw_packet_len = packet_len - VarInt::written_size(decompressed_len);
150            let decompressed_len = decompressed_len as usize;
151
152            if decompressed_len > MAX_PACKET_DATA_SIZE {
153                Err(PacketError::TooLong(decompressed_len))?;
154            }
155
156            if decompressed_len > 0 {
157                DecompressionReader::Decompress(ZlibDecoder::new(BufReader::new(bounded_reader)))
158            } else {
159                // Validate that we are not less than the compression threshold
160                if raw_packet_len > threshold.get() as _ {
161                    Err(PacketError::NotCompressed)?;
162                }
163
164                DecompressionReader::None(bounded_reader)
165            }
166        } else {
167            DecompressionReader::None(bounded_reader)
168        };
169
170        let packet_id = VarInt::read_async(&mut reader).await?;
171
172        let mut payload = Vec::new();
173        reader
174            .read_to_end(&mut payload)
175            .await
176            .map_err(|e| PacketError::DecompressionFailed(e.to_string()))?;
177
178        Ok(RawPacket {
179            id: packet_id,
180            payload,
181        })
182    }
