1use steel_macros::{ClientPacket, WriteTo};

4/// Sent to create particles on the client.
5///
6/// The client samples particle positions and velocities from the supplied
7/// distribution. A count of zero has Vanilla's special single-particle
8/// velocity behavior.
9#[derive(ClientPacket, WriteTo, Clone, Debug)]
10#[packet_id(Play = C_LEVEL_PARTICLES)]
11pub struct CLevelParticles {
12    pub override_limiter: bool,
13    pub always_show: bool,
14    pub x: f64,
15    pub y: f64,
16    pub z: f64,
17    pub x_dist: f32,
18    pub y_dist: f32,
19    pub z_dist: f32,
20    pub max_speed: f32,
21    pub count: i32,
22    pub particle: ParticleData,
23}

25#[cfg(test)]
26mod tests {
27    use std::sync::Once;
28
29    use steel_registry::{REGISTRY, Registry, RegistryEntry, vanilla_particle_types};
30    use steel_utils::{codec::VarInt, serial::WriteTo};
31
32    use super::CLevelParticles;
33
34    static INIT_REGISTRY: Once = Once::new();
35
36    fn init_registry() {
37        INIT_REGISTRY.call_once(|| {
38            let mut registry = Registry::new_vanilla();
39            registry.freeze();
40            let _ = REGISTRY.init(registry);
41        });
42    }
43
44    #[test]
45    fn writes_fields_in_vanilla_wire_order() {
46        init_registry();
47
48        let packet = CLevelParticles {
49            override_limiter: true,
50            always_show: false,
51            x: 1.25,
52            y: -2.5,
53            z: 3.75,
54            x_dist: 0.5,
55            y_dist: -1.0,
56            z_dist: 2.0,
57            max_speed: 0.125,
58            count: -7,
59            particle: steel_registry::particle_type::ParticleData::simple(
60                &vanilla_particle_types::FLAME,
61            ),
62        };
63
64        let mut encoded = Vec::new();
65        let Ok(()) = packet.write(&mut encoded) else {
66            panic!("level particles packet should encode");
67        };
68
69        let mut expected = vec![1, 0];
70        expected.extend_from_slice(&1.25_f64.to_be_bytes());
71        expected.extend_from_slice(&(-2.5_f64).to_be_bytes());
72        expected.extend_from_slice(&3.75_f64.to_be_bytes());
73        expected.extend_from_slice(&0.5_f32.to_be_bytes());
74        expected.extend_from_slice(&(-1.0_f32).to_be_bytes());
75        expected.extend_from_slice(&2.0_f32.to_be_bytes());
76        expected.extend_from_slice(&0.125_f32.to_be_bytes());
77        expected.extend_from_slice(&(-7_i32).to_be_bytes());
78
79        let Ok(flame_id) = i32::try_from(vanilla_particle_types::FLAME.id()) else {
80            panic!("flame particle id should fit in i32");
81        };
82        let Ok(()) = VarInt(flame_id).write(&mut expected) else {
83            panic!("flame particle id should encode");
84        };
85
86        assert_eq!(encoded, expected);
87    }
88}
