1use std::io::{Cursor, Read};
2
3use steel_macros::ServerPacket;
4use steel_utils::serial::ReadFrom;
5use uuid::Uuid;
6
7/// Client -> Server: Updates the player's chat session with their public key.
8///
9/// Sent when the player first joins or when their key needs to be updated.
10/// Contains the session ID and the player's public key signed by Mojang.
11///
12/// Equivalent to ServerboundChatSessionUpdatePacket in Minecraft.
13#[derive(ServerPacket, Clone, Debug)]
14pub struct SChatSessionUpdate {
15    /// The session ID for this chat session
16    pub session_id: Uuid,
17
18    /// Public key expiry timestamp (milliseconds since epoch)
19    pub expires_at: i64,
20
21    /// The player's RSA public key (DER encoded)
22    pub public_key: Vec<u8>,
23
24    /// Mojang's signature of the key (validates authenticity)
25    pub key_signature: Vec<u8>,
26}
27
28impl ReadFrom for SChatSessionUpdate {
29    fn read(reader: &mut Cursor<&[u8]>) -> std::io::Result<Self> {
30        let session_id = Uuid::read(reader)?;
31        let expires_at = i64::read(reader)?;
32
33        let key_len = steel_utils::codec::VarInt::read(reader)?.0 as usize;
34        let mut public_key = vec![0u8; key_len];
35        reader.read_exact(&mut public_key)?;
36
37        let sig_len = steel_utils::codec::VarInt::read(reader)?.0 as usize;
38        let mut key_signature = vec![0u8; sig_len];
39        reader.read_exact(&mut key_signature)?;
40
41        Ok(Self {
42            session_id,
43            expires_at,
44            public_key,
45            key_signature,
46        })
47    }
48}
