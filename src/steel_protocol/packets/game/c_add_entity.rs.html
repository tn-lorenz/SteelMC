1//! Packet sent to spawn an entity (including players) for the client.
2
3use steel_macros::ClientPacket;
4use steel_registry::packets::play::C_ADD_ENTITY;
5use steel_utils::codec::VarInt;
6use steel_utils::serial::WriteTo;
7use uuid::Uuid;
8
9/// Spawns an entity on the client.
10#[derive(ClientPacket, Clone, Debug)]
11#[packet_id(Play = C_ADD_ENTITY)]
12pub struct CAddEntity {
13    /// The entity ID (used for all future references to this entity)
14    pub id: i32,
15    /// The entity's UUID
16    pub uuid: Uuid,
17    /// The entity type (from registry)
18    pub entity_type: i32,
19    /// X position
20    pub x: f64,
21    /// Y position
22    pub y: f64,
23    /// Z position
24    pub z: f64,
25    /// Pitch (vertical rotation) as angle byte
26    pub x_rot: i8,
27    /// Yaw (horizontal rotation) as angle byte
28    pub y_rot: i8,
29    /// Head yaw as angle byte
30    pub head_y_rot: i8,
31    /// Entity data value (varies by entity type)
32    pub data: i32,
33}
34
35impl WriteTo for CAddEntity {
36    fn write(&self, writer: &mut impl std::io::Write) -> std::io::Result<()> {
37        VarInt(self.id).write(writer)?;
38        self.uuid.write(writer)?;
39        VarInt(self.entity_type).write(writer)?;
40        writer.write_all(&self.x.to_be_bytes())?;
41        writer.write_all(&self.y.to_be_bytes())?;
42        writer.write_all(&self.z.to_be_bytes())?;
43        // Velocity as LpVec3 (zero velocity = single 0 byte)
44        writer.write_all(&[0u8])?;
45        self.x_rot.write(writer)?;
46        self.y_rot.write(writer)?;
47        self.head_y_rot.write(writer)?;
48        VarInt(self.data).write(writer)
49    }
50}
51
52impl CAddEntity {
53    /// Creates a new CAddEntity packet for spawning a player.
54    #[must_use]
55    #[allow(clippy::too_many_arguments)]
56    pub fn player(
57        id: i32,
58        uuid: Uuid,
59        entity_type_id: i32,
60        x: f64,
61        y: f64,
62        z: f64,
63        yaw: f32,
64        pitch: f32,
65    ) -> Self {
66        Self {
67            id,
68            uuid,
69            entity_type: entity_type_id,
70            x,
71            y,
72            z,
73            x_rot: ((pitch / 360.0) * 256.0) as i8,
74            y_rot: ((yaw / 360.0) * 256.0) as i8,
75            head_y_rot: ((yaw / 360.0) * 256.0) as i8,
76            data: 0,
77        }
78    }
