1use std::io::{Result, Write};
2
3use steel_macros::ClientPacket;
4use steel_registry::packets::play::C_SECTION_BLOCKS_UPDATE;
5use steel_utils::{
6    BlockPos, BlockStateId, SectionPos,
7    codec::{VarInt, VarLong},
8    serial::WriteTo,
9};
10
11#[derive(ClientPacket, Clone, Debug)]
12#[packet_id(Play = C_SECTION_BLOCKS_UPDATE)]
13pub struct CSectionBlocksUpdate {
14    pub section_pos: SectionPos,
15    pub changes: Vec<BlockChange>,
16}
17
18#[derive(Clone, Debug)]
19pub struct BlockChange {
20    pub pos: BlockPos,
21    pub block_state: BlockStateId,
22}
23
24impl WriteTo for CSectionBlocksUpdate {
25    fn write(&self, writer: &mut impl Write) -> Result<()> {
26        // Write section position
27        self.section_pos.write(writer)?;
28
29        // Write number of changes
30        VarInt(self.changes.len() as i32).write(writer)?;
31
32        // Write each change as a packed VarLong
33        for change in &self.changes {
34            // Calculate section-relative position (12 bits total: 4 bits each for x, y, z)
35            let section_x = change.pos.0.x & 0xF;
36            let section_y = change.pos.0.y & 0xF;
37            let section_z = change.pos.0.z & 0xF;
38
39            // Pack as: (section_x << 8) | (section_z << 4) | section_y
40            let packed_pos = ((section_x << 8) | (section_z << 4) | section_y) as u16;
41
42            // Pack as: (block_state_id << 12) | packed_pos
43            let block_id = i32::from(change.block_state.0);
44            let packed = i64::from(block_id) << 12 | i64::from(packed_pos);
45
46            VarLong(packed).write(writer)?;
47        }
48
49        Ok(())
50    }
51}
