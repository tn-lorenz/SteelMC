20pub enum CommandNode {
21    Root {
22        children: Vec<i32>,
23    },
24    Literal {
25        children: Vec<i32>,
26        redirects_to: Option<i32>,
27        name: Cow<'static, str>,
28        is_executable: bool,
29    },
30    Argument {
31        children: Vec<i32>,
32        redirects_to: Option<i32>,
33        name: Cow<'static, str>,
34        is_executable: bool,
35        parser: ArgumentType,
36        suggestions_type: Option<SuggestionType>,
37    },
38}

40impl CommandNode {
41    const FLAG_IS_EXECUTABLE: u8 = 4;
42    const FLAG_HAS_REDIRECT: u8 = 8;
43    const FLAG_HAS_SUGGESTION_TYPE: u8 = 16;
44
45    pub fn new_root() -> Self {
46        Self::Root {
47            children: Vec::new(),
48        }
49    }
50
51    pub fn new_literal(info: CommandNodeInfo, name: impl Into<Cow<'static, str>>) -> Self {
52        Self::Literal {
53            children: info.children,
54            name: name.into(),
55            is_executable: info.is_executable,
56            redirects_to: info.redirects_to,
57        }
58    }
59
60    pub fn new_argument(
61        info: CommandNodeInfo,
62        name: impl Into<Cow<'static, str>>,
63        argument: (ArgumentType, Option<SuggestionType>),
64    ) -> Self {
65        Self::Argument {
66            children: info.children,
67            name: name.into(),
68            is_executable: info.is_executable,
69            redirects_to: info.redirects_to,
70            parser: argument.0,
71            suggestions_type: argument.1,
72        }
73    }
74
75    fn flags(&self) -> u8 {
76        let (mut flags, is_executable, has_redirect, has_suggestions_type) = match self {
77            CommandNode::Root { .. } => (0, false, false, false),
78            CommandNode::Literal {
79                is_executable,
80                redirects_to,
81                ..
82            } => (1, *is_executable, redirects_to.is_some(), false),
83            CommandNode::Argument {
84                is_executable,
85                redirects_to: r,
86                suggestions_type,
87                ..
88            } => (2, *is_executable, r.is_some(), suggestions_type.is_some()),
89        };
90
91        if is_executable {
92            flags |= Self::FLAG_IS_EXECUTABLE
93        }
94        if has_redirect {
95            flags |= Self::FLAG_HAS_REDIRECT
96        }
97        if has_suggestions_type {
98            flags |= Self::FLAG_HAS_SUGGESTION_TYPE
99        }
100        flags
101    }
102
103    pub fn set_children(&mut self, children: Vec<i32>) {
104        match self {
105            CommandNode::Root { children: c } => *c = children,
106            CommandNode::Literal { children: c, .. } => *c = children,
107            CommandNode::Argument { children: c, .. } => *c = children,
108        }
109    }
110
111    fn children(&self) -> &[i32] {
112        match self {
113            CommandNode::Root { children } => children,
114            CommandNode::Literal { children, .. } => children,
115            CommandNode::Argument { children, .. } => children,
116        }
117    }
118
119    fn redirects_to(&self) -> &Option<i32> {
120        match self {
121            CommandNode::Root { .. } => &None,
122            CommandNode::Literal { redirects_to, .. } => redirects_to,
123            CommandNode::Argument { redirects_to, .. } => redirects_to,
124        }
125    }
126
127    fn name(&self) -> Option<&str> {
128        match self {
129            CommandNode::Root { .. } => None,
130            CommandNode::Literal { name, .. } => Some(name),
131            CommandNode::Argument { name, .. } => Some(name),
132        }
133    }
134}
135
136impl WriteTo for CommandNode {
137    fn write(&self, writer: &mut impl Write) -> Result<()> {
138        writer.write_all(&self.flags().to_be_bytes())?;
139        let children = self.children();
140        VarInt(children.len() as i32).write(writer)?;
141        for child in children {
142            VarInt(*child).write(writer)?;
143        }
144
145        if let Some(redirects_to) = self.redirects_to() {
146            VarInt(*redirects_to).write(writer)?;
147        }
148
149        if let Some(name) = self.name() {
150            name.write_prefixed::<VarInt>(writer)?;
151        }
152
153        if let CommandNode::Argument {
154            parser,
155            suggestions_type,
156            ..
157        } = self
158        {
159            parser.write(writer)?;
160
161            if let Some(suggestions_type) = suggestions_type {
162                suggestions_type.as_str().write_prefixed::<VarInt>(writer)?;
163            }
164        }
165
166        Ok(())
167    }
168}

170pub struct CommandNodeInfo {
171    children: Vec<i32>,
172    is_executable: bool,
173    redirects_to: Option<i32>,
174}
175
176impl CommandNodeInfo {
177    pub fn new(children: Vec<i32>) -> Self {
178        Self {
179            children,
180            is_executable: false,
181            redirects_to: None,
182        }
183    }
184
185    pub fn new_executable() -> Self {
186        Self {
187            children: Vec::new(),
188            is_executable: true,
189            redirects_to: None,
190        }
191    }
192
193    pub fn new_redirect(redirects_to: i32) -> Self {
194        Self {
195            children: Vec::new(),
196            is_executable: false,
197            redirects_to: Some(redirects_to),
198        }
199    }
200
201    pub fn chain(mut self, mut other: Self) -> Self {
202        self.children.append(&mut other.children);
203        self.is_executable |= other.is_executable;
204        self
205    }
206}

296#[derive(Debug, Clone, Copy)]
297pub enum ArgumentStringTypeBehavior {
298    SingleWord,
299    QuotablePhrase,
300    GreedyPhrase,
301}
302
303impl WriteTo for ArgumentType {
304    fn write(&self, writer: &mut impl Write) -> Result<()> {
305        // Safety: Since Self is repr(u32), it is guaranteed to hold the discriminant in the first 4 bytes
306        // See https://doc.rust-lang.org/reference/items/enumerations.html#pointer-casting
307        let id = unsafe { *(self as *const Self as *const i32) };
308        VarInt(id).write(writer)?;
309
310        match self {
311            Self::Float { min, max } => Self::write_min_max(*min, *max, writer),
312            Self::Double { min, max } => Self::write_min_max(*min, *max, writer),
313            Self::Integer { min, max } => Self::write_min_max(*min, *max, writer),
314            Self::Long { min, max } => Self::write_min_max(*min, *max, writer),
315            Self::String { behavior } => {
316                let i = match behavior {
317                    ArgumentStringTypeBehavior::SingleWord => 0,
318                    ArgumentStringTypeBehavior::QuotablePhrase => 1,
319                    ArgumentStringTypeBehavior::GreedyPhrase => 2,
320                };
321                VarInt(i).write(writer)
322            }
323            Self::Entity { flags } => flags.write(writer),
324            Self::ScoreHolder { flags } => flags.write(writer),
325            Self::Time { min } => min.write(writer),
326            Self::ResourceOrTag { identifier } => identifier.write_prefixed::<VarInt>(writer),
327            Self::ResourceOrTagKey { identifier } => identifier.write_prefixed::<VarInt>(writer),
328            Self::Resource { identifier } => identifier.write_prefixed::<VarInt>(writer),
329            Self::ResourceKey { identifier } => identifier.write_prefixed::<VarInt>(writer),
330            _ => Ok(()),
331        }
332    }
333}
334
335impl ArgumentType {
336    fn write_min_max<T: WriteTo>(
337        min: Option<T>,
338        max: Option<T>,
339        writer: &mut impl Write,
340    ) -> Result<()> {
341        // none = 0
342        // min = 1
343        // max = 2
344        // min & max = 3
345        (min.is_some() as u8 + max.is_some() as u8 + max.is_some() as u8).write(writer)?;
346
347        if let Some(min) = min {
348            min.write(writer)?;
349        }
350        if let Some(max) = max {
351            max.write(writer)?;
352        }
353
354        Ok(())
355    }
356}
