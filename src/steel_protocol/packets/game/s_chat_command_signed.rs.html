7/// Client -> Server: Executes a signed command.
8///
9/// Commands with signable arguments have each argument individually signed.
10/// This prevents tampering with command arguments.
11///
12/// Equivalent to ServerboundChatCommandSignedPacket in Minecraft.
13#[derive(ServerPacket, Clone, Debug)]
14pub struct SChatCommandSigned {
15    /// The command string (without leading slash)
16    pub command: String,
17
18    /// Timestamp when command was issued (milliseconds since epoch)
19    pub timestamp: i64,
20
21    /// Random salt for uniqueness
22    pub salt: i64,
23
24    /// Signatures for each command argument
25    pub argument_signatures: Vec<ArgumentSignature>,
26
27    /// Acknowledgment of previously seen messages
28    pub last_seen: LastSeenMessagesUpdate,
29}
30
31impl ReadFrom for SChatCommandSigned {
32    fn read(reader: &mut Cursor<&[u8]>) -> std::io::Result<Self> {
33        let command = String::read_prefixed_bound::<VarInt>(reader, 256)?;
34        let timestamp = i64::read(reader)?;
35        let salt = i64::read(reader)?;
36
37        let arg_count = VarInt::read(reader)?.0 as usize;
38        if arg_count > 8 {
39            return Err(std::io::Error::new(
40                std::io::ErrorKind::InvalidData,
41                "Too many argument signatures",
42            ));
43        }
44        let mut argument_signatures = Vec::with_capacity(arg_count);
45        for _ in 0..arg_count {
46            argument_signatures.push(ArgumentSignature::read(reader)?);
47        }
48
49        let last_seen = LastSeenMessagesUpdate::read(reader)?;
50
51        Ok(Self {
52            command,
53            timestamp,
54            salt,
55            argument_signatures,
56            last_seen,
57        })
58    }
59}

61/// Signature for a single command argument
62#[derive(Clone, Debug)]
63pub struct ArgumentSignature {
64    /// The argument name
65    pub name: String,
66
67    /// The signature bytes (256 bytes for RSA 2048-bit)
68    pub signature: [u8; 256],
69}
70
71impl ReadFrom for ArgumentSignature {
72    fn read(reader: &mut Cursor<&[u8]>) -> std::io::Result<Self> {
73        // Read argument name (max 16 chars)
74        let name = String::read_prefixed_bound::<VarInt>(reader, 16)?;
75
76        // Read signature
77        let mut signature = [0u8; 256];
78        reader.read_exact(&mut signature)?;
79
80        Ok(Self { name, signature })
81    }
82}

84/// Last seen messages update from client
85#[derive(Clone, Debug)]
86pub struct LastSeenMessagesUpdate {
87    /// Offset to advance the message window
88    pub offset: VarInt,
89
90    /// BitSet indicating which of the last 20 messages were acknowledged
91    /// 3 bytes = 24 bits (using 20)
92    pub acknowledged: [u8; 3],
93}
94
95impl ReadFrom for LastSeenMessagesUpdate {
96    fn read(reader: &mut Cursor<&[u8]>) -> std::io::Result<Self> {
97        let offset = VarInt::read(reader)?;
98        let mut acknowledged = [0u8; 3];
99        reader.read_exact(&mut acknowledged)?;
100
101        Ok(Self {
102            offset,
103            acknowledged,
104        })
105    }
106}
