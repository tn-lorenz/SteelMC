14/// Result of a ray cast hitting a block.
15///
16/// This is kept in steel-registry because it's used by steel-protocol
17/// for packet deserialization.
18#[derive(Debug, Clone)]
19pub struct BlockHitResult {
20    /// The exact location where the ray hit the block.
21    pub location: Vector3<f64>,
22    /// The face of the block that was hit.
23    pub direction: Direction,
24    /// The position of the block that was hit.
25    pub block_pos: BlockPos,
26    /// Whether this is a miss (no block hit).
27    pub miss: bool,
28    /// Whether the hit location is inside the block.
29    pub inside: bool,
30    /// Whether the world border was hit.
31    pub world_border_hit: bool,
32}
33
34impl ReadFrom for BlockHitResult {
35    fn read(data: &mut Cursor<&[u8]>) -> io::Result<Self> {
36        let block_pos = BlockPos::read(data)?;
37        let direction = Direction::read(data)?;
38        // Click coordinates are relative to the block position (0.0 to 1.0 range)
39        let click_x = f32::read(data)?;
40        let click_y = f32::read(data)?;
41        let click_z = f32::read(data)?;
42        let inside = bool::read(data)?;
43        let world_border_hit = bool::read(data)?;
44
45        // Convert to absolute world coordinates by adding block position
46        // (matching Java's FriendlyByteBuf.readBlockHitResult)
47        let location = Vector3::new(
48            f64::from(block_pos.x()) + f64::from(click_x),
49            f64::from(block_pos.y()) + f64::from(click_y),
50            f64::from(block_pos.z()) + f64::from(click_z),
51        );
52
53        Ok(BlockHitResult {
54            location,
55            direction,
56            block_pos,
57            miss: false,
58            inside,
59            world_border_hit,
60        })
61    }
62}
