126impl NumberProvider {
127    /// Get a value from this provider using the given RNG.
128    pub fn get<R: rand::Rng>(&self, rng: &mut R, ctx: Option<&LootContextRef<'_>>) -> f32 {
129        match self {
130            Self::Constant(v) => *v,
131            Self::Uniform { min, max } => rng.random_range(*min..=*max),
132            Self::Binomial { n, p } => {
133                let mut count = 0;
134                for _ in 0..*n {
135                    if rng.random::<f32>() < *p {
136                        count += 1;
137                    }
138                }
139                count as f32
140            }
141            Self::Score { .. } => {
142                // TODO: Implement when scoreboard system is available
143                let _ = ctx;
144                0.0
145            }
146            Self::Storage { .. } => {
147                // TODO: Implement when command storage system is available
148                let _ = ctx;
149                0.0
150            }
151            Self::EnchantmentLevel { enchantment } => ctx
152                .and_then(|c| c.tool)
153                .map(|t| t.get_enchantment_level(enchantment) as f32)
154                .unwrap_or(0.0),
155        }
156    }
157
158    /// Get a value without context (for backwards compatibility).
159    pub fn get_simple(&self, rng: &mut impl rand::Rng) -> f32 {
160        match self {
161            Self::Constant(v) => *v,
162            Self::Uniform { min, max } => rng.random_range(*min..=*max),
163            Self::Binomial { n, p } => {
164                let mut count = 0;
165                for _ in 0..*n {
166                    if rng.random::<f32>() < *p {
167                        count += 1;
168                    }
169                }
170                count as f32
171            }
172            // Context-dependent providers return 0 when no context available
173            Self::Score { .. } | Self::Storage { .. } | Self::EnchantmentLevel { .. } => 0.0,
174        }
175    }
176
177    /// Get the value as an integer.
178    pub fn get_int(&self, rng: &mut impl rand::Rng) -> i32 {
179        self.get_simple(rng).floor() as i32
180    }
181
182    /// Get the value as an integer with context.
183    pub fn get_int_with_ctx<R: rand::Rng>(
184        &self,
185        rng: &mut R,
186        ctx: Option<&LootContextRef<'_>>,
187    ) -> i32 {
188        self.get(rng, ctx).floor() as i32
189    }

196    pub max: Option<NumberProvider>,
197}

199impl NumberProviderRange {
200    /// Check if a value is within this range.
201    pub fn test(&self, value: f32, rng: &mut impl rand::Rng) -> bool {
202        if let Some(min) = &self.min
203            && value < min.get_simple(rng)
204        {
205            return false;
206        }
207        if let Some(max) = &self.max
208            && value > max.get_simple(rng)
209        {
210            return false;
211        }
212        true
213    }
214
215    /// Create an exact match range.
216    pub const fn exact(value: f32) -> Self {
217        Self {
218            min: Some(NumberProvider::Constant(value)),
219            max: Some(NumberProvider::Constant(value)),
220        }
221    }

223    /// Create an at-least range.
224    pub const fn at_least(min: f32) -> Self {
225        Self {
226            min: Some(NumberProvider::Constant(min)),
227            max: None,
228        }
229    }

231    /// Create an at-most range.
232    pub const fn at_most(max: f32) -> Self {
233        Self {
234            min: None,

236        }
237    }

677impl LootCondition {
678    /// Test if this condition passes given the loot context.
679    pub fn test<R: rand::Rng>(&self, ctx: &mut LootContext<'_, R>) -> bool {
680        match self {
681            LootCondition::SurvivesExplosion => {
682                if let Some(radius) = ctx.explosion_radius {
683                    // Vanilla: 1/radius chance to survive
684                    ctx.rng.random::<f32>() <= (1.0 / radius)
685                } else {
686                    true // No explosion, always survives
687                }
688            }
689            LootCondition::BlockStateProperty { block, properties } => {
690                if let Some(state) = ctx.block_state {
691                    let state_block = state.get_block();
692                    // Check block matches
693                    if state_block.key != *block {
694                        return false;
695                    }
696                    // Check all properties match
697                    for prop in *properties {
698                        if let Some(value) = state.get_property_str(prop.name) {
699                            if value != prop.value {
700                                return false;
701                            }
702                        } else {
703                            return false; // Property doesn't exist
704                        }
705                    }
706                    true
707                } else {
708                    false // No block state in context
709                }
710            }
711            LootCondition::RandomChance(chance) => ctx.rng.random::<f32>() < *chance,
712            LootCondition::RandomChanceWithEnchantedBonus {
713                enchantment,
714                unenchanted_chance,
715                enchanted_chance,
716            } => {
717                let level = ctx.get_enchantment_level_by_id(enchantment);
718                let effective_chance = if level > 0 {
719                    match enchanted_chance {
720                        EnchantedChance::Constant(c) => *c,
721                        EnchantedChance::Linear {
722                            base,
723                            per_level_above_first,
724                        } => base + per_level_above_first * (level - 1) as f32,
725                    }
726                } else {
727                    *unenchanted_chance
728                };
729                ctx.rng.random::<f32>() < effective_chance
730            }
731            LootCondition::MatchTool(predicate) => {
732                if let Some(tool) = ctx.tool {
733                    predicate.test(tool, ctx)
734                } else {
735                    // No tool in context - only passes if predicate is Any
736                    matches!(predicate, ToolPredicate::Any)
737                }
738            }
739            LootCondition::TableBonus {
740                enchantment,
741                chances,
742            } => {
743                let level = ctx.get_enchantment_level_by_id(enchantment);
744                let index = (level as usize).min(chances.len().saturating_sub(1));
745                let chance = chances.get(index).copied().unwrap_or(0.0);
746                ctx.rng.random::<f32>() < chance
747            }
748            LootCondition::Inverted(inner) => !inner.test(ctx),
749            LootCondition::AnyOf(conditions) => conditions.iter().any(|c| c.test(ctx)),
750            LootCondition::AllOf(conditions) => conditions.iter().all(|c| c.test(ctx)),
751            LootCondition::KilledByPlayer => ctx.killed_by_player,
752            LootCondition::EntityProperties { .. } => {
753                // TODO: Implement when entity data is available in context
754                true
755            }
756            LootCondition::DamageSourceProperties { .. } => {
757                // TODO: Implement when damage source data is available in context
758                true
759            }
760            LootCondition::LocationCheck { .. } => {
761                // TODO: Implement when world position data is available in context
762                true
763            }
764            LootCondition::WeatherCheck {
765                raining,
766                thundering,
767            } => {
768                let weather = ctx.weather.unwrap_or_default();
769                raining.is_none_or(|r| r == weather.raining)
770                    && thundering.is_none_or(|t| t == weather.thundering)
771            }
772            LootCondition::TimeCheck { value, period } => {
773                let game_time = ctx.game_time.unwrap_or(0);
774                let time = if let Some(p) = period {
775                    game_time % p
776                } else {
777                    game_time
778                };
779                value.test(time as f32, ctx.rng)
780            }
781            LootCondition::ValueCheck { value, range } => {
782                let v = value.get_simple(ctx.rng);
783                range.test(v, ctx.rng)
784            }
785            LootCondition::EnchantmentActiveCheck {
786                enchantment,
787                active,
788            } => {
789                let level = ctx.get_enchantment_level_by_id(enchantment);
790                let is_active = level > 0;
791                is_active == *active
792            }
793            LootCondition::EntityScores { .. } => {
794                // TODO: Implement when scoreboard system is available
795                true
796            }
797            LootCondition::Reference(_name) => {
798                // TODO: Implement condition registry lookup
799                // For now, return true (permissive)
800                true
801            }
802        }
803    }
804}

1287impl LootTable {
1288    /// Generate random items from this loot table.
1289    ///
1290    /// # Arguments
1291    /// * `ctx` - The loot context containing RNG, luck, block state, tool, etc.
1292    ///
1293    /// This follows vanilla's approach:
1294    /// 1. For each pool, check conditions
1295    /// 2. Roll `rolls + floor(bonus_rolls * luck)` times
1296    /// 3. Each roll does weighted random selection among valid entries
1297    /// 4. Apply entry-level functions to each item
1298    /// 5. Apply pool-level functions to all items from that pool
1299    /// 6. Apply table-level functions to all items from the table
1300    pub fn get_random_items<R: rand::Rng>(&self, ctx: &mut LootContext<'_, R>) -> Vec<ItemStack> {
1301        let mut result = Vec::new();
1302        for pool in self.pools {
1303            pool.add_random_items(ctx, &mut result);
1304        }
1305
1306        // Apply table-level functions to all items
1307        if !self.functions.is_empty() {
1308            for item in &mut result {
1309                for cond_func in self.functions {
1310                    if cond_func.conditions.iter().all(|c| c.test(ctx)) {
1311                        cond_func.function.apply(item, ctx);
1312                    }
1313                }
1314            }
1315            // Remove items with zero count after applying functions
1316            result.retain(|item| item.count > 0);
1317        }
1318
1319        result
1320    }
1321}

1323impl LootPool {
1324    /// Add random items from this pool to the result.
1325    fn add_random_items<R: rand::Rng>(
1326        &self,
1327        ctx: &mut LootContext<'_, R>,
1328        result: &mut Vec<ItemStack>,
1329    ) {
1330        // Check pool conditions
1331        for condition in self.conditions {
1332            if !condition.test(ctx) {
1333                return;
1334            }
1335        }
1336
1337        // Track where items from this pool start
1338        let start_index = result.len();
1339
1340        // Calculate number of rolls
1341        let roll_count = self.rolls.get_int(ctx.rng) + (self.bonus_rolls * ctx.luck).floor() as i32;
1342
1343        for _ in 0..roll_count {
1344            self.add_random_item(ctx, result);
1345        }
1346
1347        // Apply pool-level functions to all items generated by this pool
1348        if !self.functions.is_empty() {
1349            for item in result.iter_mut().skip(start_index) {
1350                for cond_func in self.functions {
1351                    if cond_func.conditions.iter().all(|c| c.test(ctx)) {
1352                        cond_func.function.apply(item, ctx);
1353                    }
1354                }
1355            }
1356            // Remove items with zero count after applying functions
1357            result.retain(|item| item.count > 0);
1358        }
1359    }
1360
1361    /// Select and add a single random item from this pool.
1362    fn add_random_item<R: rand::Rng>(
1363        &self,
1364        ctx: &mut LootContext<'_, R>,
1365        result: &mut Vec<ItemStack>,
1366    ) {
1367        // Collect valid entries with their effective weights
1368        let mut valid_entries: Vec<(&LootEntry, i32)> = Vec::new();
1369        let mut total_weight = 0;
1370
1371        for entry in self.entries {
1372            // Check entry conditions
1373            let passes_conditions = entry.conditions().iter().all(|c| c.test(ctx));
1374
1375            if !passes_conditions {
1376                continue;
1377            }
1378
1379            let weight = entry.effective_weight(ctx.luck);
1380            if weight > 0 {
1381                valid_entries.push((entry, weight));
1382                total_weight += weight;
1383            }
1384        }
1385
1386        if total_weight == 0 || valid_entries.is_empty() {
1387            return;
1388        }
1389
1390        // Weighted random selection
1391        let selected = if valid_entries.len() == 1 {
1392            valid_entries[0].0
1393        } else {
1394            let mut index = ctx.rng.random_range(0..total_weight);
1395            let mut selected_entry = valid_entries[0].0;
1396            for (entry, weight) in &valid_entries {
1397                index -= weight;
1398                if index < 0 {
1399                    selected_entry = entry;
1400                    break;
1401                }
1402            }
1403            selected_entry
1404        };
1405
1406        // Generate item(s) from the selected entry
1407        selected.create_items(ctx, result);
1408    }
1409}

1430                    }
1431
1432                    if item.count > 0 {
1433                        result.push(item);
1434                    }
1435                }
1436            }
1437            LootEntry::LootTableRef {
1438                name, functions, ..
1439            } => {
1440                // Recursively get items from referenced loot table
1441                if let Some(table) = REGISTRY.loot_tables.by_key(name) {
1442                    let mut items = table.get_random_items(ctx);
1443                    // Apply functions to all items from the referenced table
1444                    for item in &mut items {
1445                        for cond_func in *functions {
1446                            if cond_func.conditions.iter().all(|c| c.test(ctx)) {
1447                                cond_func.function.apply(item, ctx);
1448                            }
1449                        }
1450                    }
1451                    result.extend(items.into_iter().filter(|i| i.count > 0));
1452                }
1453            }
1454            LootEntry::InlineLootTable {
1455                pools, functions, ..
1456            } => {
1457                // Process inline loot table pools directly
1458                let mut items = Vec::new();
1459                for pool in *pools {
1460                    pool.add_random_items(ctx, &mut items);
1461                }
1462                // Apply functions to all items from the inline table
1463                for item in &mut items {
1464                    for cond_func in *functions {
1465                        if cond_func.conditions.iter().all(|c| c.test(ctx)) {
1466                            cond_func.function.apply(item, ctx);
1467                        }
1468                    }
1469                }
1470                result.extend(items.into_iter().filter(|i| i.count > 0));
1471            }
1472            LootEntry::Tag {
1473                name,
1474                expand,
1475                functions,
1476                ..
1477            } => {
1478                // Get all items in the tag
1479                if let Some(items) = REGISTRY.items.get_tag(name) {
1480                    if *expand {
1481                        // Pick one random item from the tag (weighted equally)
1482                        if !items.is_empty() {
1483                            let index = ctx.rng.random_range(0..items.len());
1484                            let mut item = ItemStack::new(items[index]);
1485                            for cond_func in *functions {
1486                                if cond_func.conditions.iter().all(|c| c.test(ctx)) {
1487                                    cond_func.function.apply(&mut item, ctx);
1488                                }
1489                            }
1490                            if item.count > 0 {
1491                                result.push(item);
1492                            }
1493                        }
1494                    } else {
1495                        // Drop all items from the tag
1496                        for item_ref in items {
1497                            let mut item = ItemStack::new(item_ref);
1498                            for cond_func in *functions {
1499                                if cond_func.conditions.iter().all(|c| c.test(ctx)) {
1500                                    cond_func.function.apply(&mut item, ctx);
1501                                }
1502                            }
1503                            if item.count > 0 {
1504                                result.push(item);
1505                            }
1506                        }
1507                    }
1508                }
1509            }
1510            LootEntry::Alternatives { children, .. } => {
1511                // Try children in order, use first that passes conditions and produces items
1512                for child in *children {
1513                    // Check child's conditions first
1514                    let passes_conditions = child.conditions().iter().all(|c| c.test(ctx));
1515                    if !passes_conditions {
1516                        continue; // Try next alternative
1517                    }
1518
1519                    let before_len = result.len();
1520                    child.create_items(ctx, result);
1521                    if result.len() > before_len {
1522                        break; // First successful child that produced items, stop
1523                    }
1524                }
1525            }
1526            LootEntry::Group { children, .. } => {
1527                // Use all children that pass their conditions
1528                for child in *children {
1529                    let passes_conditions = child.conditions().iter().all(|c| c.test(ctx));
1530                    if passes_conditions {
1531                        child.create_items(ctx, result);
1532                    }
1533                }
1534            }
1535            LootEntry::Sequence { children, .. } => {
1536                // Use children in sequence until one fails its conditions
1537                // Note: Unlike Alternatives, Sequence stops when conditions FAIL,
1538                // not when items are produced. A child can produce nothing but still "succeed".
1539                for child in *children {
1540                    let passes_conditions = child.conditions().iter().all(|c| c.test(ctx));
1541                    if !passes_conditions {
1542                        break; // Condition failed, stop sequence
1543                    }
1544                    child.create_items(ctx, result);
1545                }
1546            }
1547            LootEntry::Empty { .. } => {
1548                // Empty entry produces nothing
1549            }
1550            LootEntry::Dynamic { name, .. } => {
1551                // Dynamic entries are used for block entity contents (like shulker boxes)
1552                // The name identifies what content to retrieve:
1553                // - "contents" = block entity inventory contents
1554                // - Other names may exist for specific use cases
1555                //
1556                // TODO: Implement when block entity system supports inventory retrieval
1557                // This requires:
1558                // 1. Block entity reference in LootContext
1559                // 2. Method to get inventory contents from block entity
1560                // 3. Adding those items to the result
1561                let _ = name;
1562            }
1563            LootEntry::Slots {
1564                slots, functions, ..
1565            } => {
1566                // Slots entries select items from specific block entity slots
1567                // TODO: Implement when block entity system supports slot access
1568                // This requires:
1569                // 1. Block entity reference in LootContext
1570                // 2. Method to get items from specific slots
1571                // 3. Apply functions to each retrieved item
1572                let _ = slots;
1573                let _ = functions;
1574            }
1575        }
1576    }
1577}

1581    ///
1582    /// This modifies the item in place. Functions can change:
1583    /// - Count (SetCount, ExplosionDecay, ApplyBonus, etc.)
1584    /// - Damage/durability (SetDamage)
1585    /// - Enchantments (EnchantRandomly, EnchantWithLevels, SetEnchantments)
1586    /// - Components/NBT (CopyComponents, SetComponents, CopyState)
1587    /// - Item type (FurnaceSmelt)
1588    /// - And more...
1589    pub fn apply<R: rand::Rng>(&self, item: &mut ItemStack, ctx: &mut LootContext<'_, R>) {
1590        match self {
1591            LootFunction::SetCount {
1592                count: provider,
1593                add,
1594            } => {
1595                let value = provider.get_int(ctx.rng);
1596                if *add {
1597                    item.count += value;
1598                } else {
1599                    item.count = value;
1600                }
1601            }
1602            LootFunction::ExplosionDecay => {
1603                if let Some(radius) = ctx.explosion_radius {
1604                    // Each item has 1/radius chance to survive
1605                    let probability = 1.0 / radius;
1606                    let mut result_count = 0;
1607                    for _ in 0..item.count {
1608                        if ctx.rng.random::<f32>() <= probability {
1609                            result_count += 1;
1610                        }
1611                    }
1612                    item.count = result_count;
1613                }
1614            }
1615            LootFunction::ApplyBonus {
1616                enchantment,
1617                formula,
1618            } => {
1619                let level = ctx.get_enchantment_level_by_id(enchantment);
1620                item.count = formula.apply(item.count, level, ctx.rng);
1621            }
1622            LootFunction::EnchantedCountIncrease {
1623                enchantment,
1624                count: provider,
1625                limit,
1626            } => {
1627                let level = ctx.get_enchantment_level_by_id(enchantment);
1628                if level > 0 {
1629                    let bonus = (provider.get_simple(ctx.rng) * level as f32).round() as i32;
1630                    let bonus = if *limit > 0 { bonus.min(*limit) } else { bonus };
1631                    item.count += bonus;
1632                }
1633            }
1634            LootFunction::LimitCount { min, max } => {
1635                if let Some(min_val) = min {
1636                    item.count = item.count.max(*min_val);
1637                }
1638                if let Some(max_val) = max {
1639                    item.count = item.count.min(*max_val);
1640                }
1641            }

1792impl BonusFormula {
1793    /// Apply the bonus formula to calculate new count.
1794    pub fn apply<R: rand::Rng>(&self, count: i32, level: i32, rng: &mut R) -> i32 {
1795        match self {
1796            BonusFormula::OreDrops => {
1797                if level > 0 {
1798                    // Vanilla: count * (max(0, random(0..level+2) - 1) + 1)
1799                    let bonus = rng.random_range(0..level + 2) - 1;
1800                    let multiplier = bonus.max(0) + 1;
1801                    count * multiplier
1802                } else {
1803                    count
1804                }
1805            }
1806            BonusFormula::UniformBonusCount { bonus_multiplier } => {
1807                // Vanilla: count + random(0..bonusMultiplier * level + 1)
1808                if level > 0 {
1809                    count + rng.random_range(0..bonus_multiplier * level + 1)
1810                } else {
1811                    count
1812                }
1813            }
1814            BonusFormula::BinomialWithBonusCount { extra, probability } => {
1815                // Vanilla: for each of (level + extra) trials, probability p to add 1
1816                let trials = level + extra;
1817                let mut bonus = 0;
1818                for _ in 0..trials {
1819                    if rng.random::<f32>() < *probability {
1820                        bonus += 1;
1821                    }
1822                }
1823                count + bonus
1824            }
1825        }
1826    }
1827}

1831/// Registry for loot tables.
1832pub struct LootTableRegistry {

1834    tables_by_key: FxHashMap<Identifier, usize>,
1835    allows_registering: bool,
1836}

1838impl LootTableRegistry {
1839    #[must_use]
1840    pub fn new() -> Self {
1841        Self {
1842            tables_by_id: Vec::new(),
1843            tables_by_key: FxHashMap::default(),
1844            allows_registering: true,
1845        }
1846    }
1847
1848    pub fn register(&mut self, table: LootTableRef) -> usize {
1849        assert!(
1850            self.allows_registering,
1851            "Cannot register loot tables after the registry has been frozen"
1852        );
1853
1854        let id = self.tables_by_id.len();
1855        self.tables_by_key.insert(table.key.clone(), id);
1856        self.tables_by_id.push(table);
1857        id
1858    }
1859
1860    /// Replaces a table at a given index.
1861    /// Returns true if the table was replaced and false if the table wasn't replaced
1862    #[must_use]
1863    pub fn replace(&mut self, table: LootTableRef, id: usize) -> bool {
1864        if id >= self.tables_by_id.len() {
1865            return false;
1866        }
1867        self.tables_by_id[id] = table;
1868        true
1869    }
1870
1871    #[must_use]
1872    pub fn by_id(&self, id: usize) -> Option<LootTableRef> {
1873        self.tables_by_id.get(id).copied()
1874    }
1875
1876    #[must_use]
1877    pub fn by_key(&self, key: &Identifier) -> Option<LootTableRef> {
1878        self.tables_by_key.get(key).and_then(|id| self.by_id(*id))
1879    }
1880
1881    pub fn iter(&self) -> impl Iterator<Item = (usize, LootTableRef)> + '_ {
1882        self.tables_by_id
1883            .iter()
1884            .enumerate()
1885            .map(|(id, &table)| (id, table))
1886    }
1887
1888    #[must_use]
1889    pub fn len(&self) -> usize {
1890        self.tables_by_id.len()
1891    }
1892
1893    #[must_use]
1894    pub fn is_empty(&self) -> bool {
1895        self.tables_by_id.is_empty()
1896    }
1897}
