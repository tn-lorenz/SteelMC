24// Vanilla's stream codec has no explicit guard, but deeper values already fail
25// its persistent NBT codec. Use that 512-level limit to avoid stack exhaustion.
26const MAX_TEMPLATE_CODEC_DEPTH: usize = 512;
27
28thread_local! {
29    static TEMPLATE_CODEC_DEPTH: Cell<usize> = const { Cell::new(0) };
30}
31
32struct TemplateDepthGuard;
33
34impl TemplateDepthGuard {
35    fn enter() -> Result<Self> {
36        TEMPLATE_CODEC_DEPTH.with(|depth| {
37            let current = depth.get();
38            if current >= MAX_TEMPLATE_CODEC_DEPTH {
39                return Err(Error::other(format!(
40                    "Item stack template nesting exceeds {MAX_TEMPLATE_CODEC_DEPTH}"
41                )));
42            }
43            depth.set(current + 1);
44            Ok(Self)
45        })
46    }
47}
48
49impl Drop for TemplateDepthGuard {
50    fn drop(&mut self) {
51        TEMPLATE_CODEC_DEPTH.with(|depth| depth.set(depth.get() - 1));
52    }
53}

55/// A persistable, non-empty item identity, count, and component patch.
56#[derive(Debug, Clone, PartialEq)]
57pub struct ItemStackTemplate {
58    item: ItemRef,
59    count: i32,
60    components: DataComponentPatch,
61    components_hash: Option<i32>,
62}
63
64impl ItemStackTemplate {
65    pub const MIN_COUNT: i32 = 1;
66    pub const MAX_COUNT: i32 = 99;
67
68    /// Creates the common count-one template with no component changes.
69    #[must_use]
70    pub fn new(item: ItemRef) -> Self {
71        assert!(
72            item != &*vanilla_items::AIR,
73            "Item stack template item must be non-empty"
74        );
75        Self {
76            item,
77            count: 1,
78            components: DataComponentPatch::new(),
79            components_hash: Some(empty_map_hash()),
80        }
81    }
82
83    /// Creates a template after validating its complete persistent codec shape.
84    pub fn try_with_count_and_patch(
85        item: ItemRef,
86        count: i32,
87        components: DataComponentPatch,
88    ) -> Result<Self> {
89        if item == &*vanilla_items::AIR {
90            return Err(Error::other("Item stack template item must be non-empty"));
91        }
92        if !(Self::MIN_COUNT..=Self::MAX_COUNT).contains(&count) {
93            return Err(Error::other(format!(
94                "Item stack template count {count} is outside the persistent range {}..={}",
95                Self::MIN_COUNT,
96                Self::MAX_COUNT
97            )));
98        }
99        components.try_to_nbt_tag_ref()?;
100        let components_hash = components.compute_persistent_hash()?;
101        Ok(Self {
102            item,
103            count,
104            components,
105            components_hash: Some(components_hash),
106        })
107    }
108
109    /// Copies a non-empty stack into its immutable template representation.
110    pub fn from_stack(stack: &ItemStack) -> Result<Self> {
111        if stack.is_empty() {
112            return Err(Error::other("Stack must be non-empty"));
113        }
114        Self::try_with_count_and_patch(stack.item, stack.count, stack.components_patch().clone())
115    }
116
117    pub(crate) fn validate_persistent_encoding(&self) -> Result<()> {
118        if self.item == &*vanilla_items::AIR
119            || !(Self::MIN_COUNT..=Self::MAX_COUNT).contains(&self.count)
120        {
121            return Err(Error::other("Item stack template is not persistable"));
122        }
123        self.components.try_to_nbt_tag_ref().map(|_| ())
124    }
125
126    #[must_use]
127    pub const fn item(&self) -> ItemRef {
128        self.item
129    }
130
131    #[must_use]
132    pub const fn count(&self) -> i32 {
133        self.count
134    }
135
136    #[must_use]
137    pub const fn components(&self) -> &DataComponentPatch {
138        &self.components
139    }
140
141    #[must_use]
142    pub fn create(&self) -> ItemStack {
143        let result =
144            ItemStack::with_count_and_patch(self.item, self.count, self.components.clone());
145        if let Err(error) = result.validate_strict() {
146            log::warn!("Can't create item stack with properties {self:?}, error: {error}");
147            return ItemStack::empty();
148        }
149        result
150    }
151
152    /// Creates the Vanilla hover event for this item template.
153    pub fn to_hover_event(&self) -> Result<HoverEvent> {
154        let components = if self.components.is_empty() {
155            None
156        } else {
157            Some(EncodedNbt::encode(&self.components)?)
158        };
159        Ok(HoverEvent::show_item(
160            self.item.key.to_string(),
161            Some(self.count),
162            components,
163        ))
164    }
165
166    pub(crate) fn to_nbt_tag_ref(&self) -> NbtTag {
167        let mut compound = NbtCompound::new();
168        compound.insert("id", self.item.key.to_string());
169        if self.count != 1 {
170            compound.insert("count", self.count);
171        }
172        if !self.components.is_empty() {
173            compound.insert("components", self.components.to_nbt_tag_ref());
174        }
175        NbtTag::Compound(compound)
176    }
177
178    pub(crate) fn from_nbt_identifier(value: &str) -> Option<Self> {
179        let _depth = TemplateDepthGuard::enter().ok()?;
180        let key = Identifier::from_str(value).ok()?;
181        let item = REGISTRY.items.by_key(&key)?;
182        (item != &*vanilla_items::AIR).then(|| Self::new(item))
183    }
184
185    pub(crate) fn from_nbt_compound(
186        compound: simdnbt::borrow::NbtCompound<'_, '_>,
187    ) -> Option<Self> {
188        let _depth = TemplateDepthGuard::enter().ok()?;
189        let key = Identifier::from_str(&compound.get("id")?.string()?.to_str()).ok()?;
190        let item = REGISTRY.items.by_key(&key)?;
191        let count = match compound.get("count") {
192            Some(count) => count.codec_i32()?,
193            None => 1,
194        };
195        let components = match compound.get("components") {
196            Some(components) => DataComponentPatch::from_nbt_tag(components)?,
197            None => DataComponentPatch::new(),
198        };
199        Self::try_with_count_and_patch(item, count, components).ok()
200    }
201
202    fn from_stream(item: ItemRef, count: i32, components: DataComponentPatch) -> Result<Self> {
203        if item == &*vanilla_items::AIR || count == 0 {
204            return Err(Error::other("Item stack template must be non-empty"));
205        }
206        let components_hash = components.compute_persistent_hash().ok();
207        Ok(Self {
208            item,
209            count,
210            components,
211            components_hash,
212        })
213    }
214
215    /// Gets the effective raw component value from the patch or item prototype.
216    #[must_use]
217    pub fn get_effective_value_raw(&self, key: &Identifier) -> Option<&ComponentData> {
218        match self.components.get_entry(key) {
219            Some(ComponentPatchEntry::Set(value)) => Some(value),
220            Some(ComponentPatchEntry::Removed) => None,
221            None => self.item.components.get_raw(key),
222        }
223    }
224
225    /// Gets an effective typed component value from the patch or item prototype.
226    #[must_use]
227    pub fn get<T: Component + DowncastType>(&self, component: DataComponentType<T>) -> Option<&T> {
228        self.get_effective_value_raw(&component.key)
229            .and_then(ComponentData::downcast_ref::<T>)
230    }
231
232    pub(crate) fn max_stack_size(&self) -> i32 {
233        self.get(MAX_STACK_SIZE).copied().unwrap_or(1)
234    }
235}

237impl WriteTo for ItemStackTemplate {
238    fn write(&self, writer: &mut impl Write) -> Result<()> {
239        let _depth = TemplateDepthGuard::enter()?;
240        let item_id = i32::try_from(self.item.id())
241            .map_err(|_| Error::other(format!("Item id is too large: {}", self.item.id())))?;
242        VarInt(item_id).write(writer)?;
243        VarInt(self.count).write(writer)?;
244        self.components.write(writer)
245    }
246}
247
248impl ReadFrom for ItemStackTemplate {
249    fn read(data: &mut Cursor<&[u8]>) -> Result<Self> {
250        let _depth = TemplateDepthGuard::enter()?;
251        let item_id = VarInt::read(data)?.0;
252        let item_id = usize::try_from(item_id)
253            .map_err(|_| Error::other(format!("Negative item id: {item_id}")))?;
254        let item = REGISTRY
255            .items
256            .by_id(item_id)
257            .ok_or_else(|| Error::other(format!("Unknown item id: {item_id}")))?;
258        let count = VarInt::read(data)?.0;
259        let components = DataComponentPatch::read(data)?;
260        Self::from_stream(item, count, components)
261    }
262}
263
264impl ToNbtTag for ItemStackTemplate {
265    fn to_nbt_tag(self) -> NbtTag {
266        self.to_nbt_tag_ref()
267    }
268}

270impl FromNbtTag for ItemStackTemplate {
271    fn from_nbt_tag(tag: simdnbt::borrow::NbtTag) -> Option<Self> {
272        if let Some(value) = tag.string() {
273            return Self::from_nbt_identifier(&value.to_str());
274        }
275        Self::from_nbt_compound(tag.compound()?)
276    }
277}

279impl HashComponent for ItemStackTemplate {
280    fn hash_component(&self, hasher: &mut ComponentHasher) {
281        let mut entries = Vec::with_capacity(3);
282        push_hash_entry(&mut entries, "id", self.item.key.to_string().compute_hash());
283        if self.count != 1 {
284            push_hash_entry(&mut entries, "count", self.count.compute_hash());
285        }
286        if !self.components.is_empty() {
287            let Some(components_hash) = self.components_hash else {
288                panic!("stream-only item stack template must validate before persistent hashing");
289            };
290            push_hash_entry(&mut entries, "components", components_hash);
291        }
292        sort_map_entries(&mut entries);
293        hasher.start_map();
294        for entry in &entries {
295            hasher.put_raw_bytes(&entry.key_bytes);
296            hasher.put_raw_bytes(&entry.value_bytes);
297        }
298        hasher.end_map();
299    }
300}
301
302fn empty_map_hash() -> i32 {
303    let mut hasher = ComponentHasher::new();
304    hasher.start_map();
305    hasher.end_map();
306    hasher.finish()
307}
308
309fn push_hash_entry(entries: &mut Vec<HashEntry>, key: &str, value_hash: i32) {
310    let key_hash = key.compute_hash() as u32;
311    let value_hash = value_hash as u32;
312    entries.push(HashEntry {
313        key_hash: i64::from(key_hash),
314        value_hash: i64::from(value_hash),
315        key_bytes: key_hash.to_le_bytes(),
316        value_bytes: value_hash.to_le_bytes(),
317    });
318}
