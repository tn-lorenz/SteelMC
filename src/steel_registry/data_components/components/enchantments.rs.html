12/// Enchantments stored on an item. Maps enchantment key to level.
13///
14/// Used by both the `minecraft:enchantments` component (on enchanted items)
15/// and the `minecraft:stored_enchantments` component (on enchanted books).
16///
17/// Vanilla moved tooltip visibility to the separate `TOOLTIP_DISPLAY` component.
18#[derive(Debug, Clone, PartialEq)]
19pub struct ItemEnchantments {
20    pub levels: FxHashMap<Identifier, u32>,
21}
22
23impl ItemEnchantments {
24    #[must_use]
25    pub fn empty() -> Self {
26        Self {
27            levels: FxHashMap::default(),
28        }
29    }
30
31    #[must_use]
32    pub fn get_level(&self, enchantment: &Identifier) -> u32 {
33        self.levels.get(enchantment).copied().unwrap_or(0)
34    }
35
36    pub fn set(&mut self, enchantment: Identifier, level: u32) {
37        if level == 0 {
38            self.levels.remove(&enchantment);
39        } else {
40            self.levels.insert(enchantment, level.min(255));
41        }
42    }
43
44    /// Vanilla `Mutable.upgrade`: keeps the higher of existing vs new level.
45    pub fn upgrade(&mut self, enchantment: Identifier, level: u32) {
46        if level > 0 {
47            let existing = self.get_level(&enchantment);
48            self.levels
49                .insert(enchantment, existing.max(level).min(255));
50        }
51    }
52
53    #[must_use]
54    pub fn is_empty(&self) -> bool {
55        self.levels.is_empty()
56    }
57
58    #[must_use]
59    pub fn len(&self) -> usize {
60        self.levels.len()
61    }
62
63    pub fn iter(&self) -> impl Iterator<Item = (&Identifier, &u32)> {
64        self.levels.iter()
65    }
66}
67
68impl Default for ItemEnchantments {
69    fn default() -> Self {
70        Self::empty()
71    }
72}

74/// Network format: `VarInt` count, then (`VarInt` `enchantment_id`, `VarInt` level) pairs.
75impl WriteTo for ItemEnchantments {
76    fn write(&self, writer: &mut impl std::io::Write) -> std::io::Result<()> {
77        let count = i32::try_from(self.levels.len()).map_err(|_| {
78            std::io::Error::other(format!(
79                "Enchantment map too large: {} entries",
80                self.levels.len()
81            ))
82        })?;
83        VarInt(count).write(writer)?;
84        for (key, &level) in &self.levels {
85            let id = REGISTRY
86                .enchantments
87                .id_from_key(key)
88                .ok_or_else(|| std::io::Error::other(format!("Unknown enchantment: {key}")))?;
89            let id = i32::try_from(id).map_err(|_| {
90                std::io::Error::other(format!("Enchantment id out of protocol range: {id}"))
91            })?;
92            if level > 255 {
93                return Err(std::io::Error::other(format!(
94                    "Enchantment {key} has invalid level {level}"
95                )));
96            }
97            VarInt(id).write(writer)?;
98            VarInt(level as i32).write(writer)?;
99        }
100        Ok(())
101    }
102}

104impl ReadFrom for ItemEnchantments {
105    fn read(data: &mut std::io::Cursor<&[u8]>) -> std::io::Result<Self> {
106        let count = VarInt::read(data)?.0;
107        let count = usize::try_from(count)
108            .map_err(|_| std::io::Error::other(format!("Negative enchantment count: {count}")))?;
109        let mut levels = FxHashMap::default();
110        levels.reserve(count.min(65_536));
111        for _ in 0..count {
112            let id = VarInt::read(data)?.0;
113            let id = usize::try_from(id)
114                .map_err(|_| std::io::Error::other(format!("Negative enchantment id: {id}")))?;
115            let level = VarInt::read(data)?.0;
116            if !(0..=255).contains(&level) {
117                return Err(std::io::Error::other(format!(
118                    "Enchantment level out of range: {level}"
119                )));
120            }
121            let enchantment = REGISTRY
122                .enchantments
123                .by_id(id)
124                .ok_or_else(|| std::io::Error::other(format!("Unknown enchantment id: {id}")))?;
125            levels.insert(enchantment.key.clone(), level as u32);
126        }
127        Ok(Self { levels })
128    }
129}

131/// NBT format: compound with enchantment identifiers as keys and int levels as values.
132impl ToNbtTag for ItemEnchantments {
133    fn to_nbt_tag(self) -> NbtTag {
134        let mut compound = NbtCompound::new();
135        for (key, level) in &self.levels {
136            compound.insert(key.to_string(), NbtTag::Int(*level as i32));
137        }
138        NbtTag::Compound(compound)
139    }
140}

142impl FromNbtTag for ItemEnchantments {
143    fn from_nbt_tag(tag: simdnbt::borrow::NbtTag) -> Option<Self> {
144        let compound = tag.compound()?;
145        let mut levels = FxHashMap::default();
146        for (key, value) in compound.iter() {
147            let ident = key.to_str().parse::<Identifier>().ok()?;
148            REGISTRY.enchantments.by_key(&ident)?;
149            let level = value.codec_i32()?;
150            if !(1..=255).contains(&level) {
151                return None;
152            }
153            if levels.insert(ident, level as u32).is_some() {
154                return None;
155            }
156        }
157        Some(Self { levels })
158    }
159}

161impl HashComponent for ItemEnchantments {
162    fn hash_component(&self, hasher: &mut ComponentHasher) {
163        hasher.start_map();
164        let mut entries: Vec<_> = self
165            .levels
166            .iter()
167            .map(|(key, &level)| {
168                let mut key_hasher = ComponentHasher::new();
169                key_hasher.put_string(&key.to_string());
170                let mut value_hasher = ComponentHasher::new();
171                value_hasher.put_int(level as i32);
172                HashEntry::new(key_hasher, value_hasher)
173            })
174            .collect();
175        sort_map_entries(&mut entries);
176        for entry in &entries {
177            hasher.put_raw_bytes(&entry.key_bytes);
178            hasher.put_raw_bytes(&entry.value_bytes);
179        }
180        hasher.end_map();
181    }
182}

195    fn with_borrowed_tag<R>(tag: NbtTag, visitor: impl FnOnce(BorrowedNbtTag<'_, '_>) -> R) -> R {
196        let mut bytes = Vec::new();
197        tag.write(&mut bytes);
198        let borrowed =
199            read_tag(&mut Cursor::new(bytes.as_slice())).expect("owned test tag should parse");
200        visitor(borrowed.as_tag())
201    }

203    fn parse_enchantments(compound: NbtCompound) -> Option<ItemEnchantments> {
204        with_borrowed_tag(NbtTag::Compound(compound), ItemEnchantments::from_nbt_tag)
205    }
