334/// Helper to create stub reader/writer functions for unimplemented components.
335/// These components use the Todo variant as a placeholder.
336macro_rules! register_stub {
337    ($registry:expr, $key:expr) => {{
338        fn network_reader(cursor: &mut std::io::Cursor<&[u8]>) -> std::io::Result<ComponentData> {
339            // Stub: read nothing, return Todo
340            let _ = cursor;
341            Ok(ComponentData::Todo)
342        }
343
344        fn network_writer(data: &ComponentData, _writer: &mut Vec<u8>) -> std::io::Result<()> {
345            // Stub: write nothing
346            let _ = data;
347            Ok(())
348        }
349
350        fn nbt_reader(_tag: simdnbt::borrow::NbtTag) -> Option<ComponentData> {
351            Some(ComponentData::Todo)
352        }
353
354        fn nbt_writer(_data: &ComponentData) -> simdnbt::owned::NbtTag {
355            simdnbt::owned::NbtTag::Compound(simdnbt::owned::NbtCompound::new())
356        }
357
358        $registry.register_dynamic(
359            $key,
360            crate::data_components::ComponentDataDiscriminant::Todo,
361            network_reader,
362            network_writer,
363            nbt_reader,
364            nbt_writer,
365        );
366    }};
367}

369// ==================== Custom Network Readers/Writers ====================
370
371/// Network reader for VarInt-encoded i32 components.
372fn varint_reader(cursor: &mut std::io::Cursor<&[u8]>) -> std::io::Result<ComponentData> {
373    use steel_utils::{codec::VarInt, serial::ReadFrom};
374    let value = VarInt::read(cursor)?;
375    Ok(ComponentData::I32(value.0))
376}
377
378/// Network writer for VarInt-encoded i32 components.
379fn varint_writer(data: &ComponentData, writer: &mut Vec<u8>) -> std::io::Result<()> {
380    use steel_utils::{codec::VarInt, serial::WriteTo};
381    if let ComponentData::I32(v) = data {
382        VarInt(*v).write(writer)
383    } else {
384        Err(std::io::Error::other("Component type mismatch"))
385    }
386}
