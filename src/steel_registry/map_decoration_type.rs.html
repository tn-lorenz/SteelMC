3use rustc_hash::FxHashMap;
4use steel_utils::Identifier;

6/// Registered visual and tracking behavior for a map decoration.
7#[derive(Debug)]
8pub struct MapDecorationType {
9    pub key: Identifier,
10    pub asset_id: Identifier,
11    pub show_on_item_frame: bool,
12    pub map_color: i32,
13    pub exploration_map_element: bool,
14    pub track_count: bool,
15}
16
17impl MapDecorationType {
18    pub const NO_MAP_COLOR: i32 = -1;
19
20    #[must_use]
21    pub const fn new(
22        key: Identifier,
23        asset_id: Identifier,
24        show_on_item_frame: bool,
25        map_color: i32,
26        exploration_map_element: bool,
27        track_count: bool,
28    ) -> Self {
29        Self {
30            key,
31            asset_id,
32            show_on_item_frame,
33            map_color,
34            exploration_map_element,
35            track_count,
36        }
37    }
38
39    #[must_use]
40    pub const fn has_map_color(&self) -> bool {
41        self.map_color != Self::NO_MAP_COLOR
42    }
43}
44
45pub type MapDecorationTypeRef = &'static MapDecorationType;
46
47pub struct MapDecorationTypeRegistry {
48    map_decoration_types_by_id: Vec<MapDecorationTypeRef>,
49    map_decoration_types_by_key: FxHashMap<Identifier, usize>,
50    allows_registering: bool,
51}
52
53impl MapDecorationTypeRegistry {
54    #[must_use]
55    pub fn new() -> Self {
56        Self {
57            map_decoration_types_by_id: Vec::new(),
58            map_decoration_types_by_key: FxHashMap::default(),
59            allows_registering: true,
60        }
61    }
62}

80#[cfg(test)]
81mod tests {
82    use steel_utils::Identifier;
83
84    use crate::test_support::init_test_registry;
85    use crate::{REGISTRY, RegistryExt};
86
87    #[test]
88    fn extracted_types_follow_vanilla_registry_order_and_fields() {
89        init_test_registry();
90        assert_eq!(REGISTRY.map_decoration_types.len(), 35);
91
92        let player = REGISTRY
93            .map_decoration_types
94            .by_id(0)
95            .expect("player decoration should be registered first");
96        assert_eq!(player.key, Identifier::vanilla_static("player"));
97        assert_eq!(player.asset_id, Identifier::vanilla_static("player"));
98        assert!(!player.show_on_item_frame);
99        assert!(!player.has_map_color());
100        assert!(!player.exploration_map_element);
101        assert!(player.track_count);
102
103        let trial_chambers = REGISTRY
104            .map_decoration_types
105            .by_id(34)
106            .expect("trial chambers decoration should be registered last");
107        assert_eq!(
108            trial_chambers.key,
109            Identifier::vanilla_static("trial_chambers")
110        );
111        assert_eq!(trial_chambers.map_color, 12_741_452);
112        assert!(trial_chambers.exploration_map_element);
113        assert!(!trial_chambers.track_count);
114    }
