31/// Registry operations required by a [`RegistryHolderSet`].
32///
33/// The trait keeps the holder-set codec independent from Steel's concrete
34/// registries. Plugin-owned entry types can provide the same operations through
35/// their own registry implementation.
36pub trait RegistryHolderSetEntry: RegistryEntry + Debug + Send + Sync {
37    /// Human-readable registry name used in codec errors.
38    const REGISTRY_NAME: &'static str;
39
40    /// Looks up an entry by its protocol registry ID.
41    fn holder_set_by_id(id: usize) -> Option<&'static Self>;
42
43    /// Looks up an entry by its registry key.
44    fn holder_set_by_key(key: &Identifier) -> Option<&'static Self>;
45
46    /// Returns whether the registry contains this tag.
47    fn holder_set_tag_exists(tag: &Identifier) -> bool;
48
49    /// Returns whether an entry belongs to this tag.
50    fn holder_set_tag_contains(entry: &'static Self, tag: &Identifier) -> bool;
51}
52
53/// Vanilla's homogeneous holder-set representation for a registry.
54#[derive(Debug, PartialEq)]
55pub enum RegistryHolderSet<T: RegistryHolderSetEntry> {
56    /// A named registry tag.
57    Tag(Identifier),
58    /// An ordered list of direct registry references.
59    Direct(Vec<&'static T>),
60}
61
62impl<T: RegistryHolderSetEntry> Clone for RegistryHolderSet<T> {
63    fn clone(&self) -> Self {
64        match self {
65            Self::Tag(tag) => Self::Tag(tag.clone()),
66            Self::Direct(entries) => Self::Direct(entries.clone()),
67        }
68    }
69}
70
71impl<T: RegistryHolderSetEntry> RegistryHolderSet<T> {
72    /// Returns whether this holder set contains `entry`.
73    #[must_use]
74    pub fn contains(&self, entry: &'static T) -> bool {
75        match self {
76            Self::Tag(tag) => T::holder_set_tag_contains(entry, tag),
77            Self::Direct(entries) => entries.contains(&entry),
78        }
79    }
80
81    pub(crate) fn from_owned_nbt(tag: &NbtTag) -> Option<Self> {
82        if let Some(value) = tag.string() {
83            let value = value.to_string();
84            if let Some(tag) = value.strip_prefix('#') {
85                let tag = Identifier::from_str(tag).ok()?;
86                if !T::holder_set_tag_exists(&tag) {
87                    return None;
88                }
89                return Some(Self::Tag(tag));
90            }
91
92            let key = Identifier::from_str(&value).ok()?;
93            return Some(Self::Direct(vec![T::holder_set_by_key(&key)?]));
94        }
95
96        let list = tag.list()?;
97        if list.as_nbt_tags().is_empty() {
98            return Some(Self::Direct(Vec::new()));
99        }
100        let values = list.strings()?;
101        let mut entries = Vec::with_capacity(values.len());
102        for value in values {
103            let key = Identifier::from_str(&value.to_string()).ok()?;
104            entries.push(T::holder_set_by_key(&key)?);
105        }
106        Some(Self::Direct(entries))
107    }
108}

110impl<T: RegistryHolderSetEntry> WriteTo for RegistryHolderSet<T> {
111    fn write(&self, writer: &mut impl Write) -> Result<()> {
112        match self {
113            Self::Tag(tag) => {
114                if !T::holder_set_tag_exists(tag) {
115                    return Err(Error::other(format!(
116                        "Unknown {} tag: {tag}",
117                        T::REGISTRY_NAME
118                    )));
119                }
120                VarInt(0).write(writer)?;
121                tag.write(writer)
122            }
123            Self::Direct(entries) => {
124                let count = i32::try_from(entries.len()).map_err(|_| {
125                    Error::other(format!(
126                        "{} holder set too large: {}",
127                        T::REGISTRY_NAME,
128                        entries.len()
129                    ))
130                })?;
131                let encoded_count = count.checked_add(1).ok_or_else(|| {
132                    Error::other(format!(
133                        "{} holder set count exceeds protocol range",
134                        T::REGISTRY_NAME
135                    ))
136                })?;
137                VarInt(encoded_count).write(writer)?;
138                for entry in entries {
139                    let id = entry.try_id().ok_or_else(|| {
140                        Error::other(format!("Unknown {}: {}", T::REGISTRY_NAME, entry.key()))
141                    })?;
142                    let id = i32::try_from(id).map_err(|_| {
143                        Error::other(format!(
144                            "{} id out of protocol range: {id}",
145                            T::REGISTRY_NAME
146                        ))
147                    })?;
148                    VarInt(id).write(writer)?;
149                }
150                Ok(())
151            }
152        }
153    }
154}

156impl<T: RegistryHolderSetEntry> ReadFrom for RegistryHolderSet<T> {
157    fn read(data: &mut Cursor<&[u8]>) -> Result<Self> {
158        let encoded_count = VarInt::read(data)?.0;
159        if encoded_count == 0 {
160            let tag = Identifier::read(data)?;
161            if !T::holder_set_tag_exists(&tag) {
162                return Err(Error::other(format!(
163                    "Unknown {} tag: {tag}",
164                    T::REGISTRY_NAME
165                )));
166            }
167            return Ok(Self::Tag(tag));
168        }
169
170        let count = encoded_count
171            .checked_sub(1)
172            .and_then(|count| usize::try_from(count).ok())
173            .ok_or_else(|| {
174                Error::other(format!(
175                    "Invalid {} holder set count: {encoded_count}",
176                    T::REGISTRY_NAME
177                ))
178            })?;
179        let mut entries = Vec::with_capacity(count.min(65_536));
180        for _ in 0..count {
181            let id = VarInt::read(data)?.0;
182            let id = usize::try_from(id)
183                .map_err(|_| Error::other(format!("Negative {} id: {id}", T::REGISTRY_NAME)))?;
184            let entry = T::holder_set_by_id(id)
185                .ok_or_else(|| Error::other(format!("Unknown {} id: {id}", T::REGISTRY_NAME)))?;
186            entries.push(entry);
187        }
188        Ok(Self::Direct(entries))
189    }
190}

192impl<T: RegistryHolderSetEntry> ToNbtTag for RegistryHolderSet<T> {
193    fn to_nbt_tag(self) -> NbtTag {
194        match self {
195            Self::Tag(tag) => NbtTag::String(format!("#{tag}").into()),
196            Self::Direct(entries) if entries.is_empty() => NbtTag::List(NbtList::Empty),
197            Self::Direct(entries) if entries.len() == 1 => {
198                NbtTag::String(entries[0].key().to_string().into())
199            }
200            Self::Direct(entries) => NbtTag::List(NbtList::String(
201                entries
202                    .into_iter()
203                    .map(|entry| entry.key().to_string().into())
204                    .collect(),
205            )),
206        }
207    }
208}

210impl<T: RegistryHolderSetEntry> FromNbtTag for RegistryHolderSet<T> {
211    fn from_nbt_tag(tag: simdnbt::borrow::NbtTag) -> Option<Self> {
212        Self::from_owned_nbt(&tag.to_owned())
213    }

216impl<T: RegistryHolderSetEntry> HashComponent for RegistryHolderSet<T> {
217    fn hash_component(&self, hasher: &mut ComponentHasher) {
218        match self {
219            Self::Tag(tag) => hasher.put_string(&format!("#{tag}")),
220            Self::Direct(entries) if entries.len() == 1 => {
221                hasher.put_string(&entries[0].key().to_string());
222            }
223            Self::Direct(entries) => {
224                hasher.start_list();
225                for entry in entries {
226                    hasher.put_component_hash(&entry.key().to_string());
227                }
228                hasher.end_list();
229            }
230        }
231    }
232}

234macro_rules! impl_registry_holder_set_entry {
235    ($entry:ty, $registry:ident, $name:literal) => {
236        impl RegistryHolderSetEntry for $entry {
237            const REGISTRY_NAME: &'static str = $name;
238
239            fn holder_set_by_id(id: usize) -> Option<&'static Self> {
240                REGISTRY.$registry.by_id(id)
241            }
242
243            fn holder_set_by_key(key: &Identifier) -> Option<&'static Self> {
244                REGISTRY.$registry.by_key(key)
245            }
246
247            fn holder_set_tag_exists(tag: &Identifier) -> bool {
248                REGISTRY.$registry.get_tag(tag).is_some()
249            }
250
251            fn holder_set_tag_contains(entry: &'static Self, tag: &Identifier) -> bool {
252                REGISTRY.$registry.is_in_tag(entry, tag)
253            }
254        }
255    };
256}
