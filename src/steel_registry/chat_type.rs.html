2use rustc_hash::FxHashMap;

5/// Represents a chat type definition from the data packs.
6#[derive(Debug)]
7pub struct ChatType {
8    pub key: Identifier,
9    pub chat: ChatTypeDecoration,
10    pub narration: ChatTypeDecoration,
11}
12
13/// Defines the styling and translation for a part of a chat message.
14#[derive(Debug)]
15pub struct ChatTypeDecoration {
16    pub translation_key: &'static str,
17    pub parameters: &'static [&'static str],
18    pub style: Option<ChatStyle>,
19}
20
21/// Defines optional text styling, like color and formatting.
22#[derive(Debug)]
23pub struct ChatStyle {
24    pub color: Option<&'static str>,
25    pub bold: Option<bool>,
26    pub italic: Option<bool>,
27    pub underlined: Option<bool>,
28    pub strikethrough: Option<bool>,
29    pub obfuscated: Option<bool>,
30}
31
32pub type ChatTypeRef = &'static ChatType;
33
34pub struct ChatTypeRegistry {
35    chat_types_by_id: Vec<ChatTypeRef>,
36    chat_types_by_key: FxHashMap<Identifier, usize>,
37    allows_registering: bool,
38}
39
40impl ChatTypeRegistry {
41    #[must_use]
42    pub fn new() -> Self {
43        Self {
44            chat_types_by_id: Vec::new(),
45            chat_types_by_key: FxHashMap::default(),
46            allows_registering: true,
47        }
48    }
49
50    pub fn register(&mut self, chat_type: ChatTypeRef) -> usize {
51        assert!(
52            self.allows_registering,
53            "Cannot register chat types after the registry has been frozen"
54        );
55
56        let id = self.chat_types_by_id.len();
57        self.chat_types_by_key.insert(chat_type.key.clone(), id);
58        self.chat_types_by_id.push(chat_type);
59        id
60    }
61
62    /// Replaces a chat_types at a given index.
63    /// Returns true if the chat_types was replaced and false if the chat_types wasn't replaced
64    #[must_use]
65    pub fn replace(&mut self, chat_types: ChatTypeRef, id: usize) -> bool {
66        if id >= self.chat_types_by_id.len() {
67            return false;
68        }
69        self.chat_types_by_id[id] = chat_types;
70        true
71    }
72
73    #[must_use]
74    pub fn by_id(&self, id: usize) -> Option<ChatTypeRef> {
75        self.chat_types_by_id.get(id).copied()
76    }
77
78    #[must_use]
79    pub fn get_id(&self, chat_type: ChatTypeRef) -> &usize {
80        self.chat_types_by_key
81            .get(&chat_type.key)
82            .expect("Chat type not found")
83    }
84
85    #[must_use]
86    pub fn by_key(&self, key: &Identifier) -> Option<ChatTypeRef> {
87        self.chat_types_by_key
88            .get(key)
89            .and_then(|id| self.by_id(*id))
90    }
91
92    pub fn iter(&self) -> impl Iterator<Item = (usize, ChatTypeRef)> + '_ {
93        self.chat_types_by_id
94            .iter()
95            .enumerate()
96            .map(|(id, &ct)| (id, ct))
97    }
98
99    #[must_use]
100    pub fn len(&self) -> usize {
101        self.chat_types_by_id.len()
102    }
103
104    #[must_use]
105    pub fn is_empty(&self) -> bool {
106        self.chat_types_by_id.is_empty()
107    }
108}
109
110impl RegistryExt for ChatTypeRegistry {
111    fn freeze(&mut self) {
112        self.allows_registering = false;
113    }
114}
115
116impl Default for ChatTypeRegistry {
117    fn default() -> Self {
118        Self::new()
119    }
