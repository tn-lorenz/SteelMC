27/// A stack of items with a count and component modifications.
28#[derive(Debug, Clone, PartialEq)]
29pub struct ItemStack {
30    /// The item type. AIR represents an empty stack.
31    pub item: ItemRef,
32    /// The number of items in this stack.
33    pub count: i32,
34    /// Modifications to the prototype components.
35    patch: DataComponentPatch,
36}

40        Self::empty()

115    /// Splits off the specified amount from this stack and returns it as a new stack.
116    ///
117    /// If the amount is greater than or equal to the current count, this stack becomes
118    /// empty and the entire contents are returned.
119    pub fn split(&mut self, amount: i32) -> Self {
120        let take = amount.min(self.count);
121        let result = Self {
122            item: self.item,
123            count: take,
124            patch: self.patch.clone(),
125        };
126        self.shrink(take);
127        result
128    }

192    /// Damages the item and breaks it if durability reaches zero.
193    ///
194    /// Returns `true` if the item broke and should be removed/replaced.
195    pub fn hurt_and_break(&mut self, amount: i32, has_infinite_materials: bool) -> bool {
196        if !self.is_damageable_item() || amount <= 0 {
197            return false;
198        }
199
200        if has_infinite_materials {
201            return false;
202        }
203
204        let unbreaking_level =
205            self.get_enchantment_level(&crate::vanilla_enchantments::UNBREAKING.key);
206        let mut effective_amount = 0;
207        for _ in 0..amount {
208            if should_consume_durability(unbreaking_level) {
209                effective_amount += 1;
210            }
211        }
212
213        if effective_amount == 0 {
214            return false;
215        }
216
217        let new_damage = self.get_damage_value() + effective_amount;
218
219        // TODO: Trigger ITEM_DURABILITY_CHANGED advancement criteria
220
221        self.set_damage_value(new_damage);
222
223        if self.is_broken() {
224            // TODO: Call onEquippedItemBroken callback which:
225            // - Broadcasts entity event (byte 47 for mainhand) for break sound/particles
226            // - Stops location-based effects (removes attribute modifiers)
227            self.shrink(1);
228            return true;
229        }
230
231        false
232    }

255    /// Checks if two stacks have the same item and components.
256    #[must_use]
257    pub fn is_same_item_same_components(a: &Self, b: &Self) -> bool {
258        if !Self::is_same_item(a, b) {
259            return false;
260        }
261        if a.is_empty() && b.is_empty() {
262            return true;
263        }
264        a.components_equal(b)
265    }

533    pub fn set_enchantments(&mut self, enchantments: &[(Identifier, u32)], add: bool) {
534        let mut current = self
535            .get(ENCHANTMENTS)
536            .cloned()
537            .unwrap_or_else(ItemEnchantments::empty);
538
539        for (key, level) in enchantments {
540            if add {
541                let existing = current.get_level(key);
542                current.set(key.clone(), existing + *level);
543            } else {
544                current.set(key.clone(), *level);
545            }
546        }
547
548        self.set(ENCHANTMENTS, current);
549    }

716    pub fn components_equal(&self, other: &Self) -> bool {
717        let mut all_keys = rustc_hash::FxHashSet::default();
718
719        for key in self.prototype().keys() {
720            if !self.patch.is_removed(key) {
721                all_keys.insert(key);
722            }
723        }
724        for (key, entry) in self.patch.iter() {
725            if matches!(entry, ComponentPatchEntry::Set(_)) {
726                all_keys.insert(key);
727            }
728        }
729        for key in other.prototype().keys() {
730            if !other.patch.is_removed(key) {
731                all_keys.insert(key);
732            }
733        }
734        for (key, entry) in other.patch.iter() {
735            if matches!(entry, ComponentPatchEntry::Set(_)) {
736                all_keys.insert(key);
737            }
738        }
739        for key in all_keys {
740            let val_a = self.get_effective_value_raw(key);
741            let val_b = other.get_effective_value_raw(key);
742
743            match (val_a, val_b) {
744                (Some(a), Some(b)) => {
745                    if a != b {
746                        return false;
747                    }
748                }
749                (None, None) => {}
750                _ => return false,
751            }
752        }
753
754        true
755    }
756}

758/// Vanilla unbreaking formula: `1 / (unbreaking_level + 1)` chance to consume durability.
759fn should_consume_durability(unbreaking_level: i32) -> bool {
760    if unbreaking_level <= 0 {
761        return true;
762    }
763    // TODO: Armor uses a different formula: `3 / (unbreaking_level + 3)`
764    rand::rng().random_range(0..unbreaking_level + 1) == 0
765}

836impl ToNbtTag for ItemStack {
837    /// Converts this item stack to an NBT tag for persistent storage.
838    ///
839    /// Format (matching vanilla Minecraft):
840    /// ```text
841    /// {
842    ///     id: "minecraft:stone",
843    ///     count: 64,
844    ///     components: { ... }  // Only present if patch is non-empty
845    /// }
846    /// ```
847    fn to_nbt_tag(self) -> simdnbt::owned::NbtTag {
848        if self.is_empty() {
849            // Empty stacks are represented as an empty compound
850            return simdnbt::owned::NbtTag::Compound(NbtCompound::new());
851        }
852
853        let mut compound = NbtCompound::new();
854
855        // id: The item identifier
856        compound.insert("id", self.item.key.to_string());
857
858        // count: The stack count (vanilla uses Int for NBT storage)
859        compound.insert("count", self.count);
860
861        // components: The component patch (only if non-empty)
862        if !self.patch.is_empty() {
863            compound.insert("components", self.patch.to_nbt_tag());
864        }
865
866        simdnbt::owned::NbtTag::Compound(compound)
867    }
868}
869
870impl FromNbtTag for ItemStack {
871    /// Parses an item stack from an NBT tag.
872    ///
873    /// Accepts the vanilla format:
874    /// ```text
875    /// {
876    ///     id: "minecraft:stone",
877    ///     count: 64,
878    ///     components: { ... }
879    /// }
880    /// ```
881    fn from_nbt_tag(tag: BorrowedNbtTag) -> Option<Self> {
882        let compound = tag.compound()?;
883
884        // Get the item ID
885        let id_str = compound.get("id")?.string()?.to_str();
886        let id = id_str.parse::<Identifier>().ok()?;
887
888        // Look up the item in the registry
889        let item = REGISTRY.items.by_key(&id)?;
890
891        // Get the count (default to 1 if not present)
892        let count = compound.get("count").and_then(|t| t.int()).unwrap_or(1);
893
894        // Parse components if present
895        let patch = compound
896            .get("components")
897            .and_then(DataComponentPatch::from_nbt_tag)
898            .unwrap_or_default();
899
900        Some(Self { item, count, patch })
901    }
902}

904impl ItemStack {
905    /// Parses an `ItemStack` from a borrowed `NbtCompoundView`.
906    ///
907    /// This is useful for loading items from disk where we have borrowed NBT data
908    /// and want to avoid the overhead of converting to an owned tag first.
909    #[must_use]
910    pub fn from_borrowed_compound(compound: &NbtCompoundView<'_, '_>) -> Option<Self> {
911        // Get the item ID

915        // Look up the item in the registry

918        // Get the count (default to 1 if not present)
919        let count = compound.int("count").unwrap_or(1);

923            .get("components")
924            .and_then(DataComponentPatch::from_nbt_tag)
925            .unwrap_or_default();
