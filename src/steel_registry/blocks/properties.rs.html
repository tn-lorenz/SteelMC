8pub trait Property<T>: Sync + Send {
9    fn get_value(&self, value: &str) -> Option<T>;
10    fn get_possible_values(&self) -> Box<[T]>;
11    fn get_internal_index(&self, value: &T) -> usize;
12    fn value_from_index(&self, index: usize) -> T;
13    fn as_dyn(&self) -> &dyn DynProperty;
14}

16pub trait DynProperty: Debug + Sync + Send {
17    fn get_possible_values(&self) -> Box<[&str]>;
18    fn get_name(&self) -> &'static str;
19}

99pub struct IntProperty {

101    pub max: u8,
102    pub name: &'static str,
103}
104
105impl IntProperty {
106    #[must_use]
107    pub const fn new(name: &'static str, min: u8, max: u8) -> Self {
108        Self { min, max, name }
109    }
110
111    #[must_use]
112    pub const fn value_count(&self) -> usize {
113        (self.max - self.min + 1) as usize
114    }
115}

117impl DynProperty for IntProperty {
118    fn get_possible_values(&self) -> Box<[&str]> {
119        (self.min..=self.max).map(|v| NUM_STR[v as usize]).collect()
120    }
121
122    fn get_name(&self) -> &'static str {
123        self.name
124    }
125}

127impl Property<u8> for IntProperty {
128    fn get_value(&self, value: &str) -> Option<u8> {
129        value
130            .parse()
131            .ok()
132            .filter(|v| v >= &self.min && v <= &self.max)
133    }
134
135    fn get_possible_values(&self) -> Box<[u8]> {
136        (self.min..=self.max).collect()
137    }
138
139    fn get_internal_index(&self, value: &u8) -> usize {
140        if *value <= self.max {
141            (*value - self.min) as usize
142        } else {
143            0
144        }
145    }
146
147    fn value_from_index(&self, index: usize) -> u8 {
148        self.min + index as u8
149    }
150
151    fn as_dyn(&self) -> &dyn DynProperty {
152        self
153    }
154}

173impl<T: PropertyEnum + 'static> DynProperty for EnumProperty<T> {

175        self.possible_values
176            .iter()
177            .map(PropertyEnum::as_str)
178            .collect()
179    }

186impl<T: PropertyEnum> EnumProperty<T> {
187    pub const fn new(name: &'static str, possible_values: &'static [T]) -> Self {
188        Self {
189            name,
190            possible_values,
191        }
192    }
193
194    #[must_use]
195    pub const fn value_count(&self) -> usize {
196        self.possible_values.len()
197    }
198}

200impl<T: PropertyEnum> Property<T> for EnumProperty<T> {
201    fn get_value(&self, value: &str) -> Option<T> {
202        self.possible_values
203            .iter()
204            .find(|v| v.as_str() == value)
205            .cloned()
206    }
207
208    fn get_possible_values(&self) -> Box<[T]> {
209        self.possible_values.into()
210    }
211
212    fn get_internal_index(&self, value: &T) -> usize {
213        self.possible_values
214            .iter()
215            .position(|v| v == value)
216            .unwrap()
217    }
218
219    fn value_from_index(&self, index: usize) -> T {
220        self.possible_values[index].clone()
221    }
222
223    fn as_dyn(&self) -> &dyn DynProperty {
224        self
225    }
226}
227
228impl<T: const PartialEq + PropertyEnum + 'static> EnumProperty<T> {
229    pub const fn get_internal_index_const(&self, value: &T) -> usize {
230        let mut i = 0;
231        while i < self.possible_values.len() {
232            if &self.possible_values[i] == value {
233                return i;
234            }
235            i += 1;
236        }
237        panic!("value not found in possible_values");
238    }
239}

325    /// Returns the horizontal direction from a yaw rotation.
326    ///
327    /// Yaw values follow Minecraft's convention:
328    /// - 0째 = South (+Z)
329    /// - 90째 = West (-X)
330    /// - 180째 = North (-Z)
331    /// - 270째 = East (+X)
332    #[must_use]
333    pub fn from_yaw(yaw: f32) -> Direction {
334        let adjusted = yaw.rem_euclid(360.0);
335        match adjusted {
336            y if !(45.0..315.0).contains(&y) => Direction::South,
337            y if y < 135.0 => Direction::West,
338            y if y < 225.0 => Direction::North,
339            _ => Direction::East,
340        }
341    }

412    /// Returns all directions ordered by how closely they match the player's look direction.
413    ///
414    /// This matches vanilla's `Direction.orderedByNearest(Entity)`.
415    /// - `yaw`: Player's yaw rotation in degrees (0 = South, 90 = West, 180 = North, 270 = East)
416    /// - `pitch`: Player's pitch rotation in degrees (negative = looking up, positive = looking down)
417    #[must_use]
418    pub fn ordered_by_nearest(yaw: f32, pitch: f32) -> Vec<Direction> {
419        // Convert to radians and negate yaw to match vanilla's coordinate system
420        let pitch_rad = pitch.to_radians();
421        let yaw_rad = (-yaw).to_radians();
422
423        let pitch_sin = pitch_rad.sin();
424        let pitch_cos = pitch_rad.cos();
425        let yaw_sin = yaw_rad.sin();
426        let yaw_cos = yaw_rad.cos();
427
428        // Determine which direction on each axis the player is looking
429        let x_pos = yaw_sin > 0.0;
430        let y_pos = pitch_sin < 0.0; // Negative pitch = looking up
431        let z_pos = yaw_cos > 0.0;
432
433        // Calculate magnitude of look direction on each axis
434        let x_yaw = if x_pos { yaw_sin } else { -yaw_sin };
435        let y_mag = if y_pos { -pitch_sin } else { pitch_sin };
436        let z_yaw = if z_pos { yaw_cos } else { -yaw_cos };
437        let x_mag = x_yaw * pitch_cos;
438        let z_mag = z_yaw * pitch_cos;
439
440        // Determine the primary direction on each axis
441        let axis_x = if x_pos {
442            Direction::East
443        } else {
444            Direction::West
445        };
446        let axis_y = if y_pos {
447            Direction::Up
448        } else {
449            Direction::Down
450        };
451        let axis_z = if z_pos {
452            Direction::South
453        } else {
454            Direction::North
455        };
456
457        // Sort axes by magnitude and build the direction array
458        if x_yaw > z_yaw {
459            if y_mag > x_mag {
460                Self::make_direction_array(axis_y, axis_x, axis_z)
461            } else if z_mag > y_mag {
462                Self::make_direction_array(axis_x, axis_z, axis_y)
463            } else {
464                Self::make_direction_array(axis_x, axis_y, axis_z)
465            }
466        } else if y_mag > z_mag {
467            Self::make_direction_array(axis_y, axis_z, axis_x)
468        } else if x_mag > y_mag {
469            Self::make_direction_array(axis_z, axis_x, axis_y)
470        } else {
471            Self::make_direction_array(axis_z, axis_y, axis_x)
472        }
473    }
474
475    /// Creates an array of all 6 directions, with the first 3 being the primary axes
476    /// and the last 3 being their opposites.
477    fn make_direction_array(
478        axis1: Direction,
479        axis2: Direction,
480        axis3: Direction,
481    ) -> Vec<Direction> {
482        vec![
483            axis1,
484            axis2,
485            axis3,
486            axis3.opposite(),
487            axis2.opposite(),
488            axis1.opposite(),
489        ]
490    }
