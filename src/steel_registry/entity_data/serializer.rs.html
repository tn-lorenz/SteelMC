1//! Entity data serializer registry.
2//!
3//! This registry maps serializer names to IDs and stores writer functions,
4//! following the same pattern as `DataComponentRegistry`.
5
6use rustc_hash::FxHashMap;
7use std::io;
8
9use super::EntityData;
10
11/// Writer function for serializing entity data to network format.
12///
13/// Takes a reference to the [`EntityData`] value and writes it to the buffer.
14/// Returns an error if the value doesn't match the expected serializer type.
15pub type EntityDataWriter = fn(&EntityData, &mut Vec<u8>) -> io::Result<()>;
16
17/// Entry for a registered entity data serializer.
18pub struct EntityDataSerializerEntry {
19    /// The serializer's name (e.g., "byte", "int", "float").
20    pub name: &'static str,
21    /// The writer function for this serializer.
22    pub writer: EntityDataWriter,
23}
24
25/// Registry of entity data serializers.
26///
27/// The serializer ID is determined by registration order, which must match
28/// vanilla's `EntityDataSerializers.java` exactly.
29pub struct EntityDataSerializerRegistry {
30    /// Serializer entries in registration order (index = ID).
31    entries: Vec<EntityDataSerializerEntry>,
32    /// Map from name to ID for fast lookup.
33    name_to_id: FxHashMap<&'static str, i32>,
34    /// Whether the registry has been frozen (no more registrations allowed).
35    frozen: bool,
36}
37
38impl EntityDataSerializerRegistry {
39    /// Create a new empty registry.
40    pub fn new() -> Self {
41        Self {
42            entries: Vec::new(),
43            name_to_id: FxHashMap::default(),
44            frozen: false,
45        }
46    }
47
48    /// Register a serializer with its writer function. ID is determined by registration order.
49    ///
50    /// # Panics
51    /// Panics if the registry has been frozen or if the name is already registered.
52    pub fn register(&mut self, name: &'static str, writer: EntityDataWriter) {
53        assert!(!self.frozen, "Cannot register after freezing");
54        assert!(
55            !self.name_to_id.contains_key(name),
56            "Serializer '{}' already registered",
57            name
58        );
59
60        let id = self.entries.len() as i32;
61        self.entries
62            .push(EntityDataSerializerEntry { name, writer });
63        self.name_to_id.insert(name, id);
64    }
65
66    /// Get the ID for a serializer by name.
67    pub fn get_id(&self, name: &str) -> Option<i32> {
68        self.name_to_id.get(name).copied()
69    }
70
71    /// Get the name for a serializer by ID.
72    pub fn get_name(&self, id: i32) -> Option<&'static str> {
73        self.entries.get(id as usize).map(|e| e.name)
74    }
75
76    /// Get the entry for a serializer by ID.
77    pub fn get_entry(&self, id: i32) -> Option<&EntityDataSerializerEntry> {
78        self.entries.get(id as usize)
79    }
80
81    /// Get the writer function for a serializer by ID.
82    pub fn get_writer(&self, id: i32) -> Option<EntityDataWriter> {
83        self.entries.get(id as usize).map(|e| e.writer)
84    }
85
86    /// Returns the number of registered serializers.
87    pub fn len(&self) -> usize {
88        self.entries.len()
89    }
90
91    /// Returns true if no serializers are registered.
92    pub fn is_empty(&self) -> bool {
93        self.entries.is_empty()
94    }
95
96    /// Freeze the registry, preventing further registrations.
97    pub fn freeze(&mut self) {
98        self.frozen = true;
99    }
100
101    /// Returns true if the registry has been frozen.
102    pub fn is_frozen(&self) -> bool {
103        self.frozen
104    }
105}
106
107impl Default for EntityDataSerializerRegistry {
108    fn default() -> Self {
109        Self::new()
110    }
111}
