16/// Simple serializer: extract value and call `.write(buf)`.
17macro_rules! ser_write {
18    ($name:ident, $variant:ident) => {
19        fn $name(data: &EntityData, buf: &mut Vec<u8>) -> io::Result<()> {
20            match data {
21                EntityData::$variant(v) => v.write(buf),
22                _ => Err(io::Error::other(concat!("Expected ", stringify!($variant)))),
23            }
24        }
25    };
26}
27
28/// Serializer that wraps value in VarInt.
29macro_rules! ser_varint {
30    ($name:ident, $variant:ident) => {
31        fn $name(data: &EntityData, buf: &mut Vec<u8>) -> io::Result<()> {
32            match data {
33                EntityData::$variant(v) => VarInt(*v).write(buf),
34                _ => Err(io::Error::other(concat!("Expected ", stringify!($variant)))),
35            }
36        }
37    };
38}
39
40/// Serializer that casts enum to i32 then writes as VarInt.
41macro_rules! ser_enum_varint {
42    ($name:ident, $variant:ident) => {
43        fn $name(data: &EntityData, buf: &mut Vec<u8>) -> io::Result<()> {
44            match data {
45                EntityData::$variant(v) => VarInt(*v as i32).write(buf),
46                _ => Err(io::Error::other(concat!("Expected ", stringify!($variant)))),
47            }
48        }
49    };
50}

94fn ser_optional_component(data: &EntityData, buf: &mut Vec<u8>) -> io::Result<()> {
95    match data {
96        EntityData::OptionalComponent(v) => match v {
97            Some(comp) => {
98                true.write(buf)?;
99                comp.write(buf)
100            }
101            None => false.write(buf),
102        },
103        _ => Err(io::Error::other("Expected OptionalComponent")),
104    }

152    match data {
153        EntityData::OptionalBlockState(v) => {
154            // Vanilla encodes: 0 = absent, otherwise raw block state ID

156                Some(state) => VarInt(i32::from(state.0)).write(buf),

268/// Register all vanilla entity data serializers.
269///
270/// **IMPORTANT**: The registration order MUST match vanilla's `EntityDataSerializers.java` exactly,
271/// as the serializer's network ID is determined by its registration order.
272pub fn register_vanilla_entity_data_serializers(registry: &mut EntityDataSerializerRegistry) {
273    // Order matches EntityDataSerializers.java static block
274
275    // 0: BYTE
276    registry.register("byte", ser_byte);
277    // 1: INT (VarInt)
278    registry.register("int", ser_int);
279    // 2: LONG (VarLong)
280    registry.register("long", ser_long);
281    // 3: FLOAT
282    registry.register("float", ser_float);
283    // 4: STRING
284    registry.register("string", ser_string);
285    // 5: COMPONENT
286    registry.register("component", ser_component);
287    // 6: OPTIONAL_COMPONENT
288    registry.register("optional_component", ser_optional_component);
289    // 7: ITEM_STACK
290    registry.register("item_stack", ser_item_stack);
291    // 8: BOOLEAN
292    registry.register("boolean", ser_boolean);
293    // 9: ROTATIONS
294    registry.register("rotations", ser_rotations);
295    // 10: BLOCK_POS
296    registry.register("block_pos", ser_block_pos);
297    // 11: OPTIONAL_BLOCK_POS
298    registry.register("optional_block_pos", ser_optional_block_pos);
299    // 12: DIRECTION
300    registry.register("direction", ser_direction);
301    // 13: OPTIONAL_LIVING_ENTITY_REFERENCE
302    registry.register(
303        "optional_living_entity_reference",
304        ser_optional_living_entity_reference,
305    );
306    // 14: BLOCK_STATE
307    registry.register("block_state", ser_block_state);
308    // 15: OPTIONAL_BLOCK_STATE
309    registry.register("optional_block_state", ser_optional_block_state);
310    // 16: PARTICLE
311    registry.register("particle", ser_particle);
312    // 17: PARTICLES
313    registry.register("particles", ser_particles);
314    // 18: VILLAGER_DATA
315    registry.register("villager_data", ser_villager_data);
316    // 19: OPTIONAL_UNSIGNED_INT
317    registry.register("optional_unsigned_int", ser_optional_unsigned_int);
318    // 20: POSE
319    registry.register("pose", ser_pose);
320    // 21: CAT_VARIANT
321    registry.register("cat_variant", ser_cat_variant);
322    // 22: COW_VARIANT
323    registry.register("cow_variant", ser_cow_variant);
324    // 23: WOLF_VARIANT
325    registry.register("wolf_variant", ser_wolf_variant);
326    // 24: WOLF_SOUND_VARIANT

330    // 26: PIG_VARIANT

332    // 27: CHICKEN_VARIANT

341    registry.register("sniffer_state", ser_sniffer_state);
342    // 32: ARMADILLO_STATE
343    registry.register("armadillo_state", ser_armadillo_state);

358#[cfg(test)]
359mod tests {

362    #[test]
363    fn test_serializer_registration_order() {
364        let mut registry = EntityDataSerializerRegistry::new();
365        register_vanilla_entity_data_serializers(&mut registry);
366
367        // Verify key serializers have correct IDs
368        assert_eq!(registry.get_id("byte"), Some(0));
369        assert_eq!(registry.get_id("int"), Some(1));
370        assert_eq!(registry.get_id("long"), Some(2));
371        assert_eq!(registry.get_id("float"), Some(3));
372        assert_eq!(registry.get_id("boolean"), Some(8));
373        assert_eq!(registry.get_id("pose"), Some(20));
374        assert_eq!(registry.get_id("humanoid_arm"), Some(38));
375
376        // Total count
377        assert_eq!(registry.len(), 39);
378    }
379
380    #[test]
381    fn test_serializers_write_correctly() {
382        let mut registry = EntityDataSerializerRegistry::new();
383        register_vanilla_entity_data_serializers(&mut registry);
384
385        // Test byte serializer
386        let writer = registry.get_writer(0).unwrap();
387        let mut buf = Vec::new();
388        writer(&EntityData::Byte(42), &mut buf).unwrap();
389        assert_eq!(buf, vec![42]);
390
391        // Test int serializer (VarInt)
392        let writer = registry.get_writer(1).unwrap();
393        let mut buf = Vec::new();
394        writer(&EntityData::Int(300), &mut buf).unwrap();
395        assert_eq!(buf, vec![0xAC, 0x02]); // 300 as VarInt
396
397        // Test float serializer
398        let writer = registry.get_writer(3).unwrap();
399        let mut buf = Vec::new();
400        writer(&EntityData::Float(1.5), &mut buf).unwrap();
401        assert_eq!(buf, 1.5f32.to_be_bytes().to_vec());
402
403        // Test boolean serializer
404        let writer = registry.get_writer(8).unwrap();
405        let mut buf = Vec::new();
406        writer(&EntityData::Boolean(true), &mut buf).unwrap();
407        assert_eq!(buf, vec![1]);
408    }
