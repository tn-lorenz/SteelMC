23/// The Vanilla particle factory associated with a mob effect.
24#[derive(Debug, Clone, Copy)]
25pub enum MobEffectParticle {
26    /// Builds a color payload from the effect color and instance ambience.
27    EffectColor {
28        particle_type: ParticleTypeRef,
29        regular_alpha: u8,
30        ambient_alpha: u8,
31    },
32    /// Uses a simple particle without a payload.
33    Simple(ParticleTypeRef),
34    /// Uses one fixed ARGB color payload for every instance.
35    FixedColor {
36        particle_type: ParticleTypeRef,
37        color: ArgbColor,
38    },
39}
40
41impl MobEffectParticle {
42    #[must_use]
43    pub fn create(self, effect_color: RgbColor, ambient: bool) -> ParticleData {
44        match self {
45            Self::EffectColor {
46                particle_type,
47                regular_alpha,
48                ambient_alpha,
49            } => {
50                let alpha = if ambient {
51                    ambient_alpha
52                } else {
53                    regular_alpha
54                };
55                ParticleData::new(
56                    particle_type,
57                    ColorParticleOption::new(effect_color.with_alpha(alpha)),
58                )
59            }
60            Self::Simple(particle_type) => ParticleData::simple(particle_type),
61            Self::FixedColor {
62                particle_type,
63                color,
64            } => ParticleData::new(particle_type, ColorParticleOption::new(color)),
65        }
66    }
67}

78impl MobEffect {
79    /// Creates the particle options synchronized for one effect instance.
80    #[must_use]
81    pub fn create_particle_options(&self, ambient: bool) -> ParticleData {
82        self.particle.create(self.color, ambient)
83    }

85    /// Returns the `VarInt` payload used by vanilla mob-effect holder-registry packets.
86    #[must_use]
87    pub fn packet_holder_id(&self) -> i32 {
88        let id = crate::RegistryEntry::id(self);
89        debug_assert!(i32::try_from(id).is_ok());
90        id as i32
91    }

102pub struct MobEffectRegistry {
103    effects_by_id: Vec<MobEffectRef>,
104    effects_by_key: FxHashMap<Identifier, usize>,
105    tags: FxHashMap<Identifier, Vec<Identifier>>,
106    allows_registering: bool,
107}
108
109impl Default for MobEffectRegistry {
110    fn default() -> Self {
111        Self::new()
112    }
113}
114
115impl MobEffectRegistry {
116    #[must_use]
117    pub fn new() -> Self {
118        Self {
119            effects_by_id: Vec::new(),
120            effects_by_key: FxHashMap::default(),
121            tags: FxHashMap::default(),
122            allows_registering: true,
123        }
124    }
125
126    pub fn register(&mut self, effect: MobEffectRef) {
127        assert!(
128            self.allows_registering,
129            "Cannot register mob effects after the registry has been frozen"
130        );
131        let idx = self.effects_by_id.len();
132        self.effects_by_key.insert(effect.key.clone(), idx);
133        self.effects_by_id.push(effect);
134    }
135
136    pub fn iter(&self) -> impl Iterator<Item = (usize, MobEffectRef)> + '_ {
137        self.effects_by_id
138            .iter()
139            .enumerate()
140            .map(|(id, &effect)| (id, effect))
141    }
142}

152#[cfg(test)]
153mod tests {
154    use crate::particle_type::{ColorParticleOption, SimpleParticleOptions};
155    use crate::{vanilla_mob_effects, vanilla_particle_types};
156
157    #[test]
158    fn generated_effect_particles_match_vanilla_factories() {
159        let speed = vanilla_mob_effects::SPEED;
160        let regular = speed.create_particle_options(false);
161        let ambient = speed.create_particle_options(true);
162        let Some(regular_color) = regular.downcast_ref::<ColorParticleOption>() else {
163            panic!("speed should use color particle options");
164        };
165        let Some(ambient_color) = ambient.downcast_ref::<ColorParticleOption>() else {
166            panic!("ambient speed should use color particle options");
167        };
168
169        assert_eq!(
170            regular.particle_type().key,
171            vanilla_particle_types::ENTITY_EFFECT.key
172        );
173        assert_eq!(regular_color.color().alpha(), 255);
174        assert_eq!(ambient_color.color().alpha(), 38);
175        assert_eq!(
176            regular_color.color().rgb().raw() & 0x00ff_ffff,
177            speed.color.raw()
178        );
179
180        let trial_omen = vanilla_mob_effects::TRIAL_OMEN;
181        let trial_particle = trial_omen.create_particle_options(false);
182        assert_eq!(
183            trial_particle.particle_type().key,
184            vanilla_particle_types::TRIAL_OMEN.key
185        );
186        assert!(
187            trial_particle
188                .downcast_ref::<SimpleParticleOptions>()
189                .is_some()
190        );
191    }
192}
