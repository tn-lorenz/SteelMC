1//! Login helper functions.
2//!
3//! Contains utilities for player name validation and offline UUID generation.
4
5use sha2::{Digest, Sha256};
6use uuid::Uuid;
7
8/// Checks if a player name is valid.
9///
10/// A valid player name:
11/// - Is between 3 and 16 characters long
12/// - Contains only ASCII alphanumeric characters or underscores
13#[must_use]
14pub fn is_valid_player_name(name: &str) -> bool {
15    (3..=16).contains(&name.len()) && name.chars().all(|c| c.is_ascii_alphanumeric() || c == '_')
16}
17
18/// Generates an offline mode UUID for a player.
19///
20/// This creates a deterministic UUID based on the username hash,
21/// used when the server is in offline mode.
22///
23/// # Errors
24/// Returns an error if the UUID cannot be created from the hash bytes.
25pub fn offline_uuid(username: &str) -> Result<Uuid, uuid::Error> {
26    Uuid::from_slice(&Sha256::digest(username)[..16])
27}
