1//! Server configuration loading.
2//!
3//! This module handles loading the server configuration from disk.
4//! The `ServerConfig` struct is defined in `steel-core`, this module
5//! just handles the file I/O and initialization.
6
7use base64::{Engine, prelude::BASE64_STANDARD};
8use serde::Deserialize;
9use std::{fs, path::Path, sync::LazyLock};
10
11// Re-export types from steel-core for convenience
12pub use steel_core::config::{ConfigLabel, ConfigLink, ServerConfig, ServerConfigRef, ServerLinks};

14#[cfg(feature = "stand-alone")]
15const DEFAULT_FAVICON: &[u8] = include_bytes!("../../package-content/favicon.png");
16const ICON_PREFIX: &str = "data:image/png;base64,";
17
18const DEFAULT_CONFIG: &str = include_str!("../../package-content/steel_config.json5");
19
20/// The Minecraft version this server supports.
21pub const MC_VERSION: &str = "1.21.11";

23/// The server configuration.
24///
25/// This is loaded from `config/steel_config.json5` or created if it doesn't exist.
26pub static STEEL_CONFIG: LazyLock<ExtendedConfig> =
27    LazyLock::new(|| load_or_create(Path::new("config/steel_config.json5")));

29/// A extended configuration for non-server features
30#[derive(Debug, Clone, Deserialize)]
31pub struct ExtendedConfig {
32    /// The full server configuration
33    #[serde(flatten)]
34    pub server_config: ServerConfig,
35    /// Logging configuration
36    pub log: Option<LogConfig>,
37}

39/// Logging configuration
40#[derive(Debug, Clone, Deserialize)]
41pub struct LogConfig {
42    /// Time display format: "none", "date" (HH:MM:SS:mmm), or "uptime" (seconds since start)
43    #[serde(default)]
44    pub time: LogTimeFormat,
45    /// Whether the `module_path` of the log should be displayed
46    pub module_path: bool,
47    /// Whether the extra data of the log should be displayed
48    pub extra: bool,
49}
50
51/// Time format for log entries
52#[derive(Debug, Clone, Default, Deserialize, PartialEq, Eq)]
53#[serde(rename_all = "lowercase")]
54pub enum LogTimeFormat {
55    /// No time displayed
56    None,
57    /// Current time (HH:MM:SS:mmm)
58    #[default]
59    Date,
60    /// Seconds since server start
61    Uptime,
62}

64/// Loads the server configuration from the given path, or creates it if it doesn't exist.
65///
66/// # Panics
67/// This function will panic if the config file does not exist and the directory cannot be created,
68/// or if the config file cannot be read or written.
69#[must_use]
70fn load_or_create(path: &Path) -> ExtendedConfig {
71    let mut config = if path.exists() {
72        let config_str = fs::read_to_string(path).expect("Failed to read config file");
73        let config: ExtendedConfig =
74            serde_json5::from_str(config_str.as_str()).expect("Failed to parse config");
75        validate(&config.server_config).expect("Failed to validate config");
76        config
77    } else {
78        fs::create_dir_all(path.parent().expect("Failed to get config directory"))
79            .expect("Failed to create config directory");
80        fs::write(path, DEFAULT_CONFIG).expect("Failed to write config file");
81        let config: ExtendedConfig =
82            serde_json5::from_str(DEFAULT_CONFIG).expect("Failed to parse config");
83        validate(&config.server_config).expect("Failed to validate config");
84        config
85    };
86
87    // Set the MC version (not loaded from config file)
88    config.server_config.mc_version = MC_VERSION;
89
90    // If icon file doesnt exist, write it
91    #[cfg(feature = "stand-alone")]
92    if config.server_config.use_favicon && !Path::new(&config.server_config.favicon).exists() {
93        fs::write(Path::new(&config.server_config.favicon), DEFAULT_FAVICON)
94            .expect("Failed to write favicon file");
95    }
96
97    config
98}

100/// Validates the server configuration.
101///
102/// # Errors
103/// This function will return an error if the configuration is invalid.
104fn validate(config: &ServerConfig) -> Result<(), &'static str> {
105    if !(1..=32).contains(&config.view_distance) {
106        return Err("View distance must in range 1..32");
107    }
108    if config.simulation_distance > config.view_distance {
109        return Err("Simulation distance must be less than or equal to view distance");
110    }
111    if let Some(compression) = config.compression {
112        if compression.threshold.get() < 256 {
113            return Err("Compression threshold must be greater than or equal to 256");
114        }
115        if !(1..=9).contains(&compression.level) {
116            return Err("Compression level must be between 1 and 9");
117        }
118    }
119    if config.enforce_secure_chat {
120        if !config.online_mode {
121            return Err("online_mode must be true when enforce_secure_chat is enabled");
122        }
123        if !config.encryption {
124            return Err("encryption must be true when enforce_secure_chat is enabled");
125        }
126    }
127    Ok(())
128}

130/// Loads the favicon from the path specified in the config.
131/// If the favicon doesn't exist, it will use the default favicon.
132/// If `use_favicon` is false, this will return `None`.
133///
134/// The returned string is a base64 encoded png with the `data:image/png;base64,` prefix.
135#[must_use]
136pub fn load_favicon(config: &ServerConfig) -> Option<String> {
137    if config.use_favicon {
138        let path = Path::new(&config.favicon);
139        if path.exists() {
140            let icon = fs::read(path);
141
142            if let Ok(icon) = icon {
143                let cap = ICON_PREFIX.len() + icon.len().div_ceil(3) * 4;
144                let mut base64 = String::with_capacity(cap);
145
146                base64 += ICON_PREFIX;
147                BASE64_STANDARD.encode_string(icon, &mut base64);
148
149                return Some(base64);
150            }
151            #[cfg(feature = "stand-alone")]
152            {
153                let cap = ICON_PREFIX.len() + DEFAULT_FAVICON.len().div_ceil(3) * 4;
154                let mut base64 = String::with_capacity(cap);
155
156                base64 += ICON_PREFIX;
157                BASE64_STANDARD.encode_string(DEFAULT_FAVICON, &mut base64);
158
159                return Some(base64);
160            }
161            #[cfg(not(feature = "stand-alone"))]
162            return None;
163        }
164    }
165    None
166}
