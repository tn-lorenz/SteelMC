84impl PermissionGroupStore for FilePermissionGroupStore {
85    fn save_groups(
86        &self,
87        config: PermissionGroupsConfig,
88    ) -> BoxFuture<'static, Result<(), PermissionGroupStoreError>> {
89        let path = self.path.clone();
90        Box::pin(async move {
91            let serialized = serialize_groups_config(&config).map_err(|error| {
92                PermissionGroupStoreError::new(format!(
93                    "failed to serialize groups config: {error}"
94                ))
95            })?;
96            write_atomic_config(&path, serialized)
97                .await
98                .map_err(|error| {
99                    PermissionGroupStoreError::new(format!(
100                        "failed to write groups config {}: {error}",
101                        path.display()
102                    ))
103                })
104        })
105    }
106}

108fn serialize_groups_config(config: &PermissionGroupsConfig) -> Result<String, TomlSerializeError> {
109    let mut output = String::from(GROUPS_CONFIG_HEADER);
110    push_toml_field(&mut output, "default_groups", &config.default_groups)?;
111
112    for (name, group) in &config.groups {
113        output.push('\n');
114        output.push_str("[groups.");
115        output.push_str(name);
116        output.push_str("]\n");
117        push_group_config(&mut output, group)?;
118    }
119
120    Ok(output)
121}

123fn push_group_config(
124    output: &mut String,
125    group: &PermissionGroupConfig,
126) -> Result<(), TomlSerializeError> {
127    push_toml_field(output, "priority", &group.priority)?;
128    push_string_array_field(output, "inherits", &group.inherits)?;
129    push_string_array_field(output, "allow", &group.allow)?;
130    push_string_array_field(output, "deny", &group.deny)?;
131    push_metadata_rules(output, &group.metadata)
132}
133
134fn push_string_array_field(
135    output: &mut String,
136    key: &str,
137    values: &[String],
138) -> Result<(), TomlSerializeError> {
139    if values.is_empty() {
140        output.push_str(key);
141        output.push_str(" = []\n");
142        return Ok(());
143    }
144
145    output.push_str(key);
146    output.push_str(" = [\n");
147    for value in values {
148        output.push_str("    ");
149        output.push_str(&toml_value(value)?);
150        output.push_str(",\n");
151    }
152    output.push_str("]\n");
153    Ok(())
154}
155
156fn push_metadata_rules(
157    output: &mut String,
158    metadata: &[PermissionMetadataRuleConfig],
159) -> Result<(), TomlSerializeError> {
160    if metadata.is_empty() {
161        output.push_str("metadata = []\n");
162        return Ok(());
163    }
164
165    output.push_str("metadata = [\n");
166    for entry in metadata {
167        output.push_str("    { key = ");
168        output.push_str(&toml_value(&entry.key)?);
169        output.push_str(", value = ");
170        output.push_str(&permission_metadata_value_toml(&entry.value)?);
171        output.push_str(" },\n");
172    }
173    output.push_str("]\n");
174    Ok(())
175}
176
177fn push_toml_field<T: Serialize + ?Sized>(
178    output: &mut String,
179    key: &str,
180    value: &T,
181) -> Result<(), TomlSerializeError> {
182    output.push_str(key);
183    output.push_str(" = ");
184    output.push_str(&toml_value(value)?);
185    output.push('\n');
186    Ok(())
187}
188
189fn toml_value<T: Serialize + ?Sized>(value: &T) -> Result<String, TomlSerializeError> {
190    #[derive(Serialize)]
191    struct Field<'a, T: Serialize + ?Sized> {
192        value: &'a T,
193    }
194
195    let serialized = toml::to_string(&Field { value })?;
196    let serialized = serialized.trim_end();
197    Ok(serialized
198        .strip_prefix("value = ")
199        .unwrap_or(serialized)
200        .to_owned())
201}
202
203fn permission_metadata_value_toml(
204    value: &PermissionMetadataValue,
205) -> Result<String, TomlSerializeError> {
206    match value {
207        PermissionMetadataValue::Bool(value) => toml_value(value),
208        PermissionMetadataValue::Integer(value) => toml_value(value),
209        PermissionMetadataValue::String(value) => toml_value(value),
210    }
211}

213async fn write_atomic_config(path: &Path, contents: String) -> io::Result<()> {
214    let Some(parent) = path.parent() else {
215        return Err(io::Error::new(
216            io::ErrorKind::InvalidInput,
217            "config path has no parent",
218        ));
219    };
220    async_fs::create_dir_all(parent).await?;
221    let temp_path = path.with_extension("toml.tmp");
222    async_fs::write(&temp_path, contents).await?;
223    async_fs::rename(temp_path, path).await
224}

436/// Loads the server configuration from the given path, or creates it if it doesn't exist.
437///
438pub fn load_or_create(path: &Path) -> Result<SteelConfig, String> {
439    let mut config = if path.exists() {
440        let config_str = fs::read_to_string(path)
441            .map_err(|e| format!("failed to read config file {}: {e}", path.display()))?;
442        let config: SteelConfig = toml::from_str(config_str.as_str())
443            .map_err(|e| format!("failed to parse config: {e}"))?;
444        validate(&config.server).map_err(|e| format!("failed to validate config: {e}"))?;
445        config
446    } else {
447        let parent = path
448            .parent()
449            .ok_or_else(|| format!("failed to get config directory for {}", path.display()))?;
450        fs::create_dir_all(parent).map_err(|e| {
451            format!(
452                "failed to create config directory {}: {e}",
453                parent.display()
454            )
455        })?;
456        fs::write(path, DEFAULT_CONFIG)
457            .map_err(|e| format!("failed to write config file {}: {e}", path.display()))?;
458        let config: SteelConfig = toml::from_str(DEFAULT_CONFIG)
459            .map_err(|e| format!("failed to parse default config: {e}"))?;
460        validate(&config.server).map_err(|e| format!("failed to validate default config: {e}"))?;
461        config
462    };
463
464    let worlds_path = path
465        .parent()
466        .ok_or_else(|| format!("failed to get config directory for {}", path.display()))?
467        .join("worlds.toml");
468    config.worlds = load_or_create_worlds(&worlds_path)?;
469    let groups_path = path
470        .parent()
471        .ok_or_else(|| format!("failed to get config directory for {}", path.display()))?
472        .join("groups.toml");
473    config.groups = load_or_create_groups(&groups_path)?;
474    config.groups_path = Some(groups_path);
475
476    // If icon file doesnt exist, write it
477    #[cfg(feature = "stand-alone")]
478    if config.server.use_favicon && !Path::new(&config.server.favicon).exists() {
479        fs::write(Path::new(&config.server.favicon), DEFAULT_FAVICON).map_err(|e| {
480            format!(
481                "failed to write favicon file {}: {e}",
482                config.server.favicon
483            )
484        })?;
485    }
486
487    Ok(config)
488}

490fn load_or_create_worlds(path: &Path) -> Result<WorldsConfig, String> {
491    if path.exists() {
492        let worlds_str = fs::read_to_string(path)
493            .map_err(|e| format!("failed to read worlds config file {}: {e}", path.display()))?;
494        toml::from_str(worlds_str.as_str())
495            .map_err(|e| format!("failed to parse worlds config {}: {e}", path.display()))
496    } else {
497        fs::write(path, DEFAULT_WORLDS)
498            .map_err(|e| format!("failed to write worlds config file {}: {e}", path.display()))?;
499        toml::from_str(DEFAULT_WORLDS)
500            .map_err(|e| format!("failed to parse default worlds config: {e}"))
501    }
502}

504fn load_or_create_groups(path: &Path) -> Result<PermissionGroupsConfig, String> {
505    let config: PermissionGroupsConfig = if path.exists() {
506        let contents = fs::read_to_string(path)
507            .map_err(|error| format!("failed to read groups config {}: {error}", path.display()))?;
508        toml::from_str(&contents)
509            .map_err(|error| format!("failed to parse groups config {}: {error}", path.display()))?
510    } else {
511        fs::write(path, DEFAULT_GROUPS).map_err(|error| {
512            format!("failed to write groups config {}: {error}", path.display())
513        })?;
514        toml::from_str(DEFAULT_GROUPS)
515            .map_err(|error| format!("failed to parse default groups config: {error}"))?
516    };
517    PermissionGroups::from_config(config.clone()).map_err(|error| {
518        format!(
519            "failed to validate groups config {}: {error}",
520            path.display()
521        )
522    })?;
523    Ok(config)
524}

526/// Validates the server configuration.
527///
528/// # Errors
529/// This function will return an error if the configuration is invalid.
530fn validate(config: &ServerConfig) -> Result<(), &'static str> {
531    validate_login_security(config.online_mode, config.encryption)?;
532    if !config.allow_extended_view_distance && !(1..=32).contains(&config.view_distance) {
533        return Err("View distance must in range 1..32");
534    }
535    if config.allow_extended_view_distance && !(1..=127).contains(&config.view_distance) {
536        return Err("View distance must in range 1..127");
537    }
538    if let Some(auth_server) = &config.auth_server {
539        let Ok(url) = Url::parse(auth_server) else {
540            return Err("auth_server must be an absolute URL");
541        };
542        if !matches!(url.scheme(), "http" | "https") {
543            return Err("auth_server must use http or https");
544        }
545    }
546    if let Some(profile_server) = &config.profile_server {
547        let Ok(url) = Url::parse(profile_server) else {
548            return Err("profile_server must be an absolute URL");
549        };
550        if !matches!(url.scheme(), "http" | "https") {
551            return Err("profile_server must use http or https");
552        }
553    }
554    if config.simulation_distance > config.view_distance {
555        return Err("Simulation distance must be less than or equal to view distance");
556    }
557    if let Some(compression) = config.compression {
558        if compression.threshold.get() < 256 {
559            return Err("Compression threshold must be greater than or equal to 256");
560        }
561        if !(1..=9).contains(&compression.level) {
562            return Err("Compression level must be between 1 and 9");
563        }
564    }
565    if config.enforce_secure_chat {
566        if !config.online_mode {
567            return Err("online_mode must be true when enforce_secure_chat is enabled");
568        }
569        if !config.encryption {
570            return Err("encryption must be true when enforce_secure_chat is enabled");
571        }
572    }
573    Ok(())
574}
