9/// An error that can occur during Mojang authentication.
10#[derive(Error, Debug)]
11pub enum AuthError {
12    /// Authentication servers are down.
13    #[error("Authentication servers are down")]
14    FailedResponse,
15    /// Failed to verify username.
16    #[error("Failed to verify username")]
17    UnverifiedUsername,
18    /// You are banned from Authentication servers.
19    #[error("You are banned from Authentication servers")]
20    Banned,
21    /// An error occurred with textures.
22    #[error("Texture Error {0}")]
23    TextureError(TextureError),
24    /// You have disallowed actions from Authentication servers.
25    #[error("You have disallowed actions from Authentication servers")]
26    DisallowedAction,
27    /// Failed to parse JSON into Game Profile.
28    #[error("Failed to parse JSON into Game Profile")]
29    FailedParse,
30    /// An unknown status code was returned.
31    #[error("Unknown Status Code {0}")]
32    UnknownStatusCode(StatusCode),
33}

55/// Authenticates a player with Mojang's servers.
56pub async fn mojang_authenticate(
57    username: &str,
58    server_hash: &str,
59) -> Result<GameProfile, AuthError> {
60    let cap = MOJANG_AUTH_URL.len() + SERVER_ID_ARG.len() + username.len() + server_hash.len();
61    let mut auth_url = String::with_capacity(cap);
62    auth_url += MOJANG_AUTH_URL;
63    auth_url += username;
64    auth_url += SERVER_ID_ARG;
65    auth_url += server_hash;
66
67    let response = reqwest::get(auth_url)
68        .await
69        .map_err(|_| AuthError::FailedResponse)?;
70
71    match response.status() {
72        StatusCode::OK => {}
73        StatusCode::NO_CONTENT => Err(AuthError::UnverifiedUsername)?,
74        other => Err(AuthError::UnknownStatusCode(other))?,
75    }
76
77    response.json().await.map_err(|_| AuthError::FailedParse)
78}

80/// Converts a signed bytes big endian to a hex string.
81#[must_use]
82pub fn signed_bytes_be_to_hex(bytes: &[u8]) -> String {
83    if bytes.is_empty() {
84        return "0".to_string();
85    }
86
87    // Find the first non-zero byte to handle cases like `[0x00, 0x1a]`
88    let first_digit = bytes.iter().position(|&b| b != 0);
89
90    // If all bytes are zero, the number is 0.
91    let Some(start_index) = first_digit else {
92        return "0".to_string();
93    };
94
95    let significant_bytes = &bytes[start_index..];
96    let is_negative = (significant_bytes[0] & 0x80) != 0;
97
98    if is_negative {
99        // Negative case: calculate two's complement of the original full byte array
100        // to preserve the correct number of bits for the calculation.
101        let mut inverted_bytes: Vec<u8> = bytes.iter().map(|b| !*b).collect();
102        for byte in inverted_bytes.iter_mut().rev() {
103            let (result, carry) = byte.overflowing_add(1);
104            *byte = result;
105            if !carry {
106                break;
107            }
108        }
109
110        // Now, find the first significant digit of the *result*
111        let mag_start_index = inverted_bytes.iter().position(|&b| b != 0).unwrap_or(0);
112
113        // Format the result with a leading '-'
114        format!("-{}", hex::encode(&inverted_bytes[mag_start_index..]))
115    } else {
116        // Positive case: just encode the significant bytes.
117        hex::encode(significant_bytes)
118    }
119}
