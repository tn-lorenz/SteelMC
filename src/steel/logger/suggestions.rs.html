16pub struct Completer {
17    pub enabled: bool,
18    pub error: bool,
19    pub selected: usize,
20    pub completed: String,
21    pub suggestions: Vec<String>,
22}
23impl Completer {
24    pub const fn new() -> Self {
25        Completer {
26            enabled: false,
27            error: false,
28            selected: 0,
29            completed: String::new(),
30            suggestions: vec![],
31        }
32    }
33}
34/// Modify suggestions
35impl Completer {
36    pub fn update(&mut self, out: &mut Output, pos: usize) {
37        let char_start = if out.text.is_empty() {
38            0
39        } else {
40            let (start, size) = out.char_pos(pos.saturating_sub(1));
41            start + size
42        };
43        // Gets the right chars
44        let command = &out.text[..char_start];
45
46        let Some(server) = SERVER.get() else {
47            self.completed = String::new();
48            self.selected = 0;
49            self.error = true;
50            return;
51        };
52        // Gets the suggested commands
53        self.suggestions = server
54            .command_dispatcher
55            .read()
56            .handle_suggestions(CommandSender::Console, command, server.clone())
57            .0
58            .into_iter()
59            .map(|suggestion| suggestion.text)
60            .collect();
61        if self.suggestions.is_empty() {
62            self.completed = String::new();
63            self.selected = 0;
64            self.error = true;
65        } else {
66            self.error = false;
67        }
68    }
69    pub fn rewrite(&mut self, out: &mut Output, dir: Move) -> Result<()> {
70        // Goes to the end
71        out.cursor_to(out.get_current_pos(), out.get_end())?;
72        // Clears
73        write!(out, "{}", Clear(ClearType::FromCursorDown))?;
74        if self.suggestions.is_empty() {
75            out.cursor_to(out.get_end(), out.get_current_pos())?;
76            out.flush()?;
77            return Ok(());
78        }
79
80        // Updates completion position
81        let len = self.suggestions.len();
82        match dir {
83            Move::Up => self.selected = (self.selected + len - 1) % len,
84            Move::Down => self.selected = (self.selected + 1) % len,
85            Move::None => (),
86        }
87        // Updates the screen width
88        let width = if let Ok((width, _)) = terminal::size() {
89            width as usize / 20
90        } else {
91            1
92        };
93        let grid_size = width * 3;
94        let start = (self.selected / grid_size) * grid_size;
95        let mut height = 0u16;
96        'outer: for w in 0..width {
97            for h in 0..3 {
98                let pos = start + w * 3 + h;
99                if pos >= self.suggestions.len() {
100                    break 'outer;
101                }
102
103                writeln!(out, "\r")?;
104                if w != 0 {
105                    write!(out, "{}", MoveRight(w as u16 * 20))?;
106                }
107
108                let color = if pos == self.selected {
109                    Yellow
110                } else {
111                    DarkGrey
112                };
113
114                write!(
115                    out,
116                    "{}{:<20}{}",
117                    SetForegroundColor(color),
118                    if self.suggestions[pos].len() > 20 {
119                        format!("{}...", &self.suggestions[pos][..17])
120                    } else {
121                        self.suggestions[pos].clone()
122                    },
123                    ResetColor
124                )?;
125                height += 1;
126            }
127            write!(out, "{}", MoveUp(3))?;
128            height = 0;
129        }
130        let y = height + out.get_end().1 as u16;
131        let x = out.get_current_pos().0;
132        if y != 0 {
133            write!(out, "{}", MoveUp(y))?;
134        }
135        write!(out, "\r{}", MoveRight(x as u16))?;
136
137        let char_pos = if out.is_at_start() {
138            0
139        } else {
140            let (pos, char) = out.char_pos(out.pos.saturating_sub(1));
141            pos + char
142        };
143        let text = if let Some(text) = out.text[..char_pos].split_whitespace().last()
144            && let Some(striped) = self.suggestions[self.selected].strip_prefix(text)
145        {
146            striped
147        } else {
148            &self.suggestions[self.selected]
149        };
150        self.completed = text.to_string();
151        out.flush()?;
152
153        if !out.is_at_end() {
154            return Ok(());
155        }
156        write!(
157            out,
158            "{SavePosition}{}{}{RestorePosition}",
159            SetForegroundColor(DarkGrey),
160            &self.completed
161        )?;
162        out.flush()
163    }
164}
